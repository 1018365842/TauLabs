174 void jump_to_app() {
175  const struct pios_board_info * bdinfo = &pios_board_info_blob;
176 
177  PIOS_LED_On(PIOS_LED_HEARTBEAT);
178  if (((*(__IO uint32_t*) bdinfo->fw_base) & 0x2FFE0000) == 0x20000000 ||
179  ((*(__IO uint32_t*) bdinfo->fw_base) & 0x1FFE0000) == 0x10000000) { /* Jump to user application */
180  FLASH_Lock();
181  RCC_APB2PeriphResetCmd(0xffffffff, ENABLE);
182  RCC_APB1PeriphResetCmd(0xffffffff, ENABLE);
183  RCC_APB2PeriphResetCmd(0xffffffff, DISABLE);
184  RCC_APB1PeriphResetCmd(0xffffffff, DISABLE);
185 
186  PIOS_USBHOOK_Deactivate();
187 
188  JumpAddress = *(__IO uint32_t*) (bdinfo->fw_base + 4);
189  Jump_To_Application = (pFunction) JumpAddress;
190  /* Initialize user application's Stack Pointer */
191  __set_MSP(*(__IO uint32_t*) bdinfo->fw_base);
192  Jump_To_Application();
193  } else {
194  DeviceState = failed_jump;
195  return;
196  }
197 }
198 uint32_t LedPWM(uint32_t pwm_period, uint32_t pwm_sweep_steps, uint32_t count) {
199  uint32_t curr_step = (count / pwm_period) % pwm_sweep_steps; /* 0 - pwm_sweep_steps */
200  uint32_t pwm_duty = pwm_period * curr_step / pwm_sweep_steps; /* fraction of pwm_period */
201 
202  uint32_t curr_sweep = (count / (pwm_period * pwm_sweep_steps)); /* ticks once per full sweep */
203  if (curr_sweep & 1) {
204  pwm_duty = pwm_period - pwm_duty; /* reverse direction in odd sweeps */
205  }
206  return ((count % pwm_period) > pwm_duty) ? 1 : 0;
207 }
