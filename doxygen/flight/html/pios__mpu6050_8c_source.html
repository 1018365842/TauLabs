177 static void PIOS_MPU6050_Config(struct pios_mpu60x0_cfg const *cfg)
178 {
179 #if defined(PIOS_MPU6050_SIMPLE_INIT_SEQUENCE)
180 
181  // Reset chip
182  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, PIOS_MPU60X0_PWRMGMT_IMU_RST);
183 
184  // Reset sensors signal path
185  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, PIOS_MPU60X0_USERCTL_GYRO_RST);
186 
187  // Give chip some time to initialize
188  PIOS_DELAY_WaitmS(10);
189 
190  //Power management configuration
191  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
192 
193  // User control
194  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, cfg->User_ctl);
195 
196  // Digital low-pass filter and scale
197  // set this before sample rate else sample rate calculation will fail
198  PIOS_MPU6050_SetLPF(cfg->default_filter);
199 
200  // Sample rate
201  PIOS_MPU6050_SetSampleRate(cfg->default_samplerate);
202 
203  // Set the gyro scale
204  PIOS_MPU6050_SetGyroRange(PIOS_MPU60X0_SCALE_500_DEG);
205 
206 #if defined(PIOS_MPU6050_ACCEL)
207  // Set the accel scale
208  PIOS_MPU6050_SetAccelRange(PIOS_MPU60X0_ACCEL_8G);
209 #endif /* PIOS_MPU6050_ACCEL */
210 
211  // Interrupt configuration
212  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
213 
214  // Interrupt enable
215  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
216 
217 #else /* PIOS_MPU6050_SIMPLE_INIT_SEQUENCE */
218 
219  /* This init sequence should really be dropped in favor of something
220  * less redundant but it seems to be hard to get it running well
221  * on all different targets.
222  */
223 
224  // Reset chip
225  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, 0x80 | cfg->Pwr_mgmt_clk);
226  do {
227  PIOS_DELAY_WaitmS(5);
228  } while (PIOS_MPU6050_GetReg(PIOS_MPU60X0_PWR_MGMT_REG) & 0x80);
229 
230  PIOS_DELAY_WaitmS(25);
231 
232  // Reset chip and fifo
233  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, 0x80 | 0x01 | 0x02 | 0x04);
234  do {
235  PIOS_DELAY_WaitmS(5);
236  } while (PIOS_MPU6050_GetReg(PIOS_MPU60X0_USER_CTRL_REG) & 0x07);
237 
238  PIOS_DELAY_WaitmS(25);
239 
240  //Power management configuration
241  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
242 
243  // Interrupt configuration
244  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
245 
246  // Interrupt enable
247  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
248 
249 #if defined(PIOS_MPU6050_ACCEL)
250  // Set the accel scale
251  PIOS_MPU6050_SetAccelRange(PIOS_MPU60X0_ACCEL_8G);
252 #endif
253 
254  // Digital low-pass filter and scale
255  // set this before sample rate else sample rate calculation will fail
256  PIOS_MPU6050_SetLPF(cfg->default_filter);
257 
258  // Sample rate
259  PIOS_MPU6050_SetSampleRate(cfg->default_samplerate);
260 
261  // Set the gyro scale
262  PIOS_MPU6050_SetGyroRange(PIOS_MPU60X0_SCALE_500_DEG);
263 
264  // User control
265  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, cfg->User_ctl);
266 
267  //Power management configuration
268  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
269 
270  // Interrupt configuration
271  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
272 
273  // Interrupt enable
274  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
275 
276 #endif /* PIOS_MPU6050_SIMPLE_INIT_SEQUENCE */

305 void PIOS_MPU6050_SetSampleRate(uint16_t samplerate_hz)
306 {
307  uint16_t filter_frequency = 8000;
308 
309  if (pios_mpu6050_dev->filter != PIOS_MPU60X0_LOWPASS_256_HZ)
310  filter_frequency = 1000;
311 
312  // limit samplerate to filter frequency
313  if (samplerate_hz > filter_frequency)
314  samplerate_hz = filter_frequency;
315 
316  // calculate divisor, round to nearest integeter
317  int32_t divisor = (int32_t)(((float)filter_frequency / samplerate_hz) + 0.5f) - 1;
318 
319  // limit resulting divisor to register value range
320  if (divisor < 0)
321  divisor = 0;
322 
323  if (divisor > 0xff)
324  divisor = 0xff;
325 
326  PIOS_MPU6050_SetReg(PIOS_MPU60X0_SMPLRT_DIV_REG, (uint8_t)divisor);
327 }

556 static void PIOS_MPU6050_Task(void *parameters)
557 {
558  while (1) {
559  //Wait for data ready interrupt
560  if (xSemaphoreTake(pios_mpu6050_dev->data_ready_sema, portMAX_DELAY) != pdTRUE)
561  continue;
562 
563  enum {
564  IDX_ACCEL_XOUT_H = 0,
565  IDX_ACCEL_XOUT_L,
566  IDX_ACCEL_YOUT_H,
567  IDX_ACCEL_YOUT_L,
568  IDX_ACCEL_ZOUT_H,
569  IDX_ACCEL_ZOUT_L,
570  IDX_TEMP_OUT_H,
571  IDX_TEMP_OUT_L,
572  IDX_GYRO_XOUT_H,
573  IDX_GYRO_XOUT_L,
574  IDX_GYRO_YOUT_H,
575  IDX_GYRO_YOUT_L,
576  IDX_GYRO_ZOUT_H,
577  IDX_GYRO_ZOUT_L,
578  BUFFER_SIZE,
579  };
580 
581  uint8_t mpu6050_rec_buf[BUFFER_SIZE];
582 
583  if (PIOS_MPU6050_Read(PIOS_MPU60X0_ACCEL_X_OUT_MSB, mpu6050_rec_buf, sizeof(mpu6050_rec_buf)) < 0) {
584  continue;
585  }
586 
587  // Rotate the sensor to OP convention. The datasheet defines X as towards the right
588  // and Y as forward. OP convention transposes this. Also the Z is defined negatively
589  // to our convention
590 
591 #if defined(PIOS_MPU6050_ACCEL)
592 
593  // Currently we only support rotations on top so switch X/Y accordingly
594  struct pios_sensor_accel_data accel_data;
595  struct pios_sensor_gyro_data gyro_data;
596 
597  switch (pios_mpu6050_dev->cfg->orientation) {
598  case PIOS_MPU60X0_TOP_0DEG:
599  accel_data.y = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
600  accel_data.x = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
601  gyro_data.y = (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
602  gyro_data.x = (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
603  break;
604  case PIOS_MPU60X0_TOP_90DEG:
605  accel_data.y = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
606  accel_data.x = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
607  gyro_data.y = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
608  gyro_data.x = (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
609  break;
610  case PIOS_MPU60X0_TOP_180DEG:
611  accel_data.y = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
612  accel_data.x = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
613  gyro_data.y = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
614  gyro_data.x = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
615  break;
616  case PIOS_MPU60X0_TOP_270DEG:
617  accel_data.y = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
618  accel_data.x = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
619  gyro_data.y = (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
620  gyro_data.x = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
621  break;
622  }
623 
624  gyro_data.z = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_ZOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_ZOUT_L]);
625  accel_data.z = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_ZOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_ZOUT_L]);
