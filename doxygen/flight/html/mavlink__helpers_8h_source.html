421 MAVLINK_HELPER uint8_t put_bitfield_n_by_index(int32_t b, uint8_t bits, uint8_t packet_index, uint8_t bit_index, uint8_t* r_bit_index, uint8_t* buffer)
422 {
423  uint16_t bits_remain = bits;
424  // Transform number into network order
425  int32_t v;
426  uint8_t i_bit_index, i_byte_index, curr_bits_n;
427 #if MAVLINK_NEED_BYTE_SWAP
428  union {
429  int32_t i;
430  uint8_t b[4];
431  } bin, bout;
432  bin.i = b;
433  bout.b[0] = bin.b[3];
434  bout.b[1] = bin.b[2];
435  bout.b[2] = bin.b[1];
436  bout.b[3] = bin.b[0];
437  v = bout.i;
438 #else
439  v = b;
440 #endif
441 
442  // buffer in
443  // 01100000 01000000 00000000 11110001
444  // buffer out
445  // 11110001 00000000 01000000 01100000
446 
447  // Existing partly filled byte (four free slots)
448  // 0111xxxx
449 
450  // Mask n free bits
451  // 00001111 = 2^0 + 2^1 + 2^2 + 2^3 = 2^n - 1
452  // = ((uint32_t)(1 << n)) - 1; // = 2^n - 1
453 
454  // Shift n bits into the right position
455  // out = in >> n;
456 
457  // Mask and shift bytes
458  i_bit_index = bit_index;
459  i_byte_index = packet_index;
460  if (bit_index > 0)
461  {
462  // If bits were available at start, they were available
463  // in the byte before the current index
464  i_byte_index--;
465  }
466 
467  // While bits have not been packed yet
468  while (bits_remain > 0)
469  {
470  // Bits still have to be packed
471  // there can be more than 8 bits, so
472  // we might have to pack them into more than one byte
473 
474  // First pack everything we can into the current 'open' byte
475  //curr_bits_n = bits_remain << 3; // Equals bits_remain mod 8
476  //FIXME
477  if (bits_remain <= (uint8_t)(8 - i_bit_index))
478  {
479  // Enough space
480  curr_bits_n = (uint8_t)bits_remain;
481  }
482  else
483  {
484  curr_bits_n = (8 - i_bit_index);
485  }
486 
487  // Pack these n bits into the current byte
488  // Mask out whatever was at that position with ones (xxx11111)
489  buffer[i_byte_index] &= (0xFF >> (8 - curr_bits_n));
490  // Put content to this position, by masking out the non-used part
491  buffer[i_byte_index] |= ((0x00 << curr_bits_n) & v);
492 
493  // Increment the bit index
494  i_bit_index += curr_bits_n;
495 
496  // Now proceed to the next byte, if necessary
497  bits_remain -= curr_bits_n;
498  if (bits_remain > 0)
499  {
500  // Offer another 8 bits / one byte
501  i_byte_index++;
502  i_bit_index = 0;
503  }
504  }
505 
506  *r_bit_index = i_bit_index;
507  // If a partly filled byte is present, mark this as consumed
508  if (i_bit_index != 7) i_byte_index++;
509  return i_byte_index - packet_index;
510 }
