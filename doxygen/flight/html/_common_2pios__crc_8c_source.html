132 uint8_t PIOS_CRC_updateCRC(uint8_t crc, const uint8_t* data, int32_t length)
133 {
134  // use registers for speed
135  register int32_t len = length;
136  register uint8_t crc8 = crc;
137  register const uint8_t *p = data;
138 
139  while (len--)
140  crc8 = crc_table[crc8 ^ *p++];
141 
142  return crc8;
143 }
