59 int32_t AlarmsSet(SystemAlarmsAlarmElem alarm, SystemAlarmsAlarmOptions severity)
60 {
61  SystemAlarmsData alarms;
62 
63  // Check that this is a valid alarm
64  if (alarm >= SYSTEMALARMS_ALARM_NUMELEM)
65  {
66  return -1;
67  }
68 
69  // Lock
70  xSemaphoreTakeRecursive(lock, portMAX_DELAY);
71 
72  // Read alarm and update its severity only if it was changed

76  alarms.Alarm[alarm] = severity;

78  }

181 static int32_t hasSeverity(SystemAlarmsAlarmOptions severity)
182 {
183  SystemAlarmsData alarms;
184  uint32_t n;
185 
186  // Lock
187  xSemaphoreTakeRecursive(lock, portMAX_DELAY);
188 
189  // Read alarms
190  SystemAlarmsGet(&alarms);
191 
192  // Go through alarms and check if any are of the given severity or higher
193  for (n = 0; n < SYSTEMALARMS_ALARM_NUMELEM; ++n)
194  {
195  if ( alarms.Alarm[n] >= severity)
196  {
197  xSemaphoreGiveRecursive(lock);
198  return 1;
199  }
200  }
201 
202  // If this point is reached then no alarms found
203  xSemaphoreGiveRecursive(lock);
204  return 0;
205 }
