81 void DataDownload(DownloadAction action) {
82  if ((DeviceState == downloading)) {
83 
84  uint8_t packetSize;
85  uint32_t offset;
86  uint32_t partoffset;
87  SendBuffer[0] = 0x01;
88  SendBuffer[1] = Download;
89  SendBuffer[2] = downPacketCurrent >> 24;
90  SendBuffer[3] = downPacketCurrent >> 16;
91  SendBuffer[4] = downPacketCurrent >> 8;
92  SendBuffer[5] = downPacketCurrent;
93  if (downPacketCurrent == downPacketTotal - 1) {
94  packetSize = downSizeOfLastPacket;
95  } else {
96  packetSize = 14;
97  }
98  for (uint8_t x = 0; x < packetSize; ++x) {
99  partoffset = (downPacketCurrent * 14 * 4) + (x * 4);
100  offset = baseOfAdressType(downType) + partoffset;
101  if (!flash_read(SendBuffer + (6 + x * 4), offset,
102  currentProgrammingDestination)) {
103  DeviceState = Last_operation_failed;
104  }
105  }
106  downPacketCurrent = downPacketCurrent + 1;
107  if (downPacketCurrent > downPacketTotal - 1) {
108  DeviceState = Last_operation_Success;
109  Aditionals = (uint32_t) Download;
110  }
111  sendData(SendBuffer + 1, 63);
112  }

114 void processComand(uint8_t *xReceive_Buffer) {
115 
116  Command = xReceive_Buffer[COMMAND];
117 #ifdef DEBUG_SSP
118  char str[63]= {0};
119  sprintf(str,"Received COMMAND:%d|",Command);
120  PIOS_COM_SendString(PIOS_COM_TELEM_USB,str);
121 #endif
122  EchoReqFlag = (Command >> 7);
123  EchoAnsFlag = (Command >> 6) & 0x01;
124  StartFlag = (Command >> 5) & 0x01;
125  Count = xReceive_Buffer[COUNT] << 24;
126  Count += xReceive_Buffer[COUNT + 1] << 16;
127  Count += xReceive_Buffer[COUNT + 2] << 8;
128  Count += xReceive_Buffer[COUNT + 3];
129 
130  Data = xReceive_Buffer[DATA] << 24;
131  Data += xReceive_Buffer[DATA + 1] << 16;
132  Data += xReceive_Buffer[DATA + 2] << 8;
133  Data += xReceive_Buffer[DATA + 3];
134  Data0 = xReceive_Buffer[DATA];
135  Data1 = xReceive_Buffer[DATA + 1];
136  Data2 = xReceive_Buffer[DATA + 2];
137  Data3 = xReceive_Buffer[DATA + 3];
138  Command = Command & 0b00011111;
139 
140  if (EchoReqFlag == 1) {
141  memcpy(echoBuffer, xReceive_Buffer, 64);
142  }
143  switch (Command) {
144  case EnterDFU:
145  if (((DeviceState == BLidle) && (Data0 < numberOfDevices))
146  || (DeviceState == DFUidle)) {
147  if (Data0 > 0)
148  OPDfuIni(true);
149  DeviceState = DFUidle;
150  currentProgrammingDestination = devicesTable[Data0].programmingType;
151  currentDeviceCanRead = devicesTable[Data0].readWriteFlags & 0x01;
152  currentDeviceCanWrite = devicesTable[Data0].readWriteFlags >> 1
153  & 0x01;
154  currentDevice = devicesTable[Data0];
155  uint8_t result = 0;
156  switch (currentProgrammingDestination) {
157  case Self_flash:
158  result = PIOS_BL_HELPER_FLASH_Ini();
159  break;
160  case Remote_flash_via_spi:
161  result = true;
162  break;
163  default:
164  DeviceState = Last_operation_failed;
165  Aditionals = (uint16_t) Command;
166  }
167  if (result != 1) {
168  DeviceState = Last_operation_failed;
169  Aditionals = (uint32_t) Command;
170  }
171  }
172  break;
173  case Upload:
174  if ((DeviceState == DFUidle) || (DeviceState == uploading)) {
175  if ((StartFlag == 1) && (Next_Packet == 0)) {
176  TransferType = Data0;
177  SizeOfTransfer = Count;
178  Next_Packet = 1;
179  Expected_CRC = Data2 << 24;
180  Expected_CRC += Data3 << 16;
181  Expected_CRC += xReceive_Buffer[DATA + 4] << 8;
182  Expected_CRC += xReceive_Buffer[DATA + 5];
183  SizeOfLastPacket = Data1;
184 
185  if (isBiggerThanAvailable(TransferType, (SizeOfTransfer - 1)
186  * 14 * 4 + SizeOfLastPacket * 4) == true) {
187  DeviceState = outsideDevCapabilities;
188  Aditionals = (uint32_t) Command;
189  } else {
190  uint8_t result = 1;
191  if (TransferType == FW) {
192  switch (currentProgrammingDestination) {
193  case Self_flash:
194  result = PIOS_BL_HELPER_FLASH_Start();
195  break;
196  case Remote_flash_via_spi:
197  result = false;
198  break;
199  default:
200  break;
201  }
202  }
203  if (result != 1) {
204  DeviceState = Last_operation_failed;
205  Aditionals = (uint32_t) Command;
206  } else {
207 
208  DeviceState = uploading;
209  }
210  }
211  } else if ((StartFlag != 1) && (Next_Packet != 0)) {
212  if (Count > SizeOfTransfer) {
213  DeviceState = too_many_packets;
214  Aditionals = Count;
215  } else if (Count == Next_Packet - 1) {
216  uint8_t numberOfWords = 14;
217  if (Count == SizeOfTransfer - 1)//is this the last packet?
218  {
219  numberOfWords = SizeOfLastPacket;
220  }
221  uint8_t result = 0;
222  switch (currentProgrammingDestination) {
223  case Self_flash:
224  for (uint8_t x = 0; x < numberOfWords; ++x) {
225  offset = 4 * x;
226  Data = xReceive_Buffer[DATA + offset] << 24;
227  Data += xReceive_Buffer[DATA + 1 + offset] << 16;
228  Data += xReceive_Buffer[DATA + 2 + offset] << 8;
229  Data += xReceive_Buffer[DATA + 3 + offset];
230  aux = baseOfAdressType(TransferType) + (uint32_t)(
231  Count * 14 * 4 + x * 4);
232  result = 0;
233  for (int retry = 0; retry < MAX_WRI_RETRYS; ++retry) {
234  if (result == 0) {
235  result = (FLASH_ProgramWord(aux, Data)
236  == FLASH_COMPLETE) ? 1 : 0;
237  }
238  }
239  }
240  break;
241  case Remote_flash_via_spi:
242  result = false; // No support for this for the PipX
243  break;
244  default:
245  result = 0;
246  break;
247  }
248  if (result != 1) {
249  DeviceState = Last_operation_failed;
250  Aditionals = (uint32_t) Command;
251  }
252 
253  ++Next_Packet;
254  } else {
255  DeviceState = wrong_packet_received;
256  Aditionals = Count;
257  }
258  } else {
259  DeviceState = Last_operation_failed;
260  Aditionals = (uint32_t) Command;
261  }
262  }
263  break;
264  case Req_Capabilities:
265  OPDfuIni(true);
266  Buffer[0] = 0x01;
267  Buffer[1] = Rep_Capabilities;
268  if (Data0 == 0) {
269  Buffer[2] = 0;
270  Buffer[3] = 0;
271  Buffer[4] = 0;
272  Buffer[5] = 0;
273  Buffer[6] = 0;
274  Buffer[7] = numberOfDevices;
275  uint16_t WRFlags = 0;
276  for (int x = 0; x < numberOfDevices; ++x) {
277  WRFlags = ((devicesTable[x].readWriteFlags << (x * 2))
278  | WRFlags);
279  }
280  Buffer[8] = WRFlags >> 8;
281  Buffer[9] = WRFlags;
282  } else {
283  Buffer[2] = devicesTable[Data0 - 1].sizeOfCode >> 24;
284  Buffer[3] = devicesTable[Data0 - 1].sizeOfCode >> 16;
285  Buffer[4] = devicesTable[Data0 - 1].sizeOfCode >> 8;
286  Buffer[5] = devicesTable[Data0 - 1].sizeOfCode;
287  Buffer[6] = Data0;
288  Buffer[7] = devicesTable[Data0 - 1].BL_Version;
289  Buffer[8] = devicesTable[Data0 - 1].sizeOfDescription;
290  Buffer[9] = devicesTable[Data0 - 1].devID;
291  Buffer[10] = devicesTable[Data0 - 1].FW_Crc >> 24;
292  Buffer[11] = devicesTable[Data0 - 1].FW_Crc >> 16;
293  Buffer[12] = devicesTable[Data0 - 1].FW_Crc >> 8;
294  Buffer[13] = devicesTable[Data0 - 1].FW_Crc;
295  Buffer[14] = devicesTable[Data0 - 1].devID >> 8;
296  Buffer[15] = devicesTable[Data0 - 1].devID;
297  }
298  sendData(Buffer + 1, 63);
299  break;
300  case JumpFW:
301  if (Data == 0x5AFE) {
302  /* Force board into safe mode */
303  PIOS_IAP_WriteBootCount(0xFFFF);
304  }
305  FLASH_Lock();
306  JumpToApp = 1;
307  break;
308  case Reset:
309  PIOS_SYS_Reset();
310  break;
311  case Abort_Operation:
312  Next_Packet = 0;
313  DeviceState = DFUidle;
314  JumpToApp = 0;
315  break;
316 
317  case Op_END:
318  if (DeviceState == uploading) {
319  if (Next_Packet - 1 == SizeOfTransfer) {
320  Next_Packet = 0;
321  if ((TransferType != FW) || (Expected_CRC == CalcFirmCRC())) {
322  DeviceState = Last_operation_Success;
323  } else {
324  DeviceState = CRC_Fail;
325  }
326  }
327  if (Next_Packet - 1 < SizeOfTransfer) {
328  Next_Packet = 0;
329  DeviceState = too_few_packets;
330  }
331  }
332  break;
333  case Download_Req:
334 #ifdef DEBUG_SSP
335  sprintf(str,"COMMAND:DOWNLOAD_REQ 1 Status=%d|",DeviceState);
336  PIOS_COM_SendString(PIOS_COM_TELEM_USB,str);
337 #endif
338  if (DeviceState == DFUidle) {
339 #ifdef DEBUG_SSP
340  PIOS_COM_SendString(PIOS_COM_TELEM_USB,"COMMAND:DOWNLOAD_REQ 1|");
341 #endif
342  downType = Data0;
343  downPacketTotal = Count;
344  downSizeOfLastPacket = Data1;
345  if (isBiggerThanAvailable(downType, (downPacketTotal - 1) * 14 * 4
346  + downSizeOfLastPacket * 4) == 1) {
347  DeviceState = outsideDevCapabilities;
348  Aditionals = (uint32_t) Command;
349 
350  } else {
351  downPacketCurrent = 0;
352  DeviceState = downloading;
353  }
354  } else {
355  DeviceState = Last_operation_failed;
356  Aditionals = (uint32_t) Command;
357  }
358  break;
359 
360  case Status_Request:
361  Buffer[0] = 0x01;
362  Buffer[1] = Status_Rep;
363  if (DeviceState == wrong_packet_received) {
364  Buffer[2] = Aditionals >> 24;
365  Buffer[3] = Aditionals >> 16;
366  Buffer[4] = Aditionals >> 8;
367  Buffer[5] = Aditionals;
368  } else {
369  Buffer[2] = 0;
370  Buffer[3] = ((uint16_t) Aditionals) >> 8;
371  Buffer[4] = ((uint16_t) Aditionals);
372  Buffer[5] = 0;
373  }
374  Buffer[6] = DeviceState;
375  Buffer[7] = 0;
376  Buffer[8] = 0;
377  Buffer[9] = 0;
378  sendData(Buffer + 1, 63);
379  if (DeviceState == Last_operation_Success) {
380  DeviceState = DFUidle;
381  }
382  break;
383  case Status_Rep:
384 
385  break;
386 
387  }
388  if (EchoReqFlag == 1) {
389  echoBuffer[0] = echoBuffer[0] | (1 << 6);
390  sendData(echoBuffer, 63);
391  }
392  return;

394 void OPDfuIni(uint8_t discover) {
