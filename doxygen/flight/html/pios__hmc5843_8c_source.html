223 static void PIOS_HMC5843_Config(PIOS_HMC5843_ConfigTypeDef * HMC5843_Config_Struct)
224 {
225  uint8_t CRTLA = 0x00;
226  uint8_t CRTLB = 0x00;
227  uint8_t MODE = 0x00;
228 
229  CRTLA |= (uint8_t) (HMC5843_Config_Struct->M_ODR | HMC5843_Config_Struct->Meas_Conf);
230  CRTLB |= (uint8_t) (HMC5843_Config_Struct->Gain);
231  MODE |= (uint8_t) (HMC5843_Config_Struct->Mode);
232 
233  // CRTL_REGA
234  while (!PIOS_HMC5843_Write(PIOS_HMC5843_CONFIG_REG_A, CRTLA)) ;
235 
236  // CRTL_REGB
237  while (!PIOS_HMC5843_Write(PIOS_HMC5843_CONFIG_REG_B, CRTLB)) ;
238 
239  // Mode register
240  while (!PIOS_HMC5843_Write(PIOS_HMC5843_MODE_REG, MODE)) ;
241 }

246 void PIOS_HMC5843_ReadMag(int16_t out[3])
247 {
248  uint8_t buffer[6];
249  uint8_t crtlB;
250 
251  pios_hmc5843_data_ready = false;
252 
253  while (!PIOS_HMC5843_Read(PIOS_HMC5843_CONFIG_REG_B, &crtlB, 1)) ;
254  while (!PIOS_HMC5843_Read(PIOS_HMC5843_DATAOUT_XMSB_REG, buffer, 6)) ;
255 
256  switch (crtlB & 0xE0) {
257  case 0x00:
258  for (int i = 0; i < 3; i++)
259  out[i] = ((int16_t) ((uint16_t) buffer[2 * i] << 8)
260  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_0_7Ga;
261  break;
262  case 0x20:
263  for (int i = 0; i < 3; i++)
264  out[i] = ((int16_t) ((uint16_t) buffer[2 * i] << 8)
265  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_1Ga;
266  break;
267  case 0x40:
268  for (int i = 0; i < 3; i++)
269  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
270  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_1_5Ga;
271  break;
272  case 0x60:
273  for (int i = 0; i < 3; i++)
274  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
275  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_2Ga;
276  break;
277  case 0x80:
278  for (int i = 0; i < 3; i++)
279  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
280  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_3_2Ga;
281  break;
282  case 0xA0:
283  for (int i = 0; i < 3; i++)
284  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
285  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_3_8Ga;
286  break;
287  case 0xC0:
288  for (int i = 0; i < 3; i++)
289  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
290  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_4_5Ga;
291  break;
292  case 0xE0:
293  for (int i = 0; i < 3; i++)
294  out[i] = (int16_t) (((uint16_t) buffer[2 * i] << 8)
295  + buffer[2 * i + 1]) * 1000 / PIOS_HMC5843_Sensitivity_6_5Ga;
296  break;
297  }
