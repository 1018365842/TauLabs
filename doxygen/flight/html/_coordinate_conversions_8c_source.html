36 // ****** convert Lat,Lon,Alt to ECEF ************
37 void LLA2ECEF(float LLA[3], float ECEF[3])
38 {
39  const float a = 6378137.0f; // Equatorial Radius
40  const float e = 8.1819190842622e-2f; // Eccentricity
41  float sinLat, sinLon, cosLat, cosLon;
42  float N;
43 
44  sinLat = sinf(DEG2RAD * LLA[0]);
45  sinLon = sinf(DEG2RAD * LLA[1]);
46  cosLat = cosf(DEG2RAD * LLA[0]);
47  cosLon = cosf(DEG2RAD * LLA[1]);
48 
49  N = a / sqrtf(1.0f - e * e * sinLat * sinLat); //prime vertical radius of curvature
50 
51  ECEF[0] = (N + LLA[2]) * cosLat * cosLon;
52  ECEF[1] = (N + LLA[2]) * cosLat * sinLon;
53  ECEF[2] = ((1 - e * e) * N + LLA[2]) * sinLat;
54 }

56 // ****** convert ECEF to Lat,Lon,Alt (ITERATIVE!) *********
57 uint16_t ECEF2LLA(float ECEF[3], float LLA[3])
58 {
68  const float a = 6378137.0f; // Equatorial Radius
69  const float e = 8.1819190842622e-2f; // Eccentricity
70  float x = ECEF[0], y = ECEF[1], z = ECEF[2];
71  float Lat, N, NplusH, delta, esLat;
72  uint16_t iter;
73 #define MAX_ITER 10 // should not take more than 5 for valid coordinates
74 #define ACCURACY 1.0e-11f // used to be e-14, but we don't need sub micrometer exact calculations
75 
76  LLA[1] = RAD2DEG * atan2f(y, x);
77  Lat = DEG2RAD * LLA[0];
78  esLat = e * sinf(Lat);
79  N = a / sqrtf(1 - esLat * esLat);
80  NplusH = N + LLA[2];
81  delta = 1;
82  iter = 0;
83 
84  while (((delta > ACCURACY) || (delta < -ACCURACY))
85  && (iter < MAX_ITER)) {
86  delta = Lat - atanf(z / (sqrtf(x * x + y * y) * (1 - (N * e * e / NplusH))));
87  Lat = Lat - delta;
88  esLat = e * sinf(Lat);
89  N = a / sqrtf(1 - esLat * esLat);
90  NplusH = sqrtf(x * x + y * y) / cosf(Lat);
91  iter += 1;
92  }
93 
94  LLA[0] = RAD2DEG * Lat;
95  LLA[2] = NplusH - N;
96 
97  return (iter < MAX_ITER);
98 }

101 void RneFromLLA(float LLA[3], float Rne[3][3])
102 {
103  float sinLat, sinLon, cosLat, cosLon;
104

106  sinLon = (float)sinf(DEG2RAD * LLA[1]);
107  cosLat = (float)cosf(DEG2RAD * LLA[0]);
108  cosLon = (float)cosf(DEG2RAD * LLA[1]);
109

113  Rne[1][0] = -sinLon;
114  Rne[1][1] = cosLon;
115  Rne[1][2] = 0;
116  Rne[2][0] = -cosLat * cosLon;
117  Rne[2][1] = -cosLat * sinLon;
118  Rne[2][2] = -sinLat;

121 // ****** find roll, pitch, yaw from quaternion ********
122 void Quaternion2RPY(const float q[4], float rpy[3])
123 {
124  float R13, R11, R12, R23, R33;
125  float q0s = q[0] * q[0];
126  float q1s = q[1] * q[1];
127  float q2s = q[2] * q[2];
128  float q3s = q[3] * q[3];
129 
130  R13 = 2.0f * (q[1] * q[3] - q[0] * q[2]);
131  R11 = q0s + q1s - q2s - q3s;
132  R12 = 2.0f * (q[1] * q[2] + q[0] * q[3]);
133  R23 = 2.0f * (q[2] * q[3] + q[0] * q[1]);
134  R33 = q0s - q1s - q2s + q3s;
135 
136  rpy[1] = RAD2DEG * asinf(-R13); // pitch always between -pi/2 to pi/2
137  rpy[2] = RAD2DEG * atan2f(R12, R11);
138  rpy[0] = RAD2DEG * atan2f(R23, R33);
139 
140  //TODO: consider the cases where |R13| ~= 1, |pitch| ~= pi/2
141 }

143 // ****** find quaternion from roll, pitch, yaw ********
144 void RPY2Quaternion(const float rpy[3], float q[4])
145 {
146  float phi, theta, psi;
147  float cphi, sphi, ctheta, stheta, cpsi, spsi;
148 
149  phi = DEG2RAD * rpy[0] / 2;
150  theta = DEG2RAD * rpy[1] / 2;
151  psi = DEG2RAD * rpy[2] / 2;
152  cphi = cosf(phi);
153  sphi = sinf(phi);
154  ctheta = cosf(theta);
155  stheta = sinf(theta);
156  cpsi = cosf(psi);
157  spsi = sinf(psi);
158 
159  q[0] = cphi * ctheta * cpsi + sphi * stheta * spsi;
160  q[1] = sphi * ctheta * cpsi - cphi * stheta * spsi;
161  q[2] = cphi * stheta * cpsi + sphi * ctheta * spsi;
162  q[3] = cphi * ctheta * spsi - sphi * stheta * cpsi;
163 
164  if (q[0] < 0) { // q0 always positive for uniqueness
165  q[0] = -q[0];
166  q[1] = -q[1];
167  q[2] = -q[2];
168  q[3] = -q[3];
169  }
170 }

172 //** Find Rbe, that rotates a vector from earth fixed to body frame, from quaternion **
173 void Quaternion2R(float q[4], float Rbe[3][3])
174 {
175 
176  float q0s = q[0] * q[0], q1s = q[1] * q[1], q2s = q[2] * q[2], q3s = q[3] * q[3];
177 
178  Rbe[0][0] = q0s + q1s - q2s - q3s;
179  Rbe[0][1] = 2 * (q[1] * q[2] + q[0] * q[3]);
180  Rbe[0][2] = 2 * (q[1] * q[3] - q[0] * q[2]);
181  Rbe[1][0] = 2 * (q[1] * q[2] - q[0] * q[3]);
182  Rbe[1][1] = q0s - q1s + q2s - q3s;
183  Rbe[1][2] = 2 * (q[2] * q[3] + q[0] * q[1]);
184  Rbe[2][0] = 2 * (q[1] * q[3] + q[0] * q[2]);
185  Rbe[2][1] = 2 * (q[2] * q[3] - q[0] * q[1]);
186  Rbe[2][2] = q0s - q1s - q2s + q3s;
187 }

189 //** Find Rbe, that rotates a vector from earth fixed to body frame, from euler angles, using Tait-Bryan
190 //** convention, i.e. Rbe=Rot_x(roll)*Rot_y(pitch)*Rot_z(yaw) **
191 void Euler2R(float rpy[3], float Rbe[3][3])
192 {
193 
194  float sF = sinf(rpy[0]), cF = cosf(rpy[0]);
195  float sT = sinf(rpy[1]), cT = cosf(rpy[1]);
196  float sP = sinf(rpy[2]), cP = cosf(rpy[2]);
197 
198  Rbe[0][0] = cT*cP;
199  Rbe[0][1] = cT*sP;
200  Rbe[0][2] = -sT;
201  Rbe[1][0] = sF*sT*cP - cF*sP;
202  Rbe[1][1] = sF*sT*sP + cF*cP;
203  Rbe[1][2] = cT*sF;
204  Rbe[2][0] = cF*sT*cP + sF*sP;
205  Rbe[2][1] = cF*sT*sP - sF*cP;
206  Rbe[2][2] = cT*cF;
207 }

240 // ****** convert Rotation Matrix to Quaternion ********
241 // ****** if R converts from e to b, q is rotation from e to b ****
242 void R2Quaternion(float R[3][3], float q[4])
243 {
244  float m[4], mag;
245  uint8_t index,i;
246 
247  m[0] = 1 + R[0][0] + R[1][1] + R[2][2];
248  m[1] = 1 + R[0][0] - R[1][1] - R[2][2];
249  m[2] = 1 - R[0][0] + R[1][1] - R[2][2];
250  m[3] = 1 - R[0][0] - R[1][1] + R[2][2];
251 
252  // find maximum divisor
253  index = 0;
254  mag = m[0];
255  for (i=1;i<4;i++){
256  if (m[i] > mag){
257  mag = m[i];
258  index = i;
259  }
260  }
261  mag = 2*sqrtf(mag);
262 
263  if (index == 0) {
264  q[0] = mag/4;
265  q[1] = (R[1][2]-R[2][1])/mag;
266  q[2] = (R[2][0]-R[0][2])/mag;
267  q[3] = (R[0][1]-R[1][0])/mag;
268  }
269  else if (index == 1) {
270  q[1] = mag/4;
271  q[0] = (R[1][2]-R[2][1])/mag;
272  q[2] = (R[0][1]+R[1][0])/mag;
273  q[3] = (R[0][2]+R[2][0])/mag;
274  }
275  else if (index == 2) {
276  q[2] = mag/4;
277  q[0] = (R[2][0]-R[0][2])/mag;
278  q[1] = (R[0][1]+R[1][0])/mag;
279  q[3] = (R[1][2]+R[2][1])/mag;
280  }
281  else {
282  q[3] = mag/4;
283  q[0] = (R[0][1]-R[1][0])/mag;
284  q[1] = (R[0][2]+R[2][0])/mag;
285  q[2] = (R[1][2]+R[2][1])/mag;
286  }
287 
288  // q0 positive, i.e. angle between pi and -pi
289  if (q[0] < 0){
290  q[0] = -q[0];
291  q[1] = -q[1];
292  q[2] = -q[2];
293  q[3] = -q[3];
294  }
295 }

297 // ****** Rotation Matrix from Two Vector Directions ********
298 // ****** given two vector directions (v1 and v2) known in two frames (b and e) find Rbe ***
299 // ****** solution is approximate if can't be exact ***
300 uint8_t RotFrom2Vectors(const float v1b[3], const float v1e[3], const float v2b[3], const float v2e[3], float Rbe[3][3])
301 {
302  float Rib[3][3], Rie[3][3];
303  float mag;
304  uint8_t i,j,k;
305 
306  // identity rotation in case of error
307  for (i=0;i<3;i++){
308  for (j=0;j<3;j++)
309  Rbe[i][j]=0;
310  Rbe[i][i]=1;
311  }
312 
313  // The first rows of rot matrices chosen in direction of v1
314  mag = VectorMagnitude(v1b);
315  if (fabs(mag) < 1e-30)
316  return (-1);
317  for (i=0;i<3;i++)
318  Rib[0][i]=v1b[i]/mag;
319 
320  mag = VectorMagnitude(v1e);
321  if (fabs(mag) < 1e-30)
322  return (-1);
323  for (i=0;i<3;i++)
324  Rie[0][i]=v1e[i]/mag;
325 
326  // The second rows of rot matrices chosen in direction of v1xv2
327  CrossProduct(v1b,v2b,&Rib[1][0]);
328  mag = VectorMagnitude(&Rib[1][0]);
329  if (fabs(mag) < 1e-30)
330  return (-1);
331  for (i=0;i<3;i++)
332  Rib[1][i]=Rib[1][i]/mag;
333 
334  CrossProduct(v1e,v2e,&Rie[1][0]);
335  mag = VectorMagnitude(&Rie[1][0]);
336  if (fabs(mag) < 1e-30)
337  return (-1);
338  for (i=0;i<3;i++)
339  Rie[1][i]=Rie[1][i]/mag;
340 
341  // The third rows of rot matrices are XxY (Row1xRow2)
342  CrossProduct(&Rib[0][0],&Rib[1][0],&Rib[2][0]);
343  CrossProduct(&Rie[0][0],&Rie[1][0],&Rie[2][0]);
344 
345  // Rbe = Rbi*Rie = Rib'*Rie
346  for (i=0;i<3;i++)
347  for(j=0;j<3;j++){
348  Rbe[i][j]=0;
349  for(k=0;k<3;k++)
350  Rbe[i][j] += Rib[k][i]*Rie[k][j];
351  }
352 
353  return 1;
354 }

356 void Rv2Rot(float Rv[3], float R[3][3])
357 {
358  // Compute rotation matrix from a rotation vector
359  // To save .text space, uses Quaternion2R()
360  float q[4];
361 
362  float angle = VectorMagnitude(Rv);
363  if (angle <= 0.00048828125f) {
364  // angle < sqrt(2*machine_epsilon(float)), so flush cos(x) to 1.0f
365  q[0] = 1.0f;
366 
367  // and flush sin(x/2)/x to 0.5
368  q[1] = 0.5f*Rv[0];
369  q[2] = 0.5f*Rv[1];
370  q[3] = 0.5f*Rv[2];
371  // This prevents division by zero, while retaining full accuracy
372  }
373  else {
374  q[0] = cosf(angle*0.5f);
375  float scale = sinf(angle*0.5f) / angle;
376  q[1] = scale*Rv[0];
377  q[2] = scale*Rv[1];
378  q[3] = scale*Rv[2];
379  }
380 
381  Quaternion2R(q, R);
382 }
