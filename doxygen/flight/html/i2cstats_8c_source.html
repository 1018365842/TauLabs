48 int32_t I2CStatsInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(I2CSTATS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(I2CSTATS_OBJID,
56  I2CSTATS_ISSINGLEINST, I2CSTATS_ISSETTINGS, I2CSTATS_NUMBYTES, &I2CStatsSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }
