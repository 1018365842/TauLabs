74 static void PIOS_Board_configure_com (const struct pios_usart_cfg *usart_port_cfg, size_t rx_buf_len, size_t tx_buf_len,
75  const struct pios_com_driver *com_driver, uintptr_t *pios_com_id)
76 {
77  uint32_t pios_usart_id;

79  PIOS_Assert(0);
80  }
81 
82  uint8_t * rx_buffer;
83  if (rx_buf_len > 0) {
84  rx_buffer = (uint8_t *) pvPortMalloc(rx_buf_len);
85  PIOS_Assert(rx_buffer);
86  } else {
87  rx_buffer = NULL;
88  }
89 
90  uint8_t * tx_buffer;
91  if (tx_buf_len > 0) {
92  tx_buffer = (uint8_t *) pvPortMalloc(tx_buf_len);
93  PIOS_Assert(tx_buffer);
94  } else {
95  tx_buffer = NULL;
96  }
97 
98  if (PIOS_COM_Init(pios_com_id, com_driver, pios_usart_id,
99  rx_buffer, rx_buf_len,
100  tx_buffer, tx_buf_len)) {
101  PIOS_Assert(0);
102  }
103 }
