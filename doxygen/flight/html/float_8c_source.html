93 float_op(pPmObj_t px, pPmObj_t py, pPmObj_t *r_pn, int8_t op)
94 {
95  float x;
96  float y;
97  float r;
98  PmReturn_t retval;
99 
100  /* Raise TypeError if args aren't ints or floats */
101  if (((OBJ_GET_TYPE(px) != OBJ_TYPE_INT)
102  && (OBJ_GET_TYPE(px) != OBJ_TYPE_FLT))
103  || ((OBJ_GET_TYPE(py) != OBJ_TYPE_INT)
104  && (OBJ_GET_TYPE(py) != OBJ_TYPE_FLT)))
105  {
106  PM_RAISE(retval, PM_RET_EX_TYPE);
107  return retval;
108  }
109 
110  /* Get the values as floats */
111  if (OBJ_GET_TYPE(px) == OBJ_TYPE_INT)
112  {
113  x = (float)((pPmInt_t)px)->val;
114  }
115  else
116  {
117  x = ((pPmFloat_t) px)->val;
118  }
119 
120  if (OBJ_GET_TYPE(py) == OBJ_TYPE_INT)
121  {
122  y = (float)((pPmInt_t)py)->val;
123  }
124  else
125  {
126  y = ((pPmFloat_t) py)->val;
127  }
128 
129  /* Raise ZeroDivisionError if denominator is zero */
130  if ((y == 0.0) && ((op == '/') || (op == '%')))
131  {
132  PM_RAISE(retval, PM_RET_EX_ZDIV);
133  return retval;
134  }
135 
136  /* Calculate x raised to y */
137  switch (op)
138  {
139  /* *INDENT-OFF* */
140  case '+': r = x + y; break;
141  case '-': r = x - y; break;
142  case '*': r = x * y; break;
143  case '/': r = x / y; break;
144  case '%': r = fmodf(x, y); break;
145  case 'P': r = powf(x, y); break;
146  default: r = 0.0; break;
147  /* *INDENT-ON* */
148  }
149 
150  retval = float_new(r, r_pn);
151 
152  return retval;
153 }
