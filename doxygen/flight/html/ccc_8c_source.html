50 /*
51  * Correct sensor drift, using the 3C approach by J. Cotton
52  */
53 void CottonComplementaryCorrection(float *accels, float *gyros, const float delT, GlobalAttitudeVariables *glblAtt, float *accel_err_b)
54 {
55  float grav_b[3];
56 
57  // Rotate normalized gravity reference vector to body frame and cross with measured acceleration
58  grav_b[0] = -(2 * (glblAtt->q[1] * glblAtt->q[3] - glblAtt->q[0] * glblAtt->q[2]));
59  grav_b[1] = -(2 * (glblAtt->q[2] * glblAtt->q[3] + glblAtt->q[0] * glblAtt->q[1]));
60  grav_b[2] = -(glblAtt->q[0] * glblAtt->q[0] - glblAtt->q[1] * glblAtt->q[1] -
61  glblAtt->q[2] * glblAtt->q[2] + glblAtt->q[3] * glblAtt->q[3]);
62  CrossProduct((const float *)accels, (const float *)grav_b, accel_err_b);
63 
64  // Account for accel magnitude
65  float accel_mag = VectorMagnitude(accels);
66  if (accel_mag < 1.0e-3f)
67  return;
68 
69  // Normalize error vector
70  accel_err_b[0] /= accel_mag;
71  accel_err_b[1] /= accel_mag;
72  accel_err_b[2] /= accel_mag;
73 
74  // Accumulate integral of error. Scale here so that units are (deg/s) but accelKi has units of s
75  glblAtt->gyro_correct_int[0] += accel_err_b[0] * glblAtt->accelKi;
76  glblAtt->gyro_correct_int[1] += accel_err_b[1] * glblAtt->accelKi;
77 
78  // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
79  // and make it average zero (weakly)
80  glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
81 
82  // In this step, correct rates based on proportional error. The integral component is applied in updateSensors
83  gyros[0] += accel_err_b[0] * glblAtt->accelKp / delT;
84  gyros[1] += accel_err_b[1] * glblAtt->accelKp / delT;
85  gyros[2] += accel_err_b[2] * glblAtt->accelKp / delT;
86 }
