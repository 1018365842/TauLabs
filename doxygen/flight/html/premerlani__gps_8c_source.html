101 /*
102  * Correct sensor drift, using the DCM approach from W. Premerlani et. al
103  */
104 void Premerlani_GPS(float *accels, float *gyros, float Rbe[3][3], const float delT, bool GPS_Drift_Compensation, GlobalAttitudeVariables *glblAtt, float *omegaCorrP)
105 {
106  float errYaw_b[3] = { 0, 0, 0 };
107  float errRollPitch_b[3] = { 0, 0, 0 };
108 
109  float normOmegaScalar = VectorMagnitude(gyros);
110 
111  //Correct roll-pitch drift via GPS and accelerometer
112  //The math is derived from Roll-Pitch Gyro Drift Compensation, Rev.3, by W. Premerlani
113 #if defined (PIOS_INCLUDE_GPS)
114  if (drft->gpsPresent_flag && GPS_Drift_Compensation) {
115  float accels_e[3];
116 
117  //Rotate accelerometer readings into Earth frame. Note that we need to take the transpose of Rbe.
118  rot_mult(Rbe, accels, accels_e, TRUE);
119 
120  //Integrate accelerometer measurements in Earth frame
121  drft->accels_e_integrator[0] += accels_e[0] * delT;
122  drft->accels_e_integrator[1] += accels_e[1] * delT;
123  drft->accels_e_integrator[2] += accels_e[2] * delT;
124 
125  drft->delT_between_GPS += delT;
126 
127  //Check if the GPS has new information.
128  if (!
129  (drft->
130  gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_RPY)) {
131 
132  //Compute drift correction, errRollPitch_b, from GPS
133  rollPitch_drift_GPS(Rbe, drft->accels_e_integrator,
134  drft->delT_between_GPS,
135  errRollPitch_b);
136 
137  //Reset integrator
138  memset(drft->accels_e_integrator, 0,
139  sizeof(drft->accels_e_integrator));
140 
141  //Mark GPS data as consumed by this function
142  drft->gpsVelocityDataConsumption_flag |=
143  GPS_CONSUMED_BY_RPY;
144 
145  drft->delT_between_GPS = 0;
146 
147  }
148  }
149 #endif
150 
151  if (!GPS_Drift_Compensation) {
152 #if defined (PIOS_INCLUDE_GPS) && 0 || defined (PIOS_INCLUDE_MAGNETOMETER)
153  if (!(drft->gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_Y)) {
154  // We're actually using new GPS data here, but it's already been stored in old by the previous function
155  yaw_drift_MagGPS(Rbe, true, drft->magNewData_flag, errYaw_b);
156 
157  // Mark GPS data as consumed by this function
158  drft->gpsVelocityDataConsumption_flag |= GPS_CONSUMED_BY_Y;
159  } else {
160  // In addition to calculating the roll-pitch-yaw error, we can calculate yaw drift, errYaw_b, based on GPS and attitude data
161  // We're actually using new GPS data here, but it's already been stored in old by the previous function
162  yaw_drift_MagGPS(Rbe, false, drft->magNewData_flag, errYaw_b);
163  }
164 
165  // Reset flag. Not the best place to do it, but it's messy anywhere else
166  if (drft->magNewData_flag) {
167  drft->magNewData_flag = false;
168  }
169 #endif
170  //In addition, we can calculate roll-pitch error with only the aid of an accelerometer
171 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
172  AirspeedActualData airspeedActualData;
173  AirspeedActualGet(&airspeedActualData);
174  float airspeed_tas = airspeedActualData.TrueAirspeed;
175 #else
176  float airspeed_tas = 0;
177 #endif
178  rollPitch_drift_accel(accels, gyros, Rbe, airspeed_tas,
179  errRollPitch_b);
180  }
181 
182  // Calculate gyro drift, based on all errors
183  gyro_drift(gyros, errYaw_b, errRollPitch_b, normOmegaScalar, delT, omegaCorrP, drft->omegaCorrI);
184 
185  //Calculate final drift response
186  gyros[0] += omegaCorrP[0] + drft->omegaCorrI[0];
187  gyros[1] += omegaCorrP[1] + drft->omegaCorrI[1];
188  gyros[2] += omegaCorrP[2] + drft->omegaCorrI[2];
189 
190  //Add 0.0001% of proportional error back into gyroscope bias offset. This keeps DC elements out of the raw gyroscope data.
191  glblAtt->gyro_correct_int[0] += omegaCorrP[0] / 1000000.0f;
192  glblAtt->gyro_correct_int[1] += omegaCorrP[1] / 1000000.0f;
193 
194  // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
195  // and make it average zero (weakly)
196  glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
197 }

199 /*
200  * Calculate the error, as indicated by the accelerometer. It's important to use the true airspeed here, and not CAS or EAS
201  */
202 void rollPitch_drift_accel(float accels[3], float gyros[3], float Rbe[3][3],
203  float airspeed_tas, float *errRollPitch_b)
204 {
205  float g_ref[3];
206  float errAccelsRollPitch_b[3];
207 
208  //Fuselage Z vector is simply the Z column of the rotation matrix
209  float fuselageZ[3] = { Rbe[0][2], Rbe[1][2], Rbe[2][2] };
210 
211  //Calculate centripetal acceleration
212  float acc_centripetal[3] =
213  { 0, gyros[2] * airspeed_tas, -gyros[1] * airspeed_tas };
214 
215  //Combine measurements and centripetal accelerations. This should give the gravity reference vector
216  g_ref[0] = accels[0]; //Remember that for fixed-wing, acc_centripetal[0] = 0
217  g_ref[1] = accels[1] - acc_centripetal[1];
218  g_ref[2] = accels[2] - acc_centripetal[2];
219 
220  //Normalize acceleration vector
221  float normG = sqrtf(g_ref[0] * g_ref[0] + g_ref[1] * g_ref[1] + g_ref[2] * g_ref[2]);
222  g_ref[0] /= normG;
223  g_ref[1] /= normG;
224  g_ref[2] /= normG;
225 
226  //Error is cross product of reference vector and estimated orientation. Reverse operation in order to have correct sign for application to error sum
227  CrossProduct((const float *)fuselageZ, (const float *)g_ref, errAccelsRollPitch_b);
228 
229  //Add errors into global error vector
230  errRollPitch_b[0] += errAccelsRollPitch_b[0] * drft->accelsKp;
231  errRollPitch_b[1] += errAccelsRollPitch_b[1] * drft->accelsKp;
232  errRollPitch_b[2] += errAccelsRollPitch_b[2] * drft->accelsKp;
233 }

235 /*
236  * This function takes in a number of different error rotations, weights them, and applies
237  * them to the gyroscope readings. In addition, there is a function to increase the correction
238  * at high speeds in order to keep the gyro from becoming "dizzy".
239  */
240 void gyro_drift(float gyro[3], float errYaw_b[3], float errRollPitch_b[3],
241  float normOmegaScalar, float delT, float *omegaCorrP,
242  float *omegaCorrI)
243 {
244  float kpyaw;
245  float kprollpitch;
246 
247  // boost the KPs at high spin rate, to compensate for increased error due to calibration error
248  // above 50 degrees/second, scale by rotation rate divided by 50. As per Fast Rotations, by William Premerlani.
249  if (normOmegaScalar < (50.0f)) {
250  kpyaw = drft->yawKp;
251  kprollpitch = drft->rollPitchKp;
252  } else if (normOmegaScalar < (500.0f)) {
253  kpyaw = (normOmegaScalar / 50.0f) * drft->yawKp;
254  kprollpitch = (normOmegaScalar / 50.0f) * drft->rollPitchKp;
255  } else {
256  kpyaw = 10.0f * drft->yawKp;
257  kprollpitch = 10.0f * drft->rollPitchKp;
258  }
259 
260  //Compute proportional correction.
261  omegaCorrP[0] = errRollPitch_b[0] * kprollpitch + errYaw_b[0] * kpyaw;
262  omegaCorrP[1] = errRollPitch_b[1] * kprollpitch + errYaw_b[1] * kpyaw;
263  omegaCorrP[2] = errRollPitch_b[2] * kprollpitch + errYaw_b[2] * kpyaw;
264 
265  // Compute integral correction. Turn off the offset integrator while spinning at high speeds,
266  // it doesn't work in that case, and it only causes trouble.
267  if (normOmegaScalar < MAXIMUM_SPIN_DCM_INTEGRAL) {
268  omegaCorrI[0] += (errRollPitch_b[0] * drft->rollPitchKi + errYaw_b[0] * drft->yawKi) * delT;
269  omegaCorrI[1] += (errRollPitch_b[1] * drft->rollPitchKi + errYaw_b[1] * drft->yawKi) * delT;
270  omegaCorrI[2] += (errRollPitch_b[2] * drft->rollPitchKi + errYaw_b[2] * drft->yawKi) * delT;
271  }
272 }

274 /*
275  * From Roll-Pitch Gyro Drift Compensation, Rev 3. William Premerlani, 2012.
276  */
277 static void rollPitch_drift_GPS(float Rbe[3][3], float accels_e_int[3],
278  float delT_between_updates, float *errRollPitch_b)
279 {
280  float errRollPitch_e[3];
281  float dGPSdt_e[3];
282 
283  GPSVelocityData gpsVelocity;
284  GPSVelocityGet(&gpsVelocity);
285 
286  dGPSdt_e[0] = (gpsVelocity.North - drft->GPSV_old[0]) / delT_between_updates;
287  dGPSdt_e[1] = (gpsVelocity.East - drft->GPSV_old[1]) / delT_between_updates;
288  dGPSdt_e[2] = -GRAVITY + (gpsVelocity.Down - drft->GPSV_old[2]) / delT_between_updates;
289 
290  drft->GPSV_old[0] = gpsVelocity.North;
291  drft->GPSV_old[1] = gpsVelocity.East;
292  drft->GPSV_old[2] = gpsVelocity.Down;
293 
294  float normdGPSdt_e = VectorMagnitude(dGPSdt_e);
295 
296  //Take cross product of integrated accelerometer measurements with integrated earth frame accelerations. We should be using normalized dGPSdt, but we perform that calculation in the following line(s).
297  CrossProduct((const float *)accels_e_int, (const float *)dGPSdt_e, errRollPitch_e);
298 
299  //Scale cross product
300  errRollPitch_e[0] /= (normdGPSdt_e * delT_between_updates);
301  errRollPitch_e[1] /= (normdGPSdt_e * delT_between_updates);
302  errRollPitch_e[2] /= (normdGPSdt_e * delT_between_updates);
303 
304  //Rotate earth drift error back into body frame;
305  rot_mult(Rbe, errRollPitch_e, errRollPitch_b, FALSE);
306 }

308 /*
309  * Although yaw correction is done in horizontal plane, it is
310  * computed in 3 dimensions, just in case we change our minds later.
311  */
312 void yaw_drift_MagGPS(float Rbe[3][3], bool gpsNewData_flag, bool magNewData_flag, float *errYaw_b)
313 {
314 #if defined (PIOS_INCLUDE_GPS) || defined (PIOS_INCLUDE_MAGNETOMETER)
315 
316  // Form the horizontal direction over ground based on rmat
317  float horizDirOverGndRmat[3];
318 
319  //Define forward vector in earth frame, Rbe' * [1;0;0], and eliminate vertical component from vector
320  horizDirOverGndRmat[0] = Rbe[0][0];
321  horizDirOverGndRmat[1] = Rbe[0][1];
322  horizDirOverGndRmat[2] = 0;
323 
324 #if defined (PIOS_INCLUDE_GPS)
325  if (gpsNewData_flag) {
326 
327  GPSVelocityData gpsVelocity;
328  GPSVelocityGet(&gpsVelocity);
329 
330  if (fabs(gpsVelocity.North) > GPS_SPEED_MIN
331  || fabs(gpsVelocity.East) > GPS_SPEED_MIN) {
332  float errorGPSYaw_e;
333  float errorGPSYaw_b[3];
334  float normGPS =
335  sqrtf(gpsVelocity.North * gpsVelocity.North +
336  gpsVelocity.East * gpsVelocity.East);
337  float horizDirOverGndGPS[3] = { gpsVelocity.North / normGPS, gpsVelocity.East / normGPS, 0 }; //Normalized vector
338 
339  // vector cross product to get the rotation error in ground frame. However, save several processor cycles by
340  // condensing the math to take advantage of the cross products null entries on the x and y elements.
341  // errorGPSYaw_e = horizDirOverGndRmat X horizDirOverGndGPS
342  errorGPSYaw_e =
343  horizDirOverGndRmat[0] * horizDirOverGndGPS[1] -
344  horizDirOverGndGPS[0] * horizDirOverGndRmat[1];
345 
346  // Rotate error to body frame. Again, take advantage of the yaw error vector [0;0;errorGPSYaw_e]'s null entries
347  // errorGPSYaw_b = Rbe * errorGPSYaw_e;
348  errorGPSYaw_b[0] = Rbe[0][2] * errorGPSYaw_e;
349  errorGPSYaw_b[1] = Rbe[1][2] * errorGPSYaw_e;
350  errorGPSYaw_b[2] = Rbe[2][2] * errorGPSYaw_e;
351 
352  //Sum error with existing error
353  errYaw_b[0] += errorGPSYaw_b[0] * GPS_YAW_KP;
354  errYaw_b[1] += errorGPSYaw_b[1] * GPS_YAW_KP;
355  errYaw_b[2] += errorGPSYaw_b[2] * GPS_YAW_KP;
356  }
357  }
358 #endif
359 
360  // TODO: Create a flag to indicate whether mag data is present (or check for updates of MagnetometerData)
361 #if defined (PIOS_INCLUDE_MAGNETOMETER)
362  if (magNewData_flag) {
363  MagnetometerData magnetometerData;
364  MagnetometerGet(&magnetometerData);
365 
366  float errorMagYaw_e;
367  float errorMagYaw_b[3];
368  float horizDirOverGndMag[3];
369  float mags_b[3] = { magnetometerData.x, magnetometerData.y, magnetometerData.z };
370 
371  // Rotate magnetometer to earth frame and eliminate vertical component
372  rot_mult(Rbe, mags_b, horizDirOverGndMag, true);
373  horizDirOverGndMag[2] = 0;
374 
375  //Compute norm
376  float normMag = VectorMagnitude(horizDirOverGndMag);
377 
378  //Normalize mag_vector. Recall that horizDirOverGndMag[2]=0
379  horizDirOverGndMag[0] /= normMag;
380  horizDirOverGndMag[1] /= normMag;
381 
382  // vector cross product to get the rotation error in ground frame. However, save several processor cycles by
383  // condensing the math to take advantage of the cross products null entries on the x and y elements.
384  // errorMagYaw_e = horizDirOverGndRmat X horizDirOverGndMag
385  errorMagYaw_e = horizDirOverGndRmat[0] * horizDirOverGndMag[1] -
386  horizDirOverGndMag[0] * horizDirOverGndRmat[1];
387 
388  // Rotate error to body frame. Again, take advantage of the yaw error vector [0;0;errorGPSYaw_e]'s null entries
389  // errorMagYaw_b = Rbe * errorMagYaw_e;
390  errorMagYaw_b[0] = Rbe[0][2] * errorMagYaw_e;
391  errorMagYaw_b[1] = Rbe[1][2] * errorMagYaw_e;
392  errorMagYaw_b[2] = Rbe[2][2] * errorMagYaw_e;
393 
394  errYaw_b[0] += errorMagYaw_b[0] * MAG_YAW_KP;
395  errYaw_b[1] += errorMagYaw_b[1] * MAG_YAW_KP;
396  errYaw_b[2] += errorMagYaw_b[2] * MAG_YAW_KP;
397  }
398 #endif
399 #endif
