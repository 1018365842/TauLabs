15 """
16 PyMite Image Creator
17 ====================
18 
19 Converts Python source files to a PyMite code image library.
20 Performs code filtering to ensure it will run in PyMite.
21 Formats the image as a raw binary file or a C file
22 containing a byte array.
23 
24 16- and 32-bit values are in LITTLE ENDIAN order.
25 This matches both Python and the AVR compiler's access to EEPROM.
26 
27 The order of the images in the output is undetermined.
28 
29 If the Python source contains a native code declaration
30 and '--native-file=filename" is specified, the native code
31 is formatted as C functions and an array of functions and output
32 to the given filename. When native functions are present, the user
33 should specify where the native functions should be placed-- in the
34 standard library or the user library--using the argument -s or -u,
35 respectively.
36 """

361  def _seq_to_str(self, seq):
362  """Convert a Python sequence to a PyMite image.
363 
364  The sequence is converted to a tuple of objects.
365  This handles both co_consts and co_names.
366  This is recursive to handle tuples in the const pool.
367  Return string shows type in the leading byte.
368  """
369 
370  # OPT
371  _U8_to_str = self._U8_to_str
372 
373  # ensure tuple fits within limits
374  assert len(seq) <= MAX_TUPLE_LEN
375 
376  # image string init with tuple marker and num elements
377  imgstr = _U8_to_str(OBJ_TYPE_TUP) + \
378  _U8_to_str(len(seq))
379 
380  # iterate through the sequence of objects
381  for i in range(len(seq)):
382  obj = seq[i]
383  objtype = type(obj)
384 
385  # if it is a string
386  if objtype == types.StringType:
387  # ensure string is not too long
388  assert len(obj) <= MAX_STRING_LEN
389  # marker, string length, string itself
390  imgstr += _U8_to_str(OBJ_TYPE_STR) + \
391  self._U16_to_str(len(obj)) + obj
392 
393  # if it is an integer
394  elif objtype == types.IntType:
395  # marker, int (little endian)
396  imgstr += _U8_to_str(OBJ_TYPE_INT) + \
397  _U8_to_str(obj & 0xff) + \
398  _U8_to_str((obj >> 8) & 0xff) + \
399  _U8_to_str((obj >> 16) & 0xff) + \
400  _U8_to_str((obj >> 24) & 0xff)
401 
402  # if it is a code object
403  elif objtype == types.CodeType:
404  #determine if it's native or regular
405  if (len(obj.co_consts) > 0 and
406  (type(obj.co_consts[0]) == types.StringType) and
407  (obj.co_consts[0][0:NATIVE_INDICATOR_LENGTH] ==
408  NATIVE_INDICATOR)):
409  imgstr += self.no_to_str(obj)
410  else:
411  imgstr += self.co_to_str(obj)
412 
413  # if it is a tuple
414  elif objtype == types.TupleType:
415  imgstr += self._seq_to_str(obj)
416 
417  # if it is None
418  elif objtype == types.NoneType:
419  # marker, none (0)
420  imgstr += _U8_to_str(OBJ_TYPE_NON)
421 
422  # if it is a float
423  elif objtype == types.FloatType and PM_FEATURES["HAVE_FLOAT"]:
424  imgstr += _U8_to_str(OBJ_TYPE_FLT) + self._float_to_str(obj)
425 
426  # other type?
427  else:
428  raise exceptions.NotImplementedError(
429  "Unhandled type %s." % objtype)
430  return imgstr

551  def _filter_co(self, co):
552  """Run the Python code obj, co, through various filters.
553 
554  Ensure it is compliant with PyMite restrictions.
555 
556  Consts filter:
557  Ensure num consts is less than 256.
558  Replace __doc__ with None if present.
559 
560  Flags filter:
561  Check co_flags for flags that indicate an unsupported feature
562  Supported flags: CO_NOFREE, CO_OPTIMIZED, CO_NEWLOCALS, CO_NESTED,
563  Unsupported flags: CO_VARARGS, CO_VARKEYWORDS
564  Conditionally supported flags: CO_GENERATOR if HAVE_GENERATORS
565 
566  Native code filter:
567  If this function has a native indicator,
568  extract the native code from the doc string
569  and clear the doc string.
570  Ensure num args is less or equal to
571  NATIVE_MAX_NUM_LOCALS.
572 
573  Names/varnames filter:
574  Ensure num names is less than 256.
575  If co_name is the module identifier replace it with
576  the trimmed module name
577  otherwise just append the name to co_name.
578 
579  Bcode filter:
580  Raise NotImplementedError for an invalid bcode.
581 
582  If all is well, return the filtered consts list,
583  names list, code string and native code.
584  """
585 
586  ## General filter
587  # ensure values fit within U8 type size
588  assert len(co.co_consts) < 256, "too many constants."
589  assert len(co.co_names) < 256, "too many names."
590  assert co.co_argcount < 256, "too many arguments."
591  assert co.co_stacksize < 256, "too large of a stack."
592  assert co.co_nlocals < 256, "too many local variables."
593 
594  # make consts a list so a single element can be modified
595  consts = list(co.co_consts)
596 
597  # Check co_flags
598  unsupported_flags = CO_VARARGS | CO_VARKEYWORDS
599  if not PM_FEATURES["HAVE_GENERATORS"]:
600  unsupported_flags |= CO_GENERATOR
601  assert co.co_flags & unsupported_flags == 0,\
602  "Unsupported code identified by co_flags (%s)." % hex(co.co_flags)
603 
604  # get trimmed src file name and module name
605  fn = os.path.basename(co.co_filename)
606  mn = os.path.splitext(fn)[0]
607 
608  # init native code
609  nativecode = None
610 
611  ## Bcode filter
612  # bcode string
613  s = co.co_code
614  # filtered code string
615  code = ""
616  # iterate through the string
617  lno = 0
618  i = 0
619  len_s = len(s)
620  while i < len_s:
621 
622  #get char
623  c = ord(s[i])
624 
625  #ensure no illegal bytecodes are present
626  if self.bcodes[c] == None:
627  raise NotImplementedError(
628  "Illegal bytecode (%d/%s/%s) "
629  "comes at offset %d in file %s." %
630  (c, hex(c), dis.opname[c],
631  i, co.co_filename))
632 
633  #if simple bcode, copy one byte
634  if c < dis.HAVE_ARGUMENT:
635  code += s[i]
636  i += 1
637 
638  #else copy three bytes
639  else:
640 
641  # Raise error if default arguments exist and are not configured
642  if (not PM_FEATURES["HAVE_DEFAULTARGS"]
643  and c == dis.opmap["MAKE_FUNCTION"]
644  and self._str_to_U16(s[i+1:i+3]) > 0):
645 
646  raise NotImplementedError(
647  "Bytecode (%d/%s/%s) not configured "
648  "to support default arguments; "
649  "comes at offset %d in file %s." %
650  (c, hex(c), dis.opname[c], i, co.co_filename))
651 
652  # Otherwise, copy the code (3 bytes)
653  code += s[i:i+3]
654  i += 3
655 
656  # if the first const is a String,
657  if (len(consts) > 0 and type(consts[0]) == types.StringType):
658 
659  ## Native code filter
660  # if this CO is intended to be a native func.
661  if (consts[0][:NATIVE_INDICATOR_LENGTH] ==
662  NATIVE_INDICATOR):
663 
664  # ensure num args is less or equal
665  # to NATIVE_MAX_NUM_LOCALS
666  assert co.co_nlocals <= NATIVE_MAX_NUM_LOCALS
667 
668  # extract native code and clear doc string
669  nativecode = consts[0][NATIVE_INDICATOR_LENGTH:]
670  consts[0] = None
671 
672  # If this co is a module
673  # Old #28: Module root must keep its bytecode
674  if co.co_name == MODULE_IDENTIFIER:
675  self.nativemods.append((co.co_filename, nativecode))
676 
677  # Else this co is a function;
678  # replace code with native table index
679  else:
680  # stdlib code gets a positive index
681  if self.imgtarget == "std":
682  code = self._U16_to_str(len(self.nativetable))
683  # usr code gets a negative index
684  else:
685  code = self._U16_to_str(-len(self.nativetable))
686 
687  # native function name is
688  # "nat_<modname>_<pyfuncname>".
689  # append (nat func name, nat code) to table
690  self.nativetable.append(
691  ("%s%02d_%s_%s"
692  % (NATIVE_FUNC_PREFIX, self.nfcount, mn, co.co_name),
693  nativecode))
694  self.nfcount += 1
695 
696  ## Consts filter
697  # if want to remove __doc__ string
698  # WARNING: this heuristic is not always accurate
699  elif (REMOVE_DOC_STR and len(co.co_names) > 0
700  and co.co_names[0] == "__doc__"):
701  consts[0] = None
702 
703  ## Names filter
704  names = list(co.co_names)
705 
706  # Remove __doc__ name if requested
707  if REMOVE_DOC_STR and len(names) > 0 and names[0] == "__doc__":
708  names[0] = ''
709 
710  # if co_name is the module identifier change it to module name
711  if co.co_name == MODULE_IDENTIFIER:
712  names.append(mn)
713  # else use unmodified co_name
714  else:
715  names.append(co.co_name)
716 
717 
718  return consts, names, code, nativecode

889 def parse_cmdline():
890  """Parses the command line for options.
891  """
892  try:

894  "f:bcsuo:",
895  ["memspace=", "native-file="])
896  except:
897  print __usage__
898  sys.exit(EX_USAGE)
899 
900  # Parse opts for the image type to write
901  imgtype = ".c"
902  imgtarget = "std"
903  memspace = "ram"
904  outfn = None
905  nativeFilename = None
906  for opt in opts:
907  if opt[0] == "-b":
908  imgtype = ".bin"
909  elif opt[0] == "-c":
910  imgtype = ".c"
911  elif opt[0] == "-s":
912  imgtarget = "std"
913  elif opt[0] == "-u":
914  imgtarget = "usr"

918  print "Only one of these memspace types allowed: ram, flash"
919  print __usage__
920  sys.exit(EX_USAGE)
921  memspace = opt[1]
922  elif opt[0] == "--native-file":
923  # Error if switch given without arg
924  if not opt[1]:
925  print "Specify a filename like this: --native-file=libnative.c"
926  print __usage__
927  sys.exit(EX_USAGE)
928  nativeFilename = opt[1]
929  elif opt[0] == "-f":
930  pmfeatures_filename = opt[1]
931  elif opt[0] == "-o":
932  # Error if out filename switch given without arg
933  if not opt[1]:
934  print __usage__
935  sys.exit(EX_USAGE)

937

940  print __usage__

942

944  if len(args) == 0:
945  print __usage__
