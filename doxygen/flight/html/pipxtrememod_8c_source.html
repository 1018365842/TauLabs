125 static void systemTask(void *parameters)
126 {
127  portTickType lastSysTime;
128  uint16_t prev_tx_count = 0;
129  uint16_t prev_rx_count = 0;
130  bool first_time = true;
131 
132  /* create all modules thread */
133  MODULE_TASKCREATE_ALL;
134 
135  if (mallocFailed) {
136  /* We failed to malloc during task creation,
137  * system behaviour is undefined. Reset and let
138  * the BootFault code recover for us.
139  */
140  PIOS_SYS_Reset();
141  }
142 
143  // Initialize vars
144  idleCounter = 0;
145  idleCounterClear = 0;
146  lastSysTime = xTaskGetTickCount();
147 
148  // Main system loop
149  while (1) {
150 
151  // Flash the heartbeat LED
152 #if defined(PIOS_LED_HEARTBEAT)
153  PIOS_LED_Toggle(PIOS_LED_HEARTBEAT);
154 #endif /* PIOS_LED_HEARTBEAT */
155 
156  // Update the PipXstatus UAVO
157  OPLinkStatusData oplinkStatus;
158  uint32_t pairID;
159  OPLinkStatusGet(&oplinkStatus);
160  OPLinkSettingsPairIDGet(&pairID);
161 
162  // Get the other device stats.
163  PIOS_RFM2B_GetPairStats(pios_rfm22b_id, oplinkStatus.PairIDs, oplinkStatus.PairSignalStrengths, OPLINKSTATUS_PAIRIDS_NUMELEM);
164 
165  // Get the stats from the radio device
166  struct rfm22b_stats radio_stats;
167  PIOS_RFM22B_GetStats(pios_rfm22b_id, &radio_stats);
168 
169  // Update the status
170  oplinkStatus.DeviceID = PIOS_RFM22B_DeviceID(pios_rfm22b_id);
171  oplinkStatus.RxGood = radio_stats.rx_good;
172  oplinkStatus.RxCorrected = radio_stats.rx_corrected;
173  oplinkStatus.RxErrors = radio_stats.rx_error;
174  oplinkStatus.RxMissed = radio_stats.rx_missed;
175  oplinkStatus.RxFailure = radio_stats.rx_failure;
176  oplinkStatus.TxDropped = radio_stats.tx_dropped;
177  oplinkStatus.TxResent = radio_stats.tx_resent;
178  oplinkStatus.TxFailure = radio_stats.tx_failure;
179  oplinkStatus.Resets = radio_stats.resets;
180  oplinkStatus.Timeouts = radio_stats.timeouts;
181  oplinkStatus.RSSI = radio_stats.rssi;
182  oplinkStatus.AFCCorrection = radio_stats.afc_correction;
183  oplinkStatus.LinkQuality = radio_stats.link_quality;
184  if (first_time)
185  first_time = false;
186  else
187  {
188  uint16_t tx_count = radio_stats.tx_byte_count;
189  uint16_t rx_count = radio_stats.rx_byte_count;
190  uint16_t tx_bytes = (tx_count < prev_tx_count) ? (0xffff - prev_tx_count + tx_count) : (tx_count - prev_tx_count);
191  uint16_t rx_bytes = (rx_count < prev_rx_count) ? (0xffff - prev_rx_count + rx_count) : (rx_count - prev_rx_count);
192  oplinkStatus.TXRate = (uint16_t)((float)(tx_bytes * 1000) / SYSTEM_UPDATE_PERIOD_MS);
193  oplinkStatus.RXRate = (uint16_t)((float)(rx_bytes * 1000) / SYSTEM_UPDATE_PERIOD_MS);
194  prev_tx_count = tx_count;
195  prev_rx_count = rx_count;
196  }
197  oplinkStatus.TXSeq = radio_stats.tx_seq;
198  oplinkStatus.RXSeq = radio_stats.rx_seq;
199  oplinkStatus.LinkState = radio_stats.link_state;
200  if (radio_stats.link_state == OPLINKSTATUS_LINKSTATE_CONNECTED)
201  LINK_LED_ON;
202  else
203  LINK_LED_OFF;
204 
205  // Update the object
206  OPLinkStatusSet(&oplinkStatus);
207 
208  // Wait until next period
209  vTaskDelayUntil(&lastSysTime, SYSTEM_UPDATE_PERIOD_MS / portTICK_RATE_MS);
210  }
211 }

216 void vApplicationIdleHook(void)
217 {
218  // Called when the scheduler has no tasks to run
219  if (idleCounterClear == 0) {
220  ++idleCounter;
221  } else {
222  idleCounter = 0;
223  idleCounterClear = 0;
