116 {
117  // TODO: How do we protect against the DMA buffer swapping midway through adding data
118  // to this buffer. If we were writing at the beginning it could cause a weird race.
119  if (overo_dev->tx_out_cb) {
120 
121  int32_t max_bytes = PACKET_SIZE - overo_dev->writing_offset;
122 
123  if (max_bytes > 0) {
124  uint16_t bytes_added;
125  bool tx_need_yield = false;
126  uint8_t *writing_pointer = &overo_dev->tx_buffer[overo_dev->writing_buffer][overo_dev->writing_offset];

128  bytes_added = (overo_dev->tx_out_cb)(overo_dev->tx_out_context, writing_pointer, max_bytes, NULL, &tx_need_yield);

133  overo_dev->writing_offset += bytes_added;

198 int32_t PIOS_OVERO_Init(uint32_t * overo_id, const struct pios_overo_cfg * cfg)
199 {
200  PIOS_DEBUG_Assert(overo_id);
201  PIOS_DEBUG_Assert(cfg);
202 
203  struct pios_overo_dev *overo_dev;
204 
205  overo_dev = (struct pios_overo_dev *) PIOS_OVERO_alloc();
206  if (!overo_dev) goto out_fail;
207 
208  /* Bind the configuration to the device instance */
209  overo_dev->cfg = cfg;
210  overo_dev->writing_buffer = 1; // First writes to second buffer
211 
212  /* Put buffers to a known state */
213  memset(&overo_dev->tx_buffer[0][0], 0xFF, PACKET_SIZE);
214  memset(&overo_dev->tx_buffer[1][0], 0xFF, PACKET_SIZE);
215  memset(&overo_dev->rx_buffer[0][0], 0xFF, PACKET_SIZE);
216  memset(&overo_dev->rx_buffer[1][0], 0xFF, PACKET_SIZE);
217 
218  /*
219  * Enable the SPI device
220  *
221  * 1. Enable the SPI port
222  * 2. Enable DMA with circular buffered DMA (validate config)
223  * 3. Enable the DMA Tx IRQ
224  */
225 
226  //PIOS_Assert(overo_dev->cfg->dma.tx-> == CIRCULAR);
227  //PIOS_Assert(overo_dev->cfg->dma.rx-> == CIRCULAR);
228 
229  /* only legal for single-slave config */
230  PIOS_Assert(overo_dev->cfg->slave_count == 1);
231  SPI_SSOutputCmd(overo_dev->cfg->regs, DISABLE);
232 
233  /* Initialize the GPIO pins */
234  /* note __builtin_ctz() due to the difference between GPIO_PinX and GPIO_PinSourceX */
235  GPIO_PinAFConfig(overo_dev->cfg->sclk.gpio,
236  __builtin_ctz(overo_dev->cfg->sclk.init.GPIO_Pin),
237  overo_dev->cfg->remap);
238  GPIO_PinAFConfig(overo_dev->cfg->mosi.gpio,
239  __builtin_ctz(overo_dev->cfg->mosi.init.GPIO_Pin),
240  overo_dev->cfg->remap);
241  GPIO_PinAFConfig(overo_dev->cfg->miso.gpio,
242  __builtin_ctz(overo_dev->cfg->miso.init.GPIO_Pin),
243  overo_dev->cfg->remap);
244  GPIO_PinAFConfig(overo_dev->cfg->ssel[0].gpio,
245  __builtin_ctz(overo_dev->cfg->ssel[0].init.GPIO_Pin),
246  overo_dev->cfg->remap);
247 
248  GPIO_Init(overo_dev->cfg->sclk.gpio, (GPIO_InitTypeDef*)&(overo_dev->cfg->sclk.init));
249  GPIO_Init(overo_dev->cfg->mosi.gpio, (GPIO_InitTypeDef*)&(overo_dev->cfg->mosi.init));
250  GPIO_Init(overo_dev->cfg->miso.gpio, (GPIO_InitTypeDef*)&(overo_dev->cfg->miso.init));
251 
252  /* Configure circular buffer targets. Configure 0 to be initially active */
253  DMA_InitTypeDef dma_init;
254 
255  DMA_DeInit(overo_dev->cfg->dma.rx.channel);
256  dma_init = overo_dev->cfg->dma.rx.init;
257  dma_init.DMA_Memory0BaseAddr = (uint32_t) overo_dev->rx_buffer[0];
258  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
259  dma_init.DMA_BufferSize = PACKET_SIZE;
260  DMA_Init(overo_dev->cfg->dma.rx.channel, &dma_init);
261  DMA_DoubleBufferModeConfig(overo_dev->cfg->dma.rx.channel, (uint32_t) overo_dev->rx_buffer[1], DMA_Memory_0);
262  DMA_DoubleBufferModeCmd(overo_dev->cfg->dma.rx.channel, ENABLE);
263 
264  DMA_DeInit(overo_dev->cfg->dma.tx.channel);
265  dma_init = overo_dev->cfg->dma.tx.init;
266  dma_init.DMA_Memory0BaseAddr = (uint32_t) overo_dev->tx_buffer[0];
267  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
268  dma_init.DMA_BufferSize = PACKET_SIZE;
269  DMA_Init(overo_dev->cfg->dma.tx.channel, &dma_init);
270  DMA_DoubleBufferModeConfig(overo_dev->cfg->dma.tx.channel, (uint32_t) overo_dev->tx_buffer[1], DMA_Memory_0);
271  DMA_DoubleBufferModeCmd(overo_dev->cfg->dma.tx.channel, ENABLE);
272 
273  /* Set the packet size */
274  DMA_SetCurrDataCounter(overo_dev->cfg->dma.rx.channel, PACKET_SIZE);
275  DMA_SetCurrDataCounter(overo_dev->cfg->dma.tx.channel, PACKET_SIZE);
276 
277  /* Initialize the SPI block */
278  SPI_DeInit(overo_dev->cfg->regs);
279  SPI_Init(overo_dev->cfg->regs, (SPI_InitTypeDef*)&(overo_dev->cfg->init));
280 
281  SPI_CalculateCRC(overo_dev->cfg->regs, DISABLE);
282 
283  /* Enable SPI interrupts to DMA */
284  SPI_I2S_DMACmd(overo_dev->cfg->regs, SPI_I2S_DMAReq_Tx | SPI_I2S_DMAReq_Rx, ENABLE);
285 
286  /* Configure DMA interrupt */
287  NVIC_Init((NVIC_InitTypeDef*)&(overo_dev->cfg->dma.irq.init));
288  DMA_ITConfig(overo_dev->cfg->dma.tx.channel, DMA_IT_TC, ENABLE);
289 
290  /* Enable the DMA channels */
291  DMA_Cmd(overo_dev->cfg->dma.tx.channel, ENABLE);
292  DMA_Cmd(overo_dev->cfg->dma.rx.channel, ENABLE);
293 
294  *overo_id = (uint32_t) overo_dev;
295 
296  return(0);
297 
298 out_fail:
299  return(-1);
300 }
