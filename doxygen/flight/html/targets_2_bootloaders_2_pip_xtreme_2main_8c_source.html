167 void jump_to_app() {
168  const struct pios_board_info * bdinfo = &pios_board_info_blob;
169 
170  if (((*(__IO uint32_t*) bdinfo->fw_base) & 0x2FFE0000) == 0x20000000 ||
171  ((*(__IO uint32_t*) bdinfo->fw_base) & 0x1FFE0000) == 0x10000000) { /* Jump to user application */
172  FLASH_Lock();
173  RCC_APB2PeriphResetCmd(0xffffffff, ENABLE);
174  RCC_APB1PeriphResetCmd(0xffffffff, ENABLE);
175  RCC_APB2PeriphResetCmd(0xffffffff, DISABLE);
176  RCC_APB1PeriphResetCmd(0xffffffff, DISABLE);
177  _SetCNTR(0); // clear interrupt mask
178  _SetISTR(0); // clear all requests
179  JumpAddress = *(__IO uint32_t*) (bdinfo->fw_base + 4);
180  Jump_To_Application = (pFunction) JumpAddress;
181  /* Initialize user application's Stack Pointer */
182  __set_MSP(*(__IO uint32_t*) bdinfo->fw_base);
183  Jump_To_Application();
184  } else {
185  DeviceState = failed_jump;
186  return;
187  }
188 }
189 uint32_t LedPWM(uint32_t pwm_period, uint32_t pwm_sweep_steps, uint32_t count) {
190  uint32_t curr_step = (count / pwm_period) % pwm_sweep_steps; /* 0 - pwm_sweep_steps */
191  uint32_t pwm_duty = pwm_period * curr_step / pwm_sweep_steps; /* fraction of pwm_period */
192 
193  uint32_t curr_sweep = (count / (pwm_period * pwm_sweep_steps)); /* ticks once per full sweep */
194  if (curr_sweep & 1) {
195  pwm_duty = pwm_period - pwm_duty; /* reverse direction in odd sweeps */
196  }
197  return ((count % pwm_period) > pwm_duty) ? 1 : 0;
198 }
