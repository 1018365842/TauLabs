211 {

213  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
214

216  //THIS IS BOARD SPECIFIC AND DOES NOT BELONG HERE. Can we put it in #if

224

229  SensorSettingsUpdatedCb(SensorSettingsHandle());

232 
233  if (cc3d_flag) {
234 #if defined(PIOS_INCLUDE_MPU6000)

236 #endif

238 #if defined(PIOS_INCLUDE_ADXL345)

241

244  gyro_queue = xQueueCreate(1, sizeof(float) * 4);
245  PIOS_Assert(gyro_queue != NULL);
246  PIOS_ADC_SetQueue(gyro_queue);

250  }
251

253  //enough to real temperature to have a reasonable density altitude estimate

256  float prelim_gyros[4];
257  if (cc3d_flag) {
258  getSensorsCC3D(prelim_accels, prelim_gyros, glblAtt, &gyrosBias, &sensorSettings);

262

271 
272  //Start clock for delT calculation
273  uint32_t rawtime = PIOS_DELAY_GetRaw();

275  // ----------------------------- //
276  // Main module loop. Never exits //
277  // ----------------------------- //

279  FlightStatusData flightStatus;
280  FlightStatusGet(&flightStatus);
281

287  glblAtt->accelKi = 0.9;

294  //...during arming...

296  glblAtt->accelKi = 0.9;
297  glblAtt->yawBiasRate = 0.23;
298  init = 0;
299

301  attitudeSettings.FilterChoice = ATTITUDESETTINGS_FILTERCHOICE_CCC;

304  AttitudeSettingsAccelKiGet(&glblAtt->accelKi);

317

321

323  GyrosSet(&gyros);

366

369  //http://www.engineeringtoolbox.com/air-altitude-pressure-d_462.html
370  float staticPressure =
371  homeLocation.SeaLevelPressure * powf(1.0f - 2.2555e-5f *

380  }
381 
382  gpsNew_flag = false;
383  }

394 static int32_t updateIntertialSensors(AccelsData * accels, GyrosData * gyros, bool cc3d_flag)
395 {
396  int8_t retval;
397  float prelim_accels[4];
398  float prelim_gyros[4];
399 
400  // Get the sensor data in a board specific manner
401  if (cc3d_flag) {
402  retval = getSensorsCC3D(prelim_accels, prelim_gyros, glblAtt, &gyrosBias, &sensorSettings);
403  } else {
404  retval = getSensorsCC(prelim_accels, prelim_gyros, &gyro_queue, glblAtt, &gyrosBias, &sensorSettings);
405  }
406 
407  if (retval < 0) { // No sensor data. Alarm set by calling method
408  return retval;
409  }
410 
411  // Rotate sensor board into body frame
412  if (glblAtt->rotate) {
413  float tmpVec[3];
414 
415  // Rotate the vector into a temporary vector, and then copy back into
416  // the original vectors.
417  rot_mult(glblAtt->Rbs, prelim_accels, tmpVec, false);
418  memcpy(prelim_accels, tmpVec, sizeof(tmpVec));
419 
420  rot_mult(glblAtt->Rbs, prelim_gyros, tmpVec, false);
421  memcpy(prelim_gyros, tmpVec, sizeof(tmpVec));
422  }
423 
424  // Store rotated accels
425  accels->x = prelim_accels[0];
426  accels->y = prelim_accels[1];
427  accels->z = prelim_accels[2];
428 
429  // Store rotated gyros, optionally with bias correction
430  if (glblAtt->bias_correct_gyro) {
431  // Applying integral component here so it can be seen on the gyros and correct bias
432  gyros->x = prelim_gyros[0] + glblAtt->gyro_correct_int[0];
433  gyros->y = prelim_gyros[1] + glblAtt->gyro_correct_int[1];
434  gyros->z = prelim_gyros[2] + glblAtt->gyro_correct_int[2];
435  } else {
436  gyros->x = prelim_gyros[0];
437  gyros->y = prelim_gyros[1];
438  gyros->z = prelim_gyros[2];
439  }
440 
441  // Estimate accel bias while user flies level
442  if (glblAtt->trim_requested) {
443  if (glblAtt->trim_samples >= MAX_TRIM_FLIGHT_SAMPLES) {
444  glblAtt->trim_requested = false;
445  } else {
446  uint8_t armed;
447  float throttle;
448  FlightStatusArmedGet(&armed);
449  ManualControlCommandThrottleGet(&throttle); // Until flight status indicates airborne
450  if ((armed == FLIGHTSTATUS_ARMED_ARMED) && (throttle > 0)) {
451  glblAtt->trim_samples++;
452  // Store the digitally scaled version since that is what we use for bias
453  glblAtt->trim_accels[0] += accels->x;
454  glblAtt->trim_accels[1] += accels->y;
455  glblAtt->trim_accels[2] += accels->z;
456  }
457  }
458  }
459 
460  return 0;
461

467 static void updateT3(GPSVelocityData * gpsVelocityData, PositionActualData * positionActualData)
468 {
469  //Load UAVOs
470  VelocityActualData velocityActualData;
471  VelocityActualGet(&velocityActualData);
472 
473  // Get a subset of the GPSPosition information
474  int32_t LL_int[2];
475  float altitude;
476  float geoidSeparation;
477  GPSPositionLatitudeGet(&LL_int[0]);
478  GPSPositionLongitudeGet(&LL_int[1]);
479  GPSPositionAltitudeGet(&altitude);
480  GPSPositionGeoidSeparationGet(&geoidSeparation);
481 
482  // Get NED coordinates from GPS lat-lon
483  float gps_NED[3];
484  LLA2NED(LL_int, altitude, geoidSeparation, gps_NED);
485 
486  // Calculate filter coefficients
487  float dT = .100f;
488  float tauPosNorthEast = 0.3f;
489  float tauPosDown = 0.5f;
490  float tauVelNorthEast = 0.01f;
491  float tauVelDown = 0.1f;
492  float alphaPosNorthEast = dT / (dT + tauPosNorthEast);
493  float alphaPosDown = dT / (dT + tauPosDown);
494  float alphaVelNorthEast = dT / (dT + tauVelNorthEast);
495  float alphaVelDown = dT / (dT + tauVelDown);
496 
497  //Low pass filter for velocity
498  velocityActualData.North = (1 - alphaVelNorthEast) * velocityActualData.North +
499  alphaVelNorthEast * gpsVelocityData->North;
500  velocityActualData.East = (1 - alphaVelNorthEast) * velocityActualData.East +
501  alphaVelNorthEast * gpsVelocityData->East;
502  velocityActualData.Down = (1 - alphaVelDown) * velocityActualData.Down +
503  alphaVelDown * gpsVelocityData->Down;
504 
505  //Complementary filter for position
506  positionActualData->North = (1 - alphaPosNorthEast) * (positionActualData->North +
507  (velocityActualData.North * dT)) + alphaPosNorthEast * gps_NED[0];
508  positionActualData->East = (1 - alphaPosNorthEast) * (positionActualData->East +
509  (velocityActualData.East * dT)) + alphaPosNorthEast * gps_NED[1];
510  positionActualData->Down = (1 - alphaPosDown) * (positionActualData->Down +
511  (velocityActualData.Down * dT)) + alphaPosDown * gps_NED[2];
512 
513  //Very slowly use GPS heading data to converge. This is a poor way of doing things, but will work in the short term for testing.
514  if (fabs(velocityActualData.North) > 3.0f && //Instead of calculating velocity norm, use a gauge approach
515  (fabs(velocityActualData.North) > 4.0f || fabs(velocityActualData.East) > 3.0f)) {
516  float heading = atan2f(velocityActualData.East, velocityActualData.North);
517 
518  AttitudeActualData attitudeActual;
519  AttitudeActualGet(&attitudeActual);
520 
521  while (heading - attitudeActual.Yaw < -180.0f) {
522  heading += 360.0f;
523  }
524  while (heading - attitudeActual.Yaw > 180.0f) {
525  heading -= 360.0f;
526  }
527 
528  attitudeActual.Yaw = .9f * attitudeActual.Yaw + 0.1f * heading;
529 
530  // Convert into quaternions (makes assumptions about quaternions and RPY order)
531  RPY2Quaternion(&attitudeActual.Roll, &attitudeActual.q1);
532 
533  if (!AttitudeActualReadOnly()) {
534  AttitudeActualSet(&attitudeActual);
535  }
536  }
537 
538  // Do not update position and velocity estimates when in simulation mode
539  PositionActualSet(positionActualData);
540  VelocityActualSet(&velocityActualData);
541 }

546 static void updateSO3(float *gyros, float dT)
547 {
548  {
549  // scoping variables to save memory
550  // Work out time derivative from INSAlgo writeup
551  // Also accounts for the fact that gyros are in deg/s
552  float qdot[4];
553  qdot[0] =
554  (-glblAtt->q[1] * gyros[0] - glblAtt->q[2] * gyros[1] -
555  glblAtt->q[3] * gyros[2]) * dT * DEG2RAD / 2;
556  qdot[1] =
557  (glblAtt->q[0] * gyros[0] - glblAtt->q[3] * gyros[1] +
558  glblAtt->q[2] * gyros[2]) * dT * DEG2RAD / 2;
559  qdot[2] =
560  (glblAtt->q[3] * gyros[0] + glblAtt->q[0] * gyros[1] -
561  glblAtt->q[1] * gyros[2]) * dT * DEG2RAD / 2;
562  qdot[3] =
563  (-glblAtt->q[2] * gyros[0] + glblAtt->q[1] * gyros[1] +
564  glblAtt->q[0] * gyros[2]) * dT * DEG2RAD / 2;
565 
566  // Integrate a time step
567  glblAtt->q[0] = glblAtt->q[0] + qdot[0];
568  glblAtt->q[1] = glblAtt->q[1] + qdot[1];
569  glblAtt->q[2] = glblAtt->q[2] + qdot[2];
570  glblAtt->q[3] = glblAtt->q[3] + qdot[3];
571 
572  if (glblAtt->q[0] < 0) {
573  glblAtt->q[0] = -glblAtt->q[0];
574  glblAtt->q[1] = -glblAtt->q[1];
575  glblAtt->q[2] = -glblAtt->q[2];
576  glblAtt->q[3] = -glblAtt->q[3];
577  }
578  }
579 
580  // Renomalize
581  float qmag = sqrtf(powf(glblAtt->q[0], 2.0f) + powf(glblAtt->q[1], 2.0f) +
582  powf(glblAtt->q[2], 2.0f) + powf(glblAtt->q[3], 2.0f));
583  glblAtt->q[0] = glblAtt->q[0] / qmag;
584  glblAtt->q[1] = glblAtt->q[1] / qmag;
585  glblAtt->q[2] = glblAtt->q[2] / qmag;
586  glblAtt->q[3] = glblAtt->q[3] / qmag;
587 
588  // If quaternion has become inappropriately short or is nan reinit.
589  // THIS SHOULD NEVER ACTUALLY HAPPEN
590  if ((fabs(qmag) < 1e-3) || (qmag != qmag)) {
591  glblAtt->q[0] = 1;
592  glblAtt->q[1] = 0;
593  glblAtt->q[2] = 0;
594  glblAtt->q[3] = 0;
595  }
596 
597  AttitudeActualData attitudeActual;
598  AttitudeActualGet(&attitudeActual);
599 
600  quat_copy(glblAtt->q, &attitudeActual.q1);
601 
602  // Convert into Euler angles (makes assumptions about quarternions and RPY order)
603  Quaternion2RPY(&attitudeActual.q1, &attitudeActual.Roll);
604 
605  AttitudeActualSet(&attitudeActual);
606 }

622

648  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_YAW] * DEG2RAD / 100.0f
649  };
650  Euler2R(rpy, glblAtt->Rbs);
651  glblAtt->rotate = true;
652  }
653 
654  //Check for trim flight request
655  if (attitudeSettings.TrimFlight == ATTITUDESETTINGS_TRIMFLIGHT_START) {
656  glblAtt->trim_accels[0] = 0;
657  glblAtt->trim_accels[1] = 0;
658  glblAtt->trim_accels[2] = 0;
659  glblAtt->trim_samples = 0;
660  glblAtt->trim_requested = true;
661  } else if (attitudeSettings.TrimFlight == ATTITUDESETTINGS_TRIMFLIGHT_LOAD) {
662  glblAtt->trim_requested = false;
663 
664  // Get sensor data mean
665  float a_body[3] = { glblAtt->trim_accels[0] / glblAtt->trim_samples,
666  glblAtt->trim_accels[1] / glblAtt->trim_samples,
667  glblAtt->trim_accels[2] / glblAtt->trim_samples
668  };
669 
670  // Inverse rotation of sensor data, from body frame into sensor frame
671  float a_sensor[3];
672  rot_mult(glblAtt->Rbs, a_body, a_sensor, true);
673 
674  // Temporary variables
675  float psi, theta, phi;
676 
677  psi = attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_YAW] * DEG2RAD / 100.0f;
678 
679  float cP = cosf(psi);
680  float sP = sinf(psi);
681 
682  // In case psi is too small, we have to use a different equation to solve for theta
683  if (fabs(psi) > PI / 2)
684  theta = atanf((a_sensor[1] + cP * (sP * a_sensor[0] -
685  cP * a_sensor[1])) / (sP * a_sensor[2]));
686  else
687  theta = atanf((a_sensor[0] - sP * (sP * a_sensor[0] -
688  cP * a_sensor[1])) / (cP * a_sensor[2]));
689 
690  phi = atan2f((sP * a_sensor[0] - cP * a_sensor[1]) / (-GRAVITY),
691  (a_sensor[2] / cosf(theta) / (-GRAVITY)));
692 
693  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_ROLL] = phi * RAD2DEG * 100.0f;
694  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_PITCH] = theta * RAD2DEG * 100.0f;
695 
696  attitudeSettings.TrimFlight = ATTITUDESETTINGS_TRIMFLIGHT_NORMAL;
697  AttitudeSettingsSet(&attitudeSettings);
698  } else {
699  glblAtt->trim_requested = false;
700  }
701 
702 }

713 static int32_t LLA2NED(int32_t LL[2], float altitude, float geoidSeparation, float *NED)
714 {
715  float *T = glblAtt->T;
716  float dL[3] = { (LL[0] - homeLocation.Latitude) / 10.0e6f * DEG2RAD,
717  (LL[1] - homeLocation.Longitude) / 10.0e6f * DEG2RAD,
718  (altitude + geoidSeparation - homeLocation.Altitude)
719  };
720 
721  NED[0] = T[0] * dL[0];
722  NED[1] = T[1] * dL[1];
723  NED[2] = T[2] * dL[2];
724 
725  return 0;
726 }
