208 static int32_t eventPeriodicCreate(UAVObjEvent* ev, UAVObjEventCallback cb, xQueueHandle queue, uint16_t periodMs)
209 {
210  PeriodicObjectList* objEntry;
211  // Get lock
212  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
213  // Check that the object is not already connected
214  LL_FOREACH(objList, objEntry)
215  {
216  if (objEntry->evInfo.cb == cb &&
217  objEntry->evInfo.queue == queue &&
218  objEntry->evInfo.ev.obj == ev->obj &&
219  objEntry->evInfo.ev.instId == ev->instId &&
220  objEntry->evInfo.ev.event == ev->event)
221  {
222  // Already registered, do nothing
223  xSemaphoreGiveRecursive(mutex);
224  return -1;
225  }
226  }
227  // Create handle
228  objEntry = (PeriodicObjectList*)pvPortMalloc(sizeof(PeriodicObjectList));
229  if (objEntry == NULL) return -1;
230  objEntry->evInfo.ev.obj = ev->obj;
231  objEntry->evInfo.ev.instId = ev->instId;
232  objEntry->evInfo.ev.event = ev->event;
233  objEntry->evInfo.cb = cb;
234  objEntry->evInfo.queue = queue;
235  objEntry->updatePeriodMs = periodMs;
236  objEntry->timeToNextUpdateMs = randomizePeriod(periodMs); // avoid bunching of updates
237  // Add to list
238  LL_APPEND(objList, objEntry);
239  // Release lock
240  xSemaphoreGiveRecursive(mutex);
241  return 0;
242 }

252 static int32_t eventPeriodicUpdate(UAVObjEvent* ev, UAVObjEventCallback cb, xQueueHandle queue, uint16_t periodMs)
253 {
254  PeriodicObjectList* objEntry;
255  // Get lock
256  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
257  // Find object
258  LL_FOREACH(objList, objEntry)
259  {
260  if (objEntry->evInfo.cb == cb &&
261  objEntry->evInfo.queue == queue &&
262  objEntry->evInfo.ev.obj == ev->obj &&
263  objEntry->evInfo.ev.instId == ev->instId &&
264  objEntry->evInfo.ev.event == ev->event)
265  {
266  // Object found, update period
267  objEntry->updatePeriodMs = periodMs;
268  objEntry->timeToNextUpdateMs = randomizePeriod(periodMs); // avoid bunching of updates
269  // Release lock
270  xSemaphoreGiveRecursive(mutex);
271  return 0;
272  }
273  }
274  // If this point is reached the object was not found
275  xSemaphoreGiveRecursive(mutex);
276  return -1;
277 }

282 static void eventTask()
283 {
284  int32_t timeToNextUpdateMs;
285  int32_t delayMs;
286  EventCallbackInfo evInfo;
287 
288  /* Must do this in task context to ensure that TaskMonitor has already finished its init */
289  TaskMonitorAdd(TASKINFO_RUNNING_EVENTDISPATCHER, eventTaskHandle);
290 
291  // Initialize time
292  timeToNextUpdateMs = xTaskGetTickCount()*portTICK_RATE_MS;
293 
294  // Loop forever
295  while (1)
296  {
297  // Calculate delay time
298  delayMs = timeToNextUpdateMs-(xTaskGetTickCount()*portTICK_RATE_MS);
299  if (delayMs < 0)
300  {
301  delayMs = 0;
302  }
303 
304  // Wait for queue message
305  if ( xQueueReceive(queue, &evInfo, delayMs/portTICK_RATE_MS) == pdTRUE )
306  {
307  // Invoke callback, if one
308  if ( evInfo.cb != 0)
309  {
310  evInfo.cb(&evInfo.ev); // the function is expected to copy the event information
311  }
312  }
313 
314  // Process periodic updates
315  if ((xTaskGetTickCount()*portTICK_RATE_MS) >= timeToNextUpdateMs )
316  {
317  timeToNextUpdateMs = processPeriodicUpdates();
318  }
319  }
320 }

326 static int32_t processPeriodicUpdates()
327 {
328  PeriodicObjectList* objEntry;
329  int32_t timeNow;
330  int32_t timeToNextUpdate;
331  int32_t offset;
332 
333  // Get lock
334  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
335 
336  // Iterate through each object and update its timer, if zero then transmit object.
337  // Also calculate smallest delay to next update.
338  timeToNextUpdate = xTaskGetTickCount()*portTICK_RATE_MS + MAX_UPDATE_PERIOD_MS;
339  LL_FOREACH(objList, objEntry)
340  {
341  // If object is configured for periodic updates
342  if (objEntry->updatePeriodMs > 0)
343  {
344  // Check if time for the next update
345  timeNow = xTaskGetTickCount()*portTICK_RATE_MS;
346  if (objEntry->timeToNextUpdateMs <= timeNow)
347  {
348  // Reset timer
349  offset = ( timeNow - objEntry->timeToNextUpdateMs ) % objEntry->updatePeriodMs;
350  objEntry->timeToNextUpdateMs = timeNow + objEntry->updatePeriodMs - offset;
351  // Invoke callback, if one
352  if ( objEntry->evInfo.cb != 0)
353  {
354  objEntry->evInfo.cb(&objEntry->evInfo.ev); // the function is expected to copy the event information
355  }
356  // Push event to queue, if one
357  if ( objEntry->evInfo.queue != 0)
358  {
359  if ( xQueueSend(objEntry->evInfo.queue, &objEntry->evInfo.ev, 0) != pdTRUE ) // do not block if queue is full
360  {
361  if (objEntry->evInfo.ev.obj != NULL)
362  stats.lastErrorID = UAVObjGetID(objEntry->evInfo.ev.obj);
363  ++stats.eventErrors;
364  }
365  }
366  }
367  // Update minimum delay
368  if (objEntry->timeToNextUpdateMs < timeToNextUpdate)
369  {
370  timeToNextUpdate = objEntry->timeToNextUpdateMs;
371  }
372  }
373  }
374 
375  // Done
376  xSemaphoreGiveRecursive(mutex);
377  return timeToNextUpdate;
378 }

385 static uint16_t randomizePeriod(uint16_t periodMs)
386 {
387  static uint32_t seed = 1;
388  uint32_t hi, lo;
389  lo = 16807 * (seed & 0xFFFF);
390  hi = 16807 * (seed >> 16);
391  lo += (hi & 0x7FFF) << 16;
392  lo += hi >> 15;
393  if (lo > 0x7FFFFFFF) lo -= 0x7FFFFFFF;
394  seed = lo;
395  return (uint16_t)( ((float)periodMs * (float)lo) / (float)0x7FFFFFFF );
396 }
