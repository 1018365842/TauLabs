203 static int32_t eventPeriodicCreate(UAVObjEvent* ev, UAVObjEventCallback cb, xQueueHandle queue, uint16_t periodMs)
204 {
205  PeriodicObjectList* objEntry;
206  // Get lock
207  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
208  // Check that the object is not already connected
209  LL_FOREACH(objList, objEntry)
210  {
211  if (objEntry->evInfo.cb == cb &&
212  objEntry->evInfo.queue == queue &&
213  objEntry->evInfo.ev.obj == ev->obj &&
214  objEntry->evInfo.ev.instId == ev->instId &&
215  objEntry->evInfo.ev.event == ev->event)
216  {
217  // Already registered, do nothing
218  xSemaphoreGiveRecursive(mutex);
219  return -1;
220  }
221  }
222  // Create handle
223  objEntry = (PeriodicObjectList*)pvPortMalloc(sizeof(PeriodicObjectList));
224  if (objEntry == NULL) return -1;
225  objEntry->evInfo.ev.obj = ev->obj;
226  objEntry->evInfo.ev.instId = ev->instId;
227  objEntry->evInfo.ev.event = ev->event;
228  objEntry->evInfo.cb = cb;
229  objEntry->evInfo.queue = queue;
230  objEntry->updatePeriodMs = periodMs;
231  objEntry->timeToNextUpdateMs = randomizePeriod(periodMs); // avoid bunching of updates
232  // Add to list
233  LL_APPEND(objList, objEntry);
234  // Release lock
235  xSemaphoreGiveRecursive(mutex);
236  return 0;
237 }

247 static int32_t eventPeriodicUpdate(UAVObjEvent* ev, UAVObjEventCallback cb, xQueueHandle queue, uint16_t periodMs)
248 {
249  PeriodicObjectList* objEntry;
250  // Get lock
251  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
252  // Find object
253  LL_FOREACH(objList, objEntry)
254  {
255  if (objEntry->evInfo.cb == cb &&
256  objEntry->evInfo.queue == queue &&
257  objEntry->evInfo.ev.obj == ev->obj &&
258  objEntry->evInfo.ev.instId == ev->instId &&
259  objEntry->evInfo.ev.event == ev->event)
260  {
261  // Object found, update period
262  objEntry->updatePeriodMs = periodMs;
263  objEntry->timeToNextUpdateMs = randomizePeriod(periodMs); // avoid bunching of updates
264  // Release lock
265  xSemaphoreGiveRecursive(mutex);
266  return 0;
267  }
268  }
269  // If this point is reached the object was not found
270  xSemaphoreGiveRecursive(mutex);
271  return -1;
272 }

277 static void eventTask()
278 {
279  int32_t timeToNextUpdateMs;
280  int32_t delayMs;
281  EventCallbackInfo evInfo;
282 
283  /* Must do this in task context to ensure that TaskMonitor has already finished its init */
284  TaskMonitorAdd(TASKINFO_RUNNING_EVENTDISPATCHER, eventTaskHandle);
285 
286  // Initialize time
287  timeToNextUpdateMs = xTaskGetTickCount()*portTICK_RATE_MS;
288 
289  // Loop forever
290  while (1)
291  {
292  // Calculate delay time
293  delayMs = timeToNextUpdateMs-(xTaskGetTickCount()*portTICK_RATE_MS);
294  if (delayMs < 0)
295  {
296  delayMs = 0;
297  }
298 
299  // Wait for queue message
300  if ( xQueueReceive(queue, &evInfo, delayMs/portTICK_RATE_MS) == pdTRUE )
301  {
302  // Invoke callback, if one
303  if ( evInfo.cb != 0)
304  {
305  evInfo.cb(&evInfo.ev); // the function is expected to copy the event information
306  }
307  }
308 
309  // Process periodic updates
310  if ((xTaskGetTickCount()*portTICK_RATE_MS) >= timeToNextUpdateMs )
311  {
312  timeToNextUpdateMs = processPeriodicUpdates();
313  }
314  }
315 }

321 static int32_t processPeriodicUpdates()
322 {
323  PeriodicObjectList* objEntry;
324  int32_t timeNow;
325  int32_t timeToNextUpdate;
326  int32_t offset;
327 
328  // Get lock
329  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
330 
331  // Iterate through each object and update its timer, if zero then transmit object.
332  // Also calculate smallest delay to next update.
333  timeToNextUpdate = xTaskGetTickCount()*portTICK_RATE_MS + MAX_UPDATE_PERIOD_MS;
334  LL_FOREACH(objList, objEntry)
335  {
336  // If object is configured for periodic updates
337  if (objEntry->updatePeriodMs > 0)
338  {
339  // Check if time for the next update
340  timeNow = xTaskGetTickCount()*portTICK_RATE_MS;
341  if (objEntry->timeToNextUpdateMs <= timeNow)
342  {
343  // Reset timer
344  offset = ( timeNow - objEntry->timeToNextUpdateMs ) % objEntry->updatePeriodMs;
345  objEntry->timeToNextUpdateMs = timeNow + objEntry->updatePeriodMs - offset;
346  // Invoke callback, if one
347  if ( objEntry->evInfo.cb != 0)
348  {
349  objEntry->evInfo.cb(&objEntry->evInfo.ev); // the function is expected to copy the event information
350  }
351  // Push event to queue, if one
352  if ( objEntry->evInfo.queue != 0)
353  {
354  if ( xQueueSend(objEntry->evInfo.queue, &objEntry->evInfo.ev, 0) != pdTRUE ) // do not block if queue is full
355  {
356  if (objEntry->evInfo.ev.obj != NULL)
357  stats.lastErrorID = UAVObjGetID(objEntry->evInfo.ev.obj);
358  ++stats.eventErrors;
359  }
360  }
361  }
362  // Update minimum delay
363  if (objEntry->timeToNextUpdateMs < timeToNextUpdate)
364  {
365  timeToNextUpdate = objEntry->timeToNextUpdateMs;
366  }
367  }
368  }
369 
370  // Done
371  xSemaphoreGiveRecursive(mutex);
372  return timeToNextUpdate;
373 }

380 static uint16_t randomizePeriod(uint16_t periodMs)
381 {
382  static uint32_t seed = 1;
383  uint32_t hi, lo;
384  lo = 16807 * (seed & 0xFFFF);
385  hi = 16807 * (seed >> 16);
386  lo += (hi & 0x7FFF) << 16;
387  lo += hi >> 15;
388  if (lo > 0x7FFFFFFF) lo -= 0x7FFFFFFF;
389  seed = lo;
390  return (uint16_t)( ((float)periodMs * (float)lo) / (float)0x7FFFFFFF );
391 }
