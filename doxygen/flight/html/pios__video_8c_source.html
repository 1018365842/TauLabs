78 void swap_buffers()
79 {
80  // While we could use XOR swap this is more reliable and
81  // dependable and it's only called a few times per second.
82  // Many compliers should optimise these to EXCH instructions.
83  uint8_t *tmp;
84  SWAP_BUFFS(tmp, disp_buffer_mask, draw_buffer_mask);
85  SWAP_BUFFS(tmp, disp_buffer_level, draw_buffer_level);
86 }

88 bool PIOS_Hsync_ISR() {
89 
90  if(dev_cfg->hsync->pin.gpio->IDR & dev_cfg->hsync->pin.init.GPIO_Pin) {
91  //rising
92  if(gLineType == LINE_TYPE_GRAPHICS)
93  {
94  // Activate new line
95  DMA_Cmd(dev_cfg->level.dma.tx.channel, ENABLE);
96  DMA_Cmd(dev_cfg->mask.dma.tx.channel, ENABLE);
97  }
98  } else {
99  //falling
100  gLineType = LINE_TYPE_UNKNOWN; // Default case
101  gActiveLine++;
102 
103  if ((gActiveLine >= GRAPHICS_LINE) && (gActiveLine < (GRAPHICS_LINE + GRAPHICS_HEIGHT))) {
104  gLineType = LINE_TYPE_GRAPHICS;
105  gActivePixmapLine = (gActiveLine - GRAPHICS_LINE);
106  line = gActivePixmapLine*GRAPHICS_WIDTH;
107  }
108 
109  if(gLineType == LINE_TYPE_GRAPHICS)
110  {
111  // Load new line
112  DMA_Cmd(dev_cfg->mask.dma.tx.channel, DISABLE);
113  DMA_Cmd(dev_cfg->level.dma.tx.channel, DISABLE);
114  DMA_MemoryTargetConfig(dev_cfg->level.dma.tx.channel,(uint32_t)&disp_buffer_level[line],DMA_Memory_0);
115  DMA_MemoryTargetConfig(dev_cfg->mask.dma.tx.channel,(uint32_t)&disp_buffer_mask[line],DMA_Memory_0);
116  DMA_SetCurrDataCounter(dev_cfg->level.dma.tx.channel,BUFFER_LINE_LENGTH);
117  DMA_SetCurrDataCounter(dev_cfg->mask.dma.tx.channel,BUFFER_LINE_LENGTH);
118  }
119  }
120 
121  return false;
122 }

166  cfg->level.remap);

168  __builtin_ctz(cfg->level.miso.init.GPIO_Pin),
169  cfg->level.remap);
170  }
171 
172  /* SPI3 MASTER MASKBUFFER */
173  GPIO_Init(cfg->mask.sclk.gpio, (GPIO_InitTypeDef*)&(cfg->mask.sclk.init));
174  GPIO_Init(cfg->mask.mosi.gpio, (GPIO_InitTypeDef*)&(cfg->mask.mosi.init));
175 
176  /* SPI1 SLAVE FRAMEBUFFER */
177  GPIO_Init(cfg->level.sclk.gpio, (GPIO_InitTypeDef*)&(cfg->level.sclk.init));
178  GPIO_Init(cfg->level.miso.gpio, (GPIO_InitTypeDef*)&(cfg->level.miso.init));
179 
180  /* Initialize the SPI block */
181  SPI_Init(cfg->level.regs, (SPI_InitTypeDef*)&(cfg->level.init));
182  SPI_Init(cfg->mask.regs, (SPI_InitTypeDef*)&(cfg->mask.init));
183 
184  /* Enable SPI */
185  SPI_Cmd(cfg->level.regs, ENABLE);
186  SPI_Cmd(cfg->mask.regs, ENABLE);
187 
188  /* Configure DMA for SPI Tx MASTER */
189  DMA_Cmd(cfg->mask.dma.tx.channel, DISABLE);

213

216  PIOS_EXTI_Init(cfg->vsync);
