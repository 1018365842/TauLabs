198 float32_t arm_sin_f32(
199  float32_t x)
200 {
201  float32_t sinVal, fract, in; /* Temporary variables for input, output */
202  int32_t index; /* Index variable */
203  uint32_t tableSize = (uint32_t) TABLE_SIZE; /* Initialise tablesize */
204  float32_t wa, wb, wc, wd; /* Cubic interpolation coefficients */
205  float32_t a, b, c, d; /* Four nearest output values */
206  float32_t *tablePtr; /* Pointer to table */
207  int32_t n;
208  float32_t fractsq, fractby2, fractby6, fractby3, fractsqby2;
209  float32_t oneminusfractby2;
210  float32_t frby2xfrsq, frby6xfrsq;
211 
212  /* input x is in radians */
213  /* Scale the input to [0 1] range from [0 2*PI] , divide input by 2*pi */
214  in = x * 0.159154943092f;
215 
216  /* Calculation of floor value of input */
217  n = (int32_t) in;
218 
219  /* Make negative values towards -infinity */
220  if(x < 0.0f)
221  {
222  n = n - 1;
223  }
224 
225  /* Map input value to [0 1] */
226  in = in - (float32_t) n;
227 
228  /* Calculation of index of the table */
229  index = (uint32_t) (tableSize * in);
230 
231  /* fractional value calculation */
232  fract = ((float32_t) tableSize * in) - (float32_t) index;
233 
234  /* Checking min and max index of table */
235  if(index < 0)
236  {
237  index = 0;
238  }
239  else if(index > 256)
240  {
241  index = 256;
242  }
243 
244  /* Initialise table pointer */
245  tablePtr = (float32_t *) & sinTable[index];
246 
247  /* Read four nearest values of input value from the sin table */
248  a = tablePtr[0];
249  b = tablePtr[1];
250  c = tablePtr[2];
251  d = tablePtr[3];
252 
253  /* Cubic interpolation process */
254  fractsq = fract * fract;
255  fractby2 = fract * 0.5f;
256  fractby6 = fract * 0.166666667f;
257  fractby3 = fract * 0.3333333333333f;
258  fractsqby2 = fractsq * 0.5f;
259  frby2xfrsq = (fractby2) * fractsq;
260  frby6xfrsq = (fractby6) * fractsq;
261  oneminusfractby2 = 1.0f - fractby2;
262  wb = fractsqby2 - fractby3;
263  wc = (fractsqby2 + fract);
264  wa = wb - frby6xfrsq;
265  wb = frby2xfrsq - fractsq;
266  sinVal = wa * a;
267  wc = wc - frby2xfrsq;
268  wd = (frby6xfrsq) - fractby6;
269  wb = wb + oneminusfractby2;
270 
271  /* Calculate sin value */
272  sinVal = (sinVal + (b * wb)) + ((c * wc) + (d * wd));
273 
274  /* Return the output value */
275  return (sinVal);
276 
277 }
