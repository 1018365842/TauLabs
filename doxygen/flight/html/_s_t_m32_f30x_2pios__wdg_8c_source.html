66 uint16_t PIOS_WDG_Init()
67 {
68  uint16_t delay = ((uint32_t) PIOS_WATCHDOG_TIMEOUT * 60) / 16;
69  if (delay > 0x0fff)
70  delay = 0x0fff;
71 #if defined(PIOS_INCLUDE_WDG)
72  DBGMCU_Config(DBGMCU_IWDG_STOP, ENABLE); // make the watchdog stop counting in debug mode
73  IWDG_WriteAccessCmd(IWDG_WriteAccess_Enable);
74  IWDG_SetPrescaler(IWDG_Prescaler_16);
75  IWDG_SetReload(delay);
76  IWDG_ReloadCounter();
77  IWDG_Enable();
78 
79  // watchdog flags now stored in backup registers
80  PWR_BackupAccessCmd(ENABLE);
81 
82  wdg_configuration.bootup_flags = RTC_ReadBackupRegister(PIOS_WDG_REGISTER);
83 #endif
84  return delay;
85 }

123 bool PIOS_WDG_UpdateFlag(uint16_t flag)
124 {
125  // we can probably avoid using a semaphore here which will be good for
126  // efficiency and not blocking critical tasks. race condition could
127  // overwrite their flag update, but unlikely to block _all_ of them
128  // for the timeout window
129  uint16_t cur_flags = RTC_ReadBackupRegister(PIOS_WDG_REGISTER);
130 
131  if((cur_flags | flag) == wdg_configuration.used_flags) {
132  PIOS_WDG_Clear();
133  RTC_WriteBackupRegister(PIOS_WDG_REGISTER, flag);
134  return true;
135  } else {
136  RTC_WriteBackupRegister(PIOS_WDG_REGISTER, cur_flags | flag);
137  return false;
138  }
139 
140 }
