90 void
91 mem_copy(PmMemSpace_t memspace,
92  uint8_t **pdest, uint8_t const **psrc, uint16_t count)
93 {
94  /* Copy memory from RAM */
95  if (memspace == MEMSPACE_RAM)
96  {
97  sli_memcpy(*pdest, *psrc, count);
98  *psrc += count;
99  *pdest += count;
100  return;
101  }
102 
103  /* Copy memory from non-RAM to RAM */
104  else
105  {
106  uint8_t b;
107 
108  for (; count > 0; count--)
109  {
110  b = mem_getByte(memspace, psrc);
111  **pdest = b;
112  (*pdest)++;
113  }
114  return;
115  }
116 }

120 mem_getStringLength(PmMemSpace_t memspace, uint8_t const *const pstr)
121 {
122  uint8_t const *psrc;
123 
124  /* If source is in RAM, use a possibly optimized strlen */
125  if (memspace == MEMSPACE_RAM)
126  {
127  return sli_strlen((char const *)pstr);
128  }
129 
130  /* Otherwise calculate string length */
131  psrc = pstr;
