66 static int32_t GenericI2CSensorInitialize(void)
67 {
68 
69 #ifdef MODULE_GenericI2CSensor_BUILTIN
70  module_enabled = true;
71 #else
72  uint8_t module_state[MODULESETTINGS_ADMINSTATE_NUMELEM];
73  ModuleSettingsAdminStateGet(module_state);
74  if (module_state[MODULESETTINGS_ADMINSTATE_GENERICI2CSENSOR] == MODULESETTINGS_ADMINSTATE_ENABLED) {
75  module_enabled = true;
76  } else {
77  module_enabled = false;
78  }
79 #endif
80 
81  if (!module_enabled)
82  return -1;
83 
84  /* Module is enabled, determine which program to run (if any) */
85  uint8_t selected_program;
86  ModuleSettingsI2CVMProgramSelectGet(&selected_program);
87 
88  switch (selected_program) {
89  case MODULESETTINGS_I2CVMPROGRAMSELECT_USER:
90  I2CVMUserProgramInitialize();
91  uint32_t * user_program;
92  user_program = pvPortMalloc(sizeof(((I2CVMUserProgramData *)0)->Program));
93  if (!user_program) {
94  /* Failed to allocate sufficient memory for the user program */
95  return -1;
96  }
97  I2CVMUserProgramProgramGet(user_program);
98  i2cvm_program = user_program;
99  i2cvm_program_len = I2CVMUSERPROGRAM_PROGRAM_NUMELEM;
100  break;
101  case MODULESETTINGS_I2CVMPROGRAMSELECT_OPBAROALTIMETER:
102  {
103  extern const uint32_t vmprog_op_mag_baro[];
104  extern const uint32_t vmprog_op_mag_baro_len;
105  i2cvm_program = vmprog_op_mag_baro;
106  i2cvm_program_len = vmprog_op_mag_baro_len;
107  }
108  break;
109  case MODULESETTINGS_I2CVMPROGRAMSELECT_ENDIANTEST:
110  {
111  extern const uint32_t vmprog_endiantest[];
112  extern const uint32_t vmprog_endiantest_len;
113  i2cvm_program = vmprog_endiantest;
114  i2cvm_program_len = vmprog_endiantest_len;
115  }
116  break;
117  case MODULESETTINGS_I2CVMPROGRAMSELECT_MATHTEST:
118  {
119  extern const uint32_t vmprog_mathtest[];
120  extern const uint32_t vmprog_mathtest_len;
121  i2cvm_program = vmprog_mathtest;
122  i2cvm_program_len = vmprog_mathtest_len;
123  }
124  break;
125  case MODULESETTINGS_I2CVMPROGRAMSELECT_NONE:
126  default:
127  /* No program selected, module will not start */
128  break;
129  }
130 
131  /* Make sure we have something to run */
132  if ((i2cvm_program == NULL) || (i2cvm_program_len == 0)) {
133  module_enabled = false;
134  return -1;
135  }
136 
137  I2CVMInitialize();
138 
139  return 0;
140 }

145 static void GenericI2CSensorTask(void *parameters)
146 {
147  // Main task loop
148  while (1) {
149  /* Run the selected program */
150  if (i2c_vm_run(i2cvm_program, i2cvm_program_len, PIOS_I2C_MAIN_ADAPTER)) {
151  /* Program ran to completion. This could be because the program is
152  * empty or does not infinitely loop.
153  * Delay in order to prevent these programs from consuming all CPU.
154  */
155  vTaskDelay(10 / portTICK_RATE_MS);
156  } else {
157  /* Program faulted
158  * Delay to prevent beoken programs from consuming all CPU
159  */
160  vTaskDelay(100 / portTICK_RATE_MS);
161  }
162  }
163 }
