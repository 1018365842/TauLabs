166 portSTACK_TYPE *pxPortInitialiseStack( portSTACK_TYPE *pxTopOfStack, pdTASK_CODE pxCode, void *pvParameters )
167 {
168  /* Simulate the stack frame as it would be created by a context switch
169  interrupt. */
170  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
171  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
172  pxTopOfStack--;
173  *pxTopOfStack = ( portSTACK_TYPE ) pxCode; /* PC */
174  pxTopOfStack--;
175  *pxTopOfStack = 0; /* LR */
176  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
177  *pxTopOfStack = ( portSTACK_TYPE ) pvParameters; /* R0 */
178  pxTopOfStack -= 8; /* R11, R10, R9, R8, R7, R6, R5 and R4. */
179 
180  return pxTopOfStack;
181 }

184 void vPortSVCHandler( void )
185 {
186  __asm volatile (
187  " ldr r3, pxCurrentTCBConst2 \n" /* Restore the context. */
188  " ldr r1, [r3] \n" /* Use pxCurrentTCBConst to get the pxCurrentTCB address. */
189  " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
190  " ldmia r0!, {r4-r11} \n" /* Pop the registers that are not automatically saved on exception entry and the critical nesting count. */
191  " msr psp, r0 \n" /* Restore the task stack pointer. */
192  " mov r0, #0 \n"
193  " msr basepri, r0 \n"
194  " orr r14, #0xd \n"
195  " bx r14 \n"
196  " \n"
197  " .align 2 \n"
198  "pxCurrentTCBConst2: .word pxCurrentTCB \n"
199  );
200 }

203 static void prvPortStartFirstTask( void )
204 {
205  __asm volatile(
206  " ldr r0, =0xE000ED08 \n" /* Use the NVIC offset register to locate the stack. */
207  " ldr r0, [r0] \n"
208  " ldr r0, [r0] \n"
209  " msr msp, r0 \n" /* Set the msp back to the start of the stack. */
210  " cpsie i \n" /* Globally enable interrupts. */
211  " svc 0 \n" /* System call to start first task. */
212  " nop \n"
213  );
214 }

307 void xPortPendSVHandler( void )
308 {
309  /* This is a naked function. */
310 
311  __asm volatile
312  (
313  " mrs r0, psp \n"
314  " \n"
315  " ldr r3, pxCurrentTCBConst \n" /* Get the location of the current TCB. */
316  " ldr r2, [r3] \n"
317  " \n"
318  " stmdb r0!, {r4-r11} \n" /* Save the remaining registers. */
319  " str r0, [r2] \n" /* Save the new top of stack into the first member of the TCB. */
320  " \n"
321  " stmdb sp!, {r3, r14} \n"
322  " mov r0, %0 \n"
323  " msr basepri, r0 \n"
324  " bl vTaskSwitchContext \n"
325  " mov r0, #0 \n"
326  " msr basepri, r0 \n"
327  " ldmia sp!, {r3, r14} \n"
328  " \n" /* Restore the context, including the critical nesting count. */
329  " ldr r1, [r3] \n"
330  " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
331  " ldmia r0!, {r4-r11} \n" /* Pop the registers. */
332  " msr psp, r0 \n"
333  " bx r14 \n"
334  " \n"
335  " .align 2 \n"
336  "pxCurrentTCBConst: .word pxCurrentTCB \n"
337  ::"i"(configMAX_SYSCALL_INTERRUPT_PRIORITY)
338  );
339 }

343 {
344  /* If using preemption, also force a context switch. */
345  #if configUSE_PREEMPTION == 1
346  portNVIC_INT_CTRL_REG = portNVIC_PENDSVSET_BIT;

349  /* Only reset the systick load register if configUSE_TICKLESS_IDLE is set to

352  to generate the tick interrupt. */

354  portNVIC_SYSTICK_LOAD_REG = ulTimerReloadValueForOneTick;

367  __attribute__((weak)) void vPortSuppressTicksAndSleep( portTickType xExpectedIdleTime )
368  {
369  unsigned long ulReloadValue, ulCompleteTickPeriods, ulCompletedSysTickIncrements;
370 
371  /* Make sure the SysTick reload value does not overflow the counter. */
372  if( xExpectedIdleTime > xMaximumPossibleSuppressedTicks )
373  {
374  xExpectedIdleTime = xMaximumPossibleSuppressedTicks;
375  }
376 
377  /* Calculate the reload value required to wait xExpectedIdleTime
378  tick periods. -1 is used because this code will execute part way
379  through one of the tick periods, and the fraction of a tick period is
380  accounted for later. */
381  ulReloadValue = ( ulTimerReloadValueForOneTick * ( xExpectedIdleTime - 1UL ) );
382  if( ulReloadValue > ulStoppedTimerCompensation )
383  {
384  ulReloadValue -= ulStoppedTimerCompensation;
385  }
386 
387  /* Stop the SysTick momentarily. The time the SysTick is stopped for
388  is accounted for as best it can be, but using the tickless mode will
389  inevitably result in some tiny drift of the time maintained by the
390  kernel with respect to calendar time. */
391  portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT;
392 
393  /* If a context switch is pending then abandon the low power entry as
394  the context switch might have been pended by an external interrupt that
395  requires processing. */
396  if( ( portNVIC_INT_CTRL_REG & portNVIC_PENDSVSET_BIT ) != 0 )
397  {
398  /* Restart SysTick. */
399  portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT | portNVIC_SYSTICK_ENABLE_BIT;
400  }
401  else
402  {
403  /* Adjust the reload value to take into account that the current
404  time slice is already partially complete. */
405  ulReloadValue += ( portNVIC_SYSTICK_LOAD_REG - ( portNVIC_SYSTICK_LOAD_REG - portNVIC_SYSTICK_CURRENT_VALUE_REG ) );
406  portNVIC_SYSTICK_LOAD_REG = ulReloadValue;
407 
408  /* Clear the SysTick count flag and set the count value back to
409  zero. */
410  portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
411 
412  /* Restart SysTick. */
413  portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT | portNVIC_SYSTICK_ENABLE_BIT;
414 
415  /* Sleep until something happens. */
416  configPRE_SLEEP_PROCESSING( xExpectedIdleTime );
417  if( xExpectedIdleTime > 0 )
418  {
419  __asm volatile( "wfi" );
420  }
421  configPOST_SLEEP_PROCESSING( xExpectedIdleTime );
422 
423  /* Stop SysTick. Again, the time the SysTick is stopped for is
424  accounted for as best it can be, but using the tickless mode will
425  inevitably result in some tiny drift of the time maintained by the
426  kernel with respect to calendar time. */
427  portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT;
428 
429  if( ( portNVIC_SYSTICK_CTRL_REG & portNVIC_SYSTICK_COUNT_FLAG_BIT ) != 0 )
430  {
431  /* The tick interrupt has already executed, and the SysTick
432  count reloaded with the portNVIC_SYSTICK_LOAD_REG value.
433  Reset the portNVIC_SYSTICK_LOAD_REG with whatever remains of
434  this tick period. */
435  portNVIC_SYSTICK_LOAD_REG = ulTimerReloadValueForOneTick - ( ulReloadValue - portNVIC_SYSTICK_CURRENT_VALUE_REG );
436 
437  /* The tick interrupt handler will already have pended the tick
438  processing in the kernel. As the pending tick will be
439  processed as soon as this function exits, the tick value
440  maintained by the tick is stepped forward by one less than the
441  time spent waiting. */
442  ulCompleteTickPeriods = xExpectedIdleTime - 1UL;
443  }
444  else
445  {
446  /* Something other than the tick interrupt ended the sleep.
447  Work out how long the sleep lasted. */
448  ulCompletedSysTickIncrements = ( xExpectedIdleTime * ulTimerReloadValueForOneTick ) - portNVIC_SYSTICK_CURRENT_VALUE_REG;
449 
450  /* How many complete tick periods passed while the processor
451  was waiting? */
452  ulCompleteTickPeriods = ulCompletedSysTickIncrements / ulTimerReloadValueForOneTick;
453 
454  /* The reload value is set to whatever fraction of a single tick
455  period remains. */
456  portNVIC_SYSTICK_LOAD_REG = ( ( ulCompleteTickPeriods + 1 ) * ulTimerReloadValueForOneTick ) - ulCompletedSysTickIncrements;
457  }
458 
459  /* Restart SysTick so it runs from portNVIC_SYSTICK_LOAD_REG
460  again, then set portNVIC_SYSTICK_LOAD_REG back to its standard
461  value. */
462  portNVIC_SYSTICK_CURRENT_VALUE_REG = 0UL;
463  portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT | portNVIC_SYSTICK_ENABLE_BIT;
464 
465  vTaskStepTick( ulCompleteTickPeriods );
466  }
467  }

469 #endif /* #if configUSE_TICKLESS_IDLE */

472 /*
473  * Setup the systick timer to generate the tick interrupts at the required
474  * frequency.
475  */
476 __attribute__(( weak )) void vPortSetupTimerInterrupt( void )
477 {
478  /* Calculate the constants required to configure the tick interrupt. */
479  #if configUSE_TICKLESS_IDLE == 1
480  {
481  ulTimerReloadValueForOneTick = ( configSYSTICK_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL;
482  xMaximumPossibleSuppressedTicks = 0xffffffUL / ( ( configSYSTICK_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL );
483  ulStoppedTimerCompensation = 45UL / ( configCPU_CLOCK_HZ / configSYSTICK_CLOCK_HZ );
484  }
485  #endif /* configUSE_TICKLESS_IDLE */
486 
487  /* Configure SysTick to interrupt at the requested rate. */
488  portNVIC_SYSTICK_LOAD_REG = ( configSYSTICK_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL;;
489  portNVIC_SYSTICK_CTRL_REG = portNVIC_SYSTICK_CLK_BIT | portNVIC_SYSTICK_INT_BIT | portNVIC_SYSTICK_ENABLE_BIT;
490 }
