108 int32_t PIOS_USB_Init(uint32_t * usb_id, const struct pios_usb_cfg * cfg)
109 {
110  PIOS_Assert(usb_id);
111  PIOS_Assert(cfg);
112 
113  struct pios_usb_dev * usb_dev;
114 
115  usb_dev = (struct pios_usb_dev *) PIOS_USB_alloc();
116  if (!usb_dev) goto out_fail;
117 
118  /* Bind the configuration to the device instance */
119  usb_dev->cfg = cfg;
120 
121  PIOS_USB_Reenumerate();
122 
123  /*
124  * This is a horrible hack to make this available to
125  * the interrupt callbacks. This should go away ASAP.
126  */
127  pios_usb_com_id = (uint32_t) usb_dev;
128 
129  /* Enable the USB Interrupts */
130  NVIC_Init((NVIC_InitTypeDef*)&usb_dev->cfg->irq.init);
131 
132  /* Configure USB D-/D+ (DM/DP) pins */
133  GPIO_InitTypeDef GPIO_InitStructure;
134  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_11 | GPIO_Pin_12;
135  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
136  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
137  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
138  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL;
139  GPIO_Init(GPIOA, &GPIO_InitStructure);
140 
141  GPIO_PinAFConfig(GPIOA, GPIO_PinSource11, GPIO_AF_14);
142  GPIO_PinAFConfig(GPIOA, GPIO_PinSource12, GPIO_AF_14);
143 
144  /* Configure VBUS sense pin */
145  if (usb_dev->cfg->vsense.gpio)
146  GPIO_Init(usb_dev->cfg->vsense.gpio, (GPIO_InitTypeDef*)&usb_dev->cfg->vsense.init);
147 
148  /* Select USBCLK source */
149  RCC_USBCLKConfig(RCC_USBCLKSource_PLLCLK_1Div5);
150  /* Enable the USB clock */
151  RCC_APB1PeriphClockCmd(RCC_APB1Periph_USB, ENABLE);
152 
153  USB_Init();
154  USB_SIL_Init();
155 
156  *usb_id = (uint32_t) usb_dev;
157 
158  return 0; /* No error */
159 
160 out_fail:
161  return(-1);
162 }

193 int32_t PIOS_USB_Reenumerate()
194 {
195  /* Force USB reset and power-down (this will also release the USB pins for direct GPIO control) */
196  _SetCNTR(CNTR_FRES | CNTR_PDWN);
197 
198  /* Using a "dirty" method to force a re-enumeration: */
199  /* Force DPM (Pin PA12) low for ca. 10 mS before USB Tranceiver will be enabled */
200  /* This overrules the external Pull-Up at PA12, and at least Windows & MacOS will enumerate again */
201  GPIO_InitTypeDef GPIO_InitStructure;
202  GPIO_StructInit(&GPIO_InitStructure);
203  GPIO_InitStructure.GPIO_Pin = GPIO_Pin_12;
204  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
205  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
206  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
207  GPIO_Init(GPIOA, &GPIO_InitStructure);
208 
209  PIOS_DELAY_WaitmS(50);
210 
211  /* Release power-down, still hold reset */
212  _SetCNTR(CNTR_PDWN);
213  PIOS_DELAY_WaituS(5);
214 
215  /* CNTR_FRES = 0 */
216  _SetCNTR(0);
217 
218  /* Clear pending interrupts */
219  _SetISTR(0);
220 
221  /* set back to alternate function */
222  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
223  GPIO_Init(GPIOA, &GPIO_InitStructure);
224 
225  /* Configure USB clock */
226  /* USBCLK = PLLCLK / 1.5 */
227  RCC_USBCLKConfig(RCC_USBCLKSource_PLLCLK_1Div5);
228  /* Enable USB clock */
229  RCC_APB1PeriphClockCmd(RCC_APB1Periph_USB, ENABLE);
230 
231  return 0;
232 }
