163 void PIOS_USB_RCTX_Update(uint32_t usbrctx_id, const uint16_t channel[], const int16_t channel_min[], const int16_t channel_max[], uint8_t num_channels)
164 {
165  struct pios_usb_rctx_dev * usb_rctx_dev = (struct pios_usb_rctx_dev *)usbrctx_id;
166 
167  bool valid = PIOS_USB_RCTX_validate(usb_rctx_dev);
168  PIOS_Assert(valid);
169 
170  if (!PIOS_USB_CheckAvailable(usb_rctx_dev->lower_id)) {
171  return;
172  }
173 
174  for (uint8_t i = 0;
175  i < PIOS_USB_RCTX_NUM_CHANNELS && i < num_channels;
176  i++) {
177  int16_t min = channel_min[i];
178  int16_t max = channel_max[i];
179  uint16_t val = channel[i];
180 
181  if (channel_min[i] > channel_max[i]) {
182  /* This channel is reversed, flip min and max */
183  min = channel_max[i];
184  max = channel_min[i];
185 
186  /* and flip val to be an offset from the lower end of the range */
187  val = channel_min[i] - channel[i] + channel_max[i];
188  }
189 
190  /* Scale channel linearly between min and max */
191  if (min == max) {
192  val = 0;
193  } else {
194  if (val < min) val = min;
195  if (val > max) val = max;
196 
197  val = (val - min) * (65535 / (max - min));
198  }
199 
200  usb_rctx_dev->report.vals[i] = val;
201  }
202 
203  if (GetEPTxStatus(usb_rctx_dev->cfg->data_tx_ep) == EP_TX_VALID) {
204  /* Endpoint is already transmitting */
205  return;
206  }
207 
208  PIOS_USB_RCTX_SendReport(usb_rctx_dev);
209 }
