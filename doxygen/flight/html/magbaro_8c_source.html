129 static void magbaroTask(void *parameters)
130 {
131  BaroAltitudeData data;
132  portTickType lastSysTime;
133 
134 #if defined(PIOS_INCLUDE_BMP085)
135  PIOS_BMP085_Init();
136 #endif
137 #if defined(PIOS_INCLUDE_HMC5883)
138  PIOS_HMC5883_Init(PIOS_I2C_MAIN_ADAPTER, &pios_hmc5883_cfg);
139 #endif
140 
141  mag_test = 0;
142  // Main task loop
143  lastSysTime = xTaskGetTickCount();
144  while (1)
145  {
146 #if defined(PIOS_INCLUDE_BMP085)
147  // Update the temperature data
148  PIOS_BMP085_StartADC(TemperatureConv);
149 #ifdef PIOS_BMP085_HAS_GPIOS
150  xSemaphoreTake(PIOS_BMP085_EOC, portMAX_DELAY);
151 #else
152  vTaskDelay(5 / portTICK_RATE_MS);
153 #endif
154  PIOS_BMP085_ReadADC();
155  alt_ds_temp += PIOS_BMP085_GetTemperature();
156 
157  // Update the pressure data
158  PIOS_BMP085_StartADC(PressureConv);
159 #ifdef PIOS_BMP085_HAS_GPIOS
160  xSemaphoreTake(PIOS_BMP085_EOC, portMAX_DELAY);
161 #else
162  vTaskDelay(26 / portTICK_RATE_MS);
163 #endif
164  PIOS_BMP085_ReadADC();
165  alt_ds_pres += PIOS_BMP085_GetPressure();
166 
167  if (++alt_ds_count >= alt_ds_size)
168  {
169  alt_ds_count = 0;
170 
171  // Convert from 1/10ths of degC to degC
172  data.Temperature = alt_ds_temp / (10.0 * alt_ds_size);
173  alt_ds_temp = 0;
174 
175  // Convert from Pa to kPa
176  data.Pressure = alt_ds_pres / (1000.0f * alt_ds_size);
177  alt_ds_pres = 0;
178 
179  // Compute the current altitude (all pressures in kPa)
180  data.Altitude = 44330.0 * (1.0 - powf((data.Pressure / (BMP085_P0 / 1000.0)), (1.0 / 5.255)));
181 
182  // Update the AltitudeActual UAVObject
183  BaroAltitudeSet(&data);
184  }
185 #endif
186 
187 #if defined(PIOS_INCLUDE_HMC5883)
188  struct pios_sensor_mag_data mags;
189  xQueueHandle queue = PIOS_SENSORS_GetQueue(PIOS_SENSOR_MAG);
190  if (queue != NULL && xQueueReceive(queue, (void *) &mags, 0) != errQUEUE_EMPTY) {
191  update_mags(&mags);
192  }
193 #endif
194 
195  // Delay until it is time to read the next sample
196  vTaskDelayUntil(&lastSysTime, UPDATE_PERIOD / portTICK_RATE_MS);
197  }
198 }
