153 void arm_rfft_f32(
154  const arm_rfft_instance_f32 * S,
155  float32_t * pSrc,
156  float32_t * pDst)
157 {
158  const arm_cfft_radix4_instance_f32 *S_CFFT = S->pCfft;
159 
160 
161  /* Calculation of Real IFFT of input */
162  if(S->ifftFlagR == 1u)
163  {
164  /* Real IFFT core process */
165  arm_split_rifft_f32(pSrc, S->fftLenBy2, S->pTwiddleAReal,
166  S->pTwiddleBReal, pDst, S->twidCoefRModifier);
167 
168 
169  /* Complex radix-4 IFFT process */
170  arm_radix4_butterfly_inverse_f32(pDst, S_CFFT->fftLen,
171  S_CFFT->pTwiddle,
172  S_CFFT->twidCoefModifier,
173  S_CFFT->onebyfftLen);
174 
175  /* Bit reversal process */
176  if(S->bitReverseFlagR == 1u)
177  {
178  arm_bitreversal_f32(pDst, S_CFFT->fftLen,
179  S_CFFT->bitRevFactor, S_CFFT->pBitRevTable);
180  }
181  }
182  else
183  {
184 
185  /* Calculation of RFFT of input */
186 
187  /* Complex radix-4 FFT process */
188  arm_radix4_butterfly_f32(pSrc, S_CFFT->fftLen,
189  S_CFFT->pTwiddle, S_CFFT->twidCoefModifier);
190 
191  /* Bit reversal process */
192  if(S->bitReverseFlagR == 1u)
193  {
194  arm_bitreversal_f32(pSrc, S_CFFT->fftLen,
195  S_CFFT->bitRevFactor, S_CFFT->pBitRevTable);
196  }
197 
198 
199  /* Real FFT core process */
200  arm_split_rfft_f32(pSrc, S->fftLenBy2, S->pTwiddleAReal,
201  S->pTwiddleBReal, pDst, S->twidCoefRModifier);
202  }

221 void arm_split_rfft_f32(
222  float32_t * pSrc,
223  uint32_t fftLen,
224  float32_t * pATable,
225  float32_t * pBTable,
226  float32_t * pDst,
227  uint32_t modifier)
228 {
229  uint32_t i; /* Loop Counter */
230  float32_t outR, outI; /* Temporary variables for output */
231  float32_t *pCoefA, *pCoefB; /* Temporary pointers for twiddle factors */
232  float32_t CoefA1, CoefA2, CoefB1; /* Temporary variables for twiddle coefficients */
233  float32_t *pDst1 = &pDst[2], *pDst2 = &pDst[(4u * fftLen) - 1u]; /* temp pointers for output buffer */
234  float32_t *pSrc1 = &pSrc[2], *pSrc2 = &pSrc[(2u * fftLen) - 1u]; /* temp pointers for input buffer */
235 
236  /* Init coefficient pointers */
237  pCoefA = &pATable[modifier * 2u];
238  pCoefB = &pBTable[modifier * 2u];
239 
240  i = fftLen - 1u;
241 
242  while(i > 0u)
243  {
244  /*
245  outR = (pSrc[2 * i] * pATable[2 * i] - pSrc[2 * i + 1] * pATable[2 * i + 1]
246  + pSrc[2 * n - 2 * i] * pBTable[2 * i] +
247  pSrc[2 * n - 2 * i + 1] * pBTable[2 * i + 1]);
248  */
249 
250  /* outI = (pIn[2 * i + 1] * pATable[2 * i] + pIn[2 * i] * pATable[2 * i + 1] +
251  pIn[2 * n - 2 * i] * pBTable[2 * i + 1] -
252  pIn[2 * n - 2 * i + 1] * pBTable[2 * i]); */
253 
254  /* read pATable[2 * i] */
255  CoefA1 = *pCoefA++;
256  /* pATable[2 * i + 1] */
257  CoefA2 = *pCoefA;
258 
259  /* pSrc[2 * i] * pATable[2 * i] */
260  outR = *pSrc1 * CoefA1;
261  /* pSrc[2 * i] * CoefA2 */
262  outI = *pSrc1++ * CoefA2;
263 
264  /* (pSrc[2 * i + 1] + pSrc[2 * fftLen - 2 * i + 1]) * CoefA2 */
265  outR -= (*pSrc1 + *pSrc2) * CoefA2;
266  /* pSrc[2 * i + 1] * CoefA1 */
267  outI += *pSrc1++ * CoefA1;
268 
269  CoefB1 = *pCoefB;
270 
271  /* pSrc[2 * fftLen - 2 * i + 1] * CoefB1 */
272  outI -= *pSrc2-- * CoefB1;
273  /* pSrc[2 * fftLen - 2 * i] * CoefA2 */
274  outI -= *pSrc2 * CoefA2;
275 
276  /* pSrc[2 * fftLen - 2 * i] * CoefB1 */
277  outR += *pSrc2-- * CoefB1;
278 
279  /* write output */
280  *pDst1++ = outR;
281  *pDst1++ = outI;
282 
283  /* write complex conjugate output */
284  *pDst2-- = -outI;
285  *pDst2-- = outR;
286 
287  /* update coefficient pointer */
288  pCoefB = pCoefB + (modifier * 2u);
289  pCoefA = pCoefA + ((modifier * 2u) - 1u);
290 
291  i--;
292 
293  }
294 
295  pDst[2u * fftLen] = pSrc[0] - pSrc[1];
296  pDst[(2u * fftLen) + 1u] = 0.0f;
297 
298  pDst[0] = pSrc[0] + pSrc[1];
299  pDst[1] = 0.0f;
300 
301 }

324  float32_t *pCoefA, *pCoefB; /* Temporary pointers for twiddle factors */
325  float32_t CoefA1, CoefA2, CoefB1; /* Temporary variables for twiddle coefficients */
326  float32_t *pSrc1 = &pSrc[0], *pSrc2 = &pSrc[(2u * fftLen) + 1u];
327 
328  pCoefA = &pATable[0];
329  pCoefB = &pBTable[0];
330 
331  while(fftLen > 0u)
332  {
333  /*
334  outR = (pIn[2 * i] * pATable[2 * i] + pIn[2 * i + 1] * pATable[2 * i + 1] +
335  pIn[2 * n - 2 * i] * pBTable[2 * i] -
336  pIn[2 * n - 2 * i + 1] * pBTable[2 * i + 1]);
337 
338  outI = (pIn[2 * i + 1] * pATable[2 * i] - pIn[2 * i] * pATable[2 * i + 1] -
339  pIn[2 * n - 2 * i] * pBTable[2 * i + 1] -
340  pIn[2 * n - 2 * i + 1] * pBTable[2 * i]);
341 
342  */
343 
344  CoefA1 = *pCoefA++;
345  CoefA2 = *pCoefA;
346 
347  /* outR = (pSrc[2 * i] * CoefA1 */
348  outR = *pSrc1 * CoefA1;
349 
350  /* - pSrc[2 * i] * CoefA2 */
351  outI = -(*pSrc1++) * CoefA2;
352 
353  /* (pSrc[2 * i + 1] + pSrc[2 * fftLen - 2 * i + 1]) * CoefA2 */
354  outR += (*pSrc1 + *pSrc2) * CoefA2;
355 
356  /* pSrc[2 * i + 1] * CoefA1 */
357  outI += (*pSrc1++) * CoefA1;
358 
359  CoefB1 = *pCoefB;
360 
361  /* - pSrc[2 * fftLen - 2 * i + 1] * CoefB1 */
362  outI -= *pSrc2-- * CoefB1;
363 
364  /* pSrc[2 * fftLen - 2 * i] * CoefB1 */
365  outR += *pSrc2 * CoefB1;
366 
367  /* pSrc[2 * fftLen - 2 * i] * CoefA2 */
368  outI += *pSrc2-- * CoefA2;
369 
370  /* write output */
371  *pDst++ = outR;
372  *pDst++ = outI;
373 
374  /* update coefficient pointer */
375  pCoefB = pCoefB + (modifier * 2u);
376  pCoefA = pCoefA + ((modifier * 2u) - 1u);
377 
378  /* Decrement loop count */
379  fftLen--;
380  }
