113 static int32_t uavoMavlinkBridgeInitialize(void) {
114  mavlink_port = PIOS_COM_MAVLINK;
115 
116  uint8_t module_state[MODULESETTINGS_ADMINSTATE_NUMELEM];
117  ModuleSettingsAdminStateGet(module_state);

119  if (mavlink_port
120  && (module_state[MODULESETTINGS_ADMINSTATE_UAVOMAVLINKBRIDGE]
121  == MODULESETTINGS_ADMINSTATE_ENABLED)) {

453 static bool stream_trigger(enum MAV_DATA_STREAM stream_num) {
454  uint8_t rate = (uint8_t) mav_rates[stream_num];
455 
456  if (rate == 0) {
457  return false;
458  }
459 
460  if (stream_ticks[stream_num] == 0) {
461  // we're triggering now, setup the next trigger point
462  if (rate > TASK_RATE_HZ) {
463  rate = TASK_RATE_HZ;
464  }
465  stream_ticks[stream_num] = (TASK_RATE_HZ / rate);
466  return true;
467  }
468 
469  // count down at 50Hz
470  stream_ticks[stream_num]--;
471  return false;
