103 int32_t PIOS_ADC_Init(uintptr_t *adc_id, const struct pios_adc_driver *driver, uint32_t lower_id)
104 {
105  PIOS_Assert(adc_id);
106  PIOS_Assert(driver);
107  if (sub_device_list.number_of_devices >= PIOS_ADC_SUB_DRIVER_MAX_INSTANCES)
108  return -1;
109  struct pios_adc_dev * adc_dev;
110 
111  adc_dev = (struct pios_adc_dev *) PIOS_ADC_Allocate();
112 
113  if (!adc_dev)
114  return -1;
115 
116  adc_dev->driver = driver;
117  adc_dev->lower_id = lower_id;
118 
119  sub_device_list.sub_device_pointers[sub_device_list.number_of_devices] = adc_dev;
120  sub_device_list.number_of_devices++;
121  return 0;
122 }

188 int32_t PIOS_ADC_GetChannel(uint32_t channel)
189 {
190  uint32_t offset = 0;
191  for (uint8_t x = 0; x < sub_device_list.number_of_devices; ++x) {
192  struct pios_adc_dev * adc_dev = sub_device_list.sub_device_pointers[x];
193  if (!PIOS_ADC_validate(adc_dev)) {
194  PIOS_DEBUG_Assert(0);
195  continue;
196  } else if (adc_dev->driver->number_of_channels) {
197  uint32_t num_channels_for_this_device = adc_dev->driver->number_of_channels(adc_dev->lower_id);
198  if (adc_dev->driver->get_pin && (channel < offset + num_channels_for_this_device)) {
199  return (adc_dev->driver->get_pin)(adc_dev->lower_id, channel - offset);
200  } else
201  offset += num_channels_for_this_device;
202  }
203  }
204  return -1;
205 }
