48 int32_t GPSPositionInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(GPSPOSITION_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(GPSPOSITION_OBJID,
56  GPSPOSITION_ISSINGLEINST, GPSPOSITION_ISSETTINGS, GPSPOSITION_NUMBYTES, &GPSPositionSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

74 void GPSPositionSetDefaults(UAVObjHandle obj, uint16_t instId)
75 {
76  GPSPositionData data;
77  UAVObjMetadata metadata;
78 
79  // Initialize object fields to their default values
80  UAVObjGetInstanceData(obj, instId, &data);
81  memset(&data, 0, sizeof(GPSPositionData));
82 
83  UAVObjSetInstanceData(obj, instId, &data);
84 
85  // Initialize object metadata to their default values
86  metadata.flags =
87  ACCESS_READWRITE << UAVOBJ_ACCESS_SHIFT |
88  ACCESS_READWRITE << UAVOBJ_GCS_ACCESS_SHIFT |

90  0 << UAVOBJ_GCS_TELEMETRY_ACKED_SHIFT |

92  UPDATEMODE_MANUAL << UAVOBJ_GCS_TELEMETRY_UPDATE_MODE_SHIFT;
93  metadata.telemetryUpdatePeriod = 2000;
94  metadata.gcsTelemetryUpdatePeriod = 0;
95  metadata.loggingUpdatePeriod = 1000;
