157 static void pathfollowerTask(void *parameters)
158 {
159  SystemSettingsData systemSettings;
160  FlightStatusData flightStatus;
161 
162  portTickType lastUpdateTime;
163 
164  AirspeedActualConnectCallback(airspeedActualUpdatedCb);
165  FixedWingPathFollowerSettingsConnectCallback(SettingsUpdatedCb);
166  FixedWingAirspeedsConnectCallback(SettingsUpdatedCb);
167  PathDesiredConnectCallback(SettingsUpdatedCb);
168 
169  // Force update of all the settings
170  SettingsUpdatedCb(NULL);
171 
172  FixedWingPathFollowerSettingsGet(&fixedwingpathfollowerSettings);
173  PathDesiredGet(&pathDesired);
174 
175  // Main task loop
176  lastUpdateTime = xTaskGetTickCount();
177  while (1) {
178 
179  // Conditions when this runs:
180  // 1. Must have FixedWing type airframe
181  // 2. Flight mode is PositionHold and PathDesired.Mode is Endpoint OR
182  // FlightMode is PathPlanner and PathDesired.Mode is Endpoint or Path
183 
184  SystemSettingsGet(&systemSettings);
185  if ( (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWING) &&
186  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWINGELEVON) &&
187  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_FIXEDWINGVTAIL) )
188  {
189  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
190  vTaskDelay(1000);
191  continue;
192  }
193 
194  // Continue collecting data if not enough time
195  vTaskDelayUntil(&lastUpdateTime, fixedwingpathfollowerSettings.UpdatePeriod / portTICK_RATE_MS);
196 
197 
198  FlightStatusGet(&flightStatus);
199  PathStatusGet(&pathStatus);
200 
201  uint8_t result;
202  // Check the combinations of flightmode and pathdesired mode
203  switch(flightStatus.FlightMode) {
204  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
205  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
206  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
207  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION ||
208  pathDesired.Mode == PATHDESIRED_MODE_CIRCLEPOSITIONLEFT ||
209  pathDesired.Mode == PATHDESIRED_MODE_CIRCLEPOSITIONLEFT) {
210  updatePathVelocity();
211  result = updateFixedDesiredAttitude();
212  if (result) {
213  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_OK);
214  } else {
215  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
216  }
217  } else {
218  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
219  }
220  break;
221  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
222  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
223  switch(pathDesired.Mode) {
224  case PATHDESIRED_MODE_FLYENDPOINT:
225  case PATHDESIRED_MODE_FLYVECTOR:
226  case PATHDESIRED_MODE_FLYCIRCLERIGHT:
227  case PATHDESIRED_MODE_FLYCIRCLELEFT:
228  updatePathVelocity();
229  result = updateFixedDesiredAttitude();
230  if (result) {
231  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_OK);
232  } else {
233  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
234  }
235  break;
236  default:
237  pathStatus.Status = PATHSTATUS_STATUS_CRITICAL;
238  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
239  break;
240  }
241  break;
242  default:
243  // Be cleaner and get rid of global variables
244  northVelIntegral = 0;
245  eastVelIntegral = 0;
246  downVelIntegral = 0;
247  bearingIntegral = 0;
248  speedIntegral = 0;
249  accelIntegral = 0;
250  powerIntegral = 0;
251 
252  break;
253  }
254  PathStatusSet(&pathStatus);
255  }
256 }

264 static void updatePathVelocity()
265 {
266  PositionActualData positionActual;
267  PositionActualGet(&positionActual);
268  VelocityActualData velocityActual;
269  VelocityActualGet(&velocityActual);
270 
271  float cur[3] = {positionActual.North, positionActual.East, positionActual.Down};
272  struct path_status progress;
273 
274  path_progress(&pathDesired, cur, &progress);
275 
276  float groundspeed = 0;
277  float altitudeSetpoint = 0;
278  switch (pathDesired.Mode) {
279  case PATHDESIRED_MODE_FLYCIRCLERIGHT:
280  case PATHDESIRED_MODE_DRIVECIRCLERIGHT:
281  case PATHDESIRED_MODE_FLYCIRCLELEFT:
282  case PATHDESIRED_MODE_DRIVECIRCLELEFT:
283  groundspeed = pathDesired.EndingVelocity;
284  altitudeSetpoint = pathDesired.End[2];
285  break;
286  case PATHDESIRED_MODE_FLYENDPOINT:
287  case PATHDESIRED_MODE_DRIVEENDPOINT:
288  case PATHDESIRED_MODE_FLYVECTOR:
289  case PATHDESIRED_MODE_DRIVEVECTOR:
290  default:
291  groundspeed = pathDesired.StartingVelocity + (pathDesired.EndingVelocity - pathDesired.StartingVelocity) *
292  bound_min_max(progress.fractional_progress,0,1);
293  altitudeSetpoint = pathDesired.Start[2] + (pathDesired.End[2] - pathDesired.Start[2]) *
294  bound_min_max(progress.fractional_progress,0,1);
295  break;
296  }
297  // this ensures a significant forward component at least close to the real trajectory
298  if (groundspeed<fixedWingAirspeeds.BestClimbRateSpeed/10.)
299  groundspeed=fixedWingAirspeeds.BestClimbRateSpeed/10.;
300 
301  // calculate velocity - can be zero if waypoints are too close
302  VelocityDesiredData velocityDesired;
303  velocityDesired.North = progress.path_direction[0] * groundspeed;
304  velocityDesired.East = progress.path_direction[1] * groundspeed;
305 
306  float error_speed = progress.error * fixedwingpathfollowerSettings.HorizontalPosP;
307 
308  // calculate correction - can also be zero if correction vector is 0 or no error present
309  velocityDesired.North += progress.correction_direction[0] * error_speed;
310  velocityDesired.East += progress.correction_direction[1] * error_speed;
311 
312  float downError = altitudeSetpoint - positionActual.Down;
313  velocityDesired.Down = downError * fixedwingpathfollowerSettings.VerticalPosP;
314 
315  // update pathstatus
316  pathStatus.error = progress.error;
317  pathStatus.fractional_progress = progress.fractional_progress;
318 
319  pathStatus.fractional_progress = progress.fractional_progress;
320  if (pathStatus.fractional_progress < 1)
321  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
322  else
323  pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
324  VelocityDesiredSet(&velocityDesired);
325 }

334 static uint8_t updateFixedDesiredAttitude()
335 {
336 
337  uint8_t result = 1;
338 
339  float dT = fixedwingpathfollowerSettings.UpdatePeriod / 1000.0f; //Convert from [ms] to [s]
340 
341  VelocityDesiredData velocityDesired;
342  VelocityActualData velocityActual;
343  StabilizationDesiredData stabDesired;
344  AttitudeActualData attitudeActual;
345  FixedWingPathFollowerStatusData fixedwingpathfollowerStatus;
346  AirspeedActualData airspeedActual;
347 
348  float groundspeedActual;
349  float groundspeedDesired;
350  float indicatedAirspeedActual;
351  float indicatedAirspeedDesired;
352  float airspeedError;
353 
354  float pitchCommand;
355 
356  float descentspeedDesired;
357  float descentspeedError;
358  float powerError;
359  float powerCommand;
360 
361  float bearingError;
362  float bearingCommand;
363 
364  FixedWingPathFollowerStatusGet(&fixedwingpathfollowerStatus);
365 
366  VelocityActualGet(&velocityActual);
367  StabilizationDesiredGet(&stabDesired);
368  VelocityDesiredGet(&velocityDesired);
369  AttitudeActualGet(&attitudeActual);
370  AirspeedActualGet(&airspeedActual);
371 
372 
377  // Current ground speed
378  groundspeedActual = sqrtf( velocityActual.East*velocityActual.East + velocityActual.North*velocityActual.North );
379  // note that airspeedActualBias is ( calibratedAirspeed - groundSpeed ) at the time of measurement,
380  // but thanks to accelerometers, groundspeed reacts faster to changes in direction
381  // than airspeed and gps sensors alone
382  indicatedAirspeedActual = groundspeedActual + indicatedAirspeedActualBias;
383 
384  // Desired ground speed
385  groundspeedDesired = sqrtf(velocityDesired.North*velocityDesired.North + velocityDesired.East*velocityDesired.East);
386  indicatedAirspeedDesired = bound_min_max( groundspeedDesired + indicatedAirspeedActualBias,
387  fixedWingAirspeeds.BestClimbRateSpeed,
388  fixedWingAirspeeds.CruiseSpeed);
389 
390  // Airspeed error (positive means not enough airspeed)
391  airspeedError = indicatedAirspeedDesired - indicatedAirspeedActual;
392 
393  // Vertical speed error
394  descentspeedDesired = bound_min_max (
395  velocityDesired.Down,
396  -fixedWingAirspeeds.VerticalVelMax,
397  fixedWingAirspeeds.VerticalVelMax);
398  descentspeedError = descentspeedDesired - velocityActual.Down;
399 
400  // Error condition: wind speed is higher than maximum allowed speed. We are forced backwards!
401  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_WIND] = 0;
402  if (groundspeedDesired - indicatedAirspeedActualBias <= 0 ) {
403  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_WIND] = 1;
404  result = 0;
405  }
406  // Error condition: plane too slow or too fast
407  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHSPEED] = 0;
408  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWSPEED] = 0;
409  if ( indicatedAirspeedActual > fixedWingAirspeeds.AirSpeedMax) {
410  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_OVERSPEED] = 1;
411  result = 0;
412  }
413  if ( indicatedAirspeedActual > fixedWingAirspeeds.CruiseSpeed * 1.2f) {
414  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHSPEED] = 1;
415  result = 0;
416  }
417  if (indicatedAirspeedActual < fixedWingAirspeeds.BestClimbRateSpeed * 0.8f && 1) { //The next three && 1 are placeholders for UAVOs representing LANDING and TAKEOFF
418  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWSPEED] = 1;
419  result = 0;
420  }
421  if (indicatedAirspeedActual < fixedWingAirspeeds.StallSpeedClean && 1 && 1) { //Where the && 1 represents the UAVO that will control whether the airplane is prepped for landing or not
422  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_STALLSPEED] = 1;
423  result = 0;
424  }
425  if (indicatedAirspeedActual < fixedWingAirspeeds.StallSpeedDirty && 1) {
426  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_STALLSPEED] = 1;
427  result = 0;
428  }
429 
430  if (indicatedAirspeedActual<1e-6) {
431  // prevent division by zero, abort without controlling anything. This guidance mode is not suited for takeoff or touchdown, or handling stationary planes
432  // also we cannot handle planes flying backwards, lets just wait until the nose drops
433  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWSPEED] = 1;
434  return 0;
435  }
436 
442  // compute proportional throttle response
443  powerError = -descentspeedError +
444  bound_min_max(
445  (airspeedError / fixedWingAirspeeds.BestClimbRateSpeed)* fixedwingpathfollowerSettings.AirspeedToVerticalCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_AIRSPEEDTOVERTICALCROSSFEED_KP] ,
446  -fixedwingpathfollowerSettings.AirspeedToVerticalCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_AIRSPEEDTOVERTICALCROSSFEED_MAX],
447  fixedwingpathfollowerSettings.AirspeedToVerticalCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_AIRSPEEDTOVERTICALCROSSFEED_MAX]
448  );
449 
450  // compute saturated integral error throttle response. Make integral leaky for better performance. Approximately 30s time constant.
451  if (fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI] >0) {
452  powerIntegral = bound_min_max(powerIntegral + -descentspeedError * dT,
453  -fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_ILIMIT]/fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI],
454  fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_ILIMIT]/fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI]
455  )*(1.0f-1.0f/(1.0f+30.0f/dT));
456  } else
457  powerIntegral = 0;
458 
459  // Compute final throttle response
460  powerCommand = bound_min_max(
461  (powerError * fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KP] +
462  powerIntegral* fixedwingpathfollowerSettings.PowerPI[FIXEDWINGPATHFOLLOWERSETTINGS_POWERPI_KI]) + fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_NEUTRAL],
463  fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MIN],
464  fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MAX]);
465 
466  //Output internal state to telemetry
467  fixedwingpathfollowerStatus.Error[FIXEDWINGPATHFOLLOWERSTATUS_ERROR_POWER] = powerError;
468  fixedwingpathfollowerStatus.ErrorInt[FIXEDWINGPATHFOLLOWERSTATUS_ERRORINT_POWER] = powerIntegral;
469  fixedwingpathfollowerStatus.Command[FIXEDWINGPATHFOLLOWERSTATUS_COMMAND_POWER] = powerCommand;
470 
471  // set throttle
472  stabDesired.Throttle = powerCommand;
473 
474  // Error condition: plane cannot hold altitude at current speed.
475  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWPOWER] = 0;
476  if (
477  powerCommand == fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MAX] // throttle at maximum
478  && velocityActual.Down > 0 // we ARE going down
479  && descentspeedDesired < 0 // we WANT to go up
480  && airspeedError > 0 // we are too slow already
481  )
482  {
483  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_LOWPOWER] = 1;
484  result = 0;
485  }
486  // Error condition: plane keeps climbing despite minimum throttle (opposite of above)
487  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHPOWER] = 0;
488  if (
489  powerCommand == fixedwingpathfollowerSettings.ThrottleLimit[FIXEDWINGPATHFOLLOWERSETTINGS_THROTTLELIMIT_MIN] // throttle at minimum
490  && velocityActual.Down < 0 // we ARE going up
491  && descentspeedDesired > 0 // we WANT to go down
492  && airspeedError < 0 // we are too fast already
493  )
494  {
495  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_HIGHPOWER] = 1;
496  result = 0;
497  }
498 
499 
504  if (fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI] > 0){
505  //Integrate with saturation
506  airspeedErrorInt=bound_min_max(airspeedErrorInt + airspeedError * dT,
507  -fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_ILIMIT]/fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI],
508  fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_ILIMIT]/fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI]);
509  } else
510  airspeedErrorInt = 0;
511 
512  //Compute the cross feed from vertical speed to pitch, with saturation
513  float verticalSpeedToPitchCommandComponent=bound_min_max (-descentspeedError * fixedwingpathfollowerSettings.VerticalToPitchCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_VERTICALTOPITCHCROSSFEED_KP],
514  -fixedwingpathfollowerSettings.VerticalToPitchCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_VERTICALTOPITCHCROSSFEED_MAX],
515  fixedwingpathfollowerSettings.VerticalToPitchCrossFeed[FIXEDWINGPATHFOLLOWERSETTINGS_VERTICALTOPITCHCROSSFEED_MAX]
516  );
517 
518  //Compute the pitch command as err*Kp + errInt*Ki + X_feed.
519  pitchCommand= -(airspeedError*fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KP]
520  + airspeedErrorInt*fixedwingpathfollowerSettings.SpeedPI[FIXEDWINGPATHFOLLOWERSETTINGS_SPEEDPI_KI]
521  ) + verticalSpeedToPitchCommandComponent;
522 
523  fixedwingpathfollowerStatus.Error[FIXEDWINGPATHFOLLOWERSTATUS_ERROR_SPEED] = airspeedError;
524  fixedwingpathfollowerStatus.ErrorInt[FIXEDWINGPATHFOLLOWERSTATUS_ERRORINT_SPEED] = airspeedErrorInt;
525  fixedwingpathfollowerStatus.Command[FIXEDWINGPATHFOLLOWERSTATUS_COMMAND_SPEED] = pitchCommand;
526 
527  stabDesired.Pitch = bound_min_max(fixedwingpathfollowerSettings.PitchLimit[FIXEDWINGPATHFOLLOWERSETTINGS_PITCHLIMIT_NEUTRAL] +
528  pitchCommand,
529  fixedwingpathfollowerSettings.PitchLimit[FIXEDWINGPATHFOLLOWERSETTINGS_PITCHLIMIT_MIN],
530  fixedwingpathfollowerSettings.PitchLimit[FIXEDWINGPATHFOLLOWERSETTINGS_PITCHLIMIT_MAX]);
531 
532  // Error condition: high speed dive
533  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_PITCHCONTROL] = 0;
534  if (
535  pitchCommand == fixedwingpathfollowerSettings.PitchLimit[FIXEDWINGPATHFOLLOWERSETTINGS_PITCHLIMIT_MAX] // pitch demand is full up
536  && velocityActual.Down > 0 // we ARE going down
537  && descentspeedDesired < 0 // we WANT to go up
538  && airspeedError < 0 // we are too fast already
539  ) {
540  fixedwingpathfollowerStatus.Errors[FIXEDWINGPATHFOLLOWERSTATUS_ERRORS_PITCHCONTROL] = 1;
541  result = 0;
542  }
543 
544 
548  if (groundspeedDesired> 1e-6) {
549  bearingError = RAD2DEG * (atan2f(velocityDesired.East,velocityDesired.North) - atan2f(velocityActual.East,velocityActual.North));
550  } else {
551  // if we are not supposed to move, keep going wherever we are now. Don't make things worse by changing direction.
552  bearingError = 0;
553  }
554 
555  if (bearingError<-180.0f) bearingError+=360.0f;
556  if (bearingError>180.0f) bearingError-=360.0f;
557 
558  bearingIntegral = bound_min_max(bearingIntegral + bearingError * dT * fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_KI],
559  -fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_ILIMIT],
560  fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_ILIMIT]);
561  bearingCommand = (bearingError * fixedwingpathfollowerSettings.BearingPI[FIXEDWINGPATHFOLLOWERSETTINGS_BEARINGPI_KP] +
562  bearingIntegral);
563 
564  fixedwingpathfollowerStatus.Error[FIXEDWINGPATHFOLLOWERSTATUS_ERROR_BEARING] = bearingError;
565  fixedwingpathfollowerStatus.ErrorInt[FIXEDWINGPATHFOLLOWERSTATUS_ERRORINT_BEARING] = bearingIntegral;
566  fixedwingpathfollowerStatus.Command[FIXEDWINGPATHFOLLOWERSTATUS_COMMAND_BEARING] = bearingCommand;
567 
568  stabDesired.Roll = bound_min_max( fixedwingpathfollowerSettings.RollLimit[FIXEDWINGPATHFOLLOWERSETTINGS_ROLLLIMIT_NEUTRAL] +
569  bearingCommand,
570  fixedwingpathfollowerSettings.RollLimit[FIXEDWINGPATHFOLLOWERSETTINGS_ROLLLIMIT_MIN],
571  fixedwingpathfollowerSettings.RollLimit[FIXEDWINGPATHFOLLOWERSETTINGS_ROLLLIMIT_MAX] );
572 
573  // TODO: find a check to determine loss of directional control. Likely needs some check of derivative
574 
575 
579  // TODO implement raw control mode for yaw and base on Accels.Y
580  stabDesired.Yaw = 0;
581
