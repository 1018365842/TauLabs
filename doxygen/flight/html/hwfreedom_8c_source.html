48 int32_t HwFreedomInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(HWFREEDOM_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(HWFREEDOM_OBJID,
56  HWFREEDOM_ISSINGLEINST, HWFREEDOM_ISSETTINGS, HWFREEDOM_NUMBYTES, &HwFreedomSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

74 void HwFreedomSetDefaults(UAVObjHandle obj, uint16_t instId)
75 {
76  HwFreedomData data;
77  UAVObjMetadata metadata;
78 
79  // Initialize object fields to their default values
80  UAVObjGetInstanceData(obj, instId, &data);
81  memset(&data, 0, sizeof(HwFreedomData));
82  data.Output = 1;
83  data.MainPort = 1;
84  data.FlexiPort = 2;
85  data.RcvrPort = 0;
86  data.RadioPort = 0;
87  data.USB_HIDPort = 0;
88  data.USB_VCPPort = 3;
89  data.DSMxBind = 0;
90  data.GyroRange = 1;
91  data.AccelRange = 2;
92  data.MPU6000Rate = 3;
93  data.MPU6000DLPF = 0;
94 
95  UAVObjSetInstanceData(obj, instId, &data);
96 
97  // Initialize object metadata to their default values
98  metadata.flags =
99  ACCESS_READWRITE << UAVOBJ_ACCESS_SHIFT |
100  ACCESS_READWRITE << UAVOBJ_GCS_ACCESS_SHIFT |
101  1 << UAVOBJ_TELEMETRY_ACKED_SHIFT |
102  1 << UAVOBJ_GCS_TELEMETRY_ACKED_SHIFT |
103  UPDATEMODE_ONCHANGE << UAVOBJ_TELEMETRY_UPDATE_MODE_SHIFT |
104  UPDATEMODE_ONCHANGE << UAVOBJ_GCS_TELEMETRY_UPDATE_MODE_SHIFT;
105  metadata.telemetryUpdatePeriod = 0;
106  metadata.gcsTelemetryUpdatePeriod = 0;
107  metadata.loggingUpdatePeriod = 0;
108  UAVObjSetMetadata(obj, &metadata);
109 }
