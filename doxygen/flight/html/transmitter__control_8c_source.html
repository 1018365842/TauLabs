160 int32_t transmitter_control_update()
161 {
162  lastSysTime = xTaskGetTickCount();
163 
164  float scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM];
165 
166  if (settings_updated) {
167  settings_updated = false;
168  ManualControlSettingsGet(&settings);
169  }
170 
171  /* Update channel activity monitor */
172  if (flightStatus.Armed == ARM_STATE_DISARMED) {
173  if (updateRcvrActivity(&activity_fsm)) {
174  /* Reset the aging timer because activity was detected */
175  lastActivityTime = lastSysTime;
176  }
177  }
178  if (timeDifferenceMs(lastActivityTime, lastSysTime) > 5000) {
179  resetRcvrActivity(&activity_fsm);
180  lastActivityTime = lastSysTime;
181  }
182 
183  if (ManualControlCommandReadOnly()) {
184  FlightTelemetryStatsData flightTelemStats;
185  FlightTelemetryStatsGet(&flightTelemStats);
186  if(flightTelemStats.Status != FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
187  /* trying to fly via GCS and lost connection. fall back to transmitter */
188  UAVObjMetadata metadata;
189  ManualControlCommandGetMetadata(&metadata);
190  UAVObjSetAccess(&metadata, ACCESS_READWRITE);
191  ManualControlCommandSetMetadata(&metadata);
192  }
193 
194  // Don't process anything else when GCS is overriding the objects
195  return 0;
196  }
197 
198  bool valid_input_detected = true;
199 
200  // Read channel values in us
201  for (uint8_t n = 0;
202  n < MANUALCONTROLSETTINGS_CHANNELGROUPS_NUMELEM && n < MANUALCONTROLCOMMAND_CHANNEL_NUMELEM;
203  ++n) {
204  extern uint32_t pios_rcvr_group_map[];
205 
206  if (settings.ChannelGroups[n] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
207  cmd.Channel[n] = PIOS_RCVR_INVALID;
208  } else {
209  cmd.Channel[n] = PIOS_RCVR_Read(pios_rcvr_group_map[settings.ChannelGroups[n]],
210  settings.ChannelNumber[n]);
211  }
212 
213  // If a channel has timed out this is not valid data and we shouldn't update anything
214  // until we decide to go to failsafe
215  if(cmd.Channel[n] == PIOS_RCVR_TIMEOUT)
216  valid_input_detected = false;
217  else
218  scaledChannel[n] = scaleChannel(cmd.Channel[n], settings.ChannelMax[n], settings.ChannelMin[n], settings.ChannelNeutral[n]);
219  }
220 
221  // Check settings, if error raise alarm
222  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
223  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
224  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
225  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
226  // Check all channel mappings are valid
227  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] == (uint16_t) PIOS_RCVR_INVALID ||
228  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] == (uint16_t) PIOS_RCVR_INVALID ||
229  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] == (uint16_t) PIOS_RCVR_INVALID ||
230  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] == (uint16_t) PIOS_RCVR_INVALID ||
231  // Check the driver exists
232  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL] == (uint16_t) PIOS_RCVR_NODRIVER ||
233  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH] == (uint16_t) PIOS_RCVR_NODRIVER ||
234  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW] == (uint16_t) PIOS_RCVR_NODRIVER ||
235  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE] == (uint16_t) PIOS_RCVR_NODRIVER ||
236  // Check the FlightModeNumber is valid
237  settings.FlightModeNumber < 1 || settings.FlightModeNumber > MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_NUMELEM ||
238  // Similar checks for FlightMode channel but only if more than one flight mode has been set. Otherwise don't care
239  ((settings.FlightModeNumber > 1) && (
240  settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE ||
241  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] == (uint16_t) PIOS_RCVR_INVALID ||
242  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE] == (uint16_t) PIOS_RCVR_NODRIVER))) {
243 
244  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_SETTINGS);
245 
246  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_FALSE;
247  ManualControlCommandSet(&cmd);
248 
249  // Need to do this here since we don't process armed status. Since this shouldn't happen in flight (changed config)
250  // immediately disarm
251  pending_control_event = CONTROL_EVENTS_DISARM;
252 
253  return -1;
254  }
255 
256  // decide if we have valid manual input or not
257  valid_input_detected &= validInputRange(settings.ChannelMin[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE], settings.ChannelMax[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE], cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE]) &&
258  validInputRange(settings.ChannelMin[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL], settings.ChannelMax[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL], cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL]) &&
259  validInputRange(settings.ChannelMin[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW], settings.ChannelMax[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW], cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW]) &&
260  validInputRange(settings.ChannelMin[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH], settings.ChannelMax[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH], cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH]);
261 
262  // Implement hysteresis loop on connection status
263  if (valid_input_detected && (++connected_count > 10)) {
264  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_TRUE;
265  connected_count = 0;
266  disconnected_count = 0;
267  } else if (!valid_input_detected && (++disconnected_count > 10)) {
268  cmd.Connected = MANUALCONTROLCOMMAND_CONNECTED_FALSE;
269  connected_count = 0;
270  disconnected_count = 0;
271  }
272 
273  if (cmd.Connected == MANUALCONTROLCOMMAND_CONNECTED_FALSE) {
274  // These values are not used but just put ManualControlCommand in a sane state. When
275  // Connected is false, then the failsafe submodule will be in control.
276 
277  cmd.Throttle = -1;
278  cmd.Roll = 0;
279  cmd.Yaw = 0;
280  cmd.Pitch = 0;
281  cmd.Collective = 0;
282 
283  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_NORX);
284 
285  } else if (valid_input_detected) {
286  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_NONE);
287 
288  // Scale channels to -1 -> +1 range
289  cmd.Roll = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ROLL];
290  cmd.Pitch = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_PITCH];
291  cmd.Yaw = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_YAW];
292  cmd.Throttle = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_THROTTLE];
293  flight_mode_value = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_FLIGHTMODE];
294 
295  // Apply deadband for Roll/Pitch/Yaw stick inputs
296  if (settings.Deadband) {
297  applyDeadband(&cmd.Roll, settings.Deadband);
298  applyDeadband(&cmd.Pitch, settings.Deadband);
299  applyDeadband(&cmd.Yaw, settings.Deadband);
300  }
301 
302  if(cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_INVALID &&
303  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_NODRIVER &&
304  cmd.Channel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE] != (uint16_t) PIOS_RCVR_TIMEOUT) {
305  cmd.Collective = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_COLLECTIVE];
306  }
307 
308  AccessoryDesiredData accessory;
309  // Set Accessory 0
310  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY0] !=
311  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
312  accessory.AccessoryVal = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY0];
313  if(AccessoryDesiredInstSet(0, &accessory) != 0) //These are allocated later and that allocation might fail
314  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_ACCESSORY);
315  }
316  // Set Accessory 1
317  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1] !=
318  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
319  accessory.AccessoryVal = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY1];
320  if(AccessoryDesiredInstSet(1, &accessory) != 0) //These are allocated later and that allocation might fail
321  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_ACCESSORY);
322  }
323  // Set Accessory 2
324  if (settings.ChannelGroups[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY2] !=
325  MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
326  accessory.AccessoryVal = scaledChannel[MANUALCONTROLSETTINGS_CHANNELGROUPS_ACCESSORY2];
327  if(AccessoryDesiredInstSet(2, &accessory) != 0) //These are allocated later and that allocation might fail
328  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_ACCESSORY);
329  }
330  }
331 
332  // Process arming outside conditional so system will disarm when disconnected. Notice this
333  // is processed in the _update method instead of _select method so the state system is always
334  // evalulated, even if not detected.
335  process_transmitter_events(&cmd, &settings);
336 
337  // Update cmd object
338  ManualControlCommandSet(&cmd);
339 
340 #if defined(PIOS_INCLUDE_USB_RCTX)
341  // Optionally make the hardware behave like a USB HID joystick
342  if (pios_usb_rctx_id) {
343  PIOS_USB_RCTX_Update(pios_usb_rctx_id,
344  cmd.Channel,
345  settings.ChannelMin,
346  settings.ChannelMax,
347  NELEMENTS(cmd.Channel));
348  }
349 #endif /* PIOS_INCLUDE_USB_RCTX */
350 
351  return 0;
352 }

358 int32_t transmitter_control_select(bool reset_controller)
359 {
360  // Activate the flight mode corresponding to the switch position
361  set_flight_mode();
362 
363  ManualControlCommandGet(&cmd);
364  FlightStatusGet(&flightStatus);
365 
366  // Depending on the mode update the Stabilization or Actuator objects
367  static uint8_t lastFlightMode = FLIGHTSTATUS_FLIGHTMODE_MANUAL;
368 
369  switch(flightStatus.FlightMode) {
370  case FLIGHTSTATUS_FLIGHTMODE_MANUAL:
371  update_actuator_desired(&cmd);
372  break;
373  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED1:
374  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED2:
375  case FLIGHTSTATUS_FLIGHTMODE_STABILIZED3:
376  update_stabilization_desired(&cmd, &settings);
377  break;
378  case FLIGHTSTATUS_FLIGHTMODE_AUTOTUNE:
379  // Tuning takes settings directly from manualcontrolcommand. No need to
380  // call anything else. This just avoids errors.
381  break;
382  case FLIGHTSTATUS_FLIGHTMODE_ALTITUDEHOLD:
383  altitude_hold_desired(&cmd, lastFlightMode != flightStatus.FlightMode);
384  break;
385  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
386  update_path_desired(&cmd, lastFlightMode != flightStatus.FlightMode, false);
387  break;
388  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
389  update_path_desired(&cmd, lastFlightMode != flightStatus.FlightMode, true);
390  break;
391  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
392  break;
393  default:
394  set_manual_control_error(SYSTEMALARMS_MANUALCONTROL_UNDEFINED);
395  break;
396  }
397  lastFlightMode = flightStatus.FlightMode;
398 
399  return 0;
400 }

447 static void process_transmitter_events(ManualControlCommandData * cmd, ManualControlSettingsData * settings)
448 {
449  bool lowThrottle = cmd->Throttle <= 0;
450 
451  uint8_t arm_status;
452  FlightStatusArmedGet(&arm_status);
453 
454  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_ALWAYSDISARMED) {
455  set_armed_if_changed(FLIGHTSTATUS_ARMED_DISARMED);
456  } else {
457  if (cmd->Connected == MANUALCONTROLCOMMAND_CONNECTED_FALSE)
458  lowThrottle = true;
459 
460  // The throttle is not low, in case we where arming or disarming, abort
461  if (!lowThrottle) {
462  switch(arm_state) {
463  case ARM_STATE_DISARMING_MANUAL:
464  case ARM_STATE_DISARMING_TIMEOUT:
465  arm_state = ARM_STATE_ARMED;
466  break;
467  case ARM_STATE_ARMING_MANUAL:
468  arm_state = ARM_STATE_DISARMED;
469  break;
470  default:
471  // Nothing needs to be done in the other states
472  break;
473  }
474  return;
475  }
476 
477  // The rest of these cases throttle is low
478  if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_ALWAYSARMED) {
479  set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMED);
480  return;
481  }
482 
483  // When the configuration is not "Always armed" and no "Always disarmed",
484  // the state will not be changed when the throttle is not low
485  static portTickType armedDisarmStart;
486  float armingInputLevel = 0;
487 
488  // Calc channel see assumptions7
489  int8_t sign = ((settings->Arming-MANUALCONTROLSETTINGS_ARMING_ROLLLEFT)%2) ? -1 : 1;
490  switch ( (settings->Arming-MANUALCONTROLSETTINGS_ARMING_ROLLLEFT)/2 ) {
491  case ARMING_CHANNEL_ROLL: armingInputLevel = sign * cmd->Roll; break;
492  case ARMING_CHANNEL_PITCH: armingInputLevel = sign * cmd->Pitch; break;
493  case ARMING_CHANNEL_YAW: armingInputLevel = sign * cmd->Yaw; break;
494  }
495 
496  bool manualArm = false;
497  bool manualDisarm = false;
498 
499  if (armingInputLevel <= -ARMED_THRESHOLD)
500  manualArm = true;
501  else if (armingInputLevel >= +ARMED_THRESHOLD)
502  manualDisarm = true;
503 
504  switch(arm_state) {
505  case ARM_STATE_DISARMED:
506  set_armed_if_changed(FLIGHTSTATUS_ARMED_DISARMED);
507 
508  if (manualArm) {
509  armedDisarmStart = lastSysTime;
510  arm_state = ARM_STATE_ARMING_MANUAL;
511  }
512 
513  break;
514 
515  case ARM_STATE_ARMING_MANUAL:
516  set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMING);
517 
518  if (manualArm && (timeDifferenceMs(armedDisarmStart, lastSysTime) > ARMED_TIME_MS))
519  arm_state = ARM_STATE_ARMED;
520  else if (!manualArm)
521  arm_state = ARM_STATE_DISARMED;
522  break;
523 
524  case ARM_STATE_ARMED:
525  // When we get here, the throttle is low,
526  // we go immediately to disarming due to timeout, also when the disarming mechanism is not enabled
527  armedDisarmStart = lastSysTime;
528  arm_state = ARM_STATE_DISARMING_TIMEOUT;
529  set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMED);
530  break;
531 
532  case ARM_STATE_DISARMING_TIMEOUT:
533  // We get here when armed while throttle low, even when the arming timeout is not enabled
534  if ((settings->ArmedTimeout != 0) && (timeDifferenceMs(armedDisarmStart, lastSysTime) > settings->ArmedTimeout))
535  arm_state = ARM_STATE_DISARMED;
536 
537  // Switch to disarming due to manual control when needed
538  if (manualDisarm) {
539  armedDisarmStart = lastSysTime;
540  arm_state = ARM_STATE_DISARMING_MANUAL;
541  }
542  break;
543 
544  case ARM_STATE_DISARMING_MANUAL:
545  if (manualDisarm &&(timeDifferenceMs(armedDisarmStart, lastSysTime) > ARMED_TIME_MS))
546  arm_state = ARM_STATE_DISARMED;
547  else if (!manualDisarm)
548  arm_state = ARM_STATE_ARMED;
549  break;
550  } // End Switch
551  }
552 }

556 {
557  // Convert flightMode value into the switch position in the range [0..N-1]
558  uint8_t pos = ((int16_t)(flight_mode_value * 256.0f) + 256) * settings.FlightModeNumber >> 9;

561 
562  return settings.FlightModePosition[pos];

580 static void resetRcvrActivity(struct rcvr_activity_fsm * fsm)
581 {
582  ReceiverActivityData data;
583  bool updated = false;
584 
585  /* Clear all channel activity flags */
586  ReceiverActivityGet(&data);
587  if (data.ActiveGroup != RECEIVERACTIVITY_ACTIVEGROUP_NONE &&
588  data.ActiveChannel != 255) {
589  data.ActiveGroup = RECEIVERACTIVITY_ACTIVEGROUP_NONE;
590  data.ActiveChannel = 255;
591  updated = true;
592  }
593  if (updated) {
594  ReceiverActivitySet(&data);
595  }
596 
597  /* Reset the FSM state */

609 static bool updateRcvrActivityCompare(uint32_t rcvr_id, struct rcvr_activity_fsm * fsm)
610 {
611  bool activity_updated = false;
612 
613  /* Compare the current value to the previous sampled value */
614  for (uint8_t channel = 1;
615  channel <= RCVR_ACTIVITY_MONITOR_CHANNELS_PER_GROUP;
616  channel++) {
617  uint16_t delta;
618  uint16_t prev = fsm->prev[channel - 1]; // Subtract 1 because channels are 1 indexed
619  uint16_t curr = PIOS_RCVR_Read(rcvr_id, channel);
620  if (curr > prev) {
621  delta = curr - prev;
622  } else {
623  delta = prev - curr;
624  }
625 
626  if (delta > RCVR_ACTIVITY_MONITOR_MIN_RANGE) {
627  /* Mark this channel as active */
628  ReceiverActivityActiveGroupOptions group;
629 
630  /* Don't assume manualcontrolsettings and receiveractivity are in the same order. */
631  switch (fsm->group) {
632  case MANUALCONTROLSETTINGS_CHANNELGROUPS_PWM:
633  group = RECEIVERACTIVITY_ACTIVEGROUP_PWM;
634  break;
635  case MANUALCONTROLSETTINGS_CHANNELGROUPS_PPM:
636  group = RECEIVERACTIVITY_ACTIVEGROUP_PPM;
637  break;
638  case MANUALCONTROLSETTINGS_CHANNELGROUPS_DSMMAINPORT:
639  group = RECEIVERACTIVITY_ACTIVEGROUP_DSMMAINPORT;
640  break;
641  case MANUALCONTROLSETTINGS_CHANNELGROUPS_DSMFLEXIPORT:
642  group = RECEIVERACTIVITY_ACTIVEGROUP_DSMFLEXIPORT;
643  break;
644  case MANUALCONTROLSETTINGS_CHANNELGROUPS_SBUS:
645  group = RECEIVERACTIVITY_ACTIVEGROUP_SBUS;
646  break;
647  case MANUALCONTROLSETTINGS_CHANNELGROUPS_GCS:
648  group = RECEIVERACTIVITY_ACTIVEGROUP_GCS;
649  break;
650  default:
651  PIOS_Assert(0);
652  break;
653  }
654 
655  ReceiverActivityActiveGroupSet((uint8_t*)&group);
656  ReceiverActivityActiveChannelSet(&channel);
657  activity_updated = true;
658  }
659  }
660  return (activity_updated);
661 }

663 static bool updateRcvrActivity(struct rcvr_activity_fsm * fsm)
664 {
665  bool activity_updated = false;
666 
667  if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
668  /* We're out of range, reset things */
669  resetRcvrActivity(fsm);
670  }
671 
672  extern uint32_t pios_rcvr_group_map[];
673  if (!pios_rcvr_group_map[fsm->group]) {
674  /* Unbound group, skip it */
675  goto group_completed;
676  }
677 
678  if (fsm->sample_count == 0) {
679  /* Take a sample of each channel in this group */
680  updateRcvrActivitySample(pios_rcvr_group_map[fsm->group],
681  fsm->prev,
682  NELEMENTS(fsm->prev));
683  fsm->sample_count++;
684  return (false);
685  }
686 
687  /* Compare with previous sample */
688  activity_updated = updateRcvrActivityCompare(pios_rcvr_group_map[fsm->group], fsm);
689 
690 group_completed:
691  /* Reset the sample counter */
692  fsm->sample_count = 0;
693 
694  /* Find the next active group, but limit search so we can't loop forever here */
695  for (uint8_t i = 0; i < MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE; i++) {
696  /* Move to the next group */
697  fsm->group++;
698  if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
699  /* Wrap back to the first group */
700  fsm->group = 0;
701  }
702  if (pios_rcvr_group_map[fsm->group]) {
703  /*
704  * Found an active group, take a sample here to avoid an
705  * extra 20ms delay in the main thread so we can speed up
706  * this algorithm.
707  */
708  updateRcvrActivitySample(pios_rcvr_group_map[fsm->group],
709  fsm->prev,
710  NELEMENTS(fsm->prev));
711  fsm->sample_count++;
712  break;
713  }
714  }
715 
716  return (activity_updated);
717 }

865 static void altitude_hold_desired(ManualControlCommandData * cmd, bool flightModeChanged)
866 {

868  const float DEADBAND_LOW = 0.45;

892  altitudeHoldDesired.Altitude = 0;
893  zeroed = false;
894  } else if (cmd->Throttle > DEADBAND_HIGH && zeroed)

922 {

924 
925  // Scale
926  if ((max > min && value >= neutral) || (min > max && value <= neutral))
927  {
928  if (max != neutral)
929  valueScaled = (float)(value - neutral) / (float)(max - neutral);
930  else
931  valueScaled = 0;
932  }
933  else
934  {
935  if (min != neutral)
936  valueScaled = (float)(value - neutral) / (float)(neutral - min);
937  else
938  valueScaled = 0;
939  }
940 
941  // Bound
942  if (valueScaled > 1.0) valueScaled = 1.0;
943  else
944  if (valueScaled < -1.0) valueScaled = -1.0;
945 
946  return valueScaled;

994 static void set_manual_control_error(SystemAlarmsManualControlOptions error_code)
995 {
996  // Get the severity of the alarm given the error code
997  SystemAlarmsAlarmOptions severity;
998  switch (error_code) {
999  case SYSTEMALARMS_MANUALCONTROL_NONE:
1000  severity = SYSTEMALARMS_ALARM_OK;
1001  break;
1002  case SYSTEMALARMS_MANUALCONTROL_NORX:
1003  case SYSTEMALARMS_MANUALCONTROL_ACCESSORY:
1004  severity = SYSTEMALARMS_ALARM_WARNING;
1005  break;
1006  case SYSTEMALARMS_MANUALCONTROL_SETTINGS:
1007  severity = SYSTEMALARMS_ALARM_CRITICAL;
1008  break;
1009  case SYSTEMALARMS_MANUALCONTROL_ALTITUDEHOLD:
1010  severity = SYSTEMALARMS_ALARM_ERROR;
1011  break;
1012  case SYSTEMALARMS_MANUALCONTROL_UNDEFINED:
1013  default:
1014  severity = SYSTEMALARMS_ALARM_CRITICAL;
1015  error_code = SYSTEMALARMS_MANUALCONTROL_UNDEFINED;
1016  }
1017 
1018  // Make sure not to set the error code if it didn't change
1019  SystemAlarmsManualControlOptions current_error_code;
1020  SystemAlarmsManualControlGet((uint8_t *) &current_error_code);
1021  if (current_error_code != error_code) {
1022  SystemAlarmsManualControlSet((uint8_t *) &error_code);
1023  }
1024 
1025  // AlarmSet checks only updates on toggle
1026  AlarmsSet(SYSTEMALARMS_ALARM_MANUALCONTROL, (uint8_t) severity);
1027 }
