81  uint32_t init_ssel = 0;
82 
83  PIOS_Assert(spi_id);
84  PIOS_Assert(cfg);
85 
86  struct pios_spi_dev * spi_dev;
87 
88  spi_dev = (struct pios_spi_dev *) PIOS_SPI_alloc();
89  if (!spi_dev) goto out_fail;
90 
91  /* Bind the configuration to the device instance */
92  spi_dev->cfg = cfg;
93 
94 #if defined(PIOS_INCLUDE_FREERTOS)
95  vSemaphoreCreateBinary(spi_dev->busy);
96  xSemaphoreGive(spi_dev->busy);
97 #endif
98 
99  /* Disable callback function */
100  spi_dev->callback = NULL;
101 
102  /* Set rx/tx dummy bytes to a known value */
103  spi_dev->rx_dummy_byte = 0xFF;
104  spi_dev->tx_dummy_byte = 0xFF;
105 
106  switch (spi_dev->cfg->init.SPI_NSS) {
107  case SPI_NSS_Soft:
108  if (spi_dev->cfg->init.SPI_Mode == SPI_Mode_Master) {
109  /* We're a master in soft NSS mode, make sure we see NSS high at all times. */
110  SPI_NSSInternalSoftwareConfig(spi_dev->cfg->regs, SPI_NSSInternalSoft_Set);
111  /* Init as many slave selects as the config advertises. */
112  init_ssel = spi_dev->cfg->slave_count;
113  } else {
114  /* We're a slave in soft NSS mode, make sure we see NSS low at all times. */
115  SPI_NSSInternalSoftwareConfig(spi_dev->cfg->regs, SPI_NSSInternalSoft_Reset);
116  }
117  break;
118 
119  case SPI_NSS_Hard:
120  /* only legal for single-slave config */
121  PIOS_Assert(spi_dev->cfg->slave_count == 1);
122  init_ssel = 1;
123  SPI_SSOutputCmd(spi_dev->cfg->regs, (spi_dev->cfg->init.SPI_Mode == SPI_Mode_Master) ? ENABLE : DISABLE);
124  /* FIXME: Should this also call SPI_SSOutputCmd()? */
125  break;
126 
127  default:
128  PIOS_Assert(0);
129  }
130 
131  /* Initialize the GPIO pins */
132  /* note __builtin_ctz() due to the difference between GPIO_PinX and GPIO_PinSourceX */
133  if (spi_dev->cfg->remap) {
134  GPIO_PinAFConfig(spi_dev->cfg->sclk.gpio,
135  __builtin_ctz(spi_dev->cfg->sclk.init.GPIO_Pin),
136  spi_dev->cfg->remap);
137  GPIO_PinAFConfig(spi_dev->cfg->mosi.gpio,
138  __builtin_ctz(spi_dev->cfg->mosi.init.GPIO_Pin),
139  spi_dev->cfg->remap);
140  GPIO_PinAFConfig(spi_dev->cfg->miso.gpio,
141  __builtin_ctz(spi_dev->cfg->miso.init.GPIO_Pin),
142  spi_dev->cfg->remap);

144  GPIO_PinAFConfig(spi_dev->cfg->ssel[i].gpio,
145  __builtin_ctz(spi_dev->cfg->ssel[i].init.GPIO_Pin),
146  spi_dev->cfg->remap);
147  }
148  }
149  GPIO_Init(spi_dev->cfg->sclk.gpio, (GPIO_InitTypeDef*)&(spi_dev->cfg->sclk.init));

152 
153  if(spi_dev->cfg->init.SPI_NSS != SPI_NSS_Hard) {
154  for (uint32_t i = 0; i < init_ssel; i++) {
155  /* Since we're driving the SSEL pin in software, ensure that the slave is deselected */
156  /* XXX multi-slave support - maybe have another SPI_NSS_ mode? */

165 // DMA_Init(spi_dev->cfg->dma.rx.channel, (DMA_InitTypeDef*)&(spi_dev->cfg->dma.rx.init));

435 static int32_t SPI_DMA_TransferBlock(uint32_t spi_id, const uint8_t *send_buffer, uint8_t *receive_buffer, uint16_t len, void *callback)
436 {
437  struct pios_spi_dev * spi_dev = (struct pios_spi_dev *)spi_id;
438 
439  bool valid = PIOS_SPI_validate(spi_dev);
440  PIOS_Assert(valid)
441 
442  DMA_InitTypeDef dma_init;
443 
444  /* Exit if ongoing transfer */
445  if (DMA_GetCurrDataCounter(spi_dev->cfg->dma.rx.channel)) {
446  return -3;
447  }
448 
449  /* Disable the DMA channels */
450  DMA_Cmd(spi_dev->cfg->dma.rx.channel, DISABLE);
451  DMA_Cmd(spi_dev->cfg->dma.tx.channel, DISABLE);
452 
453  //FIXME: dont know how to replace this
454 // while(DMA_GetCmdStatus(spi_dev->cfg->dma.rx.channel) == ENABLE);
455 // while(DMA_GetCmdStatus(spi_dev->cfg->dma.tx.channel) == ENABLE);
456 
457  /* Disable the SPI peripheral */
458  /* Initialize the SPI block */
459  SPI_I2S_DeInit(spi_dev->cfg->regs);
460  SPI_Init(spi_dev->cfg->regs, (SPI_InitTypeDef*)&(spi_dev->cfg->init));
461  SPI_Cmd(spi_dev->cfg->regs, DISABLE);
462  /* Configure CRC calculation */
463  if (spi_dev->cfg->use_crc) {
464  SPI_CalculateCRC(spi_dev->cfg->regs, ENABLE);
465  } else {
466  SPI_CalculateCRC(spi_dev->cfg->regs, DISABLE);
467  }
468 
469  /* Enable SPI interrupts to DMA */
470  SPI_I2S_DMACmd(spi_dev->cfg->regs, SPI_I2S_DMAReq_Tx | SPI_I2S_DMAReq_Rx, ENABLE);
471 
472  /* Set callback function */
473  spi_dev->callback = callback;
474 
475  /*
476  * Configure Rx channel
477  */
478 
479  /* Start with the default configuration for this peripheral */
480  dma_init = spi_dev->cfg->dma.rx.init;
481  DMA_DeInit(spi_dev->cfg->dma.rx.channel);
482  if (receive_buffer != NULL) {
483  /* Enable memory addr. increment - bytes written into receive buffer */
484  dma_init.DMA_MemoryBaseAddr = (uint32_t) receive_buffer;
485  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
486  } else {
487  /* Disable memory addr. increment - bytes written into dummy buffer */
488  spi_dev->rx_dummy_byte = 0xFF;
489  dma_init.DMA_MemoryBaseAddr = (uint32_t) &spi_dev->rx_dummy_byte;
490  dma_init.DMA_MemoryInc = DMA_MemoryInc_Disable;
491  }
492  if (spi_dev->cfg->use_crc) {
493  /* Make sure the CRC error flag is cleared before we start */
494  SPI_I2S_ClearFlag(spi_dev->cfg->regs, SPI_FLAG_CRCERR);
495  }
496 
497  dma_init.DMA_BufferSize = len;
498  DMA_Init(spi_dev->cfg->dma.rx.channel, &(dma_init));
499 
500  /*
501  * Configure Tx channel
502  */
503 
504  /* Start with the default configuration for this peripheral */
505  dma_init = spi_dev->cfg->dma.tx.init;
506  DMA_DeInit(spi_dev->cfg->dma.tx.channel);
507  if (send_buffer != NULL) {
508  /* Enable memory addr. increment - bytes written into receive buffer */
509  dma_init.DMA_MemoryBaseAddr = (uint32_t) send_buffer;
510  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
511  } else {
512  /* Disable memory addr. increment - bytes written into dummy buffer */
513  spi_dev->tx_dummy_byte = 0xFF;
514  dma_init.DMA_MemoryBaseAddr = (uint32_t) &spi_dev->tx_dummy_byte;
515  dma_init.DMA_MemoryInc = DMA_MemoryInc_Disable;
516  }
517 
518  if (spi_dev->cfg->use_crc) {
519  /* The last byte of the payload will be replaced with the CRC8 */
520  dma_init.DMA_BufferSize = len - 1;
521  } else {
522  dma_init.DMA_BufferSize = len;
523  }
524 
525  DMA_Init(spi_dev->cfg->dma.tx.channel, &(dma_init));
526 
527  /* Enable DMA interrupt if callback function active */
528  DMA_ITConfig(spi_dev->cfg->dma.rx.channel, DMA_IT_TC, (callback != NULL) ? ENABLE : DISABLE);
529 
530  /* Flush out the CRC registers */
531  SPI_CalculateCRC(spi_dev->cfg->regs, DISABLE);
532  (void)SPI_GetCRC(spi_dev->cfg->regs, SPI_CRC_Rx);
533  SPI_I2S_ClearFlag(spi_dev->cfg->regs, SPI_FLAG_CRCERR);
534 
535  /* Make sure to flush out the receive buffer */
536  (void)SPI_I2S_ReceiveData16(spi_dev->cfg->regs);
537 
538  if (spi_dev->cfg->use_crc) {
539  /* Need a 0->1 transition to reset the CRC logic */
540  SPI_CalculateCRC(spi_dev->cfg->regs, ENABLE);
541  }
542 
543  /* Start DMA transfers */
544  DMA_Cmd(spi_dev->cfg->dma.rx.channel, ENABLE);
545  DMA_Cmd(spi_dev->cfg->dma.tx.channel, ENABLE);
546 
547  /* Reenable the SPI device */
548  SPI_Cmd(spi_dev->cfg->regs, ENABLE);
549 
550  if (callback) {
551  /* User has requested a callback, don't wait for the transfer to complete. */
552  return 0;
553  }
554 
555  /* Wait until all bytes have been transmitted/received */
556  while (DMA_GetCurrDataCounter(spi_dev->cfg->dma.rx.channel));
557 
558  /* Wait for the final bytes of the transfer to complete, including CRC byte(s). */
559  while (!(SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_I2S_FLAG_TXE)));
560 
561  /* Wait for the final bytes of the transfer to complete, including CRC byte(s). */
562  while (SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_I2S_FLAG_BSY));
563 
564  /* Check the CRC on the transfer if enabled. */
565  if (spi_dev->cfg->use_crc) {
566  /* Check the SPI CRC error flag */
567  if (SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_FLAG_CRCERR)) {
568  return -4;
569  }
570  }
571 
572  /* No error */
573  return 0;
574 }
