15 """
16 PyMite Heap Dump
17 ================
18 
19 Parses a heap dump into human-readable format.
20 
21 The heap dump file is created by inserting a calls to heap_dump()
22 inside heap_gcRun(). Using two calls, a before and after, is ideal.
23 
24 The dump format is:
25 
26 =========== ==========================================
27 NumBytes Contents
28 =========== ==========================================
29 6 string: PMDUMP or PMUDMP depending on target endianess (little and big respectively)
30 2 uint16: pointer size
31 2 uint16: dump format version
32 2 uint16: bifield of pmfeatures enabled on target
33 4 uint32: HEAP_SIZE
34 p pointer to heap start
35 HEAP_SIZE contents of heap (byte array)
36 4 uint16: NUM_ROOTS
37 NUM_ROOTS*p pointers to root objects
38 =========== ==========================================
39 
40 The heap_dump() function names files incrementally starting from:
41 
42  pmheapdump00.bin
43  pmheapdump01.bin
44  ...
45  pmheapdumpNN.bin

99  def __init__(self, name, fmt):
100  """Initializes a new object type from name and a format string
101  The format consist of a list of field name, type and optionnal multiplicity
102  fmt = "fieldname1:fieldtype1[:multiplicity1],fieldname2:fieldtype2[:multiplicity2],..."
103  where:
104  fieldname : a string naming a field...
105  fieldtype : a type understand by python struct [c, b, B, ?, h, H, i, I, l, L, q, Q, f, d, s, p, P] or '.' for a bit
106  P (pointer) is translated to the correct size based on the dump header information
107  multiplicity : number of item for a list. multiplicity can be:
108  any positive integer
109  a string naming a field already described
110  Multiplicity can also specify:
111  '*' to read as much as possible in the limit of the size object
112  '<name' to read as long as memory offset is lower than then value of field name
113  """
114 
115  self._parse_fmt(fmt)

119  def _parse_fmt(self, fmt):
120 
121  self.fields = []
122  for f in map(lambda s: s.strip(), fmt.split(',')):
123  if f == '':
124  continue
125  name, typ = f.split(':', 1)
126  if ':' in typ:
127  typ, mul = typ.split(':')
128  try:
129  mul = int(mul)
130  except:
131  pass
132  else:
133  mul = False
134 
135  self.fields.append(PmFieldInfo(name, typ, mul))
136 
137  # Bitfield support : concatenate bit-fields in a bitfield
138  bitfield = None
139  self._rawfields = []
140 
141  for i, f in enumerate(self.fields):
142 
143  if f.type != '.':
144  self._rawfields.append(f)
145  continue
146 
147  if bitfield == None:
148  bitfield = []
149 
150  bitfield.append(f)
151 
152  # Next field is a bit in this bitfield
153  if i + 1 < len(self.fields) and self.fields[i + 1].type == '.':
154  continue
155 
156  # Last bit in field : sum the bitfield and add it to the fields list
157  bitcount = sum([sf.mul for sf in bitfield])
158  bytes = 1
159  while (bytes * 8 < bitcount):
160  bytes *= bytes
161  typechr = [None, 'B', 'H', None, 'I', None, None, None, 'Q'][bytes]
162 
163  self._rawfields.append(PmBitfieldInfo(typechr, bitfield, False))
164  bitfield = None

167  def _calc_fieldmap(self, obj):
168  """Computes fieldmap and format
169  Returns a triple(format, : a struct format string
170  fielmap, : a list of Pm(Bit)FieldInfo
171  remaining : number of field that cannot ne computed due to a missing value
172  )
173  """
174 
175  d = obj.data
176  fmt = obj.heap.endianchr + "H" # Skip object descriptor
177  fieldmap = [None]
178 
179  for i, f in enumerate(self._rawfields):
180  typechr = (f.type == 'P') and obj.heap.ptrchr or f.type
181 
182  while (struct.calcsize(fmt) % struct.calcsize(typechr)!=0):
183  fmt += 'x'
184 
185  if f.mul == False:
186  fmt += typechr
187  fieldmap.append(f)
188 
189  else:
190  d[f.name] = []
191  if isinstance(f.mul, int):
192  fmt += typechr * f.mul
193  fieldmap += [f] * f.mul
194  elif f.mul == '*':
195  while struct.calcsize(fmt)<obj.size:
196  fmt += typechr
197  fieldmap.append(f)
198  elif f.mul.startswith('<'):
199  if f.mul[1:] not in d:
200  return (fmt, fieldmap, len(self._rawfields)-i)
201 
202  while struct.calcsize(fmt + typechr) + obj.addr \
203  + obj.heap.base < d[f.mul[1:]]:
204  # and struct.calcsize(fmt+typechr)<obj.size:
205  fmt += typechr
206  fieldmap += [f]
207  else:
208  if f.mul not in d:
209  return (fmt, fieldmap, len(self._rawfields)-i)
210 
211  fmt += typechr * d[f.mul]
212  fieldmap += [f] * d[f.mul]
213 
214  return (fmt, fieldmap, 0)
