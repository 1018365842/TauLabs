92 static struct lsm303_dev *PIOS_LSM303_alloc(void)
93 {
94  struct lsm303_dev *lsm303_dev;
95 
96  lsm303_dev = (struct lsm303_dev *)pvPortMalloc(sizeof(*lsm303_dev));
97 
98  if (!lsm303_dev) return (NULL);
99 
100  lsm303_dev->magic = PIOS_LSM303_DEV_MAGIC;
101 
102  lsm303_dev->queue_accel = xQueueCreate(PIOS_LSM303_MAX_QUEUESIZE, sizeof(struct pios_sensor_accel_data));
103 
104  if (lsm303_dev->queue_accel == NULL) {
105  vPortFree(lsm303_dev);
106  return NULL;
107  }
108 
109  lsm303_dev->queue_mag = xQueueCreate(PIOS_LSM303_MAX_QUEUESIZE, sizeof(struct pios_sensor_mag_data));
110 
111  if (lsm303_dev->queue_mag == NULL) {
112  vPortFree(lsm303_dev);
113  return NULL;
114  }
115 
116  lsm303_dev->data_ready_sema = xSemaphoreCreateMutex();
117 
118  if (lsm303_dev->data_ready_sema == NULL) {
119  vPortFree(lsm303_dev);
120  return NULL;
121  }
122 
123  return(lsm303_dev);
124 }

148 int32_t PIOS_LSM303_Init(uint32_t i2c_id, const struct pios_lsm303_cfg *cfg)
149 {
150  pios_lsm303_dev = PIOS_LSM303_alloc();
151 
152  if (pios_lsm303_dev == NULL)
153  return -1;
154 
155  pios_lsm303_dev->i2c_id = i2c_id;
156 
157  switch (cfg->devicetype) {
158  case PIOS_LSM303DLHC_DEVICE:
159  pios_lsm303_dev->i2c_addr_accel = 0x19;
160  pios_lsm303_dev->i2c_addr_mag = 0x1e;
161  break;
162 
163  default:
164  //not implemented
165  PIOS_Assert(false);
166  }
167 
168  pios_lsm303_dev->cfg = cfg;
169 
170  /* Configure the LSM303 Sensor */
171  PIOS_LSM303_Config(cfg);
172 
173  int result = xTaskCreate(PIOS_LSM303_Task, (const signed char *)"pios_lsm303",
174  LSM303_TASK_STACK, NULL, LSM303_TASK_PRIORITY,
175  &pios_lsm303_dev->TaskHandle);
176  PIOS_Assert(result == pdPASS);
177 
178  /* Set up EXTI line */
179  PIOS_EXTI_Init(cfg->exti_cfg);
180 
181  PIOS_SENSORS_Register(PIOS_SENSOR_ACCEL, pios_lsm303_dev->queue_accel);
182  PIOS_SENSORS_Register(PIOS_SENSOR_MAG, pios_lsm303_dev->queue_mag);
183 
184  return 0;
185 }

193 static void PIOS_LSM303_Config(struct pios_lsm303_cfg const *cfg)
194 {
195  /*
196  * accel
197  */
198 
199  // Reset
200  while (PIOS_LSM303_Accel_SetReg(PIOS_LSM303_CTRL_REG5_A, PIOS_LSM303_CTRL5_BOOT) != 0);
201 
202  // This register enables the channels and sets the bandwidth
203  while (PIOS_LSM303_Accel_SetReg(PIOS_LSM303_CTRL_REG1_A,
204  PIOS_LSM303_CTRL1_400HZ |
205  PIOS_LSM303_CTRL1_ZEN |
206  PIOS_LSM303_CTRL1_YEN |
207  PIOS_LSM303_CTRL1_XEN) != 0);
208 
209  // Disable the high pass filters
210  while (PIOS_LSM303_Accel_SetReg(PIOS_LSM303_CTRL_REG2_A, 0) != 0);
211 
212  // Set INT1 to go high on data ready
213  while (PIOS_LSM303_Accel_SetReg(PIOS_LSM303_CTRL_REG3_A, PIOS_LSM303_CTRL3_I1_DRDY1) != 0);
214 
215  // Set High Resolution mode
216  while (PIOS_LSM303_Accel_SetReg(PIOS_LSM303_CTRL_REG4_A, PIOS_LSM303_CTRL4_HR) != 0);
217 
218  // Set the accel scale
219  PIOS_LSM303_Accel_SetRange(PIOS_LSM303_ACCEL_8G);
220 
221  // Enabling the FIFO reduces resolution to 10-bits. It's default behaviour is disabled.
222  // This was found through trial and error. Is not documented in datasheet.
223 
224  /*
225  * mag
226  */
227  // set the mag bandwidth
228  while (PIOS_LSM303_Mag_SetReg(PIOS_LSM303_CRA_REG_M, PIOS_LSM303_CRA_220HZ) != 0);
229 
230  // set mag to continuous operation
231  while (PIOS_LSM303_Mag_SetReg(PIOS_LSM303_MR_REG_M, PIOS_LSM303_MR_CONTINUOUS) != 0);
232 
233  // Set the mag range
234  PIOS_LSM303_Mag_SetRange(PIOS_LSM303_MAG_1_9GA);
235 }

437 static int32_t PIOS_LSM303_Mag_ReadData(struct pios_lsm303_mag_data *data)
438 {
439  uint8_t temp[6];
440 
441  if (PIOS_LSM303_Mag_Read(PIOS_LSM303_OUT_X_H_M, temp, sizeof(temp)) < 0) {
442  return -2;
443  }
444 
445  data->mag_x = temp[0] << 8 | temp[1];
446  data->mag_z = temp[2] << 8 | temp[3];
447  data->mag_y = temp[4] << 8 | temp[5];
448 
449  return 0;

470 void PIOS_LSM303_Accel_SetRange(enum pios_lsm303_accel_range accel_range)
471 {
472  if (PIOS_LSM303_Validate(pios_lsm303_dev) != 0)
473  return;
474 
475  uint8_t buf;
476  PIOS_LSM303_Accel_Read(PIOS_LSM303_CTRL_REG4_A, &buf, 1);
477 
478  // We reset the scale bits, so when OR'ing new value it won't get OR'd with old scale
479  buf &= 0xCF;
480 
481  while (PIOS_LSM303_Accel_SetReg(PIOS_LSM303_CTRL_REG4_A, accel_range | buf) != 0);
482 
483  pios_lsm303_dev->accel_range = accel_range;
484 }

499 static float PIOS_LSM303_Accel_GetScale()
500 {
501  // Not validating device here for efficiency
502 
503  switch (pios_lsm303_dev->accel_range) {
504  case PIOS_LSM303_ACCEL_2G:
505  return GRAVITY / (16 * 1000.0f); //1mg/LSB, left shifted by four bits
506  case PIOS_LSM303_ACCEL_4G:
507  return GRAVITY / (16 * 500.0f); //2mg/LSB, left shifted by four bits
508  case PIOS_LSM303_ACCEL_8G:
509  return GRAVITY / (16 * 250.0f); //4mg/LSB, left shifted by four bits
510  case PIOS_LSM303_ACCEL_16G:
511  return GRAVITY / (16 * 250.0f / 3); //12mg/LSB, left shifted by four bits
512  }
513 
514  return 0;
515 }

607 static void PIOS_LSM303_Task(void *parameters)
608 {
609  while (1) {
610  //Wait for data ready interrupt
611  if (xSemaphoreTake(pios_lsm303_dev->data_ready_sema, 5 * portTICK_RATE_MS) != pdTRUE) {
612  // If this expires kick start the sensor
613  struct pios_lsm303_accel_data data;
614  PIOS_LSM303_Accel_ReadData(&data);
615  continue;
616  }
617 
618  /*
619  * Process accel data
620  */
621  {
622  struct pios_lsm303_accel_data data;
623 
624  if (PIOS_LSM303_Accel_ReadData(&data) < 0) {
625  continue;
626  }
627 
628  float accel_scale = PIOS_LSM303_Accel_GetScale();
629 
630  struct pios_sensor_accel_data normalized_data;
631 
632  switch (pios_lsm303_dev->cfg->orientation) {
633  default:
634  case PIOS_LSM303_TOP_0DEG:
635  normalized_data.x = +data.accel_x * accel_scale;
636  normalized_data.y = -data.accel_y * accel_scale;
637  break;
638  case PIOS_LSM303_TOP_90DEG:
639  normalized_data.x = +data.accel_y * accel_scale;
640  normalized_data.y = +data.accel_x * accel_scale;
641  break;
642  case PIOS_LSM303_TOP_180DEG:
643  normalized_data.x = -data.accel_x * accel_scale;
644  normalized_data.y = +data.accel_y * accel_scale;
645  break;
646  case PIOS_LSM303_TOP_270DEG:
647  normalized_data.x = -data.accel_y * accel_scale;
648  normalized_data.y = -data.accel_x * accel_scale;
649  break;
650  }
651 
652  normalized_data.z = -data.accel_z * accel_scale;
653  normalized_data.temperature = 0;
654 
655  xQueueSendToBack(pios_lsm303_dev->queue_accel, (void *)&normalized_data, 0);
656  }
657 
658  /*
659  * Process mag data
660  */
661  {
662  if ((PIOS_LSM303_Mag_GetReg(PIOS_LSM303_SR_REG_M) & PIOS_LSM303_SR_DRDY) != 0) {
663  struct pios_lsm303_mag_data data;
664 
665  if (PIOS_LSM303_Mag_ReadData(&data) < 0) {
666  continue;
667  }
668 
669  float mag_scale_xy = PIOS_LSM303_Mag_GetScaleXY();
670  float mag_scale_z = PIOS_LSM303_Mag_GetScaleZ();
671  struct pios_sensor_mag_data normalized_data;
672 
673  switch (pios_lsm303_dev->cfg->orientation) {
674  default:
675  case PIOS_LSM303_TOP_0DEG:
676  normalized_data.x = +data.mag_x * mag_scale_xy;
677  normalized_data.y = -data.mag_y * mag_scale_xy;
678  break;
679  case PIOS_LSM303_TOP_90DEG:
680  normalized_data.x = +data.mag_y * mag_scale_xy;
681  normalized_data.y = +data.mag_x * mag_scale_xy;
682  break;
683  case PIOS_LSM303_TOP_180DEG:
684  normalized_data.x = -data.mag_x * mag_scale_xy;
685  normalized_data.y = +data.mag_y * mag_scale_xy;
686  break;
687  case PIOS_LSM303_TOP_270DEG:
688  normalized_data.x = -data.mag_y * mag_scale_xy;
689  normalized_data.y = -data.mag_x * mag_scale_xy;
690  break;
691  }
692 
693  normalized_data.z = -data.mag_z * mag_scale_z;
694 
695  xQueueSendToBack(pios_lsm303_dev->queue_mag, (void *)&normalized_data, 0);
696  }
697  }
698  }
