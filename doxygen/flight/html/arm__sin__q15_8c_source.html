109 q15_t arm_sin_q15(
110  q15_t x)
111 {
112  q31_t sinVal; /* Temporary variables output */
113  q15_t *tablePtr; /* Pointer to table */
114  q15_t fract, in, in2; /* Temporary variables for input, output */
115  q31_t wa, wb, wc, wd; /* Cubic interpolation coefficients */
116  q15_t a, b, c, d; /* Four nearest output values */
117  q15_t fractCube, fractSquare; /* Temporary values for fractional value */
118  q15_t oneBy6 = 0x1555; /* Fixed point value of 1/6 */
119  q15_t tableSpacing = TABLE_SPACING_Q15; /* Table spacing */
120  int32_t index; /* Index variable */
121 
122  in = x;
123 
124  /* Calculate the nearest index */
125  index = (int32_t) in / tableSpacing;
126 
127  /* Calculate the nearest value of input */
128  in2 = (q15_t) ((index) * tableSpacing);
129 
130  /* Calculation of fractional value */
131  fract = (in - in2) << 8;
132 
133  /* fractSquare = fract * fract */
134  fractSquare = (q15_t) ((fract * fract) >> 15);
135 
136  /* fractCube = fract * fract * fract */
137  fractCube = (q15_t) ((fractSquare * fract) >> 15);
138 
139  /* Checking min and max index of table */
140  if(index < 0)
141  {
142  index = 0;
143  }
144  else if(index > 256)
145  {
146  index = 256;
147  }
148 
149  /* Initialise table pointer */
150  tablePtr = (q15_t *) & sinTableQ15[index];
151 
152  /* Cubic interpolation process */
153  /* Calculation of wa */
154  /* wa = -(oneBy6)*fractCube + (fractSquare >> 1u) - (0x2AAA)*fract; */
155  wa = (q31_t) oneBy6 *fractCube;
156  wa += (q31_t) 0x2AAA *fract;
157  wa = -(wa >> 15);
158  wa += ((q31_t) fractSquare >> 1u);
159 
160  /* Read first nearest value of output from the sin table */
161  a = *tablePtr++;
162 
163  /* sinVal = a * wa */
164  sinVal = a * wa;
165 
166  /* Calculation of wb */
167  wb = (((q31_t) fractCube >> 1u) - (q31_t) fractSquare) -
168  (((q31_t) fract >> 1u) - 0x7FFF);
169 
170  /* Read second nearest value of output from the sin table */
171  b = *tablePtr++;
172 
173  /* sinVal += b*wb */
174  sinVal += b * wb;
175 
176 
177  /* Calculation of wc */
178  wc = -(q31_t) fractCube + fractSquare;
179  wc = (wc >> 1u) + fract;
180 
181  /* Read third nearest value of output from the sin table */
182  c = *tablePtr++;
183 
184  /* sinVal += c*wc */
185  sinVal += c * wc;
186 
187  /* Calculation of wd */
188  /* wd = (oneBy6)*fractCube - (oneBy6)*fract; */
189  fractCube = fractCube - fract;
190  wd = ((q15_t) (((q31_t) oneBy6 * fractCube) >> 15));
191 
192  /* Read fourth nearest value of output from the sin table */
193  d = *tablePtr++;
194 
195  /* sinVal += d*wd; */
196  sinVal += d * wd;
197 
198  /* Convert output value in 1.15(q15) format and saturate */
199  sinVal = __SSAT((sinVal >> 15), 16);
200 
201  /* Return the output value in 1.15(q15) format */
202  return ((q15_t) sinVal);
203 
204 }
