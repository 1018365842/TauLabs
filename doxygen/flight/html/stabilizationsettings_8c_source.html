48 int32_t StabilizationSettingsInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(STABILIZATIONSETTINGS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(STABILIZATIONSETTINGS_OBJID,
56  STABILIZATIONSETTINGS_ISSINGLEINST, STABILIZATIONSETTINGS_ISSETTINGS, STABILIZATIONSETTINGS_NUMBYTES, &StabilizationSettingsSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }
