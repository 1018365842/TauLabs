80 int32_t TxPIDInitialize(void)
81 {
82  bool module_enabled;
83 
84 #ifdef MODULE_TxPID_BUILTIN
85  module_enabled = true;
86 #else
87  uint8_t module_state[MODULESETTINGS_ADMINSTATE_NUMELEM];
88  ModuleSettingsAdminStateGet(module_state);
89  if (module_state[MODULESETTINGS_ADMINSTATE_TXPID] == MODULESETTINGS_ADMINSTATE_ENABLED) {
90  module_enabled = true;
91  } else {
92  module_enabled = false;
93  }
94 #endif
95 
96  if (module_enabled) {
97 
98  TxPIDSettingsInitialize();
99  AccessoryDesiredInitialize();
100 
101  UAVObjEvent ev = {
102  .obj = AccessoryDesiredHandle(),
103  .instId = 0,
104  .event = 0,
105  };
106  EventPeriodicCallbackCreate(&ev, updatePIDs, SAMPLE_PERIOD_MS / portTICK_RATE_MS);
107 
108 #if (TELEMETRY_UPDATE_PERIOD_MS != 0)
109  // Change StabilizationSettings update rate from OnChange to periodic
110  // to prevent telemetry link flooding with frequent updates in case of
111  // control channel jitter.
112  // Warning: saving to flash with this code active will change the
113  // StabilizationSettings update rate permanently. Use Metadata via
114  // browser to reset to defaults (telemetryAcked=true, OnChange).
115  UAVObjMetadata metadata;
116  StabilizationSettingsInitialize();
117  StabilizationSettingsGetMetadata(&metadata);
118  metadata.telemetryAcked = 0;
119  metadata.telemetryUpdateMode = UPDATEMODE_PERIODIC;
120  metadata.telemetryUpdatePeriod = TELEMETRY_UPDATE_PERIOD_MS;
121  StabilizationSettingsSetMetadata(&metadata);
122 #endif
123 
124  return 0;
125  }
126 
127  return -1;
128 }

135 static void updatePIDs(UAVObjEvent* ev)
136 {
137  if (ev->obj != AccessoryDesiredHandle())
138  return;
139 
140  TxPIDSettingsData inst;
141  TxPIDSettingsGet(&inst);
142 
143  if (inst.UpdateMode == TXPIDSETTINGS_UPDATEMODE_NEVER)
144  return;
145 
146  uint8_t armed;
147  FlightStatusArmedGet(&armed);
148  if ((inst.UpdateMode == TXPIDSETTINGS_UPDATEMODE_WHENARMED) &&
149  (armed == FLIGHTSTATUS_ARMED_DISARMED))
150  return;
151 
152  StabilizationSettingsData stab;
153  StabilizationSettingsGet(&stab);
154  AccessoryDesiredData accessory;
155 
156  uint8_t needsUpdate = 0;
157 
158  // Loop through every enabled instance
159  for (uint8_t i = 0; i < TXPIDSETTINGS_PIDS_NUMELEM; i++) {
160  if (inst.PIDs[i] != TXPIDSETTINGS_PIDS_DISABLED) {
161 
162  float value;
163  if (inst.Inputs[i] == TXPIDSETTINGS_INPUTS_THROTTLE) {
164  ManualControlCommandThrottleGet(&value);
165  value = scale(value,
166  inst.ThrottleRange[TXPIDSETTINGS_THROTTLERANGE_MIN],
167  inst.ThrottleRange[TXPIDSETTINGS_THROTTLERANGE_MAX],
168  inst.MinPID[i], inst.MaxPID[i]);
169  } else if (AccessoryDesiredInstGet(inst.Inputs[i] - TXPIDSETTINGS_INPUTS_ACCESSORY0, &accessory) == 0) {
170  value = scale(accessory.AccessoryVal, -1.0, 1.0, inst.MinPID[i], inst.MaxPID[i]);
171  } else {
172  continue;
173  }
174 
175  switch (inst.PIDs[i]) {
176  case TXPIDSETTINGS_PIDS_ROLLRATEKP:
177  needsUpdate |= update(&stab.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP], value);
178  break;
179  case TXPIDSETTINGS_PIDS_ROLLRATEKI:
180  needsUpdate |= update(&stab.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI], value);
181  break;
182  case TXPIDSETTINGS_PIDS_ROLLRATEKD:
183  needsUpdate |= update(&stab.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KD], value);
184  break;
185  case TXPIDSETTINGS_PIDS_ROLLRATEILIMIT:
186  needsUpdate |= update(&stab.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_ILIMIT], value);
187  break;
188  case TXPIDSETTINGS_PIDS_ROLLATTITUDEKP:
189  needsUpdate |= update(&stab.RollPI[STABILIZATIONSETTINGS_ROLLPI_KP], value);
190  break;
191  case TXPIDSETTINGS_PIDS_ROLLATTITUDEKI:
192  needsUpdate |= update(&stab.RollPI[STABILIZATIONSETTINGS_ROLLPI_KI], value);
193  break;
194  case TXPIDSETTINGS_PIDS_ROLLATTITUDEILIMIT:
195  needsUpdate |= update(&stab.RollPI[STABILIZATIONSETTINGS_ROLLPI_ILIMIT], value);
196  break;
197  case TXPIDSETTINGS_PIDS_PITCHRATEKP:
198  needsUpdate |= update(&stab.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KP], value);
199  break;
200  case TXPIDSETTINGS_PIDS_PITCHRATEKI:
201  needsUpdate |= update(&stab.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KI], value);
202  break;
203  case TXPIDSETTINGS_PIDS_PITCHRATEKD:
204  needsUpdate |= update(&stab.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KD], value);
205  break;
206  case TXPIDSETTINGS_PIDS_PITCHRATEILIMIT:
207  needsUpdate |= update(&stab.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_ILIMIT], value);
208  break;
209  case TXPIDSETTINGS_PIDS_PITCHATTITUDEKP:
210  needsUpdate |= update(&stab.PitchPI[STABILIZATIONSETTINGS_PITCHPI_KP], value);
211  break;
212  case TXPIDSETTINGS_PIDS_PITCHATTITUDEKI:
213  needsUpdate |= update(&stab.PitchPI[STABILIZATIONSETTINGS_PITCHPI_KI], value);

221  break;

227  needsUpdate |= update(&stab.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KD], value);

245  break;
246  case TXPIDSETTINGS_PIDS_YAWRATEKP:
247  needsUpdate |= update(&stab.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KP], value);
248  break;
249  case TXPIDSETTINGS_PIDS_YAWRATEKI:
250  needsUpdate |= update(&stab.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KI], value);
251  break;
252  case TXPIDSETTINGS_PIDS_YAWRATEKD:
253  needsUpdate |= update(&stab.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KD], value);
254  break;
255  case TXPIDSETTINGS_PIDS_YAWRATEILIMIT:
256  needsUpdate |= update(&stab.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_ILIMIT], value);
257  break;
258  case TXPIDSETTINGS_PIDS_YAWATTITUDEKP:
259  needsUpdate |= update(&stab.YawPI[STABILIZATIONSETTINGS_YAWPI_KP], value);
260  break;
261  case TXPIDSETTINGS_PIDS_YAWATTITUDEKI:
262  needsUpdate |= update(&stab.YawPI[STABILIZATIONSETTINGS_YAWPI_KI], value);
263  break;
264  case TXPIDSETTINGS_PIDS_YAWATTITUDEILIMIT:
265  needsUpdate |= update(&stab.YawPI[STABILIZATIONSETTINGS_YAWPI_ILIMIT], value);
266  break;
267  case TXPIDSETTINGS_PIDS_GYROTAU:
268  needsUpdate |= update(&stab.GyroTau, value);
269  break;
270  default:
271  PIOS_Assert(0);
272  }
273  }
274  }
275  if (needsUpdate)
276  StabilizationSettingsSet(&stab);
277 }

286 static float scale(float val, float inMin, float inMax, float outMin, float outMax)
287 {
288  // bound input value
289  if (val > inMax) val = inMax;
290  if (val < inMin) val = inMin;
291 
292  // normalize input value to [0..1]
293  if (inMax <= inMin)
294  val = 0.0;
295  else
296  val = (val - inMin) / (inMax - inMin);
297 
298  // update output bounds
299  if (outMin > outMax) {
300  float t = outMin;
301  outMin = outMax;
302  outMax = t;
303  val = 1.0 - val;
304  }
305 
306  return (outMax - outMin) * val + outMin;
307 }
