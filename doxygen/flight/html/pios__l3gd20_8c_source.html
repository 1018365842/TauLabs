163 static void PIOS_L3GD20_Config(const struct pios_l3gd20_cfg *cfg)
164 {
165  // This register enables the channels and sets the bandwidth
166  while (PIOS_L3GD20_SetReg(PIOS_L3GD20_CTRL_REG1, PIOS_L3GD20_CTRL1_FASTEST |
167  PIOS_L3GD20_CTRL1_PD | PIOS_L3GD20_CTRL1_ZEN |
168  PIOS_L3GD20_CTRL1_YEN | PIOS_L3GD20_CTRL1_XEN) != 0);
169 
170  // Disable the high pass filters
171  while (PIOS_L3GD20_SetReg(PIOS_L3GD20_CTRL_REG2, 0) != 0);
172 
173  // Set int2 to go high on data ready
174  while (PIOS_L3GD20_SetReg(PIOS_L3GD20_CTRL_REG3, 0x08) != 0);
175 
176  // Select SPI interface, 500 deg/s, endianness?
177  while (PIOS_L3GD20_SetRange(cfg->range) != 0);
178 
179  // disable HPF
180  while (PIOS_L3GD20_SetReg(PIOS_L3GD20_CTRL_REG5, 0x00) != 0);
181 
182  pios_l3gd20_dev->configured = true;
183 }

333 static int32_t PIOS_L3GD20_ReadGyros(struct pios_l3gd20_data *data)
334 {
335  uint8_t buf[7] = { PIOS_L3GD20_GYRO_X_OUT_LSB | 0x80 | 0x40, 0, 0, 0, 0, 0, 0 };
336  uint8_t rec[7];
337 
338  if (PIOS_L3GD20_ClaimBus() != 0)
339  return -1;
340 
341  if (PIOS_SPI_TransferBlock(pios_l3gd20_dev->spi_id, &buf[0], &rec[0], sizeof(buf), NULL) < 0) {
342  PIOS_L3GD20_ReleaseBus();
343  data->gyro_x = 0;
344  data->gyro_y = 0;
345  data->gyro_z = 0;
346  data->temperature = 0;
347  return -2;
348  }
349 
350  PIOS_L3GD20_ReleaseBus();
351 
352  memcpy((uint8_t *)&data->gyro_x, &rec[1], 6);
353  data->temperature = PIOS_L3GD20_GetReg(PIOS_L3GD20_OUT_TEMP);
354 
355  return 0;
356 }

412 bool PIOS_L3GD20_IRQHandler(void)
413 {
414  if (PIOS_L3GD20_Validate(pios_l3gd20_dev) != 0 || pios_l3gd20_dev->configured == false)
415  return false;
416 
417  struct pios_l3gd20_data data;
418  uint8_t buf[7] = { PIOS_L3GD20_GYRO_X_OUT_LSB | 0x80 | 0x40, 0, 0, 0, 0, 0, 0 };
419  uint8_t rec[7];
420 
421  /* This code duplicates ReadGyros above but uses ClaimBusIsr */
422  bool woken = false;
423 
424  if (PIOS_L3GD20_ClaimBusIsr(&woken) != 0)
425  return woken;
426 
427  if (PIOS_SPI_TransferBlock(pios_l3gd20_dev->spi_id, &buf[0], &rec[0], sizeof(buf), NULL) < 0) {
428  PIOS_L3GD20_ReleaseBusIsr(&woken);
429  return woken;
430  }
431 
432  PIOS_L3GD20_ReleaseBusIsr(&woken);
433 
434  memcpy((uint8_t *)&data.gyro_x, &rec[1], 6);
435 
436  // TODO: This reordering is specific to the FlyingF3 chip placement. Whenever
437  // this code is used on another board add an orientation mapping to the configuration
438  struct pios_sensor_gyro_data normalized_data;
439  float scale = PIOS_L3GD20_GetScale();
440  normalized_data.y = data.gyro_x * scale;
441  normalized_data.x = data.gyro_y * scale;
442  normalized_data.z = -data.gyro_z * scale;
443  normalized_data.temperature = PIOS_L3GD20_GetRegIsr(PIOS_L3GD20_OUT_TEMP, &woken);
444 
445  portBASE_TYPE xHigherPriorityTaskWoken;
446  xQueueSendToBackFromISR(pios_l3gd20_dev->queue, (void *)&normalized_data, &xHigherPriorityTaskWoken);
447 
448  return woken || (xHigherPriorityTaskWoken == pdTRUE);
449 }
