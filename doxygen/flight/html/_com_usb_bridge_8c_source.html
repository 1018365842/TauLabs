132 static void com2UsbBridgeTask(void *parameters)
133 {
134  /* Handle usart -> vcp direction */
135  volatile uint32_t tx_errors = 0;
136  while (1) {
137  uint32_t rx_bytes;
138 
139  rx_bytes = PIOS_COM_ReceiveBuffer(usart_port, com2usb_buf, BRIDGE_BUF_LEN, 500);
140  if (rx_bytes > 0) {
141  /* Bytes available to transfer */
142  if (PIOS_COM_SendBuffer(vcp_port, com2usb_buf, rx_bytes) != rx_bytes) {
143  /* Error on transmit */
144  tx_errors++;
145  }
146  }
147  }
148 }
