85 static void exampleTask(void *parameters)
86 {
87  UAVObjEvent ev;
88  ExampleSettingsData settings;
89  ExampleObject1Data data1;
90  ExampleObject1Data data2;
91  int32_t step;
92 
93  // Main task loop
94  while (1) {
95  // Check the event queue for any object updates. In this case
96  // we are only listening for the ExampleSettings object. However
97  // the module could listen to multiple objects.
98  // Since this object only executes on object updates, the task
99  // will block until an object event is pushed in the queue.
100  while (xQueueReceive(queue, &ev, portMAX_DELAY) != pdTRUE) ;
101 
102  // Make sure that the object update is for ExampleObject1
103  // This is redundant in this example since we have only
104  // registered a single object in the queue, however
105  // in most practical cases there will be more than one
106  // object connected in the queue.
107  if (ev.obj == ExampleObject1Handle()) {
108  // Update settings with latest value
109  ExampleSettingsGet(&settings);
110 
111  // Get the input object
112  ExampleObject1Get(&data1);
113 
114  // Determine how to update the output object
115  if (settings.StepDirection == EXAMPLESETTINGS_STEPDIRECTION_UP) {
116  step = settings.StepSize;
117  } else {
118  step = -settings.StepSize;
119  }
120 
121  // Update data
122  data2.Field1 = data1.Field1 + step;
123  data2.Field2 = data1.Field2 + step;
124  data2.Field3 = data1.Field3 + step;
125  data2.Field4[0] = data1.Field4[0] + step;
126  data2.Field4[1] = data1.Field4[1] + step;
127 
128  // Update the ExampleObject2, after this function is called
129  // notifications to any other modules listening to that object
130  // will be sent and the GCS object will be updated through the
131  // telemetry link. All operations will take place asynchronously
132  // and the following call will return immediately.
133  ExampleObject2Set(&data2);
134  }
135  }
136 }
