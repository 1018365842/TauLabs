145 static void vtolPathFollowerTask(void *parameters)
146 {
147  SystemSettingsData systemSettings;
148  FlightStatusData flightStatus;
149 
150  portTickType lastUpdateTime;
151 
152  VtolPathFollowerSettingsConnectCallback(SettingsUpdatedCb);
153  PathDesiredConnectCallback(SettingsUpdatedCb);
154 
155  VtolPathFollowerSettingsGet(&guidanceSettings);
156  PathDesiredGet(&pathDesired);
157 
158  // Main task loop
159  lastUpdateTime = xTaskGetTickCount();
160  while (1) {
161 
162  // Conditions when this runs:
163  // 1. Must have VTOL type airframe
164  // 2. Flight mode is PositionHold and PathDesired.Mode is Endpoint OR
165  // FlightMode is PathPlanner and PathDesired.Mode is Endpoint or Path
166 
167  SystemSettingsGet(&systemSettings);
168  if ( (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_VTOL) &&
169  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_QUADP) &&
170  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_QUADX) &&
171  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HEXA) &&
172  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HEXAX) &&
173  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_HEXACOAX) &&
174  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTO) &&
175  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTOV) &&
176  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXP) &&
177  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXX) &&
178  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_TRI) )
179  {
180  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
181  vTaskDelay(1000);
182  continue;
183  }
184 
185  // Continue collecting data if not enough time
186  vTaskDelayUntil(&lastUpdateTime, guidanceSettings.UpdatePeriod / portTICK_RATE_MS);
187 
188  // Convert the accels into the NED frame
189  updateNedAccel();
190 
191  FlightStatusGet(&flightStatus);
192 
193  // Check the combinations of flightmode and pathdesired mode
194  switch(flightStatus.FlightMode) {
195  /* This combination of RETURNTOHOME and HOLDPOSITION looks strange but
196  * is correct. RETURNTOHOME mode uses HOLDPOSITION with the position
197  * set to home */
198  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
199  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
200  updateEndpointVelocity();
201  updateVtolDesiredAttitude();
202  } else {
203  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
204  }
205  break;
206  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
207  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
208  updateEndpointVelocity();
209  updateVtolDesiredAttitude();
210  } else {
211  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
212  }
213  break;
214  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
215  if (pathDesired.Mode == PATHDESIRED_MODE_FLYENDPOINT ||
216  pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
217  updateEndpointVelocity();
218  updateVtolDesiredAttitude();
219  } else if (pathDesired.Mode == PATHDESIRED_MODE_FLYVECTOR ||
220  pathDesired.Mode == PATHDESIRED_MODE_FLYCIRCLELEFT ||
221  pathDesired.Mode == PATHDESIRED_MODE_FLYCIRCLERIGHT) {
222  updatePathVelocity();
223  updateVtolDesiredAttitude();
224  } else {
225  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
226  }
227  break;
228  default:
229 
230  for (uint32_t i = 0; i < VTOL_PID_NUM; i++)
231  pid_zero(&vtol_pids[i]);
232 
233  // Track throttle before engaging this mode. Cheap system ident
234  StabilizationDesiredData stabDesired;
235  StabilizationDesiredGet(&stabDesired);
236  throttleOffset = stabDesired.Throttle;
237 
238  break;
239  }
240 
241  AlarmsClear(SYSTEMALARMS_ALARM_PATHFOLLOWER);
242 
243  }
244 }

252 static void updatePathVelocity()
253 {
254  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
255 
256  PositionActualData positionActual;
257  PositionActualGet(&positionActual);
258 
259  float cur[3] = {positionActual.North, positionActual.East, positionActual.Down};
260  struct path_status progress;
261 
262  path_progress(&pathDesired, cur, &progress);
263 
264  // Update the path status UAVO
265  PathStatusData pathStatus;
266  PathStatusGet(&pathStatus);
267  pathStatus.fractional_progress = progress.fractional_progress;
268  if (pathStatus.fractional_progress < 1)
269  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
270  else
271  pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
272  PathStatusSet(&pathStatus);
273 
274  float groundspeed = pathDesired.StartingVelocity +
275  (pathDesired.EndingVelocity - pathDesired.StartingVelocity) * progress.fractional_progress;
276  if(progress.fractional_progress > 1)
277  groundspeed = 0;
278 
279  VelocityDesiredData velocityDesired;
280  velocityDesired.North = progress.path_direction[0] * groundspeed;
281  velocityDesired.East = progress.path_direction[1] * groundspeed;
282 
283  float error_speed = progress.error * guidanceSettings.HorizontalPosPI[VTOLPATHFOLLOWERSETTINGS_HORIZONTALPOSPI_KP];
284  float correction_velocity[2] = {progress.correction_direction[0] * error_speed,
285  progress.correction_direction[1] * error_speed};
286 
287  float total_vel = sqrtf(powf(correction_velocity[0],2) + powf(correction_velocity[1],2));
288  float scale = 1;
289  if(total_vel > guidanceSettings.HorizontalVelMax)
290  scale = guidanceSettings.HorizontalVelMax / total_vel;
291 
292  // Currently not apply a PID loop for horizontal corrections
293  velocityDesired.North += progress.correction_direction[0] * error_speed * scale;
294  velocityDesired.East += progress.correction_direction[1] * error_speed * scale;
295 
296  // Interpolate desired velocity along the path
297  float altitudeSetpoint = pathDesired.Start[2] + (pathDesired.End[2] - pathDesired.Start[2]) *
298  bound_min_max(progress.fractional_progress,0,1);
299 
300  float downError = altitudeSetpoint - positionActual.Down;
301  velocityDesired.Down = pid_apply_antiwindup(&vtol_pids[DOWN_POSITION], downError,
302  -guidanceSettings.VerticalVelMax, guidanceSettings.VerticalVelMax, dT);
303 
304  VelocityDesiredSet(&velocityDesired);
305 }

313 void updateEndpointVelocity()
314 {
315  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
316 
317  PositionActualData positionActual;
318  VelocityDesiredData velocityDesired;
319 
320  PositionActualGet(&positionActual);
321  VelocityDesiredGet(&velocityDesired);
322 
323  float northError;
324  float eastError;
325  float downError;
326  float northCommand;
327  float eastCommand;
328 
329  float northPos = positionActual.North;
330  float eastPos = positionActual.East;
331  float downPos = positionActual.Down;
332 
333  // Compute desired north command velocity from position error
334  northError = pathDesired.End[PATHDESIRED_END_NORTH] - northPos;
335  northCommand = pid_apply_antiwindup(&vtol_pids[NORTH_POSITION], northError,
336  -guidanceSettings.HorizontalVelMax, guidanceSettings.HorizontalVelMax, dT);
337 
338  // Compute desired east command velocity from position error
339  eastError = pathDesired.End[PATHDESIRED_END_EAST] - eastPos;
340  eastCommand = pid_apply_antiwindup(&vtol_pids[EAST_POSITION], eastError,
341  -guidanceSettings.HorizontalVelMax, guidanceSettings.HorizontalVelMax, dT);
342 
343  // Limit the maximum velocity any direction (not north and east separately)
344  float total_vel = sqrtf(powf(northCommand,2) + powf(eastCommand,2));
345  float scale = 1;
346  if(total_vel > guidanceSettings.HorizontalVelMax)
347  scale = guidanceSettings.HorizontalVelMax / total_vel;
348 
349  velocityDesired.North = northCommand * scale;
350  velocityDesired.East = eastCommand * scale;
351 
352  // Compute the desired velocity from the position difference
353  downError = pathDesired.End[PATHDESIRED_END_DOWN] - downPos;
354  velocityDesired.Down = pid_apply_antiwindup(&vtol_pids[DOWN_POSITION], downError,
355  -guidanceSettings.VerticalVelMax, guidanceSettings.VerticalVelMax, dT);
356 
357  VelocityDesiredSet(&velocityDesired);
358 
359  // Indicate whether we are in radius of this endpoint
360  uint8_t path_status = PATHSTATUS_STATUS_INPROGRESS;
361  float distance2 = powf(northError, 2) + powf(eastError, 2);
362  if (distance2 < (guidanceSettings.EndpointRadius * guidanceSettings.EndpointRadius)) {
363  path_status = PATHSTATUS_STATUS_COMPLETED;
364  }
365  PathStatusStatusSet(&path_status);
366 }

375 static void updateVtolDesiredAttitude()
376 {
377  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
378 
379  VelocityDesiredData velocityDesired;
380  VelocityActualData velocityActual;
381  StabilizationDesiredData stabDesired;
382  AttitudeActualData attitudeActual;
383  NedAccelData nedAccel;
384  StabilizationSettingsData stabSettings;
385 
386  float northError;
387  float northCommand;
388 
389  float eastError;
390  float eastCommand;
391 
392  float downError;
393  float downCommand;
394 
395  VtolPathFollowerSettingsGet(&guidanceSettings);
396 
397  VelocityActualGet(&velocityActual);
398  VelocityDesiredGet(&velocityDesired);
399  StabilizationDesiredGet(&stabDesired);
400  VelocityDesiredGet(&velocityDesired);
401  AttitudeActualGet(&attitudeActual);
402  StabilizationSettingsGet(&stabSettings);
403  NedAccelGet(&nedAccel);
404 
405  float northVel = velocityActual.North;
406  float eastVel = velocityActual.East;
407  float downVel = velocityActual.Down;
408 
409  // Compute desired north command from velocity error
410  northError = velocityDesired.North - northVel;
411  northCommand = pid_apply_antiwindup(&vtol_pids[NORTH_VELOCITY], northError,
412  -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch, dT) + velocityDesired.North * guidanceSettings.VelocityFeedforward;
413 
414  // Compute desired east command from velocity error
415  eastError = velocityDesired.East - eastVel;
416  eastCommand = pid_apply_antiwindup(&vtol_pids[NORTH_VELOCITY], eastError,
417  -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch, dT) + velocityDesired.East * guidanceSettings.VelocityFeedforward;
418 
419  // Compute desired down command. Using NED accel as the damping term
420  downError = velocityDesired.Down - downVel;
421  // Negative is critical here since throttle is negative with down
422  downCommand = -pid_apply_antiwindup(&vtol_pids[DOWN_VELOCITY], downError, -1, 1, dT) +
423  nedAccel.Down * guidanceSettings.VerticalVelPID[VTOLPATHFOLLOWERSETTINGS_VERTICALVELPID_KD];
424 
425  // If this setting is zero then the throttle level available when enabled is used for hover:wq
426  float used_throttle_offset = (guidanceSettings.HoverThrottle == 0) ? throttleOffset : guidanceSettings.HoverThrottle;
427  stabDesired.Throttle = bound_min_max(downCommand + used_throttle_offset, 0, 1);
428 
429  // Project the north and east command signals into the pitch and roll based on yaw.
430  // For this to behave well the craft should move similarly for 5 deg roll versus 5 deg pitch.
431  // Notice the inputs are crudely bounded by the anti-winded but if both N and E were
432  // saturated and the craft were at 45 degrees that would result in a value greater than
433  // the limit, so apply limit again here.
434  stabDesired.Pitch = bound_min_max(-northCommand * cosf(attitudeActual.Yaw * M_PI / 180) +
435  -eastCommand * sinf(attitudeActual.Yaw * M_PI / 180),
436  -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch);
437  stabDesired.Roll = bound_min_max(-northCommand * sinf(attitudeActual.Yaw * DEG2RAD) +
438  eastCommand * cosf(attitudeActual.Yaw * DEG2RAD),
439  -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch);
440 
441  if(guidanceSettings.ThrottleControl == VTOLPATHFOLLOWERSETTINGS_THROTTLECONTROL_FALSE) {
442  // For now override throttle with manual control. Disable at your risk, quad goes to China.
443  ManualControlCommandData manualControl;
444  ManualControlCommandGet(&manualControl);
445  stabDesired.Throttle = manualControl.Throttle;
446  }
447 
448  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
449  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
450 
451  float yaw;
452  switch(guidanceSettings.YawMode) {
453  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_RATE:
454  /* This is awkward. This allows the transmitter to control the yaw while flying navigation */
455  ManualControlCommandYawGet(&yaw);
456  stabDesired.Yaw = stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_YAW] * yaw;
457  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
458  break;
459  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_AXISLOCK:
460  ManualControlCommandYawGet(&yaw);
461  stabDesired.Yaw = stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_YAW] * yaw;
462  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK;
463  break;
464  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_ATTITUDE:
465  ManualControlCommandYawGet(&yaw);
466  stabDesired.Yaw = stabSettings.YawMax * yaw;
467  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
468  break;
469  case VTOLPATHFOLLOWERSETTINGS_YAWMODE_POI:
470  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_POI;
471  break;
472  }
473 
474  StabilizationDesiredSet(&stabDesired);
475 }

480 static void updateNedAccel()
481 {
482  float accel[3];
483  float q[4];
484  float Rbe[3][3];
485  float accel_ned[3];
486 
487  // Collect downsampled attitude data
488  AccelsData accels;
489  AccelsGet(&accels);
490  accel[0] = accels.x;
491  accel[1] = accels.y;
492  accel[2] = accels.z;
493 
494  //rotate avg accels into earth frame and store it
495  AttitudeActualData attitudeActual;
496  AttitudeActualGet(&attitudeActual);
497  q[0]=attitudeActual.q1;
498  q[1]=attitudeActual.q2;
499  q[2]=attitudeActual.q3;
500  q[3]=attitudeActual.q4;
501  Quaternion2R(q, Rbe);
502  for (uint8_t i=0; i<3; i++){
503  accel_ned[i]=0;
504  for (uint8_t j=0; j<3; j++)
505  accel_ned[i] += Rbe[j][i]*accel[j];
506  }
507  accel_ned[2] += GRAVITY;
508 
509  NedAccelData accelData;
510  NedAccelGet(&accelData);
511  accelData.North = accel_ned[0];
512  accelData.East = accel_ned[1];
513  accelData.Down = accel_ned[2];
514  NedAccelSet(&accelData);
515 }
