158 int32_t PIOS_TCP_Init(uint32_t *tcp_id, const struct pios_tcp_cfg * cfg)
159 {
160 
161  pios_tcp_dev *tcp_dev = &pios_tcp_devices[pios_tcp_num_devices];
162 
163  pios_tcp_num_devices++;
164 
165 
166  /* initialize */
167  tcp_dev->rx_in_cb = NULL;
168  tcp_dev->tx_out_cb = NULL;
169  tcp_dev->cfg=cfg;
170 
171  /* assign socket */
172  tcp_dev->socket = socket(PF_INET, SOCK_STREAM, IPPROTO_TCP);
173  memset(&tcp_dev->server,0,sizeof(tcp_dev->server));
174  memset(&tcp_dev->client,0,sizeof(tcp_dev->client));
175  tcp_dev->server.sin_family = AF_INET;
176  tcp_dev->server.sin_addr.s_addr = INADDR_ANY; //inet_addr(tcp_dev->cfg->ip);
177  tcp_dev->server.sin_port = htons(tcp_dev->cfg->port);
178  int res= bind(tcp_dev->socket, (struct sockaddr *)&tcp_dev->server,sizeof(tcp_dev->server));
179  if (res == -1) {
180  perror("Binding socket failed\n");
181  exit(EXIT_FAILURE);
182  }
183 
184  res = listen(tcp_dev->socket, 10);
185  if (res == -1) {
186  perror("Socket listen failed\n");
187  exit(EXIT_FAILURE);
188  }
189 
190  fifoBuf_init(&tcp_dev->rx_fifo, tcp_dev->rx_buffer, PIOS_TCP_RX_BUFFER_SIZE);
191 
192  pthread_create(&tcp_dev->rxThread, NULL, PIOS_TCP_RxThread, (void*)tcp_dev);
193 
194  xTaskCreate(PIOS_TCP_RxTask, (signed char *)"TcpRx", 1024, (void*)tcp_dev, 2, &tcpRxTaskHandle);
195 
196  printf("udp dev %i - socket %i opened - result %i\n",pios_tcp_num_devices-1,tcp_dev->socket,res);
197 
198  *tcp_id = pios_tcp_num_devices-1;
199 
200  return res;
201 }

220 static void PIOS_TCP_TxStart(uint32_t tcp_id, uint16_t tx_bytes_avail)
221 {
222  pios_tcp_dev *tcp_dev = find_tcp_dev_by_id(tcp_id);
223 
224  PIOS_Assert(tcp_dev);
225 
226  int32_t length,len,rem;
227 
231  if (tcp_dev->tx_out_cb) {
232  while (tx_bytes_avail>0) {
233  bool tx_need_yield = false;
234  length = (tcp_dev->tx_out_cb)(tcp_dev->tx_out_context, tcp_dev->tx_buffer, PIOS_TCP_RX_BUFFER_SIZE, NULL, &tx_need_yield);
235  rem = length;
236  while (rem>0) {
237  if(tcp_dev->socket_connection != 0) {
238  len = write(tcp_dev->socket_connection, tcp_dev->tx_buffer, length);
239  }
240  if (len<=0) {
241  rem=0;
242  } else {
243  rem -= len;
244  }
245  }
246  tx_bytes_avail -= length;
247 #if defined(PIOS_INCLUDE_FREERTOS)
248  // Not sure about this
249  if (tx_need_yield) {
250  vPortYieldFromISR();
251  }
252 #endif /* PIOS_INCLUDE_FREERTOS */
253  }
254  }
