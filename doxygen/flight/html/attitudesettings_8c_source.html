48 int32_t AttitudeSettingsInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(ATTITUDESETTINGS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(ATTITUDESETTINGS_OBJID,
56  ATTITUDESETTINGS_ISSINGLEINST, ATTITUDESETTINGS_ISSETTINGS, ATTITUDESETTINGS_NUMBYTES, &AttitudeSettingsSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

74 void AttitudeSettingsSetDefaults(UAVObjHandle obj, uint16_t instId)
75 {
76  AttitudeSettingsData data;
77  UAVObjMetadata metadata;
78 
79  // Initialize object fields to their default values
80  UAVObjGetInstanceData(obj, instId, &data);
81  memset(&data, 0, sizeof(AttitudeSettingsData));
82  data.MagKp = 0.05;
83  data.MagKi = 0.0001;
84  data.AccelKp = 0.05;
85  data.AccelKi = 0.0001;
86  data.AccelTau = 0.1;
87  data.YawBiasRate = 1e-06;
88  data.BoardRotation[0] = 0;
89  data.BoardRotation[1] = 0;
90  data.BoardRotation[2] = 0;
91  data.ZeroDuringArming = 1;
92  data.BiasCorrectGyro = 1;
93  data.FilterChoice = 0;
94  data.TrimFlight = 0;
95 
96  UAVObjSetInstanceData(obj, instId, &data);
97 
98  // Initialize object metadata to their default values
99  metadata.flags =
100  ACCESS_READWRITE << UAVOBJ_ACCESS_SHIFT |
101  ACCESS_READWRITE << UAVOBJ_GCS_ACCESS_SHIFT |
102  1 << UAVOBJ_TELEMETRY_ACKED_SHIFT |
103  1 << UAVOBJ_GCS_TELEMETRY_ACKED_SHIFT |
104  UPDATEMODE_ONCHANGE << UAVOBJ_TELEMETRY_UPDATE_MODE_SHIFT |
105  UPDATEMODE_ONCHANGE << UAVOBJ_GCS_TELEMETRY_UPDATE_MODE_SHIFT;
106  metadata.telemetryUpdatePeriod = 0;
107  metadata.gcsTelemetryUpdatePeriod = 0;
108  metadata.loggingUpdatePeriod = 0;
109  UAVObjSetMetadata(obj, &metadata);
110 }
