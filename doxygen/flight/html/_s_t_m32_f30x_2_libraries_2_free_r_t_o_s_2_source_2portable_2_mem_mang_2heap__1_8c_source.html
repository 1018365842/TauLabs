71  * The simplest possible implementation of pvPortMalloc(). Note that this
72  * implementation does NOT allow allocated memory to be freed again.
73  *
74  * See heap_2.c, heap_3.c and heap_4.c for alternative implementations, and the
75  * memory management pages of http://www.FreeRTOS.org for more information.
76  */

89 /* Allocate the memory for the heap. The struct is used to force byte
90 alignment without using any non-portable code. */
91 static union xRTOS_HEAP
92 {
93  #if portBYTE_ALIGNMENT == 8
94  volatile portDOUBLE dDummy;
95  #else
96  volatile unsigned long ulDummy;
97  #endif
98  unsigned char ucHeap[ configTOTAL_HEAP_SIZE ];
99 } xHeap __attribute__ ((section (".heap")));
100 
101 static size_t xNextFreeByte = ( size_t ) 0;
102 static size_t currentTOTAL_HEAP_SIZE = configTOTAL_HEAP_SIZE;
103 
104 /*-----------------------------------------------------------*/
105 
106 void *pvPortMalloc( size_t xWantedSize )
107 {
108 void *pvReturn = NULL;
109 
110  /* Ensure that blocks are always aligned to the required number of bytes. */
111  #if portBYTE_ALIGNMENT != 1
112  if( xWantedSize & portBYTE_ALIGNMENT_MASK )
113  {
114  /* Byte alignment required. */
115  xWantedSize += ( portBYTE_ALIGNMENT - ( xWantedSize & portBYTE_ALIGNMENT_MASK ) );
116  }
117  #endif
118 
119  vTaskSuspendAll();
120  {
121  /* Check there is enough room left for the allocation. */
122  if( ( ( xNextFreeByte + xWantedSize ) < currentTOTAL_HEAP_SIZE ) &&
123  ( ( xNextFreeByte + xWantedSize ) > xNextFreeByte ) )/* Check for overflow. */
124  {
125  /* Return the next free byte then increment the index past this
126  block. */
127  pvReturn = &( xHeap.ucHeap[ xNextFreeByte ] );
128  xNextFreeByte += xWantedSize;
129  }
130  }
131  xTaskResumeAll();
132 
133  #if( configUSE_MALLOC_FAILED_HOOK == 1 )
134  {
135  if( pvReturn == NULL )
136  {
137  extern void vApplicationMallocFailedHook( void );
138  vApplicationMallocFailedHook();
139  }
140  }
141  #endif
142 
143  return pvReturn;
144 }

173 {
174  vTaskSuspendAll();
175  currentTOTAL_HEAP_SIZE = configTOTAL_HEAP_SIZE + bytes;
