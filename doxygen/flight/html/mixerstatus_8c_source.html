48 int32_t MixerStatusInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(MIXERSTATUS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(MIXERSTATUS_OBJID,
56  MIXERSTATUS_ISSINGLEINST, MIXERSTATUS_ISSETTINGS, MIXERSTATUS_NUMBYTES, &MixerStatusSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
