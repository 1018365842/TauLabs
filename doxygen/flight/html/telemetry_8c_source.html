151 static void registerObject(UAVObjHandle obj)
152 {
153  if (UAVObjIsMetaobject(obj)) {
154  /* Only connect change notifications for meta objects. No periodic updates */
155  UAVObjConnectQueue(obj, priorityQueue, EV_MASK_ALL_UPDATES);
156  return;
157  } else {
158  UAVObjMetadata metadata;
159  UAVObjUpdateMode updateMode;
160  UAVObjGetMetadata(obj, &metadata);
161  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
162 
163  /* Only create a periodic event for objects that are periodic */
164  if ((updateMode == UPDATEMODE_PERIODIC) ||
165  (updateMode == UPDATEMODE_THROTTLED)) {
166  // Setup object for periodic updates
167  UAVObjEvent ev = {
168  .obj = obj,
169  .instId = UAVOBJ_ALL_INSTANCES,
170  .event = EV_UPDATED_PERIODIC,
171  };
172  EventPeriodicQueueCreate(&ev, queue, 0);
173  }
174 
175  // Setup object for telemetry updates
176  updateObject(obj, EV_NONE);
177  }
178 }

184 static void updateObject(UAVObjHandle obj, int32_t eventType)
185 {
186  UAVObjMetadata metadata;
187  UAVObjUpdateMode updateMode;
188  int32_t eventMask;
189 
190  if (UAVObjIsMetaobject(obj)) {
191  /* This function updates the periodic updates for the object.
192  * Meta Objects cannot have periodic updates.
193  */
194  PIOS_Assert(false);
195  return;
196  }
197 
198  // Get metadata
199  UAVObjGetMetadata(obj, &metadata);
200  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
201 
202  // Setup object depending on update mode
203  switch (updateMode) {
204  case UPDATEMODE_PERIODIC:
205  // Set update period
206  setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
207  // Connect queue
208  eventMask = EV_UPDATED_PERIODIC | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
209  UAVObjConnectQueue(obj, priorityQueue, eventMask);
210  break;
211  case UPDATEMODE_ONCHANGE:
212  // Set update period
213  setUpdatePeriod(obj, 0);
214  // Connect queue
215  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
216  UAVObjConnectQueue(obj, priorityQueue, eventMask);
217  break;
218  case UPDATEMODE_THROTTLED:
219  if ((eventType == EV_UPDATED_PERIODIC) || (eventType == EV_NONE)) {
220  // If we received a periodic update, we can change back to update on change
221  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
222  // Set update period on initialization and metadata change
223  if (eventType == EV_NONE)
224  setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
225  } else {
226  eventMask = getEventMask(obj, priorityQueue);
227  if (eventMask & EV_UPDATED_PERIODIC) {
228  // If periodic flag is already set then we have previously sent an update during
229  // the timeout period and this update would be missed, so set the dirty flag.
230  // Once setting EV_UPDATED_THROTTLED_DIRTY, there is no need to listen for EV_UPDATED.
231  eventMask = EV_UPDATED_PERIODIC | EV_UPDATE_REQ | EV_UPDATED_THROTTLED_DIRTY;
232  } else { //If periodic is not set then we just received an object
233  // update so switch to periodic for the timeout period to prevent
234  // sending more updates. Listen to the EV_UPDATED flag still to
235  // catch any updates that would overwise never get sent.
236  eventMask = EV_UPDATED_PERIODIC | EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
237  }
238  }
239  UAVObjConnectQueue(obj, priorityQueue, eventMask);
240  break;
241  case UPDATEMODE_MANUAL:
242  // Set update period
243  setUpdatePeriod(obj, 0);
244  // Connect queue
245  eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ;
246  UAVObjConnectQueue(obj, priorityQueue, eventMask);
247  break;
248  }
249 }

254 static void processObjEvent(UAVObjEvent * ev)
255 {
256  UAVObjMetadata metadata;
257  UAVObjUpdateMode updateMode;
258  FlightTelemetryStatsData flightStats;
259  int32_t retries;
260  int32_t success;
261 
262  if (ev->obj == 0) {
263  updateTelemetryStats();
264  } else if (ev->obj == GCSTelemetryStatsHandle()) {
265  gcsTelemetryStatsUpdated();
266  } else {
267  FlightTelemetryStatsGet(&flightStats);
268  // Get object metadata
269  UAVObjGetMetadata(ev->obj, &metadata);
270  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
271 
272  // Act on event
273  retries = 0;
274  success = -1;
275  if (ev->event == EV_UPDATED || ev->event == EV_UPDATED_MANUAL || ((ev->event == EV_UPDATED_PERIODIC) && (updateMode != UPDATEMODE_THROTTLED))) {
276  // Send update to GCS (with retries)
277  while (retries < MAX_RETRIES && success == -1) {
278  success = UAVTalkSendObject(uavTalkCon, ev->obj, ev->instId, UAVObjGetTelemetryAcked(&metadata), REQ_TIMEOUT_MS); // call blocks until ack is received or timeout
279  ++retries;
280  }
281  // Update stats
282  txRetries += (retries - 1);
283  if (success == -1) {
284  ++txErrors;
285  }
286  } else if (ev->event == EV_UPDATE_REQ) {
287  // Request object update from GCS (with retries)
288  while (retries < MAX_RETRIES && success == -1) {
289  success = UAVTalkSendObjectRequest(uavTalkCon, ev->obj, ev->instId, REQ_TIMEOUT_MS); // call blocks until update is received or timeout
290  ++retries;
291  }
292  // Update stats
293  txRetries += (retries - 1);
294  if (success == -1) {
295  ++txErrors;
296  }
297  } else if (ev->event == EV_UPDATED_PERIODIC && updateMode == UPDATEMODE_THROTTLED) {
298  // Get the event mask
299  int32_t eventMask = getEventMask(ev->obj, priorityQueue);
300 
301  if (eventMask & EV_UPDATED_THROTTLED_DIRTY) { // If EV_UPDATED_THROTTLED_DIRTY flag is set then send the data like normal.
302  // Send update to GCS (with retries)
303  while (retries < MAX_RETRIES && success == -1) {
304  success = UAVTalkSendObject(uavTalkCon, ev->obj, ev->instId, UAVObjGetTelemetryAcked(&metadata), REQ_TIMEOUT_MS); // call blocks until ack is received or timeout
305  ++retries;
306  }
307  // Update stats
308  txRetries += (retries - 1);
309  if (success == -1) {
310  ++txErrors;
311  }
312  }
313  }
314  // If this is a metaobject then make necessary telemetry updates
315  if (UAVObjIsMetaobject(ev->obj)) {
316  updateObject(UAVObjGetLinkedObj(ev->obj), EV_NONE); // linked object will be the actual object the metadata are for
317  } else {
318  if (updateMode == UPDATEMODE_THROTTLED) {
319  // If this is UPDATEMODE_THROTTLED, the event mask changes on every event.
320  updateObject(ev->obj, ev->event);
321  }
322  }
323  }
324 }

365 static void telemetryRxTask(void *parameters)
366 {
367 
368  // Task loop
369  while (1) {
370  uint32_t inputPort = getComPort();
371 
372  if (inputPort) {
373  // Block until data are available
374  uint8_t serial_data[1];
375  uint16_t bytes_to_process;
376 
377  bytes_to_process = PIOS_COM_ReceiveBuffer(inputPort, serial_data, sizeof(serial_data), 500);
378  if (bytes_to_process > 0) {
379  for (uint8_t i = 0; i < bytes_to_process; i++) {
380  UAVTalkProcessInputStream(uavTalkCon,serial_data[i]);
381  }
382  }
383  } else {
384  vTaskDelay(5);
385  }
386  }
387 }

443 static void updateTelemetryStats()
444 {
445  UAVTalkStats utalkStats;
446  FlightTelemetryStatsData flightStats;
447  GCSTelemetryStatsData gcsStats;
448  uint8_t forceUpdate;
449  uint8_t connectionTimeout;
450  uint32_t timeNow;
451 
452  // Get stats
453  UAVTalkGetStats(uavTalkCon, &utalkStats);
454  UAVTalkResetStats(uavTalkCon);
455 
456  // Get object data
457  FlightTelemetryStatsGet(&flightStats);
458  GCSTelemetryStatsGet(&gcsStats);
459 
460  // Update stats object
461  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
462  flightStats.RxDataRate = (float)utalkStats.rxBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0);
463  flightStats.TxDataRate = (float)utalkStats.txBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0);
464  flightStats.RxFailures += utalkStats.rxErrors;
465  flightStats.TxFailures += txErrors;
466  flightStats.TxRetries += txRetries;
467  txErrors = 0;
468  txRetries = 0;
469  } else {
470  flightStats.RxDataRate = 0;
471  flightStats.TxDataRate = 0;
472  flightStats.RxFailures = 0;
473  flightStats.TxFailures = 0;
474  flightStats.TxRetries = 0;
475  txErrors = 0;
476  txRetries = 0;
477  }
478 
479  // Check for connection timeout
480  timeNow = xTaskGetTickCount() * portTICK_RATE_MS;
481  if (utalkStats.rxObjects > 0) {
482  timeOfLastObjectUpdate = timeNow;
483  }
484  if ((timeNow - timeOfLastObjectUpdate) > CONNECTION_TIMEOUT_MS) {
485  connectionTimeout = 1;
486  } else {
487  connectionTimeout = 0;
488  }
489 
490  // Update connection state
491  forceUpdate = 1;
492  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED) {
493  // Wait for connection request
494  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_HANDSHAKEREQ) {
495  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK;
496  }
497  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK) {
498  // Wait for connection
499  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_CONNECTED) {
500  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_CONNECTED;
501  } else if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_DISCONNECTED) {
502  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
503  }
504  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
505  if (gcsStats.Status != GCSTELEMETRYSTATS_STATUS_CONNECTED || connectionTimeout) {
506  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
507  } else {
508  forceUpdate = 0;
509  }
510  } else {
511  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
512  }
513 
514  // Update the telemetry alarm
515  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
516  AlarmsClear(SYSTEMALARMS_ALARM_TELEMETRY);
517  } else {
518  AlarmsSet(SYSTEMALARMS_ALARM_TELEMETRY, SYSTEMALARMS_ALARM_ERROR);
519  }
520 
521  // Update object
522  FlightTelemetryStatsSet(&flightStats);
523 
524  // Force telemetry update if not connected
525  if (forceUpdate) {
526  FlightTelemetryStatsUpdated();
527  }
528 }
