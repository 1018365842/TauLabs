182 {
183  uint32_t index = 0;
184  uint32_t Temp = 0;
185 
186  for (index = 0; index < NbrOfBytes; index++)
187  {
188  Temp |= buffer[idx + index] << (index * 8);
189  }
190 
191  if (BytesFormat == BigEndian)
192  {
193  Temp = __REV(Temp);
194  }
195  return Temp;

217 static ErrorCode WavePlayer_WaveParsing(uint8_t *DirName, uint8_t *FileName, uint32_t *FileLen)
218 {
219  uint32_t Temp = 0x00;
220  uint32_t ExtraFormatBytes = 0;
221  __IO uint32_t err = 0;
222  uint32_t number_of_clusters;
223  uint32_t i;
224 
225  /* Directory enumeration test */
226  if (PIOS_FOPEN_READ(FileName, file))
227  {
228  err = 1;
229  }
230  else
231  {
232  *FileLen = file.filelen;
233  number_of_clusters = file.filelen / 512;
234  if ((file.filelen % SECTOR_SIZE) > 0)
235  {
236  number_of_clusters ++;
237  }
238  }
239  PIOS_FREAD(&file, buffer1, 44, &i);
240  //DFS_ReadFile(&file, sector, buffer1, &i, SECTOR_SIZE);
241 
242  /* Read chunkID, must be 'RIFF' ----------------------------------------------*/
243  Temp = ReadUnit(buffer1, 0, 4, BigEndian);
244  if (Temp != CHUNK_ID)
245  {
246  return(Unvalid_RIFF_ID);
247  }
248 
249  /* Read the file length ----------------------------------------------------*/
250  WAVE_Format.RIFFchunksize = ReadUnit(buffer1, 4, 4, LittleEndian);
251 
252  /* Read the file format, must be 'WAVE' ------------------------------------*/
253  Temp = ReadUnit(buffer1, 8, 4, BigEndian);
254  if (Temp != FILE_FORMAT)
255  {
256  return(Unvalid_WAVE_Format);
257  }
258 
259  /* Read the format chunk, must be'fmt ' --------------------------------------*/
260  Temp = ReadUnit(buffer1, 12, 4, BigEndian);
261  if (Temp != FORMAT_ID)
262  {
263  return(Unvalid_FormatChunk_ID);
264  }
265  /* Read the length of the 'fmt' data, must be 0x10 -------------------------*/
266  Temp = ReadUnit(buffer1, 16, 4, LittleEndian);
267  if (Temp != 0x10)
268  {
269  ExtraFormatBytes = 1;
270  }
271  /* Read the audio format, must be 0x01 (PCM) -------------------------------*/
272  WAVE_Format.FormatTag = ReadUnit(buffer1, 20, 2, LittleEndian);
273  if (WAVE_Format.FormatTag != WAVE_FORMAT_PCM)
274  {
275  return(Unsupporetd_FormatTag);
276  }
277 
278  /* Read the number of channels, must be 0x01 (Mono) ------------------------*/
279  WAVE_Format.NumChannels = ReadUnit(buffer1, 22, 2, LittleEndian);
280  if (WAVE_Format.NumChannels != CHANNEL_MONO)
281  {
282  return(Unsupporetd_Number_Of_Channel);
283  }
284 
285  /* Read the Sample Rate ----------------------------------------------------*/
286  WAVE_Format.SampleRate = ReadUnit(buffer1, 24, 4, LittleEndian);
287  /* Update the OCA value according to the .WAV file Sample Rate */
288  switch (WAVE_Format.SampleRate)
289  {
290  case SAMPLE_RATE_8000 :
291  TIM6ARRValue = (PIOS_PERIPHERAL_APB1_CLOCK)/8000;
292  break; /* 8KHz = 24MHz / 3000 */
293  case SAMPLE_RATE_11025:
294  TIM6ARRValue = (PIOS_PERIPHERAL_APB1_CLOCK)/11025;
295  break; /* 11.025KHz = 24MHz / 2176 */
296  case SAMPLE_RATE_22050:
297  TIM6ARRValue = (PIOS_PERIPHERAL_APB1_CLOCK)/22050;
298  break; /* 22.05KHz = 24MHz / 1088 */
299  case SAMPLE_RATE_44100:
300  TIM6ARRValue = (PIOS_PERIPHERAL_APB1_CLOCK)/44100;
301  break; /* 44.1KHz = 24MHz / 544 */
302  default:
303  return(Unsupporetd_Sample_Rate);
304  }
305 
306  /* Read the Byte Rate ------------------------------------------------------*/
307  WAVE_Format.ByteRate = ReadUnit(buffer1, 28, 4, LittleEndian);
308 
309  /* Read the block alignment ------------------------------------------------*/
310  WAVE_Format.BlockAlign = ReadUnit(buffer1, 32, 2, LittleEndian);
311 
312  /* Read the number of bits per sample --------------------------------------*/
313  WAVE_Format.BitsPerSample = ReadUnit(buffer1, 34, 2, LittleEndian);
314  if (WAVE_Format.BitsPerSample != BITS_PER_SAMPLE_8)
315  {
316  return(Unsupporetd_Bits_Per_Sample);
317  }
318  SpeechDataOffset = 36;
319  /* If there is Extra format bytes, these bytes will be defined in "Fact Chunk" */
320  if (ExtraFormatBytes == 1)
321  {
322  /* Read th Extra format bytes, must be 0x00 ------------------------------*/
323  Temp = ReadUnit(buffer1, 36, 2, LittleEndian);
324  if (Temp != 0x00)
325  {
326  return(Unsupporetd_ExtraFormatBytes);
327  }
328  /* Read the Fact chunk, must be 'fact' -----------------------------------*/
329  Temp = ReadUnit(buffer1, 38, 4, BigEndian);
330  if (Temp != FACT_ID)
331  {
332  return(Unvalid_FactChunk_ID);
333  }
334  /* Read Fact chunk data Size ---------------------------------------------*/
335  Temp = ReadUnit(buffer1, 42, 4, LittleEndian);
336 
337  SpeechDataOffset += 10 + Temp;
338  }
339  /* Read the Data chunk, must be 'data' ---------------------------------------*/
340  Temp = ReadUnit(buffer1, SpeechDataOffset, 4, BigEndian);
341  SpeechDataOffset += 4;
342  if (Temp != DATA_ID)
343  {
344  return(Unvalid_DataChunk_ID);
345  }
346 
347  /* Read the number of sample data ------------------------------------------*/
348  WAVE_Format.DataSize = ReadUnit(buffer1, SpeechDataOffset, 4, LittleEndian);
349  SpeechDataOffset += 4;
350  wavecounter = SpeechDataOffset;
351  PIOS_FREAD(&file, buffer1, SECTOR_SIZE, &i);
352  PIOS_FREAD(&file, buffer2, SECTOR_SIZE, &i);
353  return(Valid_WAVE_File);
354 }
