162 int32_t PIOS_USART_Init(uint32_t * usart_id, const struct pios_usart_cfg * cfg)
163 {
164  PIOS_DEBUG_Assert(usart_id);
165  PIOS_DEBUG_Assert(cfg);
166 
167  struct pios_usart_dev * usart_dev;
168 
169  usart_dev = (struct pios_usart_dev *) PIOS_USART_alloc();
170  if (!usart_dev) goto out_fail;
171 
172  /* Bind the configuration to the device instance */
173  usart_dev->cfg = cfg;
174 
175  /* Map pins to USART function */
176  /* note __builtin_ctz() due to the difference between GPIO_PinX and GPIO_PinSourceX */
177  if (usart_dev->cfg->remap) {
178  if (usart_dev->cfg->rx.gpio != 0)
179  GPIO_PinAFConfig(usart_dev->cfg->rx.gpio,
180  __builtin_ctz(usart_dev->cfg->rx.init.GPIO_Pin),
181  usart_dev->cfg->remap);
182  if (usart_dev->cfg->tx.gpio != 0)
183  GPIO_PinAFConfig(usart_dev->cfg->tx.gpio,
184  __builtin_ctz(usart_dev->cfg->tx.init.GPIO_Pin),
185  usart_dev->cfg->remap);
186  }
187 
188  /* Initialize the USART Rx and Tx pins */
189  if (usart_dev->cfg->rx.gpio != 0)
190  GPIO_Init(usart_dev->cfg->rx.gpio, (GPIO_InitTypeDef *)&usart_dev->cfg->rx.init);
191  if (usart_dev->cfg->tx.gpio != 0)
192  GPIO_Init(usart_dev->cfg->tx.gpio, (GPIO_InitTypeDef *)&usart_dev->cfg->tx.init);
193 
194  /* Configure the USART */
195  USART_Init(usart_dev->cfg->regs, (USART_InitTypeDef *)&usart_dev->cfg->init);
196 
197  *usart_id = (uint32_t)usart_dev;
198 
199  /* Configure USART Interrupts */
200  switch ((uint32_t)usart_dev->cfg->regs) {
201  case (uint32_t)USART1:
202  PIOS_USART_1_id = (uint32_t)usart_dev;
203  break;
204  case (uint32_t)USART2:
205  PIOS_USART_2_id = (uint32_t)usart_dev;
206  break;
207  case (uint32_t)USART3:
208  PIOS_USART_3_id = (uint32_t)usart_dev;
209  break;
210  case (uint32_t)UART4:
211  PIOS_USART_4_id = (uint32_t)usart_dev;
212  break;
213  case (uint32_t)UART5:
214  PIOS_USART_5_id = (uint32_t)usart_dev;
215  break;
216  case (uint32_t)USART6:
217  PIOS_USART_6_id = (uint32_t)usart_dev;
218  break;
219  }
220  NVIC_Init((NVIC_InitTypeDef *)&(usart_dev->cfg->irq.init));
221  USART_ITConfig(usart_dev->cfg->regs, USART_IT_RXNE, ENABLE);
222  USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, ENABLE);
223 
224  // FIXME XXX Clear / reset uart here - sends NUL char else
225 
226  /* Enable USART */
227  USART_Cmd(usart_dev->cfg->regs, ENABLE);
228 
229  return(0);
230 
231 out_fail:
232  return(-1);
233 }

308 static void PIOS_USART_generic_irq_handler(uint32_t usart_id)
309 {
310  struct pios_usart_dev * usart_dev = (struct pios_usart_dev *)usart_id;
311 
312  bool valid = PIOS_USART_validate(usart_dev);
313  PIOS_Assert(valid);
314 
315  /* Force read of dr after sr to make sure to clear error flags */
316  volatile uint16_t sr = usart_dev->cfg->regs->SR;
317  volatile uint8_t dr = usart_dev->cfg->regs->DR;
318 
319  /* Check if RXNE flag is set */
320  bool rx_need_yield = false;
321  if (sr & USART_SR_RXNE) {
322  uint8_t byte = dr;
323  if (usart_dev->rx_in_cb) {
324  (void) (usart_dev->rx_in_cb)(usart_dev->rx_in_context, &byte, 1, NULL, &rx_need_yield);
325  }
326  }
327 
328  /* Check if TXE flag is set */
329  bool tx_need_yield = false;
330  if (sr & USART_SR_TXE) {
331  if (usart_dev->tx_out_cb) {
332  uint8_t b;
333  uint16_t bytes_to_send;
334 
335  bytes_to_send = (usart_dev->tx_out_cb)(usart_dev->tx_out_context, &b, 1, NULL, &tx_need_yield);
336 
337  if (bytes_to_send > 0) {
338  /* Send the byte we've been given */
339  usart_dev->cfg->regs->DR = b;
340  } else {
341  /* No bytes to send, disable TXE interrupt */
342  USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, DISABLE);
343  }
344  } else {
345  /* No bytes to send, disable TXE interrupt */
346  USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, DISABLE);
347  }
348  }
349 
350 #if defined(PIOS_INCLUDE_FREERTOS)
351  portEND_SWITCHING_ISR(rx_need_yield || tx_need_yield);
352 #endif /* PIOS_INCLUDE_FREERTOS */
353 }
