97 static void flightPlanTask(void *parameters)
98 {
99  UAVObjEvent ev;
100  PmReturn_t retval;
101  FlightPlanStatusData status;
102  FlightPlanControlData control;
103 
104  // Setup status object
105  status.Status = FLIGHTPLANSTATUS_STATUS_STOPPED;
106  status.ErrorFileID = 0;
107  status.ErrorLineNum = 0;
108  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_NONE;
109  status.Debug[0] = 0.0;
110  status.Debug[1] = 0.0;
111  FlightPlanStatusSet(&status);
112 
113  // Main thread loop
114  while (1)
115  {
116  // Wait for FlightPlanControl updates
117  while (xQueueReceive(queue, &ev, portMAX_DELAY) != pdTRUE) ;
118 
119  // Get object and check if a start command was sent
120  FlightPlanControlGet(&control);
121  if ( control.Command == FLIGHTPLANCONTROL_COMMAND_START )
122  {
123  // Init PyMite
124  retval = pm_init(MEMSPACE_PROG, usrlib_img);
125  if (retval == PM_RET_OK)
126  {
127  // Update status
128  FlightPlanStatusGet(&status);
129  status.Status = FLIGHTPLANSTATUS_STATUS_RUNNING;
130  FlightPlanStatusSet(&status);
131  // Run the test script (TODO: load from SD card)
132  retval = pm_run((uint8_t *)"test");
133  // Check if an error or exception was thrown
134  if (retval == PM_RET_OK || retval == PM_RET_EX_EXIT)
135  {
136  status.Status = FLIGHTPLANSTATUS_STATUS_STOPPED;
137  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_NONE;
138  }
139  else if (retval == PM_RET_EX)
140  {
141  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
142  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_EXCEPTION;
143  }
144  else if (retval == PM_RET_EX_IO)
145  {
146  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
147  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_IOERROR;
148  }
149  else if (retval == PM_RET_EX_ZDIV)
150  {
151  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
152  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_DIVBYZERO;
153  }
154  else if (retval == PM_RET_EX_ASSRT)
155  {
156  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
157  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_ASSERTERROR;
158  }
159  else if (retval == PM_RET_EX_ATTR)
160  {
161  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
162  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_ATTRIBUTEERROR;
163  }
164  else if (retval == PM_RET_EX_IMPRT)
165  {
166  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
167  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_IMPORTERROR;
168  }
169  else if (retval == PM_RET_EX_INDX)
170  {
171  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
172  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_INDEXERROR;
173  }
174  else if (retval == PM_RET_EX_KEY)
175  {
176  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
177  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_KEYERROR;
178  }
179  else if (retval == PM_RET_EX_MEM)
180  {
181  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
182  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_MEMORYERROR;
183  }
184  else if (retval == PM_RET_EX_NAME)
185  {
186  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
187  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_NAMEERROR;
188  }
189  else if (retval == PM_RET_EX_SYNTAX)
190  {
191  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
192  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_SYNTAXERROR;
193  }
194  else if (retval == PM_RET_EX_SYS)
195  {
196  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
197  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_SYSTEMERROR;
198  }
199  else if (retval == PM_RET_EX_TYPE)
200  {
201  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
202  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_TYPEERROR;
203  }
204  else if (retval == PM_RET_EX_VAL)
205  {
206  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
207  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_VALUEERROR;
208  }
209  else if (retval == PM_RET_EX_STOP)
210  {
211  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
212  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_STOPITERATION;
213  }
214  else if (retval == PM_RET_EX_WARN)
215  {
216  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
217  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_WARNING;
218  }
219  else
220  {
221  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
222  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_UNKNOWNERROR;
223  }
224  // Get file ID and line number of error (if one)
225  status.ErrorFileID = gVmGlobal.errFileId;
226  status.ErrorLineNum = gVmGlobal.errLineNum;
227  }
228  else
229  {
230  status.Status = FLIGHTPLANSTATUS_STATUS_ERROR;
231  status.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_VMINITERROR;
232  }
233 
234  // Update status object
235  FlightPlanStatusSet(&status);
236  }
237  }

244 static void objectUpdatedCb(UAVObjEvent * ev)
245 {
246  FlightPlanControlData controlData;
247  FlightPlanStatusData statusData;
248 
249  // If the object updated was the FlightPlanControl execute requested action
250  if ( ev->obj == FlightPlanControlHandle() )
251  {
252  // Get data
253  FlightPlanControlGet(&controlData);
254  // Execute command
255  if ( controlData.Command == FLIGHTPLANCONTROL_COMMAND_START )
256  {
257  // Start VM task if not running already
258  if ( taskHandle == NULL )
259  {
260  xTaskCreate(flightPlanTask, (signed char *)"FlightPlan", STACK_SIZE_BYTES/4, NULL, TASK_PRIORITY, &taskHandle);
261  TaskMonitorAdd(TASKINFO_RUNNING_FLIGHTPLAN, taskHandle);
262  }
263  }
264  else if ( controlData.Command == FLIGHTPLANCONTROL_COMMAND_KILL )
265  {
266  // Force kill VM task if it is already running
267  // (NOTE: the STOP command is preferred as it allows the script to terminate without killing the VM)
268  if ( taskHandle != NULL )
269  {
270  // Kill VM
271  TaskMonitorRemove(TASKINFO_RUNNING_FLIGHTPLAN);
272  vTaskDelete(taskHandle);
273  taskHandle = NULL;
274  // Update status object
275  statusData.Status = FLIGHTPLANSTATUS_STATUS_STOPPED;
276  statusData.ErrorFileID = 0;
277  statusData.ErrorLineNum = 0;
278  statusData.ErrorType = FLIGHTPLANSTATUS_ERRORTYPE_NONE;
279  statusData.Debug[0] = 0.0;
280  statusData.Debug[1] = 0.0;
281  FlightPlanStatusSet(&statusData);
282  }
283  }
284  }
