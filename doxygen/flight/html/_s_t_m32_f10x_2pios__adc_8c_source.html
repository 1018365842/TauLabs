100 int32_t PIOS_ADC_Init(const struct pios_adc_cfg * cfg)
101 {
102  pios_adc_dev = PIOS_ADC_Allocate();
103  if (pios_adc_dev == NULL)
104  return -1;
105 
106  pios_adc_dev->cfg = cfg;
107  pios_adc_dev->callback_function = NULL;

115  GPIO_StructInit(&GPIO_InitStructure);
116  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
117  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AIN;

127  return 0;

134 void PIOS_ADC_Config(uint32_t oversampling)
135 {
136  pios_adc_dev->adc_oversample = (oversampling > PIOS_ADC_MAX_OVERSAMPLING) ? PIOS_ADC_MAX_OVERSAMPLING : oversampling;
137 
138  ADC_DeInit(ADC1);
139  ADC_DeInit(ADC2);
140 
141  /* Disable interrupts */
142  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->dma.irq.flags, DISABLE);
143 
144  /* Enable ADC clocks */
145  PIOS_ADC_CLOCK_FUNCTION;
146 
147  /* Map channels to conversion slots depending on the channel selection mask */
148  for (int32_t i = 0; i < PIOS_ADC_NUM_PINS; i++) {
149  ADC_RegularChannelConfig(ADC_MAPPING[i], ADC_CHANNEL[i],
150  ADC_CHANNEL_MAPPING[i],
151  PIOS_ADC_SAMPLE_TIME);
152  }
153 
154 #if (PIOS_ADC_USE_TEMP_SENSOR)
155  ADC_TempSensorVrefintCmd(ENABLE);
156  ADC_RegularChannelConfig(PIOS_ADC_TEMP_SENSOR_ADC, ADC_Channel_16,
157  PIOS_ADC_TEMP_SENSOR_ADC_CHANNEL,
158  PIOS_ADC_SAMPLE_TIME);
159 #endif
160  // return
161  /* Configure ADCs */
162  ADC_InitTypeDef ADC_InitStructure;
163  ADC_StructInit(&ADC_InitStructure);
164  ADC_InitStructure.ADC_Mode = ADC_Mode_RegSimult;
165  ADC_InitStructure.ADC_ScanConvMode = ENABLE;
166  ADC_InitStructure.ADC_ContinuousConvMode = ENABLE;
167  ADC_InitStructure.ADC_ExternalTrigConv = ADC_ExternalTrigConv_None;
168  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Right;
169  ADC_InitStructure.ADC_NbrOfChannel = ((PIOS_ADC_NUM_CHANNELS + 1) >> 1);
170  ADC_Init(ADC1, &ADC_InitStructure);
171 
172 #if (PIOS_ADC_USE_ADC2)
173  ADC_Init(ADC2, &ADC_InitStructure);
174 
175  /* Enable ADC2 external trigger conversion (to synch with ADC1) */
176  ADC_ExternalTrigConvCmd(ADC2, ENABLE);
177 #endif
178 
179  RCC_ADCCLKConfig(PIOS_ADC_ADCCLK);
180 
181  /* Enable ADC1->DMA request */
182  ADC_DMACmd(ADC1, ENABLE);
183 
184  /* ADC1 calibration */
185  ADC_Cmd(ADC1, ENABLE);
186  ADC_ResetCalibration(ADC1);
187  while (ADC_GetResetCalibrationStatus(ADC1)) ;
188  ADC_StartCalibration(ADC1);
189  while (ADC_GetCalibrationStatus(ADC1)) ;
190 
191 #if (PIOS_ADC_USE_ADC2)
192  /* ADC2 calibration */
193  ADC_Cmd(ADC2, ENABLE);
194  ADC_ResetCalibration(ADC2);
195  while (ADC_GetResetCalibrationStatus(ADC2)) ;
196  ADC_StartCalibration(ADC2);
197  while (ADC_GetCalibrationStatus(ADC2)) ;
198 #endif
199 
200  /* This makes sure we have an even number of transfers if using ADC2 */
201  pios_adc_dev->dma_block_size = ((PIOS_ADC_NUM_CHANNELS + PIOS_ADC_USE_ADC2) >> PIOS_ADC_USE_ADC2) << PIOS_ADC_USE_ADC2;
202  pios_adc_dev->dma_half_buffer_size = pios_adc_dev->dma_block_size * pios_adc_dev->adc_oversample;
203 
204  /* Configure DMA channel */
205  DMA_InitTypeDef dma_init = pios_adc_dev->cfg->dma.rx.init;
206  dma_init.DMA_MemoryBaseAddr = (uint32_t) &pios_adc_dev->raw_data_buffer[0];
207  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
208  dma_init.DMA_BufferSize = pios_adc_dev->dma_half_buffer_size; /* x2 for double buffer /2 for 32-bit xfr */
209  DMA_Init(pios_adc_dev->cfg->dma.rx.channel, &dma_init);
210  DMA_Cmd(pios_adc_dev->cfg->dma.rx.channel, ENABLE);
211 
212  /* Trigger interrupt when for half conversions too to indicate double buffer */
213  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, DMA_IT_TC, ENABLE);
214  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, DMA_IT_HT, ENABLE);
215 
216  /* Configure DMA interrupt */
217  NVIC_Init(&pios_adc_dev->cfg->dma.irq.init);
218 
219  /* Finally start initial conversion */
220  ADC_SoftwareStartConvCmd(ADC1, ENABLE);
221 
222  /* Use simple averaging filter for now */
223  for (int32_t i = 0; i < pios_adc_dev->adc_oversample; i++)
224  pios_adc_dev->fir_coeffs[i] = 1;
225  pios_adc_dev->fir_coeffs[pios_adc_dev->adc_oversample] = pios_adc_dev->adc_oversample;
226 
227  /* Enable DMA1 clock */
228  RCC_AHBPeriphClockCmd(pios_adc_dev->cfg->dma.ahb_clk, ENABLE);
229 }

309 void PIOS_ADC_downsample_data()
310 {
311  uint16_t chan;
312  uint16_t sample;
313  float * downsampled_buffer = &pios_adc_dev->downsampled_buffer[0];
314 
315  for (chan = 0; chan < PIOS_ADC_NUM_CHANNELS; chan++) {
316  int32_t sum = 0;
317  for (sample = 0; sample < pios_adc_dev->adc_oversample; sample++) {
318  sum += pios_adc_dev->valid_data_buffer[chan + sample * pios_adc_dev->dma_block_size] * pios_adc_dev->fir_coeffs[sample];
319  }
320  downsampled_buffer[chan] = (float) sum / pios_adc_dev->fir_coeffs[pios_adc_dev->adc_oversample];
321  }
322 
323 #if defined(PIOS_INCLUDE_FREERTOS)
324  if(pios_adc_dev->data_queue) {
325  static portBASE_TYPE xHigherPriorityTaskWoken;
326  xQueueSendFromISR(pios_adc_dev->data_queue, pios_adc_dev->downsampled_buffer, &xHigherPriorityTaskWoken);
327  portEND_SWITCHING_ISR(xHigherPriorityTaskWoken);
328  }
329 #endif
330  if(pios_adc_dev->callback_function)
331  pios_adc_dev->callback_function(pios_adc_dev->downsampled_buffer);
332 }

343 void PIOS_ADC_DMA_Handler(void)
344 {
345  if (!PIOS_ADC_validate(pios_adc_dev))
346  return;
347 
348  if (DMA_GetFlagStatus(pios_adc_dev->cfg->full_flag /*DMA1_IT_TC1*/)) { // whole double buffer filled
349  pios_adc_dev->valid_data_buffer = &pios_adc_dev->raw_data_buffer[pios_adc_dev->dma_half_buffer_size];
350  DMA_ClearFlag(pios_adc_dev->cfg->full_flag);
351  PIOS_ADC_downsample_data();
352  }
353  else if (DMA_GetFlagStatus(pios_adc_dev->cfg->half_flag /*DMA1_IT_HT1*/)) {
354  pios_adc_dev->valid_data_buffer = &pios_adc_dev->raw_data_buffer[0];
355  DMA_ClearFlag(pios_adc_dev->cfg->half_flag);
356  PIOS_ADC_downsample_data();
357  }
358  else {
359  // This should not happen, probably due to transfer errors
360  DMA_ClearFlag(pios_adc_dev->cfg->dma.irq.flags /*DMA1_FLAG_GL1*/);
361  }
362 }
