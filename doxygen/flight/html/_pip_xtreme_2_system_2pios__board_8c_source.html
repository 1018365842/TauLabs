78 
79  /* Set up the SPI interface to the rfm22b */
80  if (PIOS_SPI_Init(&pios_spi_rfm22b_id, &pios_spi_rfm22b_cfg)) {
81  PIOS_DEBUG_Assert(0);
82  }
83 
84 #ifdef PIOS_INCLUDE_WDG
85  /* Initialize watchdog as early as possible to catch faults during init */
86  PIOS_WDG_Init();
87 #endif /* PIOS_INCLUDE_WDG */
88 
89 #if defined(PIOS_INCLUDE_RTC)
90  /* Initialize the real-time clock and its associated tick */
91  PIOS_RTC_Init(&pios_rtc_main_cfg);
92 #endif /* PIOS_INCLUDE_RTC */
93 
94  OPLinkSettingsInitialize();
95 
96 #if defined(PIOS_INCLUDE_LED)
97  PIOS_LED_Init(&pios_led_cfg);
98 #endif /* PIOS_INCLUDE_LED */
99 
100  OPLinkSettingsData oplinkSettings;
101 #if defined(PIOS_INCLUDE_FLASH_EEPROM)
102  PIOS_EEPROM_Init(&pios_eeprom_cfg);
103 
104  /* Read the settings from flash. */
105  /* NOTE: We probably need to save/restore the objID here incase the object changed but the size doesn't */
106  if (PIOS_EEPROM_Load((uint8_t*)&oplinkSettings, sizeof(OPLinkSettingsData)) == 0)
107  OPLinkSettingsSet(&oplinkSettings);
108  else
109  OPLinkSettingsGet(&oplinkSettings);
110 #else
111  OPLinkSettingsGet(&oplinkSettings);
112 #endif /* PIOS_INCLUDE_FLASH_EEPROM */
113 
114  /* Initialize the task monitor library */
115  TaskMonitorInitialize();
116 
117 #if defined(PIOS_INCLUDE_TIM)
118  /* Set up pulse timers */
119  PIOS_TIM_InitClock(&tim_1_cfg);
120  PIOS_TIM_InitClock(&tim_2_cfg);
121  PIOS_TIM_InitClock(&tim_3_cfg);
122  PIOS_TIM_InitClock(&tim_4_cfg);
123 #endif /* PIOS_INCLUDE_TIM */

125  /* Initialize board specific USB data */
126  PIOS_USB_BOARD_DATA_Init();
127 
128  /* Flags to determine if various USB interfaces are advertised */
129  bool usb_cdc_present = false;
130 
131 #if defined(PIOS_INCLUDE_USB_CDC)
132  if (PIOS_USB_DESC_HID_CDC_Init()) {
133  PIOS_Assert(0);
134  }
135  usb_cdc_present = true;
136 #else
137  if (PIOS_USB_DESC_HID_ONLY_Init()) {
138  PIOS_Assert(0);
139  }
140 #endif
141 
142  /*Initialize the USB device */
143  uint32_t pios_usb_id;
144  PIOS_USB_Init(&pios_usb_id, &pios_usb_main_cfg);
145 
146  /* Configure the USB HID port */
147  {
148  uint32_t pios_usb_hid_id;
149  if (PIOS_USB_HID_Init(&pios_usb_hid_id, &pios_usb_hid_cfg, pios_usb_id)) {
150  PIOS_Assert(0);
151  }
152  uint8_t *rx_buffer = (uint8_t *)pvPortMalloc(PIOS_COM_TELEM_USB_RX_BUF_LEN);
153  uint8_t *tx_buffer = (uint8_t *)pvPortMalloc(PIOS_COM_TELEM_USB_TX_BUF_LEN);
154  PIOS_Assert(rx_buffer);
155  PIOS_Assert(tx_buffer);
156  if (PIOS_COM_Init(&pios_com_telem_usb_id, &pios_usb_hid_com_driver, pios_usb_hid_id,
157  rx_buffer, PIOS_COM_TELEM_USB_RX_BUF_LEN,
158  tx_buffer, PIOS_COM_TELEM_USB_TX_BUF_LEN)) {
159  PIOS_Assert(0);
160  }
161  }
162 
163  /* Configure the USB virtual com port (VCP) */
164 #if defined(PIOS_INCLUDE_USB_CDC)
165  if (usb_cdc_present)
166  {
167  uint32_t pios_usb_cdc_id;
168  if (PIOS_USB_CDC_Init(&pios_usb_cdc_id, &pios_usb_cdc_cfg, pios_usb_id)) {
169  PIOS_Assert(0);
170  }
171  uint8_t *rx_buffer = (uint8_t *)pvPortMalloc(PIOS_COM_TELEM_VCP_RX_BUF_LEN);
172  uint8_t *tx_buffer = (uint8_t *)pvPortMalloc(PIOS_COM_TELEM_VCP_TX_BUF_LEN);
173  PIOS_Assert(rx_buffer);
174  PIOS_Assert(tx_buffer);
175  if (PIOS_COM_Init(&pios_com_telem_vcp_id, &pios_usb_cdc_com_driver, pios_usb_cdc_id,
176  rx_buffer, PIOS_COM_TELEM_VCP_RX_BUF_LEN,
177  tx_buffer, PIOS_COM_TELEM_VCP_TX_BUF_LEN)) {
178  PIOS_Assert(0);
179  }
180  }
181 #endif
182

184 #ifndef PIOS_RFM22B_DEBUG_ON_TELEM

197  PIOS_Assert(0);
198  }
199  }
200 #endif
201 
202  /* Configure PPM input */
