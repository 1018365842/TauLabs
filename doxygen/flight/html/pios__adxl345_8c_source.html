282 uint8_t PIOS_ADXL345_Read(struct pios_adxl345_data * data)
283 {
284  if(PIOS_ADXL345_Validate(dev) != 0)
285  return -1;
286 
287  if(PIOS_ADXL345_ClaimBus() != 0)
288  return -2;
289 
290  // To save memory use same buffer for in and out but offset by
291  // a byte
292  uint8_t buf[9] = {0,0,0,0,0,0,0,0};
293  uint8_t rec[9] = {0,0,0,0,0,0,0,0};
294  buf[0] = ADXL_X0_ADDR | ADXL_MULTI_BIT | ADXL_READ_BIT ; // Multibyte read starting at X0
295 
296  if(PIOS_SPI_TransferBlock(dev->spi_id,&buf[0],&rec[0],9,NULL) < 0) {
297  PIOS_ADXL345_ReleaseBus();
298  return -3;
299  }
300 
301  PIOS_ADXL345_ReleaseBus();
302 
303  data->x = rec[1] + (rec[2] << 8);
304  data->y = rec[3] + (rec[4] << 8);
305  data->z = rec[5] + (rec[6] << 8);
306 
307  return rec[8] & 0x7F; // return number of remaining entries
308 }
