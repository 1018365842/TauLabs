94 static int32_t PIOS_PCF8591_ADC_DevicePinGet(uint32_t adc_id, uint32_t device_pin) {
95  struct pios_pcf8591_adc_dev * adc_dev = (struct pios_pcf8591_adc_dev *) adc_id;
96  if(!PIOS_PCF8591_ADC_validate(adc_dev))
97  {
98  return -1;
99  }
100  /* Check if pin exists */
101  if (device_pin >= NELEMENTS(ADC_CHANNEL)) {
102  return -1;
103  }
104  uint8_t readBuffer[2];
105  uint8_t config;
106  config = ADC_CHANNEL[device_pin] | (adc_dev->cfg->use_auto_increment ? PIOS_PCF8591_ADC_AUTO_INCREMENT : 0x00) | adc_dev->cfg->adc_input_type | (adc_dev->cfg->enable_dac ? PIOS_PCF8591_DAC_ENABLE : 0x00);
