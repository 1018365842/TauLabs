154 static void systemTask(void *parameters)
155 {
156  /* create all modules thread */
157  MODULE_TASKCREATE_ALL;
158 
159  if (mallocFailed) {
160  /* We failed to malloc during task creation,
161  * system behaviour is undefined. Reset and let
162  * the BootFault code recover for us.
163  */
164  PIOS_SYS_Reset();
165  }
166 
167 #if defined(PIOS_INCLUDE_IAP)
168  /* Record a successful boot */
169  PIOS_IAP_WriteBootCount(0);
170 #endif
171 
172  // Initialize vars
173  idleCounter = 0;
174  idleCounterClear = 0;
175 
176  // Listen for SettingPersistance object updates, connect a callback function
177  ObjectPersistenceConnectQueue(objectPersistenceQueue);
178 
179 #if (defined(COPTERCONTROL) || defined(REVOLUTION) || defined(SIM_OSX)) && ! (defined(SIM_POSIX))
180  // Run this initially to make sure the configuration is checked
181  configuration_check();
182 
183  // Whenever the configuration changes, make sure it is safe to fly
184  if (SystemSettingsHandle())
185  SystemSettingsConnectCallback(configurationUpdatedCb);
186  if (ManualControlSettingsHandle())
187  ManualControlSettingsConnectCallback(configurationUpdatedCb);
188 #endif
189 
190  // Main system loop
191  while (1) {
192  // Update the system statistics
193  updateStats();
194 
195  // Update the system alarms
196  updateSystemAlarms();
197 #if defined(I2C_WDG_STATS_DIAGNOSTICS)
198  updateI2Cstats();
199  updateWDGstats();
200 #endif
201 
202 #if defined(DIAG_TASKS)
203  // Update the task status object
204  TaskMonitorUpdateAll();
205 #endif
206 
207  // Flash the heartbeat LED
208 #if defined(PIOS_LED_HEARTBEAT)
209  PIOS_LED_Toggle(PIOS_LED_HEARTBEAT);
210  DEBUG_MSG("+ 0x%08x\r\n", 0xDEADBEEF);
211 #endif /* PIOS_LED_HEARTBEAT */
212 
213  // Turn on the error LED if an alarm is set
214 #if defined (PIOS_LED_ALARM)
215  if (AlarmsHasWarnings()) {
216  PIOS_LED_On(PIOS_LED_ALARM);
217  } else {
218  PIOS_LED_Off(PIOS_LED_ALARM);
219  }
220 #endif /* PIOS_LED_ALARM */
221 
222  FlightStatusData flightStatus;
223  FlightStatusGet(&flightStatus);
224 
225  UAVObjEvent ev;
226  int delayTime = flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED ?
227  SYSTEM_UPDATE_PERIOD_MS / portTICK_RATE_MS / (LED_BLINK_RATE_HZ * 2) :
228  SYSTEM_UPDATE_PERIOD_MS / portTICK_RATE_MS;
229 
230  if(xQueueReceive(objectPersistenceQueue, &ev, delayTime) == pdTRUE) {
231  // If object persistence is updated call the callback
232  objectUpdatedCb(&ev);
233  }
234  }
235 }

240 static void objectUpdatedCb(UAVObjEvent * ev)
241 {
242  ObjectPersistenceData objper;
243  UAVObjHandle obj;
244 
245  // If the object updated was the ObjectPersistence execute requested action
246  if (ev->obj == ObjectPersistenceHandle()) {
247  // Get object data
248  ObjectPersistenceGet(&objper);
249 
250  int retval = 1;
251 
252  // When this is called because of this method don't do anything
253  if (objper.Operation == OBJECTPERSISTENCE_OPERATION_ERROR ||
254  objper.Operation == OBJECTPERSISTENCE_OPERATION_COMPLETED) {
255  return;
256  }
257 
258  if (objper.Operation == OBJECTPERSISTENCE_OPERATION_LOAD) {
259  if (objper.Selection == OBJECTPERSISTENCE_SELECTION_SINGLEOBJECT) {
260  // Get selected object
261  obj = UAVObjGetByID(objper.ObjectID);
262  if (obj == 0) {
263  return;
264  }
265  // Load selected instance
266  retval = UAVObjLoad(obj, objper.InstanceID);
267  } else if (objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLSETTINGS
268  || objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLOBJECTS) {
269  retval = UAVObjLoadSettings();
270  } else if (objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLMETAOBJECTS
271  || objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLOBJECTS) {
272  retval = UAVObjLoadMetaobjects();
273  }
274  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_SAVE) {
275  if (objper.Selection == OBJECTPERSISTENCE_SELECTION_SINGLEOBJECT) {
276  // Get selected object
277  obj = UAVObjGetByID(objper.ObjectID);
278  if (obj == 0) {
279  return;
280  }
281  // Save selected instance
282  retval = UAVObjSave(obj, objper.InstanceID);
283 
284  // Not sure why this is needed
285  vTaskDelay(10);
286 
287  // Verify saving worked
288  if (retval == 0)
289  retval = UAVObjLoad(obj, objper.InstanceID);
290  } else if (objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLSETTINGS
291  || objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLOBJECTS) {
292  retval = UAVObjSaveSettings();
293  } else if (objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLMETAOBJECTS
294  || objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLOBJECTS) {
295  retval = UAVObjSaveMetaobjects();
296  }
297  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_DELETE) {
298  if (objper.Selection == OBJECTPERSISTENCE_SELECTION_SINGLEOBJECT) {
299  // Delete selected instance
300  retval = UAVObjDeleteById(objper.ObjectID, objper.InstanceID);
301  } else if (objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLSETTINGS
302  || objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLOBJECTS) {
303  retval = UAVObjDeleteSettings();
304  } else if (objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLMETAOBJECTS
305  || objper.Selection == OBJECTPERSISTENCE_SELECTION_ALLOBJECTS) {
306  retval = UAVObjDeleteMetaobjects();
307  }
308  } else if (objper.Operation == OBJECTPERSISTENCE_OPERATION_FULLERASE) {
309  retval = -1;
310 #if defined(PIOS_INCLUDE_FLASH_SECTOR_SETTINGS)
311  extern uintptr_t pios_uavo_settings_fs_id;
312  retval = PIOS_FLASHFS_Format(pios_uavo_settings_fs_id);
313 #endif
314  }
315  switch(retval) {
316  case 0:
317  objper.Operation = OBJECTPERSISTENCE_OPERATION_COMPLETED;
318  ObjectPersistenceSet(&objper);
319  break;
320  case -1:
321  objper.Operation = OBJECTPERSISTENCE_OPERATION_ERROR;
322  ObjectPersistenceSet(&objper);
323  break;
324  default:
325  break;
326  }
327  }
328 }

377 static uint16_t GetFreeIrqStackSize(void)
378 {
379  uint32_t i = 0x200;
380 
381 #if !defined(ARCH_POSIX) && !defined(ARCH_WIN32) && defined(CHECK_IRQ_STACK)
382 extern uint32_t _irq_stack_top;
383 extern uint32_t _irq_stack_end;
384 uint32_t pattern = 0x0000A5A5;
385 uint32_t *ptr = &_irq_stack_end;
386 
387 #if 1 /* the ugly way accurate but takes more time, useful for debugging */
388  uint32_t stack_size = (((uint32_t)&_irq_stack_top - (uint32_t)&_irq_stack_end) & ~3 ) / 4;
389 
390  for (i=0; i< stack_size; i++)
391  {
392  if (ptr[i] != pattern)
393  {
394  i=i*4;
395  break;
396  }
397  }
398 #else /* faster way but not accurate */
399  if (*(volatile uint32_t *)((uint32_t)ptr + IRQSTACK_LIMIT_CRITICAL) != pattern)
400  {
401  i = IRQSTACK_LIMIT_CRITICAL - 1;
402  }
403  else if (*(volatile uint32_t *)((uint32_t)ptr + IRQSTACK_LIMIT_WARNING) != pattern)
404  {
405  i = IRQSTACK_LIMIT_WARNING - 1;
406  }
407  else
408  {
409  i = IRQSTACK_LIMIT_WARNING;
410  }
411 #endif
412 #endif
413  return i;
414 }

419 static void updateStats()
420 {
421  static portTickType lastTickCount = 0;
422  SystemStatsData stats;
423 
424  // Get stats and update
425  SystemStatsGet(&stats);
426  stats.FlightTime = xTaskGetTickCount() * portTICK_RATE_MS;
427 #if defined(ARCH_POSIX) || defined(ARCH_WIN32)
428  // POSIX port of FreeRTOS doesn't have xPortGetFreeHeapSize()
429  stats.HeapRemaining = 10240;
430 #else
431  stats.HeapRemaining = xPortGetFreeHeapSize();
432 #endif
433 
434  // Get Irq stack status
435  stats.IRQStackRemaining = GetFreeIrqStackSize();
436 
437  // When idleCounterClear was not reset by the idle-task, it means the idle-task did not run
438  if (idleCounterClear) {
439  idleCounter = 0;
440  }
441 
442  portTickType now = xTaskGetTickCount();
443  if (now > lastTickCount) {
444  uint32_t dT = (xTaskGetTickCount() - lastTickCount) * portTICK_RATE_MS; // in ms
445 
446  // In the case of a slightly miscalibrated max idle count, make sure CPULoad does
447  // not go negative and set an alarm inappropriately.
448  float idleFraction = ((float)idleCounter / ((float)dT / 1000.0f)) / (float)IDLE_COUNTS_PER_SEC_AT_NO_LOAD;
449  if (idleFraction > 1)
450  stats.CPULoad = 0;
451  else
452  stats.CPULoad = 100 - roundf(100.0f * idleFraction);
453  } //else: TickCount has wrapped, do not calc now
454  lastTickCount = now;
455  idleCounterClear = 1;
456 
457 #if defined(PIOS_INCLUDE_ADC) && defined(PIOS_ADC_USE_TEMP_SENSOR)
458  float temp_voltage = 3.3 * PIOS_ADC_PinGet(0) / ((1 << 12) - 1);
459  const float STM32_TEMP_V25 = 1.43; /* V */
460  const float STM32_TEMP_AVG_SLOPE = 4.3; /* mV/C */
461  stats.CPUTemp = (temp_voltage-STM32_TEMP_V25) * 1000 / STM32_TEMP_AVG_SLOPE + 25;
462 #endif
463  SystemStatsSet(&stats);
464 }

469 static void updateSystemAlarms()
470 {
471  SystemStatsData stats;
472  UAVObjStats objStats;
473  EventStats evStats;
474  SystemStatsGet(&stats);
475 
476  // Check heap, IRQ stack and malloc failures
477  if ( mallocFailed
478  || (stats.HeapRemaining < HEAP_LIMIT_CRITICAL)
479 #if !defined(ARCH_POSIX) && !defined(ARCH_WIN32) && defined(CHECK_IRQ_STACK)
480  || (stats.IRQStackRemaining < IRQSTACK_LIMIT_CRITICAL)
481 #endif
482  ) {
483  AlarmsSet(SYSTEMALARMS_ALARM_OUTOFMEMORY, SYSTEMALARMS_ALARM_CRITICAL);
484  } else if (
485  (stats.HeapRemaining < HEAP_LIMIT_WARNING)
486 #if !defined(ARCH_POSIX) && !defined(ARCH_WIN32) && defined(CHECK_IRQ_STACK)
487  || (stats.IRQStackRemaining < IRQSTACK_LIMIT_WARNING)
488 #endif
489  ) {
490  AlarmsSet(SYSTEMALARMS_ALARM_OUTOFMEMORY, SYSTEMALARMS_ALARM_WARNING);
491  } else {
492  AlarmsClear(SYSTEMALARMS_ALARM_OUTOFMEMORY);
493  }
494 
495  // Check CPU load
496  if (stats.CPULoad > CPULOAD_LIMIT_CRITICAL) {
497  AlarmsSet(SYSTEMALARMS_ALARM_CPUOVERLOAD, SYSTEMALARMS_ALARM_CRITICAL);
498  } else if (stats.CPULoad > CPULOAD_LIMIT_WARNING) {
499  AlarmsSet(SYSTEMALARMS_ALARM_CPUOVERLOAD, SYSTEMALARMS_ALARM_WARNING);
500  } else {
501  AlarmsClear(SYSTEMALARMS_ALARM_CPUOVERLOAD);
502  }
503 
504  // Check for stack overflow
505  if (stackOverflow) {
506  AlarmsSet(SYSTEMALARMS_ALARM_STACKOVERFLOW, SYSTEMALARMS_ALARM_CRITICAL);
507  } else {
508  AlarmsClear(SYSTEMALARMS_ALARM_STACKOVERFLOW);
509  }
510 
511  // Check for event errors
512  UAVObjGetStats(&objStats);
513  EventGetStats(&evStats);
514  UAVObjClearStats();
515  EventClearStats();
516  if (objStats.eventCallbackErrors > 0 || objStats.eventQueueErrors > 0 || evStats.eventErrors > 0) {
517  AlarmsSet(SYSTEMALARMS_ALARM_EVENTSYSTEM, SYSTEMALARMS_ALARM_WARNING);
518  } else {
519  AlarmsClear(SYSTEMALARMS_ALARM_EVENTSYSTEM);
520  }
521 
522  if (objStats.lastCallbackErrorID || objStats.lastQueueErrorID || evStats.lastErrorID) {
523  SystemStatsData sysStats;
524  SystemStatsGet(&sysStats);
525  sysStats.EventSystemWarningID = evStats.lastErrorID;
526  sysStats.ObjectManagerCallbackID = objStats.lastCallbackErrorID;
527  sysStats.ObjectManagerQueueID = objStats.lastQueueErrorID;
528  SystemStatsSet(&sysStats);
529  }
