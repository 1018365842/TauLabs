159 static void overoSyncTask(void *parameters)
160 {
161  UAVObjEvent ev;
162 
163  // Kick off SPI transfers (once one is completed another will automatically transmit)
164  overosync->transaction_done = true;
165  overosync->sent_objects = 0;
166  overosync->failed_objects = 0;
167  overosync->received_objects = 0;
168 
169  portTickType lastUpdateTime = xTaskGetTickCount();
170  portTickType updateTime;
171 
172  fid = fopen("sim_log.opl", "w");
173 
174  // Loop forever
175  while (1) {
176  // Wait for queue message
177  if (xQueueReceive(queue, &ev, portMAX_DELAY) == pdTRUE) {
178 
179  // Check it will fit before packetizing
180  if ((overosync->write_pointer + UAVObjGetNumBytes(ev.obj) + 12) >=
181  sizeof(overosync->transactions[overosync->loading_transaction_id].tx_buffer)) {
182  overosync->failed_objects ++;
183  } else {
184  // Process event. This calls transmitData
185  UAVTalkSendObject(uavTalkCon, ev.obj, ev.instId, false, 0);
186  }
187 
188  updateTime = xTaskGetTickCount();
189  if(((portTickType) (updateTime - lastUpdateTime)) > 1000) {
190  // Update stats. This will trigger a local send event too
191  OveroSyncStatsData syncStats;
192  syncStats.Send = overosync->sent_bytes;
193  syncStats.Received = 0;
194  syncStats.Connected = syncStats.Send > 500 ? OVEROSYNCSTATS_CONNECTED_TRUE : OVEROSYNCSTATS_CONNECTED_FALSE;
195  syncStats.DroppedUpdates = overosync->failed_objects;
196  OveroSyncStatsSet(&syncStats);
197  overosync->failed_objects = 0;
198  overosync->sent_bytes = 0;
199  lastUpdateTime = updateTime;
200  }
201  }
202  }
203 }

212 static int32_t packData(uint8_t * data, int32_t length)
213 {
214  // Get the lock for manipulating the buffer
215  xSemaphoreTake(overosync->buffer_lock, portMAX_DELAY);
216 
217  portTickType tickTime = xTaskGetTickCount();
218  uint64_t packetSize = data[2] + (data[3] << 8);
219  fwrite((void *) &tickTime, 1, sizeof(tickTime), fid);
220  fwrite((void *) &packetSize, sizeof(packetSize), 1, fid);
221  fwrite((void *) data, 1, length, fid);
222  overosync->sent_bytes += length;
223  overosync->sent_objects++;
224 
225  xSemaphoreGive(overosync->buffer_lock);
226 
227  return length;
228 }
