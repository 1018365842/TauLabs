65 uint16_t PIOS_WDG_Init()
66 {
67  uint16_t delay = ((uint32_t) PIOS_WATCHDOG_TIMEOUT * 60) / 16;
68  if (delay > 0x0fff)
69  delay = 0x0fff;
70 #if defined(PIOS_INCLUDE_WDG)
71  DBGMCU_Config(DBGMCU_IWDG_STOP, ENABLE); // make the watchdog stop counting in debug mode
72  IWDG_WriteAccessCmd(IWDG_WriteAccess_Enable);
73  IWDG_SetPrescaler(IWDG_Prescaler_16);
74  IWDG_SetReload(delay);
75  IWDG_ReloadCounter();
76  IWDG_Enable();
77 
78  // watchdog flags now stored in backup registers
79  PWR_BackupAccessCmd(ENABLE);
80 
81  wdg_configuration.bootup_flags = RTC_ReadBackupRegister(PIOS_WDG_REGISTER);
82 #endif
83  return delay;
84 }

122 bool PIOS_WDG_UpdateFlag(uint16_t flag)
123 {
124  // we can probably avoid using a semaphore here which will be good for
125  // efficiency and not blocking critical tasks. race condition could
126  // overwrite their flag update, but unlikely to block _all_ of them
127  // for the timeout window
128  uint16_t cur_flags = RTC_ReadBackupRegister(PIOS_WDG_REGISTER);
129 
130  if((cur_flags | flag) == wdg_configuration.used_flags) {
131  PIOS_WDG_Clear();
132  RTC_WriteBackupRegister(PIOS_WDG_REGISTER, flag);
133  return true;
134  } else {
135  RTC_WriteBackupRegister(PIOS_WDG_REGISTER, cur_flags | flag);
136  return false;
137  }
138 
139 }
