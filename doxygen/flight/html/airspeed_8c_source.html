181 static void airspeedTask(void *parameters)
182 {
183  AirspeedSettingsUpdatedCb(AirspeedSettingsHandle());
184 
185  BaroAirspeedData airspeedData;
186  AirspeedActualData airspeedActualData;
187  BaroAirspeedGet(&airspeedData);
188  AirspeedActualGet(&airspeedActualData);
189 
190 
191  airspeedData.BaroConnected = BAROAIRSPEED_BAROCONNECTED_FALSE;
192 
193 #ifdef BARO_AIRSPEED_PRESENT
194  portTickType lastGPSTime = xTaskGetTickCount(); //Time since last GPS-derived airspeed calculation
195  portTickType lastLoopTime= xTaskGetTickCount(); //Time since last loop
196 
197  float airspeedErrInt=0;
198 #endif
199 
200  //GPS airspeed calculation variables
201 #ifdef GPS_AIRSPEED_PRESENT
202  GPSVelocityConnectCallback(GPSVelocityUpdatedCb);
203  gps_airspeedInitialize();
204 #endif
205 
206  // Main task loop
207  portTickType lastSysTime = xTaskGetTickCount();
208  while (1)
209  {
210  // Update the airspeed object
211  BaroAirspeedGet(&airspeedData);
212  AirspeedActualGet(&airspeedActualData);
213 
214 #ifdef BARO_AIRSPEED_PRESENT
215  float airspeed_tas_baro=0;
216 
217  if(airspeedSensorType!=AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_GPSONLY){
218  //Fetch calibrated airspeed from sensors
219  baro_airspeedGet(&airspeedData, &lastSysTime, airspeedSensorType, airspeedADCPin);
220 
221  //Calculate true airspeed, not taking into account compressibility effects
222  float baroTemperature;
223 
224  BaroAltitudeTemperatureGet(&baroTemperature);
225  baroTemperature-=BARO_TEMPERATURE_OFFSET; //Do this just because we suspect that the board heats up relative to its surroundings. THIS IS BAD(tm)
226  #ifdef GPS_AIRSPEED_PRESENT
227  //GPS present, so use baro sensor to filter TAS
228  airspeed_tas_baro = airspeedData.CalibratedAirspeed * sqrtf((baroTemperature + CELSIUS2KELVIN) / STANDARD_AIR_TEMPERATURE) + airspeedErrInt * GPS_AIRSPEED_BIAS_KI;
229  #else
230  //No GPS, so TAS comes only from baro sensor
231  airspeedData.TrueAirspeed = airspeedData.CalibratedAirspeed * sqrtf((baroTemperature + CELSIUS2KELVIN) / STANDARD_AIR_TEMPERATURE) + airspeedErrInt * GPS_AIRSPEED_BIAS_KI;
232  #endif
233 
234  }
235  else
236 #endif
237  { //Have to catch the fallthrough, or else this loop will monopolize the processor!
238  airspeedData.BaroConnected=BAROAIRSPEED_BAROCONNECTED_FALSE;
239  airspeedData.SensorValue=12345;
240 
241  //Likely, we have a GPS, so let's configure the fallthrough at close to GPS refresh rates
242  vTaskDelayUntil(&lastSysTime, SAMPLING_DELAY_MS_FALLTHROUGH / portTICK_RATE_MS);
243  }
244 
245 #ifdef GPS_AIRSPEED_PRESENT
246  float v_air_GPS=-1.0f;
247 
248  //Check if sufficient time has passed. This will depend on whether we have a pitot tube
249  //sensor or not. In the case we do, shoot for about once per second. Otherwise, consume GPS
250  //as quickly as possible.
251  #ifdef BARO_AIRSPEED_PRESENT
252  float delT = (lastSysTime - lastLoopTime)/(portTICK_RATE_MS*1000.0f);
253  lastLoopTime=lastSysTime;
254  if ( ((lastSysTime - lastGPSTime) > 1000*portTICK_RATE_MS || airspeedSensorType==AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_GPSONLY)
255  && gpsNew) {
256  lastGPSTime=lastSysTime;
257  #else
258  if (gpsNew) {
259  #endif
260  gpsNew=false; //Do this first
261 
262  //Calculate airspeed as a function of GPS groundspeed and vehicle attitude. From "IMU Wind Estimation (Theory)", by William Premerlani
263  gps_airspeedGet(&v_air_GPS);
264  }
265 
266 
267  //Use the GPS error to correct the airspeed estimate.
268  if (v_air_GPS > 0) //We have valid GPS estimate...
269  {
270  airspeedData.GPSAirspeed=v_air_GPS;
271 
272  #ifdef BARO_AIRSPEED_PRESENT
273  if(airspeedData.BaroConnected==BAROAIRSPEED_BAROCONNECTED_TRUE){ //Check if there is an airspeed sensors present...
274  //Calculate error and error integral
275  float airspeedErr=v_air_GPS - airspeed_tas_baro;
276  airspeedErrInt+=airspeedErr * delT;
277 
278  //Saturate integral component at 5 m/s
279  airspeedErrInt = airspeedErrInt > (5.0f / GPS_AIRSPEED_BIAS_KI) ? (5.0f / GPS_AIRSPEED_BIAS_KI) : airspeedErrInt;
280  airspeedErrInt = airspeedErrInt < -(5.0f / GPS_AIRSPEED_BIAS_KI) ? -(5.0f / GPS_AIRSPEED_BIAS_KI) : airspeedErrInt;
281 
282  //There's already an airspeed sensor, so instead correct it for bias with P correction. The I correction happened earlier in the function.
283  airspeedData.TrueAirspeed = airspeed_tas_baro + airspeedErr * GPS_AIRSPEED_BIAS_KP;
284 
285 
286  /* Note:
287  This would be a good place to change the airspeed calibration, so that it matches the GPS computed values. However,
288  this might be a bad idea, as their are two degrees of freedom here: temperature and sensor calibration. I don't
289  know how to control for temperature bias.
290  */
291  }
292  else
293  #endif
294  {
295  //...there's no airspeed sensor, so everything comes from GPS. In this
296  //case, filter the airspeed for smoother output
297  float alpha=gpsSamplePeriod_ms/(gpsSamplePeriod_ms + GPS_AIRSPEED_TIME_CONSTANT_MS); //Low pass filter.
298  airspeedData.TrueAirspeed=v_air_GPS*(alpha) + airspeedData.TrueAirspeed*(1.0f-alpha);
299 
300  //Calculate calibrated airspeed from GPS, since we're not getting it from a discrete airspeed sensor
301  float baroTemperature;
302  BaroAltitudeTemperatureGet(&baroTemperature);
303  baroTemperature-=BARO_TEMPERATURE_OFFSET; //Do this just because we suspect that the board heats up relative to its surroundings. THIS IS BAD(tm)
304  airspeedData.CalibratedAirspeed = airspeedData.TrueAirspeed / sqrtf((baroTemperature + CELSIUS2KELVIN) / STANDARD_AIR_TEMPERATURE);
305  }
306  }
307  #ifdef BARO_AIRSPEED_PRESENT
308  else if (airspeedData.BaroConnected==BAROAIRSPEED_BAROCONNECTED_TRUE){
309  //No GPS velocity estimate this loop, so filter true airspeed data with baro airspeed
310  float alpha=delT/(delT + BARO_TRUEAIRSPEED_TIME_CONSTANT_S); //Low pass filter.
311  airspeedData.TrueAirspeed=airspeed_tas_baro*(alpha) + airspeedData.TrueAirspeed*(1.0f-alpha);
312  }
313  #endif
314 #endif
315  //Set the UAVO
316  airspeedActualData.TrueAirspeed = airspeedData.TrueAirspeed;
317  airspeedActualData.CalibratedAirspeed = airspeedData.CalibratedAirspeed;
318  BaroAirspeedSet(&airspeedData);
319  AirspeedActualSet(&airspeedActualData);
320 
321  }
322 }

333 
334  //Find out which sensor we're using.
335  switch (airspeedSensorType) {
336 #if defined(PIOS_INCLUDE_MPXV7002)
337  case AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_DIYDRONESMPXV7002:
338 #endif
339 #if defined(PIOS_INCLUDE_MPXV5004)
340  case AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_DIYDRONESMPXV5004:
341 #endif
342 #if defined(PIOS_INCLUDE_MPXV7002) || defined(PIOS_INCLUDE_MPXV5004)
343  //MPXV5004 and MPXV7002 sensors
344  baro_airspeedGetAnalog(baroAirspeedData, lastSysTime, airspeedSensorType, airspeedADCPin);

346 #endif
