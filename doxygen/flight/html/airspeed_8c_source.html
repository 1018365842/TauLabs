182 static void airspeedTask(void *parameters)
183 {
184  AirspeedSettingsUpdatedCb(AirspeedSettingsHandle());
185 
186  BaroAirspeedData airspeedData;
187  AirspeedActualData airspeedActualData;
188  BaroAirspeedGet(&airspeedData);
189  AirspeedActualGet(&airspeedActualData);
190 
191 
192  airspeedData.BaroConnected = BAROAIRSPEED_BAROCONNECTED_FALSE;
193 
194 #ifdef BARO_AIRSPEED_PRESENT
195  portTickType lastGPSTime = xTaskGetTickCount(); //Time since last GPS-derived airspeed calculation
196  portTickType lastLoopTime= xTaskGetTickCount(); //Time since last loop
197 
198  float airspeedErrInt=0;
199 #endif
200 
201  //GPS airspeed calculation variables
202 #ifdef GPS_AIRSPEED_PRESENT
203  GPSVelocityConnectCallback(GPSVelocityUpdatedCb);
204  gps_airspeedInitialize();
205 #endif
206 
207  // Main task loop
208  portTickType lastSysTime = xTaskGetTickCount();
209  while (1)
210  {
211  // Update the airspeed object
212  BaroAirspeedGet(&airspeedData);
213  AirspeedActualGet(&airspeedActualData);
214 
215 #ifdef BARO_AIRSPEED_PRESENT
216  float airspeed_tas_baro=0;
217 
218  if(airspeedSensorType!=AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_GPSONLY){
219  //Fetch calibrated airspeed from sensors
220  baro_airspeedGet(&airspeedData, &lastSysTime, airspeedSensorType, airspeedADCPin);
221 
222  //Calculate true airspeed, not taking into account compressibility effects
223  float baroTemperature;
224 
225  BaroAltitudeTemperatureGet(&baroTemperature);
226  baroTemperature-=BARO_TEMPERATURE_OFFSET; //Do this just because we suspect that the board heats up relative to its surroundings. THIS IS BAD(tm)
227  #ifdef GPS_AIRSPEED_PRESENT
228  //GPS present, so use baro sensor to filter TAS
229  airspeed_tas_baro = airspeedData.CalibratedAirspeed * sqrtf((baroTemperature + CELSIUS2KELVIN) / STANDARD_AIR_TEMPERATURE) + airspeedErrInt * GPS_AIRSPEED_BIAS_KI;
230  #else
231  //No GPS, so TAS comes only from baro sensor
232  airspeedData.TrueAirspeed = airspeedData.CalibratedAirspeed * sqrtf((baroTemperature + CELSIUS2KELVIN) / STANDARD_AIR_TEMPERATURE) + airspeedErrInt * GPS_AIRSPEED_BIAS_KI;
233  #endif
234 
235  }
236  else
237 #endif
238  { //Have to catch the fallthrough, or else this loop will monopolize the processor!
239  airspeedData.BaroConnected=BAROAIRSPEED_BAROCONNECTED_FALSE;
240  airspeedData.SensorValue=12345;
241 
242  //Likely, we have a GPS, so let's configure the fallthrough at close to GPS refresh rates
243  vTaskDelayUntil(&lastSysTime, SAMPLING_DELAY_MS_FALLTHROUGH / portTICK_RATE_MS);
244  }
245 
246 #ifdef GPS_AIRSPEED_PRESENT
247  float v_air_GPS=-1.0f;
248 
249  //Check if sufficient time has passed. This will depend on whether we have a pitot tube
250  //sensor or not. In the case we do, shoot for about once per second. Otherwise, consume GPS
251  //as quickly as possible.
252  #ifdef BARO_AIRSPEED_PRESENT
253  float delT = (lastSysTime - lastLoopTime)/(portTICK_RATE_MS*1000.0f);
254  lastLoopTime=lastSysTime;
255  if ( ((lastSysTime - lastGPSTime) > 1000*portTICK_RATE_MS || airspeedSensorType==AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_GPSONLY)
256  && gpsNew) {
257  lastGPSTime=lastSysTime;
258  #else
259  if (gpsNew) {
260  #endif
261  gpsNew=false; //Do this first
262 
263  //Calculate airspeed as a function of GPS groundspeed and vehicle attitude. From "IMU Wind Estimation (Theory)", by William Premerlani
264  gps_airspeedGet(&v_air_GPS);
265  }
266 
267 
268  //Use the GPS error to correct the airspeed estimate.
269  if (v_air_GPS > 0) //We have valid GPS estimate...
270  {
271  airspeedData.GPSAirspeed=v_air_GPS;
272 
273  #ifdef BARO_AIRSPEED_PRESENT
274  if(airspeedData.BaroConnected==BAROAIRSPEED_BAROCONNECTED_TRUE){ //Check if there is an airspeed sensors present...
275  //Calculate error and error integral
276  float airspeedErr=v_air_GPS - airspeed_tas_baro;
277  airspeedErrInt+=airspeedErr * delT;
278 
279  //Saturate integral component at 5 m/s
280  airspeedErrInt = airspeedErrInt > (5.0f / GPS_AIRSPEED_BIAS_KI) ? (5.0f / GPS_AIRSPEED_BIAS_KI) : airspeedErrInt;
281  airspeedErrInt = airspeedErrInt < -(5.0f / GPS_AIRSPEED_BIAS_KI) ? -(5.0f / GPS_AIRSPEED_BIAS_KI) : airspeedErrInt;
282 
283  //There's already an airspeed sensor, so instead correct it for bias with P correction. The I correction happened earlier in the function.
284  airspeedData.TrueAirspeed = airspeed_tas_baro + airspeedErr * GPS_AIRSPEED_BIAS_KP;
285 
286 
287  /* Note:
288  This would be a good place to change the airspeed calibration, so that it matches the GPS computed values. However,
289  this might be a bad idea, as their are two degrees of freedom here: temperature and sensor calibration. I don't
290  know how to control for temperature bias.
291  */
292  }
293  else
294  #endif
295  {
296  //...there's no airspeed sensor, so everything comes from GPS. In this
297  //case, filter the airspeed for smoother output
298  float alpha=gpsSamplePeriod_ms/(gpsSamplePeriod_ms + GPS_AIRSPEED_TIME_CONSTANT_MS); //Low pass filter.
299  airspeedData.TrueAirspeed=v_air_GPS*(alpha) + airspeedData.TrueAirspeed*(1.0f-alpha);
300 
301  //Calculate calibrated airspeed from GPS, since we're not getting it from a discrete airspeed sensor
302  float baroTemperature;
303  BaroAltitudeTemperatureGet(&baroTemperature);
304  baroTemperature-=BARO_TEMPERATURE_OFFSET; //Do this just because we suspect that the board heats up relative to its surroundings. THIS IS BAD(tm)
305  airspeedData.CalibratedAirspeed = airspeedData.TrueAirspeed / sqrtf((baroTemperature + CELSIUS2KELVIN) / STANDARD_AIR_TEMPERATURE);
306  }
307  }
308  #ifdef BARO_AIRSPEED_PRESENT
309  else if (airspeedData.BaroConnected==BAROAIRSPEED_BAROCONNECTED_TRUE){
310  //No GPS velocity estimate this loop, so filter true airspeed data with baro airspeed
311  float alpha=delT/(delT + BARO_TRUEAIRSPEED_TIME_CONSTANT_S); //Low pass filter.
312  airspeedData.TrueAirspeed=airspeed_tas_baro*(alpha) + airspeedData.TrueAirspeed*(1.0f-alpha);
313  }
314  #endif
315 #endif
316  //Set the UAVO
317  airspeedActualData.TrueAirspeed = airspeedData.TrueAirspeed;
318  airspeedActualData.CalibratedAirspeed = airspeedData.CalibratedAirspeed;
319  BaroAirspeedSet(&airspeedData);
320  AirspeedActualSet(&airspeedActualData);
321 
322  }
323 }

333 void baro_airspeedGet(BaroAirspeedData *baroAirspeedData, portTickType *lastSysTime, uint8_t airspeedSensorType, int8_t airspeedADCPin){
334 
335  //Find out which sensor we're using.
336  switch (airspeedSensorType) {
337 #if defined(PIOS_INCLUDE_MPXV7002)
338  case AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_DIYDRONESMPXV7002:
339 #endif
340 #if defined(PIOS_INCLUDE_MPXV5004)
341  case AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_DIYDRONESMPXV5004:
342 #endif
343 #if defined(PIOS_INCLUDE_MPXV7002) || defined(PIOS_INCLUDE_MPXV5004)
344  //MPXV5004 and MPXV7002 sensors
345  baro_airspeedGetAnalog(baroAirspeedData, lastSysTime, airspeedSensorType, airspeedADCPin);
346  break;
347 #endif
348 #if defined(PIOS_INCLUDE_ETASV3)
349  case AIRSPEEDSETTINGS_AIRSPEEDSENSORTYPE_EAGLETREEAIRSPEEDV3:
350  //Eagletree Airspeed v3
351  baro_airspeedGetETASV3(baroAirspeedData, lastSysTime, airspeedSensorType, airspeedADCPin);
352  break;
353 #endif
354  default:
355  baroAirspeedData->BaroConnected = BAROAIRSPEED_BAROCONNECTED_FALSE;
356  }
357 }
