159 static void groundPathFollowerTask(void *parameters)
160 {
161  SystemSettingsData systemSettings;
162  FlightStatusData flightStatus;
163 
164  portTickType lastUpdateTime;
165 
166  GroundPathFollowerSettingsConnectCallback(SettingsUpdatedCb);
167  PathDesiredConnectCallback(SettingsUpdatedCb);
168 
169  GroundPathFollowerSettingsGet(&guidanceSettings);
170  PathDesiredGet(&pathDesired);
171 
172  // Main task loop
173  lastUpdateTime = xTaskGetTickCount();
174  while (1) {
175 
176  // Conditions when this runs:
177  // 1. Must have GROUND type airframe
178  // 2. Flight mode is PositionHold and PathDesired.Mode is Endpoint OR
179  // FlightMode is PathPlanner and PathDesired.Mode is Endpoint or Path
180 
181  SystemSettingsGet(&systemSettings);
182  if ( (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_GROUNDVEHICLECAR) &&
183  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_GROUNDVEHICLEDIFFERENTIAL) &&
184  (systemSettings.AirframeType != SYSTEMSETTINGS_AIRFRAMETYPE_GROUNDVEHICLEMOTORCYCLE) )
185  {
186  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_WARNING);
187  vTaskDelay(1000);
188  continue;
189  }
190 
191  // Continue collecting data if not enough time
192  vTaskDelayUntil(&lastUpdateTime, guidanceSettings.UpdatePeriod / portTICK_RATE_MS);
193 
194  // Convert the accels into the NED frame
195  updateNedAccel();
196 
197  FlightStatusGet(&flightStatus);
198 
199  // Check the combinations of flightmode and pathdesired mode
200  switch(flightStatus.FlightMode) {
201  /* This combination of RETURNTOHOME and HOLDPOSITION looks strange but
202  * is correct. RETURNTOHOME mode uses HOLDPOSITION with the position
203  * set to home */
204  case FLIGHTSTATUS_FLIGHTMODE_RETURNTOHOME:
205  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
206  updateEndpointVelocity();
207  updateGroundDesiredAttitude();
208  } else {
209  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
210  }
211  break;
212  case FLIGHTSTATUS_FLIGHTMODE_POSITIONHOLD:
213  if (pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
214  updateEndpointVelocity();
215  updateGroundDesiredAttitude();
216  } else {
217  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
218  }
219  break;
220  case FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER:
221  if (pathDesired.Mode == PATHDESIRED_MODE_FLYENDPOINT ||
222  pathDesired.Mode == PATHDESIRED_MODE_HOLDPOSITION) {
223  updateEndpointVelocity();
224  updateGroundDesiredAttitude();
225  } else if (pathDesired.Mode == PATHDESIRED_MODE_FLYVECTOR ||
226  pathDesired.Mode == PATHDESIRED_MODE_FLYCIRCLELEFT ||
227  pathDesired.Mode == PATHDESIRED_MODE_FLYCIRCLERIGHT) {
228  updatePathVelocity();
229  updateGroundDesiredAttitude();
230  } else {
231  AlarmsSet(SYSTEMALARMS_ALARM_PATHFOLLOWER,SYSTEMALARMS_ALARM_ERROR);
232  }
233  break;
234  default:
235  // Be cleaner and get rid of global variables
236  northVelIntegral = 0;
237  eastVelIntegral = 0;
238  northPosIntegral = 0;
239  eastPosIntegral = 0;
240 
241  // Track throttle before engaging this mode. Cheap system ident
242  StabilizationDesiredData stabDesired;
243  StabilizationDesiredGet(&stabDesired);
244  throttleOffset = stabDesired.Throttle;
245 
246  break;
247  }
248 
249  AlarmsClear(SYSTEMALARMS_ALARM_PATHFOLLOWER);
250 
251  }
252 }

260 static void updatePathVelocity()
261 {
262  PositionActualData positionActual;
263  PositionActualGet(&positionActual);
264 
265  float cur[3] = {positionActual.North, positionActual.East, positionActual.Down};
266  struct path_status progress;
267 
268  path_progress(&pathDesired, cur, &progress);
269 
270  // Update the path status UAVO
271  PathStatusData pathStatus;
272  PathStatusGet(&pathStatus);
273  pathStatus.fractional_progress = progress.fractional_progress;
274  if (pathStatus.fractional_progress < 1)
275  pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
276  else
277  pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
278  PathStatusSet(&pathStatus);
279 
280  float groundspeed = pathDesired.StartingVelocity +
281  (pathDesired.EndingVelocity - pathDesired.StartingVelocity) * progress.fractional_progress;
282  if(progress.fractional_progress > 1)
283  groundspeed = 0;
284 
285  VelocityDesiredData velocityDesired;
286  velocityDesired.North = progress.path_direction[0] * groundspeed;
287  velocityDesired.East = progress.path_direction[1] * groundspeed;
288 
289  float error_speed = progress.error * guidanceSettings.HorizontalPosPI[GROUNDPATHFOLLOWERSETTINGS_HORIZONTALPOSPI_KP];
290  float correction_velocity[2] = {progress.correction_direction[0] * error_speed,
291  progress.correction_direction[1] * error_speed};
292 
293  float total_vel = sqrtf(powf(correction_velocity[0],2) + powf(correction_velocity[1],2));
294  float scale = 1;
295  if(total_vel > guidanceSettings.HorizontalVelMax)
296  scale = guidanceSettings.HorizontalVelMax / total_vel;
297 
298  // Currently not apply a PID loop for horizontal corrections
299  velocityDesired.North += progress.correction_direction[0] * error_speed * scale;
300  velocityDesired.East += progress.correction_direction[1] * error_speed * scale;
301 
302  // No altitude control on a ground vehicle
303  velocityDesired.Down = 0;
304 
305  VelocityDesiredSet(&velocityDesired);
306 }

314 void updateEndpointVelocity()
315 {
316  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
317 
318  PositionActualData positionActual;
319  VelocityDesiredData velocityDesired;
320 
321  PositionActualGet(&positionActual);
322  VelocityDesiredGet(&velocityDesired);
323 
324  float northError;
325  float eastError;
326  float northCommand;
327  float eastCommand;
328 
329  float northPos = positionActual.North;
330  float eastPos = positionActual.East;
331 
332  // Compute desired north command velocity from position error
333  northError = pathDesired.End[PATHDESIRED_END_NORTH] - northPos;
334  northCommand = pid_apply(&ground_pids[NORTH_POSITION], northError, dT);
335 
336  // Compute desired east command velocity from position error
337  eastError = pathDesired.End[PATHDESIRED_END_EAST] - eastPos;
338  eastCommand = pid_apply(&ground_pids[EAST_POSITION], eastError, dT);
339 
340  // Limit the maximum velocity any direction (not north and east separately)
341  float total_vel = sqrtf(powf(northCommand,2) + powf(eastCommand,2));
342  float scale = 1;
343  if(total_vel > guidanceSettings.HorizontalVelMax)
344  scale = guidanceSettings.HorizontalVelMax / total_vel;
345 
346  velocityDesired.North = northCommand * scale;
347  velocityDesired.East = eastCommand * scale;
348 
349  // No altitude control on a ground vehicle
350  velocityDesired.Down = 0;
351 
352  VelocityDesiredSet(&velocityDesired);
353 
354  // Indicate whether we are in radius of this endpoint
355  uint8_t path_status = PATHSTATUS_STATUS_INPROGRESS;
356  float distance2 = powf(northError, 2) + powf(eastError, 2);
357  if (distance2 < (guidanceSettings.EndpointRadius * guidanceSettings.EndpointRadius)) {
358  path_status = PATHSTATUS_STATUS_COMPLETED;
359  }
360  PathStatusStatusSet(&path_status);
361 }

370 static void updateGroundDesiredAttitude()
371 {
372  float dT = guidanceSettings.UpdatePeriod / 1000.0f;
373 
374  VelocityDesiredData velocityDesired;
375  VelocityActualData velocityActual;
376  StabilizationDesiredData stabDesired;
377  AttitudeActualData attitudeActual;
378  NedAccelData nedAccel;
379  GroundPathFollowerSettingsData guidanceSettings;
380  StabilizationSettingsData stabSettings;
381  SystemSettingsData systemSettings;
382 
383  SystemSettingsGet(&systemSettings);
384  GroundPathFollowerSettingsGet(&guidanceSettings);
385  VelocityActualGet(&velocityActual);
386  VelocityDesiredGet(&velocityDesired);
387  StabilizationDesiredGet(&stabDesired);
388  VelocityDesiredGet(&velocityDesired);
389  AttitudeActualGet(&attitudeActual);
390  StabilizationSettingsGet(&stabSettings);
391  NedAccelGet(&nedAccel);
392 
393  float northVel = velocityActual.North;
394  float eastVel = velocityActual.East;
395 
396  // Calculate direction from velocityDesired and set stabDesired.Yaw
397  stabDesired.Yaw = atan2f( velocityDesired.East, velocityDesired.North ) * RAD2DEG;
398 
399  // Calculate throttle and set stabDesired.Throttle
400  float velDesired = sqrtf(powf(velocityDesired.East,2) + powf(velocityDesired.North,2));
401  float velActual = sqrtf(powf(eastVel,2) + powf(northVel,2));
402  ManualControlCommandData manualControlData;
403  ManualControlCommandGet(&manualControlData);
404  switch (guidanceSettings.ThrottleControl) {
405  case GROUNDPATHFOLLOWERSETTINGS_THROTTLECONTROL_MANUAL:
406  {
407  stabDesired.Throttle = manualControlData.Throttle;
408  break;
409  }
410  case GROUNDPATHFOLLOWERSETTINGS_THROTTLECONTROL_PROPORTIONAL:
411  {
412  float velRatio = velDesired / guidanceSettings.HorizontalVelMax;
413  stabDesired.Throttle = guidanceSettings.MaxThrottle * velRatio;
414  if (guidanceSettings.ManualOverride == GROUNDPATHFOLLOWERSETTINGS_MANUALOVERRIDE_TRUE) {
415  stabDesired.Throttle = stabDesired.Throttle * manualControlData.Throttle;
416  }
417  break;
418  }
419  case GROUNDPATHFOLLOWERSETTINGS_THROTTLECONTROL_AUTO:
420  {
421  float velError = velDesired - velActual;
422  stabDesired.Throttle = pid_apply(&ground_pids[VELOCITY], velError, dT) + velDesired * guidanceSettings.VelocityFeedforward;
423  if (guidanceSettings.ManualOverride == GROUNDPATHFOLLOWERSETTINGS_MANUALOVERRIDE_TRUE) {
424  stabDesired.Throttle = stabDesired.Throttle * manualControlData.Throttle;
425  }
426  break;
427  }
428  default:
429  {
430  PIOS_Assert(0);
431  break;
432  }
433  }
434 
435  // Limit throttle as per settings
436  stabDesired.Throttle = bound_min_max(stabDesired.Throttle, 0, guidanceSettings.MaxThrottle);
437 
438  // Set StabilizationDesired object
439  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
440  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
441  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
442  StabilizationDesiredSet(&stabDesired);
443 }

448 static void updateNedAccel()
449 {
450  float accel[3];
451  float q[4];
452  float Rbe[3][3];
453  float accel_ned[3];
454 
455  // Collect downsampled attitude data
456  AccelsData accels;
457  AccelsGet(&accels);
458  accel[0] = accels.x;
459  accel[1] = accels.y;
460  accel[2] = accels.z;
461 
462  //rotate avg accels into earth frame and store it
463  AttitudeActualData attitudeActual;
464  AttitudeActualGet(&attitudeActual);
465  q[0]=attitudeActual.q1;
466  q[1]=attitudeActual.q2;
467  q[2]=attitudeActual.q3;
468  q[3]=attitudeActual.q4;
469  Quaternion2R(q, Rbe);
470  for (uint8_t i=0; i<3; i++){
471  accel_ned[i]=0;
472  for (uint8_t j=0; j<3; j++)
473  accel_ned[i] += Rbe[j][i]*accel[j];
474  }
475  accel_ned[2] += GRAVITY;
476 
477  NedAccelData accelData;
478  NedAccelGet(&accelData);
479  accelData.North = accel_ned[0];
480  accelData.East = accel_ned[1];
481  accelData.Down = accel_ned[2];
482  NedAccelSet(&accelData);
483 }
