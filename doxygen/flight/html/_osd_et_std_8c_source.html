320 static void UpdateConfig(void)
321 {
322  uint8_t buf[120];
323  uint32_t addr = 0;
324  uint32_t n;
325  FILEINFO file;
326  uint32_t res;
327 
328  // Try to open the file that contains a new config
329  res = DFS_OpenFile(&PIOS_SDCARD_VolInfo, (uint8_t *) UpdateConfFilePath, DFS_READ, PIOS_SDCARD_Sector, &file);
330  if (res == DFS_OK) {
331  uint32_t bytesRead;
332  bool ok = TRUE;
333 
334  DEBUG_MSG("Updating Config ");
335 
336  // Write the config-data in blocks to OSD
337  while (addr < CONFIG_LENGTH && ok) {
338  n = MIN(CONFIG_LENGTH - addr, sizeof(buf));
339  res = DFS_ReadFile(&file, PIOS_SDCARD_Sector, buf, &bytesRead, n);
340  if (res == DFS_OK && bytesRead == n) {
341  ok = Write(addr, n, buf);
342  if (ok) {
343  //DEBUG_MSG(" %d %d\n", addr, n);
344  DEBUG_MSG(".");
345  addr += n;
346  }
347  } else {
348  DEBUG_MSG(" FILEREAD FAILED ");
349  }
350  }
351 
352  DEBUG_MSG(ok ? " OK\n" : "FAILURE\n");
353 
354  // If writing is OK, read the data back and verify
355  if (ok) {
356  DEBUG_MSG("Verify Config ");
357  DFS_Seek(&file, 0, PIOS_SDCARD_Sector);
358 
359  addr = 0;
360  while (addr < CONFIG_LENGTH && ok) {
361  // First half of the buffer is used to store the data read from the OSD, the second half will contain the data from the file
362  n = MIN(CONFIG_LENGTH - addr, sizeof(buf) / 2);
363  ok = Read(addr, n, buf);
364  if (ok) {
365  uint32_t bytesRead;
366  res = DFS_ReadFile(&file, PIOS_SDCARD_Sector, buf + sizeof(buf) / 2, &bytesRead, n);
367  if (res == DFS_OK && bytesRead == n) {
368  DEBUG_MSG(".");
369  addr += n;
370 
371  if (memcmp(buf, buf + sizeof(buf) / 2, n) != 0) {
372  DEBUG_MSG(" MISMATCH ");
373  ok = FALSE;
374  }
375  }
376  }
377  }
378  DEBUG_MSG(ok ? " OK\n" : "FAILURE\n");
379  }
380 
381  DFS_Close(&file);
382 
383  // When the config was updated correctly, remove the config-file
384  if (ok) {
385  DFS_UnlinkFile(&PIOS_SDCARD_VolInfo, (uint8_t *) UpdateConfFilePath, PIOS_SDCARD_Sector);
386  }
387 
388  }
389 }

429 static void Run(void)
430 {
431  static uint32_t cnt = 0;
432 
433  if (newBattData) {
434  FlightBatteryStateData flightBatteryData;
435 
436  FlightBatteryStateGet(&flightBatteryData);
437 
438  //DEBUG_MSG("%5d Batt: V=%dmV\n\r", cnt, (uint32_t)(flightBatteryData.Voltage*1000));
439 
440  SetVoltage((uint32_t) (flightBatteryData.Voltage * 1000));
441  SetCurrent((uint32_t) (flightBatteryData.Current * 1000));
442  newBattData = FALSE;
443  }
444 
445  if (newPosData) {
446  GPSPositionData positionData;
447  AttitudeActualData attitudeActualData;
448 
449  GPSPositionGet(&positionData);
450  AttitudeActualGet(&attitudeActualData);
451 
452  //DEBUG_MSG("%5d Pos: #stat=%d #sats=%d alt=%d\n\r", cnt,
453  // positionData.Status, positionData.Satellites, (uint32_t)positionData.Altitude);
454 
455  // GPS Status
456  if (positionData.Status == GPSPOSITION_STATUS_FIX3D)
457  msg[OSDMSG_GPS_STAT] = OSDMSG_GPS_STAT_FIX;
458  else
459  msg[OSDMSG_GPS_STAT] = OSDMSG_GPS_STAT_NOFIX;
460  msg[OSDMSG_GPS_STAT] |= OSDMSG_GPS_STAT_HB_FLAG;
461 
462  // GPS info
463  SetCoord(OSDMSG_LAT_IDX, positionData.Latitude);
464  SetCoord(OSDMSG_LON_IDX, positionData.Longitude);
465  SetAltitude(positionData.Altitude);
466  SetNbSats(positionData.Satellites);
467  SetCourse(attitudeActualData.Yaw);
468 
469  newPosData = FALSE;
470  } else {
471  msg[OSDMSG_GPS_STAT] &= ~OSDMSG_GPS_STAT_HB_FLAG;
472  }
473  if (newBaroData) {
474  BaroAltitudeData baroData;
475 
476  BaroAltitudeGet(&baroData);
477  SetBaroAltitude(baroData.Altitude);
478 
479  newBaroData = FALSE;
480  }
481 
482  DEBUG_MSG("SendMsg %d\n",cnt);
483  {
484  DebugPinHigh(DEBUG_PIN_I2C);
485  const struct pios_i2c_txn txn_list[] = {
486  {
487  .addr = OSD_ADDRESS,
488  .rw = PIOS_I2C_TXN_WRITE,
489  .len = sizeof(msg),
490  .buf = msg,
491  }
492  ,
493  };
494 
495  PIOS_I2C_Transfer(PIOS_I2C_MAIN_ADAPTER, txn_list, NELEMENTS(txn_list));
496  DebugPinLow(DEBUG_PIN_I2C);
497  }
498 
499  cnt++;
500 }

502 static void onTimer(UAVObjEvent * ev)
503 {
504  DebugPinHigh(DEBUG_PIN_RUNNING);
505 
506  #ifdef ENABLE_DEBUG_MSG
507  PIOS_COM_ChangeBaud(DEBUG_PORT, 57600);
508  #endif
509 
510  if (state == STATE_DETECT) {
511  version = ReadSwVersion();
512  DEBUG_MSG("SW: %d ", version);
513 
514  if (version == SUPPORTED_VERSION) {
515  DEBUG_MSG("OK\n");
516  state++;
517  } else {
518  DEBUG_MSG("INVALID\n");
519  }
520  } else if (state == STATE_UPDATE_CONF) {
521  UpdateConfig();
522  state++;
523  } else if (state == STATE_DUMP_CONF) {
524  DumpConfig();
525  state++;
526  } else if (state == STTE_RUNNING) {
527  Run();
528  } else {
529  // should not happen..
530  state = STATE_DETECT;
531  }
532 
533  DebugPinLow(DEBUG_PIN_RUNNING);
534 
535 }
