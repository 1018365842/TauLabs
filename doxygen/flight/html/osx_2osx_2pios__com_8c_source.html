117 int32_t PIOS_COM_Init(uint32_t * com_id, const struct pios_com_driver * driver, uint32_t lower_id, uint8_t * rx_buffer, uint16_t rx_buffer_len, uint8_t * tx_buffer, uint16_t tx_buffer_len)
118 {
119  PIOS_Assert(com_id);
120  PIOS_Assert(driver);
121 
122  bool has_rx = (rx_buffer && rx_buffer_len > 0);
123  bool has_tx = (tx_buffer && tx_buffer_len > 0);
124  PIOS_Assert(has_rx || has_tx);
125  PIOS_Assert(driver->bind_tx_cb || !has_tx);
126  PIOS_Assert(driver->bind_rx_cb || !has_rx);
127 
128  uint32_t com_dev_id;
129  struct pios_com_dev * com_dev;
130 
131  com_dev_id = PIOS_COM_create();
132  com_dev = PIOS_COM_find_dev(com_dev_id);
133  if (!com_dev) goto out_fail;
134 
135  com_dev->driver = driver;
136  com_dev->lower_id = lower_id;
137 
138  com_dev->has_rx = has_rx;
139  com_dev->has_tx = has_tx;
140 
141  if (has_rx) {
142  fifoBuf_init(&com_dev->rx, rx_buffer, rx_buffer_len);
143 #if defined(PIOS_INCLUDE_FREERTOS)
144  vSemaphoreCreateBinary(com_dev->rx_sem);
145 #endif /* PIOS_INCLUDE_FREERTOS */
146  (com_dev->driver->bind_rx_cb)(lower_id, PIOS_COM_RxInCallback, com_dev_id);
147  if (com_dev->driver->rx_start) {
148  /* Start the receiver */
149  (com_dev->driver->rx_start)(com_dev->lower_id,
150  fifoBuf_getFree(&com_dev->rx));
151  }
152  }
153 
154  if (has_tx) {
155  fifoBuf_init(&com_dev->tx, tx_buffer, tx_buffer_len);
156 #if defined(PIOS_INCLUDE_FREERTOS)
157  vSemaphoreCreateBinary(com_dev->tx_sem);
158 #endif /* PIOS_INCLUDE_FREERTOS */
159  (com_dev->driver->bind_tx_cb)(lower_id, PIOS_COM_TxOutCallback, com_dev_id);
160  }
161 
162  *com_id = com_dev_id;
163  return(0);
164 
165 out_fail:
166  return(-1);
167 }

201 static uint16_t PIOS_COM_RxInCallback(uint32_t context, uint8_t * buf, uint16_t buf_len, uint16_t * headroom, bool * need_yield)
202 {
203  struct pios_com_dev * com_dev = PIOS_COM_find_dev(context);
204 
205  bool valid = PIOS_COM_validate(com_dev);
206  PIOS_Assert(valid);
207  PIOS_Assert(com_dev->has_rx);
208 
209  PIOS_IRQ_Disable();
210  uint16_t bytes_into_fifo = fifoBuf_putData(&com_dev->rx, buf, buf_len);
211  PIOS_IRQ_Enable();
212 
213  if (bytes_into_fifo > 0) {
214  /* Data has been added to the buffer */
215  PIOS_COM_UnblockRx(com_dev, need_yield);
216  }
217 
218  if (headroom) {
219  *headroom = fifoBuf_getFree(&com_dev->rx);
220  }
221 
222  return (bytes_into_fifo);
223 }

285 int32_t PIOS_COM_SendBufferNonBlocking(uint32_t com_id, const uint8_t *buffer, uint16_t len)
286 {
287  struct pios_com_dev * com_dev = PIOS_COM_find_dev(com_id);
288 
289  if (!PIOS_COM_validate(com_dev)) {
290  /* Undefined COM port for this board (see pios_board.c) */
291  return -1;
292  }
293 
294  PIOS_Assert(com_dev->has_tx);
295 
296  if (len >= fifoBuf_getFree(&com_dev->tx)) {
297  /* Buffer cannot accept all requested bytes (retry) */
298  return -2;
299  }
300 
301  PIOS_IRQ_Disable();
302  uint16_t bytes_into_fifo = fifoBuf_putData(&com_dev->tx, buffer, len);
303  PIOS_IRQ_Enable();
304 
305  if (bytes_into_fifo > 0) {
306  /* More data has been put in the tx buffer, make sure the tx is started */
307  if (com_dev->driver->tx_start) {
308  com_dev->driver->tx_start(com_dev->lower_id,
309  fifoBuf_getUsed(&com_dev->tx));
310  }
311  }
312 
313  return (bytes_into_fifo);
314 }

325 int32_t PIOS_COM_SendBuffer(uint32_t com_id, const uint8_t *buffer, uint16_t len)
326 {
327  struct pios_com_dev * com_dev = PIOS_COM_find_dev(com_id);
328 
329  if (!PIOS_COM_validate(com_dev)) {
330  /* Undefined COM port for this board (see pios_board.c) */
331  return -1;
332  }
333 
334  PIOS_Assert(com_dev->has_tx);
335 
336  int32_t rc;
337  do {
338  rc = PIOS_COM_SendBufferNonBlocking(com_id, buffer, len);
339 
340 #if defined(PIOS_INCLUDE_FREERTOS)
341  if (rc == -2) {
342  /* Make sure the transmitter is running while we wait */
343  if (com_dev->driver->tx_start) {
344  (com_dev->driver->tx_start)(com_dev->lower_id,
345  fifoBuf_getUsed(&com_dev->tx));
346  }
347  if (xSemaphoreTake(com_dev->tx_sem, portMAX_DELAY) != pdTRUE) {
348  return -3;
349  }
350  }
351 #endif
352  } while (rc == -2);
353 
354  return len;
355 }

456 uint16_t PIOS_COM_ReceiveBuffer(uint32_t com_id, uint8_t * buf, uint16_t buf_len, uint32_t timeout_ms)
457 {
458  PIOS_Assert(buf);
459  PIOS_Assert(buf_len);
460 
461  struct pios_com_dev * com_dev = PIOS_COM_find_dev(com_id);
462 
463  if (!PIOS_COM_validate(com_dev)) {
464  /* Undefined COM port for this board (see pios_board.c) */
465  PIOS_Assert(0);
466  }
467  PIOS_Assert(com_dev->has_rx);
468 
469  check_again:
470  PIOS_IRQ_Disable();
471  uint16_t bytes_from_fifo = fifoBuf_getData(&com_dev->rx, buf, buf_len);
472  PIOS_IRQ_Enable();
473 
474  if (bytes_from_fifo == 0 && timeout_ms > 0) {
475  /* No more bytes in receive buffer */
476  /* Make sure the receiver is running while we wait */
477  if (com_dev->driver->rx_start) {
478  /* Notify the lower layer that there is now room in the rx buffer */
479  (com_dev->driver->rx_start)(com_dev->lower_id,
480  fifoBuf_getFree(&com_dev->rx));
481  }
482 #if defined(PIOS_INCLUDE_FREERTOS)
483  if (xSemaphoreTake(com_dev->rx_sem, timeout_ms / portTICK_RATE_MS) == pdTRUE) {
484  /* Make sure we don't come back here again */
485  timeout_ms = 0;
486  goto check_again;
487  }
488 #else
489  PIOS_DELAY_WaitmS(1);
490  timeout_ms--;
491  goto check_again;
492 #endif
493  }
494 
495  /* Return received byte */
496  return (bytes_from_fifo);
497 }
