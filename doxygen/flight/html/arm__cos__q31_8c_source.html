139 q31_t arm_cos_q31(
140  q31_t x)
141 {
142  q31_t cosVal, in, in2; /* Temporary variables for input, output */
143  q31_t wa, wb, wc, wd; /* Cubic interpolation coefficients */
144  q31_t a, b, c, d; /* Four nearest output values */
145  q31_t *tablePtr; /* Pointer to table */
146  q31_t fract, fractCube, fractSquare; /* Temporary values for fractional values */
147  q31_t oneBy6 = 0x15555555; /* Fixed point value of 1/6 */
148  q31_t tableSpacing = TABLE_SPACING_Q31; /* Table spacing */
149  q31_t temp; /* Temporary variable for intermediate process */
150  int32_t index; /* Index variable */
151 
152  in = x;
153 
154  /* Calculate the nearest index */
155  index = in / tableSpacing;
156 
157  /* Calculate the nearest value of input */
158  in2 = ((q31_t) index) * tableSpacing;
159 
160  /* Calculation of fractional value */
161  fract = (in - in2) << 8;
162 
163  /* fractSquare = fract * fract */
164  fractSquare = ((q31_t) (((q63_t) fract * fract) >> 32));
165  fractSquare = fractSquare << 1;
166 
167  /* fractCube = fract * fract * fract */
168  fractCube = ((q31_t) (((q63_t) fractSquare * fract) >> 32));
169  fractCube = fractCube << 1;
170 
171  /* Checking min and max index of table */
172  if(index < 0)
173  {
174  index = 0;
175  }
176  else if(index > 256)
177  {
178  index = 256;
179  }
180 
181  /* Initialise table pointer */
182  tablePtr = (q31_t *) & cosTableQ31[index];
183 
184  /* Cubic interpolation process */
185  /* Calculation of wa */
186  /* wa = -(oneBy6)*fractCube + (fractSquare >> 1u) - (0x2AAAAAAA)*fract; */
187  wa = ((q31_t) (((q63_t) oneBy6 * fractCube) >> 32));
188  temp = 0x2AAAAAAA;
189  wa = (q31_t) ((((q63_t) wa << 32) + ((q63_t) temp * fract)) >> 32);
190  wa = -(wa << 1u);
191  wa += (fractSquare >> 1u);
192 
193  /* Read first nearest value of output from the cos table */
194  a = *tablePtr++;
195 
196  /* cosVal = a*wa */
197  cosVal = ((q31_t) (((q63_t) a * wa) >> 32));
198 
199  /* q31(1.31) Fixed point value of 1 */
200  temp = 0x7FFFFFFF;
201 
202  /* Calculation of wb */
203  wb = ((fractCube >> 1u) - (fractSquare + (fract >> 1u))) + temp;
204  /* Read second nearest value of output from the cos table */
205  b = *tablePtr++;
206 
207  /* cosVal += b*wb */
208  cosVal = (q31_t) ((((q63_t) cosVal << 32) + ((q63_t) b * (wb))) >> 32);
209 
210  /* Calculation of wc */
211  wc = -fractCube + fractSquare;
212  wc = (wc >> 1u) + fract;
213  /* Read third nearest values of output value from the cos table */
214  c = *tablePtr++;
215 
216  /* cosVal += c*wc */
217  cosVal = (q31_t) ((((q63_t) cosVal << 32) + ((q63_t) c * (wc))) >> 32);
218 
219  /* Calculation of wd */
220  /* wd = (oneBy6)*fractCube - (oneBy6)*fract; */
221  fractCube = fractCube - fract;
222  wd = ((q31_t) (((q63_t) oneBy6 * fractCube) >> 32));
223  wd = (wd << 1u);
224 
225  /* Read fourth nearest value of output from the cos table */
226  d = *tablePtr++;
227 
228  /* cosVal += d*wd; */
229  cosVal = (q31_t) ((((q63_t) cosVal << 32) + ((q63_t) d * (wd))) >> 32);
230 
231 
232  /* convert cosVal in 2.30 format to 1.31 format */
233  return (__QADD(cosVal, cosVal));
234 
235 }
