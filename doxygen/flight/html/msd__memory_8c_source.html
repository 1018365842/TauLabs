80 /*******************************************************************************
81 * Function Name : Read_Memory
82 * Description : Handle the Read operation from the microSD card.
83 * Input : None.
84 * Output : None.
85 * Return : None.
86 *******************************************************************************/
87 void MSD_Read_Memory(uint8_t lun, uint32_t Memory_Offset, uint32_t Transfer_Length)
88 {
89  static uint32_t Offset, Length;
90 
91  if (TransferState == TXFR_IDLE )
92  {
93  Offset = Memory_Offset * MSD_Mass_Block_Size[lun];
94  Length = Transfer_Length * MSD_Mass_Block_Size[lun];
95  TransferState = TXFR_ONGOING;
96  }
97 
98  if (TransferState == TXFR_ONGOING )
99  {
100  if (!Block_Read_count)
101  {
102  s32 status;
103 
104  switch( lun ) {
105  case 0:
106  if( MSD_Mass_Block_Size[lun] != 512 )
107  break;
108  status = PIOS_SDCARD_SectorRead(Offset/512, (u8 *)Data_Buffer);
109  // TK: how to handle an error here?
110  break;
111 
112  default:
113  status = -1;
114  // TK: how to handle an error here?
115  }
116 
117  UserToPMABufferCopy((uint8_t *)Data_Buffer, MSD_ENDP1_TXADDR, MSD_BULK_MAX_PACKET_SIZE);
118  Block_Read_count = MSD_Mass_Block_Size[lun] - MSD_BULK_MAX_PACKET_SIZE;
119  Block_offset = MSD_BULK_MAX_PACKET_SIZE;
120  }
121  else
122  {
123  UserToPMABufferCopy((uint8_t *)Data_Buffer + Block_offset, MSD_ENDP1_TXADDR, MSD_BULK_MAX_PACKET_SIZE);
124  Block_Read_count -= MSD_BULK_MAX_PACKET_SIZE;
125  Block_offset += MSD_BULK_MAX_PACKET_SIZE;
126  }
127 
128  SetEPTxCount(ENDP1, MSD_BULK_MAX_PACKET_SIZE);
129  SetEPTxStatus(ENDP1, EP_TX_VALID);
130  Offset += MSD_BULK_MAX_PACKET_SIZE;
131  Length -= MSD_BULK_MAX_PACKET_SIZE;
132 
133  MSD_CSW.dDataResidue -= MSD_BULK_MAX_PACKET_SIZE;
134  }
135  if (Length == 0)
136  {
137  Block_Read_count = 0;
138  Block_offset = 0;
139  Offset = 0;
140  MSD_Bot_State = BOT_DATA_IN_LAST;
141  TransferState = TXFR_IDLE;
142  }
143 }

145 /*******************************************************************************
146 * Function Name : Write_Memory
147 * Description : Handle the Write operation to the microSD card.
148 * Input : None.
149 * Output : None.
150 * Return : None.
151 *******************************************************************************/
152 void MSD_Write_Memory (uint8_t lun, uint32_t Memory_Offset, uint32_t Transfer_Length)
153 {
154 
155  static uint32_t W_Offset, W_Length;
156 
157  uint32_t temp = Counter + 64;
158 
159  if (TransferState == TXFR_IDLE )
160  {
161  W_Offset = Memory_Offset * MSD_Mass_Block_Size[lun];
162  W_Length = Transfer_Length * MSD_Mass_Block_Size[lun];
163  TransferState = TXFR_ONGOING;
164  }
165 
166  if (TransferState == TXFR_ONGOING )
167  {
168 
169  for (Idx = 0 ; Counter < temp; Counter++)
170  {
171  *((uint8_t *)Data_Buffer + Counter) = MSD_Bulk_Data_Buff[Idx++];
172  }
173 
174  W_Offset += MSD_Data_Len;
175  W_Length -= MSD_Data_Len;
176 
177  if (!(W_Length % MSD_Mass_Block_Size[lun]))
178  {
179  Counter = 0;
180 
181  s32 status;
182  u32 Offset = W_Offset - MSD_Mass_Block_Size[lun];
183 
184  switch( lun ) {
185  case 0:
186  if( MSD_Mass_Block_Size[lun] != 512 )
187  break;
188  status = PIOS_SDCARD_SectorWrite(Offset/512, (u8 *)Data_Buffer);
189  // TK: how to handle an error here?
190  break;
191 
192  default:
193  status = -1;
194  // TK: how to handle an error here?
195  }
196  }
197 
198  MSD_CSW.dDataResidue -= MSD_Data_Len;
199  SetEPRxStatus(ENDP2, EP_RX_VALID); /* enable the next transaction*/
200  }
201 
202  if ((W_Length == 0) || (MSD_Bot_State == BOT_CSW_Send))
203  {
204  Counter = 0;
205  MSD_Set_CSW (CSW_CMD_PASSED, SEND_CSW_ENABLE);
206  TransferState = TXFR_IDLE;
207  }
208 }

211 /*******************************************************************************
212 * Function Name : MAL_GetStatus
213 * Description : Get status
214 * Input : None
215 * Output : None
216 * Return : None
217 *******************************************************************************/
218 uint16_t MSD_MAL_GetStatus (uint8_t lun)
219 {
220  // LUN removed or disabled?
221  if( !MSD_LUN_AvailableGet(lun) )
222  return 1;
223 
224  if( lun == 0 ) {
225  SDCARDCsdTypeDef csd;
226  if( PIOS_SDCARD_CSDRead(&csd) < 0 )
227  return 1;
228 
229  u32 DeviceSizeMul = csd.DeviceSizeMul + 2;
230  u32 temp_block_mul = (1 << csd.RdBlockLen)/ 512;
231  MSD_Mass_Block_Count[lun] = ((csd.DeviceSize + 1) * (1 << (DeviceSizeMul))) * temp_block_mul;
232  MSD_Mass_Block_Size[lun] = 512;
233  MSD_Mass_Memory_Size[lun] = (MSD_Mass_Block_Count[lun] * MSD_Mass_Block_Size[0]);
234 
235  return 0;
236  }
237 
238  return 1;
239 }
