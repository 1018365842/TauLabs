200  const struct pios_com_driver *com_driver, uintptr_t *pios_com_id)

213  }
214

216  if (tx_buf_len > 0) {
217  tx_buffer = (uint8_t *) pvPortMalloc(tx_buf_len);
218  PIOS_Assert(tx_buffer);
219  } else {
220  tx_buffer = NULL;
221  }
222 
223  if (PIOS_COM_Init(pios_com_id, com_driver, pios_usart_id,
224  rx_buffer, rx_buf_len,
225  tx_buffer, tx_buf_len)) {
226  PIOS_Assert(0);
227  }
228 }

264 static void panic(int32_t code) {
265  while(1){
266  for (int32_t i = 0; i < code; i++) {
267  PIOS_WDG_Clear();
268  PIOS_LED_Toggle(PIOS_LED_ALARM);
269  PIOS_DELAY_WaitmS(200);
270  PIOS_WDG_Clear();
271  PIOS_LED_Toggle(PIOS_LED_ALARM);
272  PIOS_DELAY_WaitmS(200);
273  }
274  PIOS_DELAY_WaitmS(200);
275  PIOS_WDG_Clear();
276  PIOS_DELAY_WaitmS(200);
277  PIOS_WDG_Clear();
278  PIOS_DELAY_WaitmS(100);
279  PIOS_WDG_Clear();
280  }

291 void PIOS_Board_Init(void) {
292 
293  /* Delay system */
294  PIOS_DELAY_Init();
295 
296  const struct pios_board_info * bdinfo = &pios_board_info_blob;
297 
298 #if defined(PIOS_INCLUDE_LED)
299  const struct pios_led_cfg * led_cfg = PIOS_BOARD_HW_DEFS_GetLedCfg(bdinfo->board_rev);
300  PIOS_Assert(led_cfg);
301  PIOS_LED_Init(led_cfg);
302 #endif /* PIOS_INCLUDE_LED */
303 
304 #if defined(PIOS_INCLUDE_FLASH)
305  if (PIOS_SPI_Init(&pios_spi_flash_id, &pios_spi_flash_cfg)) {
306  PIOS_DEBUG_Assert(0);
307  }
308 
309  /* Connect flash to the approrpiate interface and configure it */
310  uintptr_t flash_id;
311  if (PIOS_Flash_Jedec_Init(&flash_id, pios_spi_flash_id, 0, &flash_m25p_cfg) != 0)
312  panic(1);
313  if (PIOS_FLASHFS_Logfs_Init(&pios_uavo_settings_fs_id, &flashfs_m25p_settings_cfg, &pios_jedec_flash_driver, flash_id) != 0)
314  panic(1);
315  if (PIOS_FLASHFS_Logfs_Init(&pios_waypoints_settings_fs_id, &flashfs_m25p_waypoints_cfg, &pios_jedec_flash_driver, flash_id) != 0)
316  panic(1);
317 #endif
318 
319  /* Initialize UAVObject libraries */
320  EventDispatcherInitialize();
321  UAVObjInitialize();
322 
323  HwFlyingF4Initialize();
324  ModuleSettingsInitialize();
325 
326 #if defined(PIOS_INCLUDE_RTC)
327  /* Initialize the real-time clock and its associated tick */
328  PIOS_RTC_Init(&pios_rtc_main_cfg);
329 #endif
330 
331 #ifndef ERASE_FLASH
332  /* Initialize watchdog as early as possible to catch faults during init
333  * but do it only if there is no debugger connected
334  */
335  if ((CoreDebug->DHCSR & CoreDebug_DHCSR_C_DEBUGEN_Msk) == 0) {
336  PIOS_WDG_Init();
337  }
338 #endif
339 
340  /* Initialize the alarms library */
341  AlarmsInitialize();
342 
343  /* Initialize the task monitor library */
344  TaskMonitorInitialize();
345 
346  /* Set up pulse timers */
347  //inputs
348  PIOS_TIM_InitClock(&tim_2_cfg);
349  PIOS_TIM_InitClock(&tim_4_cfg);
350  PIOS_TIM_InitClock(&tim_8_cfg);
351  PIOS_TIM_InitClock(&tim_9_cfg);
352  //outputs
353  PIOS_TIM_InitClock(&tim_1_cfg);
354  PIOS_TIM_InitClock(&tim_3_cfg);
355 
356  /* IAP System Setup */
357  PIOS_IAP_Init();
358  uint16_t boot_count = PIOS_IAP_ReadBootCount();
359  if (boot_count < 3) {
360  PIOS_IAP_WriteBootCount(++boot_count);
361  AlarmsClear(SYSTEMALARMS_ALARM_BOOTFAULT);
362  } else {
363  /* Too many failed boot attempts, force hw config to defaults */
364  HwFlyingF4SetDefaults(HwFlyingF4Handle(), 0);
365  ModuleSettingsSetDefaults(ModuleSettingsHandle(),0);
366  AlarmsSet(SYSTEMALARMS_ALARM_BOOTFAULT, SYSTEMALARMS_ALARM_CRITICAL);
367  }
368 
369 #if defined(PIOS_INCLUDE_USB)
370  /* Initialize board specific USB data */
371  PIOS_USB_BOARD_DATA_Init();
372 
373  /* Flags to determine if various USB interfaces are advertised */
374  bool usb_hid_present = false;
375  bool usb_cdc_present = false;
376 
377 #if defined(PIOS_INCLUDE_USB_CDC)
378  if (PIOS_USB_DESC_HID_CDC_Init()) {
379  PIOS_Assert(0);
380  }
381  usb_hid_present = true;
382  usb_cdc_present = true;
383 #else
384  if (PIOS_USB_DESC_HID_ONLY_Init()) {
385  PIOS_Assert(0);
386  }
387  usb_hid_present = true;
388 #endif
389 
390  uint32_t pios_usb_id;
391  PIOS_USB_Init(&pios_usb_id, PIOS_BOARD_HW_DEFS_GetUsbCfg(bdinfo->board_rev));
392 
393 #if defined(PIOS_INCLUDE_USB_CDC)
394 
395  uint8_t hw_usb_vcpport;
396  /* Configure the USB VCP port */
397  HwFlyingF4USB_VCPPortGet(&hw_usb_vcpport);
398 
399  if (!usb_cdc_present) {
400  /* Force VCP port function to disabled if we haven't advertised VCP in our USB descriptor */
401  hw_usb_vcpport = HWFLYINGF4_USB_VCPPORT_DISABLED;
402  }
403 
404  uint32_t pios_usb_cdc_id;
405  if (PIOS_USB_CDC_Init(&pios_usb_cdc_id, &pios_usb_cdc_cfg, pios_usb_id)) {
406  PIOS_Assert(0);
407  }
408 
409  switch (hw_usb_vcpport) {
410  case HWFLYINGF4_USB_VCPPORT_DISABLED:
411  break;
412  case HWFLYINGF4_USB_VCPPORT_USBTELEMETRY:
413 #if defined(PIOS_INCLUDE_COM)
414  {
415  uint8_t * rx_buffer = (uint8_t *) pvPortMalloc(PIOS_COM_TELEM_USB_RX_BUF_LEN);
416  uint8_t * tx_buffer = (uint8_t *) pvPortMalloc(PIOS_COM_TELEM_USB_TX_BUF_LEN);

418  PIOS_Assert(tx_buffer);

420  rx_buffer, PIOS_COM_TELEM_USB_RX_BUF_LEN,
421  tx_buffer, PIOS_COM_TELEM_USB_TX_BUF_LEN)) {
422  PIOS_Assert(0);
423  }
424  }
425 #endif /* PIOS_INCLUDE_COM */
426  break;
427  case HWFLYINGF4_USB_VCPPORT_COMBRIDGE:
428 #if defined(PIOS_INCLUDE_COM)
429  {
430  uint8_t * rx_buffer = (uint8_t *) pvPortMalloc(PIOS_COM_BRIDGE_RX_BUF_LEN);
431  uint8_t * tx_buffer = (uint8_t *) pvPortMalloc(PIOS_COM_BRIDGE_TX_BUF_LEN);
432  PIOS_Assert(rx_buffer);
433  PIOS_Assert(tx_buffer);

438  }
439  }
440 #endif /* PIOS_INCLUDE_COM */

443 #if defined(PIOS_INCLUDE_COM)

494  }
495 
496 #endif /* PIOS_INCLUDE_USB_HID */

499  PIOS_USBHOOK_Activate();

501 #endif /* PIOS_INCLUDE_USB */
