59 static void PIOS_RFM22B_COM_ChangeBaud(uint32_t rfm22b_id, uint32_t baud)
60 {
61  struct pios_rfm22b_dev *rfm22b_dev = (struct pios_rfm22b_dev *)rfm22b_id;
62  if (!PIOS_RFM22B_validate(rfm22b_dev))
63  return;
64  // This is only allowed for coordinators.
65  if (!rfm22b_dev->coordinator)
66  return;
67  // Set the RF data rate on the modem to ~2X the selected buad rate because the modem is half duplex.
68  enum rfm22b_datarate datarate = RFM22_datarate_64000;
69  if (baud <= 1024)
70  datarate = RFM22_datarate_500;
71  else if (baud <= 2048)
72  datarate = RFM22_datarate_1000;
73  else if (baud <= 4096)
74  datarate = RFM22_datarate_8000;
75  else if (baud <= 9600)
76  datarate = RFM22_datarate_16000;
77  else if (baud <= 19200)
78  datarate = RFM22_datarate_32000;
79  else if (baud <= 38400)
80  datarate = RFM22_datarate_64000;
81  else if (baud <= 57600)
82  datarate = RFM22_datarate_128000;
83  else if (baud <= 115200)
84  datarate = RFM22_datarate_192000;
85  PIOS_RFM22B_SetDatarate(rfm22b_id, datarate, true);
86 }
