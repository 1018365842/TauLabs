50 int32_t InertialSensorSettingsInitialize(void)
51 {
52  // Don't set the handle to null if already registered
53  if(UAVObjGetByID(INERTIALSENSORSETTINGS_OBJID) != NULL)
54  return -2;
55 
56  // Register object with the object manager
57  handle = UAVObjRegister(INERTIALSENSORSETTINGS_OBJID,
58  INERTIALSENSORSETTINGS_ISSINGLEINST, INERTIALSENSORSETTINGS_ISSETTINGS, INERTIALSENSORSETTINGS_NUMBYTES, &InertialSensorSettingsSetDefaults);
59 
60  // Done
61  if (handle != 0)
62  {
63  return 0;
64  }
65  else
66  {
67  return -1;
68  }
