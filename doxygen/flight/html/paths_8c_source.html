56 void path_progress(PathDesiredData *pathDesired,
57  float *cur_point,
58  struct path_status *status)
59 {
60  uint8_t mode = pathDesired->Mode;
61  float start_point[2] = {pathDesired->Start[0],pathDesired->Start[1]};
62  float end_point[2] = {pathDesired->End[0],pathDesired->End[1]};
63 
64  switch(mode) {
65  case PATHDESIRED_MODE_FLYVECTOR:
66  case PATHDESIRED_MODE_DRIVEVECTOR:
67  return path_vector(start_point, end_point, cur_point, status);
68  break;
69  case PATHDESIRED_MODE_FLYCIRCLERIGHT:
70  case PATHDESIRED_MODE_DRIVECIRCLERIGHT:
71  return path_curve(start_point, end_point, pathDesired->ModeParameters, cur_point, status, 1);
72  break;
73  case PATHDESIRED_MODE_FLYCIRCLELEFT:
74  case PATHDESIRED_MODE_DRIVECIRCLELEFT:
75  return path_curve(start_point, end_point, pathDesired->ModeParameters, cur_point, status, 0);
76  break;
77  case PATHDESIRED_MODE_CIRCLEPOSITIONLEFT:
78  return path_circle(end_point, pathDesired->ModeParameters, cur_point, status, 0);
79  break;
80  case PATHDESIRED_MODE_CIRCLEPOSITIONRIGHT:
81  return path_circle(end_point, pathDesired->ModeParameters, cur_point, status, 1);
82  break;
83  case PATHDESIRED_MODE_FLYENDPOINT:
84  case PATHDESIRED_MODE_DRIVEENDPOINT:
85  default:
86  // use the endpoint as default failsafe if called in unknown modes
87  return path_endpoint(start_point, end_point, cur_point, status);
88  break;
89  }
90 }

99 static void path_endpoint(float *start_point,
100  float *end_point,
101  float *cur_point,
102  struct path_status *status)
103 {
104  float path_north, path_east, diff_north, diff_east;
105  float dist_path, dist_diff;
106 
107  // we do not correct in this mode
108  status->correction_direction[0] = status->correction_direction[1] = 0;
109 
110  // Distance to go
111  path_north = end_point[0] - start_point[0];
112  path_east = end_point[1] - start_point[1];
113 
114  // Current progress location relative to end
115  diff_north = end_point[0] - cur_point[0];
116  diff_east = end_point[1] - cur_point[1];
117 
118  dist_diff = sqrtf( diff_north * diff_north + diff_east * diff_east );
119  dist_path = sqrtf( path_north * path_north + path_east * path_east );
120 
121  if(dist_diff < 1e-6 ) {
122  status->fractional_progress = 1;
123  status->error = 0;
124  status->path_direction[0] = status->path_direction[1] = 0;
125  return;
126  }
127 
128  status->fractional_progress = 1 - dist_diff / (1 + dist_path);
129  status->error = dist_diff;
130 
131  // Compute direction to travel
132  status->path_direction[0] = diff_north / dist_diff;
133  status->path_direction[1] = diff_east / dist_diff;
134 
135 }

144 static void path_vector(float *start_point,
145  float *end_point,
146  float *cur_point,
147  struct path_status *status)
148 {
149  float path_north, path_east, diff_north, diff_east;
150  float dist_path;
151  float dot;
152  float normal[2];
153 
154  // Distance to go
155  path_north = end_point[0] - start_point[0];
156  path_east = end_point[1] - start_point[1];
157 
158  // Current progress location relative to start
159  diff_north = cur_point[0] - start_point[0];
160  diff_east = cur_point[1] - start_point[1];
161 
162  dot = path_north * diff_north + path_east * diff_east;
163  dist_path = sqrtf( path_north * path_north + path_east * path_east );
164 
165  if(dist_path < 1e-6) {
166  // if the path is too short, we cannot determine vector direction.
167  // Fly towards the endpoint to prevent flying away,
168  // but assume progress=1 either way.
169  path_endpoint( start_point, end_point, cur_point, status );
170  status->fractional_progress = 1;
171  return;
172  }
173 
174  // Compute the normal to the path
175  normal[0] = -path_east / dist_path;
176  normal[1] = path_north / dist_path;
177 
178  status->fractional_progress = dot / (dist_path * dist_path);
179  status->error = normal[0] * diff_north + normal[1] * diff_east;
180 
181  // Compute direction to correct error
182  status->correction_direction[0] = (status->error > 0) ? -normal[0] : normal[0];
183  status->correction_direction[1] = (status->error > 0) ? -normal[1] : normal[1];
184 
185  // Now just want magnitude of error
186  status->error = fabs(status->error);
187 
188  // Compute direction to travel
189  status->path_direction[0] = path_north / dist_path;
190  status->path_direction[1] = path_east / dist_path;
191 
192 }

201 static void path_circle(float * center_point,
202  float radius,
203  float * cur_point,
204  struct path_status * status,
205  bool clockwise)
206 {
207  float diff_north, diff_east;
208  float cradius;
209  float normal[2];
210 
211  // Current location relative to center
212  diff_north = cur_point[0] - center_point[0];
213  diff_east = cur_point[1] - center_point[1];
214 
215  cradius = sqrtf( diff_north * diff_north + diff_east * diff_east );
216 
217  if (cradius < 1e-6) {
218  // cradius is zero, just fly somewhere and make sure correction is still a normal
219  status->fractional_progress = 1;
220  status->error = radius;
221  status->correction_direction[0] = 0;
222  status->correction_direction[1] = 1;
223  status->path_direction[0] = 1;
224  status->path_direction[1] = 0;
225  return;
226  }
227 
228  if (clockwise) {
229  // Compute the normal to the radius clockwise
230  normal[0] = -diff_east / cradius;
231  normal[1] = diff_north / cradius;
232  } else {
233  // Compute the normal to the radius counter clockwise
234  normal[0] = diff_east / cradius;
235  normal[1] = -diff_north / cradius;
236  }
237 
238  status->fractional_progress = 0;
239 
240  // error is current radius minus wanted radius - positive if too close
241  status->error = radius - cradius;
242 
243  // Compute direction to correct error
244  status->correction_direction[0] = (status->error>0?1:-1) * diff_north / cradius;
245  status->correction_direction[1] = (status->error>0?1:-1) * diff_east / cradius;
246 
247  // Compute direction to travel
248  status->path_direction[0] = normal[0];
249  status->path_direction[1] = normal[1];
250 
251  status->error = fabs(status->error);
252 }

262 static void path_curve(float * start_point,
263  float * end_point,
264  float radius,
265  float * cur_point,
266  struct path_status *status,
267  bool clockwise)
268 {
269  float diff_north, diff_east;
270  float path_north, path_east;
271  float cradius;
272  float normal[2];
273 
274  // Compute the center of the circle connecting the two points as the intersection of two circles
275  // around the two points from
276  // http://www.mathworks.com/matlabcentral/newsreader/view_thread/255121
277  float m_n, m_e, p_n, p_e, d, center[2];
278 
279  // Center between start and end
280  m_n = (start_point[0] + end_point[0]) / 2;
281  m_e = (start_point[1] + end_point[1]) / 2;
282 
283  // Normal vector the line between start and end.
284  if (clockwise) {
285  p_n = -(end_point[1] - start_point[1]);
286  p_e = (end_point[0] - start_point[0]);
287  } else {
288  p_n = (end_point[1] - start_point[1]);
289  p_e = -(end_point[0] - start_point[0]);
290  }
291 
292  // Work out how far to go along the perpendicular bisector
293  d = sqrtf(radius * radius / (p_n * p_n + p_e * p_e) - 0.25f);
294 
295  float radius_sign = (radius > 0) ? 1 : -1;
296  radius = fabs(radius);
297 
298  if (fabs(p_n) < 1e-3 && fabs(p_e) < 1e-3) {
299  center[0] = m_n;
300  center[1] = m_e;
301  } else {
302  center[0] = m_n + p_n * d * radius_sign;
303  center[1] = m_e + p_e * d * radius_sign;
304  }
305 
306  // Current location relative to center
307  diff_north = cur_point[0] - center[0];
308  diff_east = cur_point[1] - center[1];
309 
310  // Compute current radius from the center
311  cradius = sqrtf( diff_north * diff_north + diff_east * diff_east );
312 
313  // Compute error in terms of meters from the curve (the distance projected
314  // normal onto the path i.e. cross-track distance)
315  status->error = radius - cradius;
316 
317  if (cradius < 1e-6) {
318  // cradius is zero, just fly somewhere and make sure correction is still a normal
319  status->fractional_progress = 1;
320  status->error = radius;
321  status->correction_direction[0] = 0;
322  status->correction_direction[1] = 1;
323  status->path_direction[0] = 1;
324  status->path_direction[1] = 0;
325  return;
326  }
327 
328  if (clockwise) {
329  // Compute the normal to the radius clockwise
330  normal[0] = -diff_east / cradius;
331  normal[1] = diff_north / cradius;
332  } else {
333  // Compute the normal to the radius counter clockwise
334  normal[0] = diff_east / cradius;
335  normal[1] = -diff_north / cradius;
336  }
337 
338  // Compute direction to correct error
339  status->correction_direction[0] = (status->error>0?1:-1) * diff_north / cradius;
340  status->correction_direction[1] = (status->error>0?1:-1) * diff_east / cradius;
341 
342  // Compute direction to travel
343  status->path_direction[0] = normal[0];
344  status->path_direction[1] = normal[1];
345 
346  path_north = end_point[0] - start_point[0];
347  path_east = end_point[1] - start_point[1];
348  diff_north = cur_point[0] - start_point[0];
349  diff_east = cur_point[1] - start_point[1];
350  float dist_path = sqrtf( path_north * path_north + path_east * path_east );
351  float dot = path_north * diff_north + path_east * diff_east;
352 
353  status->fractional_progress = dot / (dist_path * dist_path);
354 
355  status->error = fabs(status->error);
356 }
