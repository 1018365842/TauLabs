53 int32_t configuration_check()
54 {
55  SystemAlarmsConfigErrorOptions error_code = SYSTEMALARMS_CONFIGERROR_NONE;
56 
57  // Get board type
58  const struct pios_board_info * bdinfo = &pios_board_info_blob;
59  bool coptercontrol = bdinfo->board_type == 0x04;
60 
61  // For when modules are not running we should explicitly check the objects are
62  // valid
63  if (ManualControlSettingsHandle() == NULL ||
64  SystemSettingsHandle() == NULL) {
65  AlarmsSet(SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION, SYSTEMALARMS_ALARM_CRITICAL);
66  return 0;
67  }
68 
69  // Classify airframe type
70  bool multirotor = true;
71  uint8_t airframe_type;
72  SystemSettingsAirframeTypeGet(&airframe_type);
73  switch(airframe_type) {
74  case SYSTEMSETTINGS_AIRFRAMETYPE_QUADX:
75  case SYSTEMSETTINGS_AIRFRAMETYPE_QUADP:
76  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXA:
77  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTO:
78  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXAX:
79  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTOV:
80  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXP:
81  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXACOAX:
82  case SYSTEMSETTINGS_AIRFRAMETYPE_TRI:
83  multirotor = true;
84  break;
85  default:
86  multirotor = false;
87  }
88 
89  // For each available flight mode position sanity check the available
90  // modes
91  uint8_t num_modes;
92  uint8_t modes[MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_NUMELEM];
93  ManualControlSettingsFlightModeNumberGet(&num_modes);
94  ManualControlSettingsFlightModePositionGet(modes);
95 
96 
97  for(uint32_t i = 0; i < num_modes; i++) {
98  switch(modes[i]) {
99  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_MANUAL:
100  if (multirotor) {
101  error_code = SYSTEMALARMS_CONFIGERROR_STABILIZATION;
102  }
103  break;
104  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED1:
105  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(1, multirotor) : error_code;
106  break;
107  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED2:
108  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(2, multirotor) : error_code;
109  break;
110  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED3:
111  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(3, multirotor) : error_code;
112  break;
113  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_AUTOTUNE:
114  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE))
115  error_code = SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
116  break;
117  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_ALTITUDEHOLD:
118  if (coptercontrol)
119  error_code = SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD;
120  else {
121  if ( !TaskMonitorQueryRunning(TASKINFO_RUNNING_ALTITUDEHOLD) )
122  error_code = SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD;
123  }
124  break;
125  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_VELOCITYCONTROL:
126  if (coptercontrol) {
127  error_code = SYSTEMALARMS_CONFIGERROR_VELOCITYCONTROL;
128  }
129  else {
130  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER)) {
131  error_code = SYSTEMALARMS_CONFIGERROR_VELOCITYCONTROL;
132  }
133  }
134  break;
135  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_POSITIONHOLD:
136  if (coptercontrol) {
137  error_code = SYSTEMALARMS_CONFIGERROR_POSITIONHOLD;
138  }
139  else {
140  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER)) {
141  error_code = SYSTEMALARMS_CONFIGERROR_POSITIONHOLD;
142  }
143  }
144  break;
145  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_PATHPLANNER:
146  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_TABLETCONTROL:
147  if (coptercontrol) {
148  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
149  }
150  else {
151  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER) ||
152  !TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHPLANNER)) {
153  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
154  }
155  }
156  break;
157  default:
158  // Uncovered modes are automatically an error
159  error_code = SYSTEMALARMS_CONFIGERROR_UNDEFINED;
160  }
161  }
162 
163  set_config_error(error_code);
164 
165  return 0;
166 }

175 static int32_t check_stabilization_settings(int index, bool multirotor)
176 {
177  // Make sure the modes have identical sizes
178  if (MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION2SETTINGS_NUMELEM ||
179  MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION3SETTINGS_NUMELEM)
180  return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
181 
182  uint8_t modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM];
183 
184  // Get the different axis modes for this switch position
185  switch(index) {
186  case 1:
187  ManualControlSettingsStabilization1SettingsGet(modes);
188  break;
189  case 2:
190  ManualControlSettingsStabilization2SettingsGet(modes);
191  break;
192  case 3:
193  ManualControlSettingsStabilization3SettingsGet(modes);
194  break;
195  default:
196  return SYSTEMALARMS_CONFIGERROR_NONE;
197  }
198 
199  // For multirotors verify that nothing is set to "none"
200  if (multirotor) {
201  for(uint32_t i = 0; i < NELEMENTS(modes); i++) {
202  if (modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NONE)
203  return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
204 
205  // If this axis allows enabling an autotune behavior without the module
206  // running then set an alarm now that aututune module initializes the
207  // appropriate objects
208  if ((modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_RELAYRATE ||
209  modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_RELAYATTITUDE) &&
210  (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE)))
211  return SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
212  }
213  }
214 
215  // Warning: This assumes that certain conditions in the XML file are met. That
216  // MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NONE has the same numeric value for each channel
217  // and is the same for STABILIZATIONDESIRED_STABILIZATIONMODE_NONE
218 
219  return SYSTEMALARMS_CONFIGERROR_NONE;
220 }

226 static void set_config_error(SystemAlarmsConfigErrorOptions error_code)
227 {
228  // Get the severity of the alarm given the error code
229  SystemAlarmsAlarmOptions severity;
230  switch (error_code) {
231  case SYSTEMALARMS_CONFIGERROR_NONE:
232  severity = SYSTEMALARMS_ALARM_OK;
233  break;
234  case SYSTEMALARMS_CONFIGERROR_STABILIZATION:
235  case SYSTEMALARMS_CONFIGERROR_MULTIROTOR:
236  case SYSTEMALARMS_CONFIGERROR_AUTOTUNE:
237  case SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD:
238  case SYSTEMALARMS_CONFIGERROR_VELOCITYCONTROL:
239  case SYSTEMALARMS_CONFIGERROR_POSITIONHOLD:
240  case SYSTEMALARMS_CONFIGERROR_PATHPLANNER:
241  severity = SYSTEMALARMS_ALARM_ERROR;
242  break;
243  default:
244  severity = SYSTEMALARMS_ALARM_ERROR;
245  error_code = SYSTEMALARMS_CONFIGERROR_UNDEFINED;
246  break;
247  }
248 
249  // Make sure not to set the error code if it didn't change
250  SystemAlarmsConfigErrorOptions current_error_code;
251  SystemAlarmsConfigErrorGet((uint8_t *) &current_error_code);
252  if (current_error_code != error_code) {
253  SystemAlarmsConfigErrorSet((uint8_t *) &error_code);
254  }
255 
256  // AlarmSet checks only updates on toggle
257  AlarmsSet(SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION, (uint8_t) severity);
258 }
