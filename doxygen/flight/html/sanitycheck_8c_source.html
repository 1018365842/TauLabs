51 int32_t configuration_check()
52 {
53  SystemAlarmsConfigErrorOptions error_code = SYSTEMALARMS_CONFIGERROR_NONE;
54 
55  // Get board type
56  const struct pios_board_info * bdinfo = &pios_board_info_blob;
57  bool coptercontrol = bdinfo->board_type == 0x04;
58 
59  // For when modules are not running we should explicitly check the objects are
60  // valid
61  if (ManualControlSettingsHandle() == NULL ||
62  SystemSettingsHandle() == NULL) {
63  AlarmsSet(SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION, SYSTEMALARMS_ALARM_CRITICAL);
64  return 0;
65  }
66 
67  // Classify airframe type
68  bool multirotor = true;
69  uint8_t airframe_type;
70  SystemSettingsAirframeTypeGet(&airframe_type);
71  switch(airframe_type) {
72  case SYSTEMSETTINGS_AIRFRAMETYPE_QUADX:
73  case SYSTEMSETTINGS_AIRFRAMETYPE_QUADP:
74  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXA:
75  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTO:
76  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXAX:
77  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTOV:
78  case SYSTEMSETTINGS_AIRFRAMETYPE_OCTOCOAXP:
79  case SYSTEMSETTINGS_AIRFRAMETYPE_HEXACOAX:
80  case SYSTEMSETTINGS_AIRFRAMETYPE_TRI:
81  multirotor = true;
82  break;
83  default:
84  multirotor = false;
85  }
86 
87  // For each available flight mode position sanity check the available
88  // modes
89  uint8_t num_modes;
90  uint8_t modes[MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_NUMELEM];
91  ManualControlSettingsFlightModeNumberGet(&num_modes);
92  ManualControlSettingsFlightModePositionGet(modes);
93 
94 
95  for(uint32_t i = 0; i < num_modes; i++) {
96  switch(modes[i]) {
97  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_MANUAL:
98  if (multirotor) {
99  error_code = SYSTEMALARMS_CONFIGERROR_STABILIZATION;
100  }
101  break;
102  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED1:
103  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(1, multirotor) : error_code;
104  break;
105  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED2:
106  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(2, multirotor) : error_code;
107  break;
108  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_STABILIZED3:
109  error_code = (error_code == SYSTEMALARMS_CONFIGERROR_NONE) ? check_stabilization_settings(3, multirotor) : error_code;
110  break;
111  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_AUTOTUNE:
112  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE))
113  error_code = SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
114  break;
115  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_ALTITUDEHOLD:
116  if (coptercontrol)
117  error_code = SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD;
118  else {
119  if ( !TaskMonitorQueryRunning(TASKINFO_RUNNING_ALTITUDEHOLD) )
120  error_code = SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD;
121  }
122  break;
123  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_VELOCITYCONTROL:
124  if (coptercontrol) {
125  error_code = SYSTEMALARMS_CONFIGERROR_VELOCITYCONTROL;
126  }
127  else {
128  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER)) {
129  error_code = SYSTEMALARMS_CONFIGERROR_VELOCITYCONTROL;
130  }
131  }
132  break;
133  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_POSITIONHOLD:
134  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_RETURNTOHOME:
135  if (coptercontrol) {
136  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
137  }
138  else {
139  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER)) {
140  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
141  }
142  }
143  break;
144  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_PATHPLANNER:
145  case MANUALCONTROLSETTINGS_FLIGHTMODEPOSITION_TABLETCONTROL:
146  if (coptercontrol) {
147  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
148  }
149  else {
150  if (!TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER) ||
151  !TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHPLANNER)) {
152  error_code = SYSTEMALARMS_CONFIGERROR_PATHPLANNER;
153  }
154  }
155  break;
156  default:
157  // Uncovered modes are automatically an error
158  error_code = SYSTEMALARMS_CONFIGERROR_UNDEFINED;
159  }
160  }
161 
162  set_config_error(error_code);
163 
164  return 0;
165 }

174 static int32_t check_stabilization_settings(int index, bool multirotor)
175 {
176  // Make sure the modes have identical sizes
177  if (MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION2SETTINGS_NUMELEM ||
178  MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION3SETTINGS_NUMELEM)
179  return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
180 
181  uint8_t modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM];
182 
183  // Get the different axis modes for this switch position
184  switch(index) {
185  case 1:
186  ManualControlSettingsStabilization1SettingsGet(modes);
187  break;
188  case 2:
189  ManualControlSettingsStabilization2SettingsGet(modes);
190  break;
191  case 3:
192  ManualControlSettingsStabilization3SettingsGet(modes);
193  break;
194  default:
195  return SYSTEMALARMS_CONFIGERROR_NONE;
196  }
197 
198  // For multirotors verify that nothing is set to "none"
199  if (multirotor) {
200  for(uint32_t i = 0; i < NELEMENTS(modes); i++) {
201  if (modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NONE)
202  return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
203 
204  // If this axis allows enabling an autotune behavior without the module
205  // running then set an alarm now that aututune module initializes the
206  // appropriate objects
207  if ((modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_RELAYRATE ||
208  modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_RELAYATTITUDE) &&
209  (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE)))
210  return SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
211  }
212  }
213 
214  // Warning: This assumes that certain conditions in the XML file are met. That
215  // MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NONE has the same numeric value for each channel
216  // and is the same for STABILIZATIONDESIRED_STABILIZATIONMODE_NONE
217 
218  return SYSTEMALARMS_CONFIGERROR_NONE;
219 }

225 static void set_config_error(SystemAlarmsConfigErrorOptions error_code)
226 {
227  // Get the severity of the alarm given the error code
228  SystemAlarmsAlarmOptions severity;
229  switch (error_code) {
230  case SYSTEMALARMS_CONFIGERROR_NONE:
231  severity = SYSTEMALARMS_ALARM_OK;
232  break;
233  case SYSTEMALARMS_CONFIGERROR_STABILIZATION:
234  case SYSTEMALARMS_CONFIGERROR_MULTIROTOR:
235  case SYSTEMALARMS_CONFIGERROR_AUTOTUNE:
236  case SYSTEMALARMS_CONFIGERROR_ALTITUDEHOLD:
237  case SYSTEMALARMS_CONFIGERROR_VELOCITYCONTROL:
238  case SYSTEMALARMS_CONFIGERROR_POSITIONHOLD:
239  case SYSTEMALARMS_CONFIGERROR_PATHPLANNER:
240  severity = SYSTEMALARMS_ALARM_ERROR;
241  break;
242  default:
243  severity = SYSTEMALARMS_ALARM_ERROR;
244  error_code = SYSTEMALARMS_CONFIGERROR_UNDEFINED;
245  break;
246  }
247 
248  // Make sure not to set the error code if it didn't change
249  SystemAlarmsConfigErrorOptions current_error_code;
250  SystemAlarmsConfigErrorGet((uint8_t *) &current_error_code);
251  if (current_error_code != error_code) {
252  SystemAlarmsConfigErrorSet((uint8_t *) &error_code);
253  }
254 
255  // AlarmSet checks only updates on toggle
256  AlarmsSet(SYSTEMALARMS_ALARM_SYSTEMCONFIGURATION, (uint8_t) severity);
257 }
