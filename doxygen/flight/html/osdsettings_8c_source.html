48 int32_t OsdSettingsInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(OSDSETTINGS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(OSDSETTINGS_OBJID,
56  OSDSETTINGS_ISSINGLEINST, OSDSETTINGS_ISSETTINGS, OSDSETTINGS_NUMBYTES, &OsdSettingsSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

74 void OsdSettingsSetDefaults(UAVObjHandle obj, uint16_t instId)
75 {
76  OsdSettingsData data;
77  UAVObjMetadata metadata;
78 
79  // Initialize object fields to their default values
80  UAVObjGetInstanceData(obj, instId, &data);
81  memset(&data, 0, sizeof(OsdSettingsData));
82  data.AttitudeSetup[0] = 168;
83  data.AttitudeSetup[1] = 135;
84  data.TimeSetup[0] = 10;
85  data.TimeSetup[1] = 250;
86  data.BatterySetup[0] = 316;
87  data.BatterySetup[1] = 210;
88  data.SpeedSetup[0] = 2;
89  data.SpeedSetup[1] = 145;
90  data.AltitudeSetup[0] = 2;
91  data.AltitudeSetup[1] = 145;
92  data.HeadingSetup[0] = 168;
93  data.HeadingSetup[1] = 240;
94  data.Attitude = 1;
95  data.Time = 1;
96  data.Battery = 1;
97  data.Speed = 1;
98  data.Altitude = 1;
99  data.Heading = 1;
100  data.Screen = 0;
101 
102  UAVObjSetInstanceData(obj, instId, &data);
103 
104  // Initialize object metadata to their default values
105  metadata.flags =
106  ACCESS_READWRITE << UAVOBJ_ACCESS_SHIFT |
107  ACCESS_READWRITE << UAVOBJ_GCS_ACCESS_SHIFT |
108  1 << UAVOBJ_TELEMETRY_ACKED_SHIFT |
109  1 << UAVOBJ_GCS_TELEMETRY_ACKED_SHIFT |
110  UPDATEMODE_ONCHANGE << UAVOBJ_TELEMETRY_UPDATE_MODE_SHIFT |
111  UPDATEMODE_ONCHANGE << UAVOBJ_GCS_TELEMETRY_UPDATE_MODE_SHIFT;
112  metadata.telemetryUpdatePeriod = 0;
113  metadata.gcsTelemetryUpdatePeriod = 0;
114  metadata.loggingUpdatePeriod = 0;
115  UAVObjSetMetadata(obj, &metadata);
116 }
