129 int32_t PIOS_INTERNAL_ADC_Init(uint32_t * internal_adc_id, const struct pios_internal_adc_cfg * cfg)
130 {
131  PIOS_DEBUG_Assert(internal_adc_id);
132  PIOS_DEBUG_Assert(cfg);
133 
134  struct pios_internal_adc_dev * adc_dev;
135  adc_dev = PIOS_INTERNAL_ADC_Allocate();
136  if (adc_dev == NULL)
137  return -1;
138 
139  adc_dev->cfg = cfg;
140  adc_dev->callback_function = NULL;
141 
142 #if defined(PIOS_INCLUDE_FREERTOS)
143  adc_dev->data_queue = NULL;
144 #endif
145 
146  /* Setup analog pins */
147  GPIO_InitTypeDef GPIO_InitStructure;
148  GPIO_StructInit(&GPIO_InitStructure);
149  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
150  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AIN;
151 
152  /* Enable each ADC pin in the array */
153  for (int32_t i = 0; i < PIOS_ADC_NUM_PINS; i++) {
154  GPIO_InitStructure.GPIO_Pin = ADC_GPIO_PIN[i];
155  GPIO_Init(ADC_GPIO_PORT[i], &GPIO_InitStructure);
156  }
157 
158  *internal_adc_id = (uint32_t)adc_dev;
159 
160  PIOS_INTERNAL_ADC_Config(*internal_adc_id, cfg->oversampling);
161 
162  return 0;
163 }

170 static void PIOS_INTERNAL_ADC_Config(uint32_t internal_adc_id, uint32_t oversampling)
171 {
172  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *)internal_adc_id;
173  if(!PIOS_INTERNAL_ADC_validate(adc_dev))
174  {
175  return;
176  }
177  adc_dev->adc_oversample = (oversampling > PIOS_ADC_MAX_OVERSAMPLING) ? PIOS_ADC_MAX_OVERSAMPLING : oversampling;
178 
179  ADC_DeInit(ADC1);
180  ADC_DeInit(ADC2);
181 
182  /* Disable interrupts */
183  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, adc_dev->cfg->dma.irq.flags, DISABLE);
184 
185  /* Enable ADC clocks */
186  PIOS_ADC_CLOCK_FUNCTION;
187 
188  /* Map channels to conversion slots depending on the channel selection mask */
189  for (int32_t i = 0; i < PIOS_ADC_NUM_PINS; i++) {
190  ADC_RegularChannelConfig(ADC_MAPPING[i], ADC_CHANNEL[i],
191  ADC_CHANNEL_MAPPING[i],
192  PIOS_ADC_SAMPLE_TIME);
193  }
194 
195 #if (PIOS_ADC_USE_TEMP_SENSOR)
196  ADC_TempSensorVrefintCmd(ENABLE);
197  ADC_RegularChannelConfig(PIOS_ADC_TEMP_SENSOR_ADC, ADC_Channel_16,
198  PIOS_ADC_TEMP_SENSOR_ADC_CHANNEL,
199  PIOS_ADC_SAMPLE_TIME);
200 #endif
201  // return
202  /* Configure ADCs */
203  ADC_InitTypeDef ADC_InitStructure;
204  ADC_StructInit(&ADC_InitStructure);
205  ADC_InitStructure.ADC_Mode = ADC_Mode_RegSimult;
206  ADC_InitStructure.ADC_ScanConvMode = ENABLE;
207  ADC_InitStructure.ADC_ContinuousConvMode = ENABLE;
208  ADC_InitStructure.ADC_ExternalTrigConv = ADC_ExternalTrigConv_None;
209  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Right;
210  ADC_InitStructure.ADC_NbrOfChannel = ((PIOS_ADC_NUM_CHANNELS + 1) >> 1);
211  ADC_Init(ADC1, &ADC_InitStructure);
212 
213 #if (PIOS_ADC_USE_ADC2)
214  ADC_Init(ADC2, &ADC_InitStructure);
215 
216  /* Enable ADC2 external trigger conversion (to synch with ADC1) */
217  ADC_ExternalTrigConvCmd(ADC2, ENABLE);
218 #endif
219 
220  RCC_ADCCLKConfig(PIOS_ADC_ADCCLK);
221 
222  /* Enable ADC1->DMA request */
223  ADC_DMACmd(ADC1, ENABLE);
224 
225  /* ADC1 calibration */
226  ADC_Cmd(ADC1, ENABLE);
227  ADC_ResetCalibration(ADC1);
228  while (ADC_GetResetCalibrationStatus(ADC1)) ;
229  ADC_StartCalibration(ADC1);
230  while (ADC_GetCalibrationStatus(ADC1)) ;
231 
232 #if (PIOS_ADC_USE_ADC2)
233  /* ADC2 calibration */
234  ADC_Cmd(ADC2, ENABLE);
235  ADC_ResetCalibration(ADC2);
236  while (ADC_GetResetCalibrationStatus(ADC2)) ;
237  ADC_StartCalibration(ADC2);
238  while (ADC_GetCalibrationStatus(ADC2)) ;
239 #endif
240 
241  /* This makes sure we have an even number of transfers if using ADC2 */
242  adc_dev->dma_block_size = ((PIOS_ADC_NUM_CHANNELS + PIOS_ADC_USE_ADC2) >> PIOS_ADC_USE_ADC2) << PIOS_ADC_USE_ADC2;
243  adc_dev->dma_half_buffer_size = adc_dev->dma_block_size * adc_dev->adc_oversample;
244 
245  /* Configure DMA channel */
246  DMA_InitTypeDef dma_init = adc_dev->cfg->dma.rx.init;
247  dma_init.DMA_MemoryBaseAddr = (uint32_t) &adc_dev->raw_data_buffer[0];
248  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
249  dma_init.DMA_BufferSize = adc_dev->dma_half_buffer_size; /* x2 for double buffer /2 for 32-bit xfr */
250  DMA_Init(adc_dev->cfg->dma.rx.channel, &dma_init);
251  DMA_Cmd(adc_dev->cfg->dma.rx.channel, ENABLE);
252 
253  /* Trigger interrupt when for half conversions too to indicate double buffer */
254  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, DMA_IT_TC, ENABLE);
255  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, DMA_IT_HT, ENABLE);
256 
257  /* Configure DMA interrupt */
258  NVIC_Init(&adc_dev->cfg->dma.irq.init);
259 
260  /* Finally start initial conversion */
261  ADC_SoftwareStartConvCmd(ADC1, ENABLE);
262 
263  /* Use simple averaging filter for now */
264  for (int32_t i = 0; i < adc_dev->adc_oversample; i++)
265  adc_dev->fir_coeffs[i] = 1;
266  adc_dev->fir_coeffs[adc_dev->adc_oversample] = adc_dev->adc_oversample;
267 
268  /* Enable DMA1 clock */
269  RCC_AHBPeriphClockCmd(adc_dev->cfg->dma.ahb_clk, ENABLE);
270 }

280 static int32_t PIOS_INTERNAL_ADC_PinGet(uint32_t internal_adc_id, uint32_t pin)

317 static void PIOS_INTERNAL_ADC_downsample_data(uint32_t internal_adc_id)
318 {
319  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *)internal_adc_id;
320  if(!PIOS_INTERNAL_ADC_validate(adc_dev))
321  {
322  return;
323  }
324  uint16_t chan;
325  uint16_t sample;
326  float * downsampled_buffer = &adc_dev->downsampled_buffer[0];
327 
328  for (chan = 0; chan < PIOS_ADC_NUM_CHANNELS; chan++) {
329  int32_t sum = 0;
330  for (sample = 0; sample < adc_dev->adc_oversample; sample++) {
331  sum += adc_dev->valid_data_buffer[chan + sample * adc_dev->dma_block_size] * adc_dev->fir_coeffs[sample];
332  }
333  downsampled_buffer[chan] = (float) sum / adc_dev->fir_coeffs[adc_dev->adc_oversample];
334  }
335 
336 #if defined(PIOS_INCLUDE_FREERTOS)
337  if(adc_dev->data_queue) {
338  static portBASE_TYPE xHigherPriorityTaskWoken;
339  xQueueSendFromISR(adc_dev->data_queue, adc_dev->downsampled_buffer, &xHigherPriorityTaskWoken);
340  portEND_SWITCHING_ISR(xHigherPriorityTaskWoken);
341  }
342 #endif
343  if(adc_dev->callback_function)
344  adc_dev->callback_function(adc_dev->downsampled_buffer);
345 }

356 void PIOS_INTERNAL_ADC_DMA_Handler()
357 {
358  if(!PIOS_INTERNAL_ADC_validate(static_adc_dev))
359  return;
360  if (DMA_GetFlagStatus(static_adc_dev->cfg->full_flag /*DMA1_IT_TC1*/)) { // whole double buffer filled
361  static_adc_dev->valid_data_buffer = &static_adc_dev->raw_data_buffer[static_adc_dev->dma_half_buffer_size];
362  DMA_ClearFlag(static_adc_dev->cfg->full_flag);
363  PIOS_INTERNAL_ADC_downsample_data((uint32_t)static_adc_dev);
364  }
365  else if (DMA_GetFlagStatus(static_adc_dev->cfg->half_flag /*DMA1_IT_HT1*/)) {
366  static_adc_dev->valid_data_buffer = &static_adc_dev->raw_data_buffer[0];
367  DMA_ClearFlag(static_adc_dev->cfg->half_flag);
368  PIOS_INTERNAL_ADC_downsample_data((uint32_t)static_adc_dev);
369  }
370  else {
371  // This should not happen, probably due to transfer errors
372  DMA_ClearFlag(static_adc_dev->cfg->dma.irq.flags /*DMA1_FLAG_GL1*/);
373  }
374 }
