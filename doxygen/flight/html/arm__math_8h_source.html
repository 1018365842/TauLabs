499  __STATIC_INLINE uint32_t arm_recip_q31(
500  q31_t in,
501  q31_t * dst,
502  q31_t * pRecipTable)
503  {
504 
505  uint32_t out, tempVal;
506  uint32_t index, i;
507  uint32_t signBits;
508 
509  if(in > 0)
510  {
511  signBits = __CLZ(in) - 1;
512  }
513  else
514  {
515  signBits = __CLZ(-in) - 1;
516  }
517 
518  /* Convert input sample to 1.31 format */
519  in = in << signBits;
520 
521  /* calculation of index for initial approximated Val */
522  index = (uint32_t) (in >> 24u);
523  index = (index & INDEX_MASK);
524 
525  /* 1.31 with exp 1 */
526  out = pRecipTable[index];
527 
528  /* calculation of reciprocal value */
529  /* running approximation for two iterations */
530  for (i = 0u; i < 2u; i++)
531  {
532  tempVal = (q31_t) (((q63_t) in * out) >> 31u);
533  tempVal = 0x7FFFFFFF - tempVal;
534  /* 1.31 with exp 1 */
535  //out = (q31_t) (((q63_t) out * tempVal) >> 30u);
536  out = (q31_t) clip_q63_to_q31(((q63_t) out * tempVal) >> 30u);
537  }
538 
539  /* write output */
540  *dst = out;
541 
542  /* return num of signbits of out = 1/in value */
543  return (signBits + 1u);
544
