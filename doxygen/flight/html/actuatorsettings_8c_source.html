48 int32_t ActuatorSettingsInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(ACTUATORSETTINGS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(ACTUATORSETTINGS_OBJID,
56  ACTUATORSETTINGS_ISSINGLEINST, ACTUATORSETTINGS_ISSETTINGS, ACTUATORSETTINGS_NUMBYTES, &ActuatorSettingsSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
