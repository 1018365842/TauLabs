169 static void overoSyncTask(void *parameters)
170 {
171  UAVObjEvent ev;
172 
173  // Kick off SPI transfers (once one is completed another will automatically transmit)
174  overosync->sent_objects = 0;
175  overosync->failed_objects = 0;
176  overosync->received_objects = 0;
177 
178  portTickType lastUpdateTime = xTaskGetTickCount();
179  portTickType updateTime;
180 
181  bool initialized = false;
182  uint8_t last_connected = OVEROSYNCSTATS_CONNECTED_FALSE;
183 
184  // Loop forever
185  while (1) {
186  // Wait for queue message
187  if (xQueueReceive(queue, &ev, portMAX_DELAY) == pdTRUE) {
188 
189  // For the first seconds do not send updates to allow the
190  // overo to boot. Then enable it and act normally.
191  if (!initialized && xTaskGetTickCount() < 5000) {
192  continue;
193  } else if (!initialized) {
194  initialized = true;
195  PIOS_OVERO_Enable(pios_overo_id);
196  }
197 
198  // Process event. This calls transmitData
199  UAVTalkSendObjectTimestamped(uavTalkCon, ev.obj, ev.instId, false, 0);
200 
201  updateTime = xTaskGetTickCount();
202  if(((portTickType) (updateTime - lastUpdateTime)) > 1000) {
203  // Update stats. This will trigger a local send event too
204  OveroSyncStatsData syncStats;
205  syncStats.Send = overosync->sent_bytes;
206  syncStats.Connected = syncStats.Send > 500 ? OVEROSYNCSTATS_CONNECTED_TRUE : OVEROSYNCSTATS_CONNECTED_FALSE;
207  syncStats.DroppedUpdates = overosync->failed_objects;
208  syncStats.Packets = PIOS_OVERO_GetPacketCount(pios_overo_id);
209  OveroSyncStatsSet(&syncStats);
210  overosync->failed_objects = 0;
211  overosync->sent_bytes = 0;
212  lastUpdateTime = updateTime;
213 
214  // When first connected, send all the settings. Right now this
215  // will fail since all the settings will overfill the buffer and
216  if (last_connected == OVEROSYNCSTATS_CONNECTED_FALSE &&
217  syncStats.Connected == OVEROSYNCSTATS_CONNECTED_TRUE) {
218  UAVObjIterate(&send_settings);
219  }
220 
221  // Because the previous code only happens on connection and the
222  // remote logging program doesn't send the settings to the log
223  // when arming starts we send all settings every thirty seconds
224  static uint32_t second_count = 0;
225  if (second_count ++ > 30) {
226  UAVObjIterate(&send_settings);
227  second_count = 0;
228  }
229  last_connected = syncStats.Connected;
230  }
231 
232  // TODO: Check the receive buffer
233  }
234  }
235 }

244 static int32_t pack_data(uint8_t * data, int32_t length)
245 {
246  if( PIOS_COM_SendBufferNonBlocking(pios_com_overo_id, data, length) < 0)
247  goto fail;
248 
249  overosync->sent_bytes += length;
250 
251  return length;
252 
253 fail:
254  overosync->failed_objects++;
255  return -1;
256 }
