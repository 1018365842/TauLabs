123 static uint8_t iap_state = IAP_STATE_READY;
124 static void FirmwareIAPCallback(UAVObjEvent* ev)
125 {
126  const struct pios_board_info * bdinfo = &pios_board_info_blob;
127  static uint32_t last_time = 0;
128  uint32_t this_time;
129  uint32_t delta;
130 
131  if(iap_state == IAP_STATE_RESETTING)
132  return;
133 
134  FirmwareIAPObjData data;
135  FirmwareIAPObjGet(&data);
136 
137  if ( ev->obj == FirmwareIAPObjHandle() ) {
138  // Get the input object data
139  FirmwareIAPObjGet(&data);
140  this_time = get_time();
141  delta = this_time - last_time;
142  last_time = this_time;
143  if((data.BoardType==bdinfo->board_type)&&(data.crc != PIOS_BL_HELPER_CRC_Memory_Calc()))
144  {
145  PIOS_BL_HELPER_FLASH_Read_Description(data.Description,FIRMWAREIAPOBJ_DESCRIPTION_NUMELEM);
146  PIOS_SYS_SerialNumberGetBinary(data.CPUSerial);
147  data.BoardRevision=bdinfo->board_rev;
148  data.crc = PIOS_BL_HELPER_CRC_Memory_Calc();
149  FirmwareIAPObjSet( &data );
150  }
151  if((data.ArmReset==1)&&(iap_state!=IAP_STATE_RESETTING))
152  {
153  data.ArmReset=0;
154  FirmwareIAPObjSet( &data );
155  }
156  switch(iap_state) {
157  case IAP_STATE_READY:
158  if( data.Command == IAP_CMD_STEP_1 ) {
159  iap_state = IAP_STATE_STEP_1;
160  } break;
161  case IAP_STATE_STEP_1:
162  if( data.Command == IAP_CMD_STEP_2 ) {
163  if( delta > iap_time_2_low_end && delta < iap_time_2_high_end ) {
164  iap_state = IAP_STATE_STEP_2;
165  } else {
166  iap_state = IAP_STATE_READY;
167  }
168  } else {
169  iap_state = IAP_STATE_READY;
170  }
171  break;
172  case IAP_STATE_STEP_2:
173  if( data.Command == IAP_CMD_STEP_3 ) {
174  if( delta > iap_time_3_low_end && delta < iap_time_3_high_end ) {
175 
176  FlightStatusData flightStatus;
177  FlightStatusGet(&flightStatus);
178 
179  if(flightStatus.Armed != FLIGHTSTATUS_ARMED_DISARMED) {
180  // Abort any attempts if not disarmed
181  iap_state = IAP_STATE_READY;
182  break;
183  }
184 
185  // we've met the three sequence of command numbers
186  // we've met the time requirements.
187  PIOS_IAP_SetRequest1();
188  PIOS_IAP_SetRequest2();
189 
190  /* Note: Cant just wait timeout value, because first time is randomized */
191  reset_count = 0;
192  lastResetSysTime = xTaskGetTickCount();
193  UAVObjEvent * ev = pvPortMalloc(sizeof(UAVObjEvent));
194  memset(ev,0,sizeof(UAVObjEvent));
195  EventPeriodicCallbackCreate(ev, resetTask, 100);
196  iap_state = IAP_STATE_RESETTING;
197  } else {
198  iap_state = IAP_STATE_READY;
199  }
200  } else {
201  iap_state = IAP_STATE_READY;
202  }
203  break;
204  case IAP_STATE_RESETTING:
205  // stay here permanentally, should reboot
206  break;
207  default:
208  iap_state = IAP_STATE_READY;
209  last_time = 0; // Reset the time counter, as we are not doing a IAP reset
210  break;
211  }
212  }
213 }

240 static void resetTask(UAVObjEvent * ev)
241 {
242 #if defined (PIOS_LED_HEARTBEAT)
243  PIOS_LED_Toggle(PIOS_LED_HEARTBEAT);
244 #endif /* PIOS_LED_HEARTBEAT */
245 
246 #if defined (PIOS_LED_ALARM)
247  PIOS_LED_Toggle(PIOS_LED_ALARM);
248 #endif /* PIOS_LED_ALARM */
249 
250  FirmwareIAPObjData data;
251  FirmwareIAPObjGet(&data);
252 
253  if((portTickType) (xTaskGetTickCount() - lastResetSysTime) > RESET_DELAY / portTICK_RATE_MS) {
254  lastResetSysTime = xTaskGetTickCount();
255  data.BoardType=0xFF;
256  data.ArmReset=1;
257  data.crc=reset_count; /* Must change a value for this to get to INS */
258  FirmwareIAPObjSet(&data);
259  ++reset_count;
260  if(reset_count>3)
261  {
262  PIOS_SYS_Reset();
263  }
264  }
265 }
