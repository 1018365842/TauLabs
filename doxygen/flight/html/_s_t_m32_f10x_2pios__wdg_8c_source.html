60 uint16_t PIOS_WDG_Init()
61 {
62  uint16_t delay = ((uint32_t) PIOS_WATCHDOG_TIMEOUT * 60) / 16;
63  if (delay > 0x0fff)
64  delay = 0x0fff;
65 #if defined(PIOS_INCLUDE_WDG)
66  DBGMCU_Config(DBGMCU_IWDG_STOP, ENABLE); // make the watchdog stop counting in debug mode
67  IWDG_WriteAccessCmd(IWDG_WriteAccess_Enable);
68  IWDG_SetPrescaler(IWDG_Prescaler_16);
69  IWDG_SetReload(delay);
70  IWDG_ReloadCounter();
71  IWDG_Enable();
72 
73  // watchdog flags now stored in backup registers
74  PWR_BackupAccessCmd(ENABLE);
75 
76  BKP_WriteBackupRegister(PIOS_WDG_REGISTER, 0x0);
77  wdg_configuration.bootup_flags = BKP_ReadBackupRegister(PIOS_WDG_REGISTER);
78 #endif
79  return delay;
80 }

118 bool PIOS_WDG_UpdateFlag(uint16_t flag)
119 {
120  // we can probably avoid using a semaphore here which will be good for
121  // efficiency and not blocking critical tasks. race condition could
122  // overwrite their flag update, but unlikely to block _all_ of them
123  // for the timeout window
124  uint16_t cur_flags = BKP_ReadBackupRegister(PIOS_WDG_REGISTER);
125 
126  if((cur_flags | flag) == wdg_configuration.used_flags) {
127  PIOS_WDG_Clear();
128  BKP_WriteBackupRegister(PIOS_WDG_REGISTER, flag);
129  return true;
130  } else {
131  BKP_WriteBackupRegister(PIOS_WDG_REGISTER, cur_flags | flag);
132  return false;
133  }
134 
135 }
