168 
169  /*
170  * Make sure we refuse OUT transactions until we explicitly
171  * connect a receive buffer with PIOS_USBHOOK_EndpointRx().
172  *
173  * Without this, the ST USB code will receive on this endpoint
174  * and blindly write the data to a NULL pointer which will
175  * have the side effect of placing the internal flash into an
176  * errored state. Address 0x0000_0000 is aliased into internal
177  * flash via the "Section 2.4 Boot configuration" BOOT0/1 pins.
178  */
179  DCD_SetEPStatus(&pios_usb_otg_core_handle,
180  epnum,
181  USB_OTG_EP_RX_NAK);
182 }

320 static uint8_t PIOS_USBHOOK_CLASS_Init(void *pdev, uint8_t cfgidx)
321 {
322  /* Call all of the registered init callbacks */
323  for (uint8_t i = 0; i < NELEMENTS(usb_if_table); i++) {
324  struct usb_if_entry * usb_if = &(usb_if_table[i]);

344 static struct usb_setup_request usb_ep0_active_req;
345 static uint8_t PIOS_USBHOOK_CLASS_Setup(void *pdev, USB_SETUP_REQ *req)
346 {
347  switch (req->bmRequest & (USB_REQ_TYPE_MASK | USB_REQ_RECIPIENT_MASK)) {
348  case (USB_REQ_TYPE_STANDARD | USB_REQ_RECIPIENT_INTERFACE):
349  case (USB_REQ_TYPE_CLASS | USB_REQ_RECIPIENT_INTERFACE):
350  {
351  uint8_t ifnum = LOBYTE(req->wIndex);
352  if ((ifnum < NELEMENTS(usb_if_table)) &&
353  (usb_if_table[ifnum].ifops && usb_if_table[ifnum].ifops->setup)) {
354  usb_if_table[ifnum].ifops->setup(usb_if_table[ifnum].context,
355  (struct usb_setup_request *)req);
356  if (!(req->bmRequest & 0x80) && req->wLength > 0) {
357  /* Request is a host-to-device data setup packet, keep track of the request details for the EP0_RxReady call */
358  usb_ep0_active_req.bmRequestType = req->bmRequest;
359  usb_ep0_active_req.bRequest = req->bRequest;
360  usb_ep0_active_req.wValue = req->wValue;
361  usb_ep0_active_req.wIndex = req->wIndex;
362  usb_ep0_active_req.wLength = req->wLength;
363  }
364  } else {
365  /* No Setup handler or Setup handler failed */
366  USBD_CtlError (&pios_usb_otg_core_handle, req);
367  }
368  break;
369  }
370  default:
371  /* Unhandled Setup */
372  USBD_CtlError (&pios_usb_otg_core_handle, req);
373  break;
374  }
375 
376  return USBD_OK;
377 }

397 static uint8_t PIOS_USBHOOK_CLASS_DataIn(void *pdev, uint8_t epnum)
398 {
399  /* Remove the direction bit so we can use this as an index */
400  uint8_t epnum_idx = epnum & 0x7F;
401 
402  if ((epnum_idx < NELEMENTS(usb_epin_table)) && usb_epin_table[epnum_idx].cb) {
403  struct usb_ep_entry *ep = &(usb_epin_table[epnum_idx]);
404  if (!ep->cb(ep->context, epnum_idx, ep->max_len)) {
405  /* NOTE: use real endpoint number including direction bit */
406  DCD_SetEPStatus(pdev, epnum, USB_OTG_EP_TX_NAK);
407  }
408  }
409 
410  return USBD_OK;
411 }
