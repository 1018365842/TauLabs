172 static void SensorsTask(void *parameters)
173 {
174  portTickType lastSysTime;
175 
176  AlarmsSet(SYSTEMALARMS_ALARM_SENSORS, SYSTEMALARMS_ALARM_CRITICAL);
177 
178  UAVObjEvent ev;
179  settingsUpdatedCb(&ev);
180 
181 
182  // Main task loop
183  lastSysTime = xTaskGetTickCount();
184  uint32_t good_runs = 1;
185 
186  while (1) {
187  if (good_runs == 0) {
188  PIOS_WDG_UpdateFlag(PIOS_WDG_SENSORS);
189  lastSysTime = xTaskGetTickCount();
190  AlarmsSet(SYSTEMALARMS_ALARM_SENSORS, SYSTEMALARMS_ALARM_CRITICAL);
191  vTaskDelayUntil(&lastSysTime, SENSOR_PERIOD / portTICK_RATE_MS);
192  }

194  struct pios_sensor_gyro_data gyros;
195  struct pios_sensor_accel_data accels;
196  struct pios_sensor_mag_data mags;
197  struct pios_sensor_baro_data baro;
198 
199  uint32_t timeval = PIOS_DELAY_GetRaw();
200 
201  //Block on gyro data but nothing else
202  xQueueHandle queue;
203  queue = PIOS_SENSORS_GetQueue(PIOS_SENSOR_GYRO);
204  if(queue == NULL || xQueueReceive(queue, (void *) &gyros, SENSOR_PERIOD) == errQUEUE_EMPTY) {
205  good_runs = 0;
206  continue;
207  }
208 
209  queue = PIOS_SENSORS_GetQueue(PIOS_SENSOR_ACCEL);
210  if(queue == NULL || xQueueReceive(queue, (void *) &accels, 0) == errQUEUE_EMPTY) {
211  //If no new accels data is ready, reuse the latest sample
212  AccelsSet(&accelsData);
213  }
214  else
215  update_accels(&accels);
216 
217  // Update gyros after the accels since the rest of the code expects
218  // the accels to be available first
219  update_gyros(&gyros);
220 
221  queue = PIOS_SENSORS_GetQueue(PIOS_SENSOR_MAG);
222  if(queue != NULL && xQueueReceive(queue, (void *) &mags, 0) != errQUEUE_EMPTY) {
223  update_mags(&mags);
224  }
225 
226  queue = PIOS_SENSORS_GetQueue(PIOS_SENSOR_BARO);
227  if (queue != NULL && xQueueReceive(queue, (void *) &baro, 0) != errQUEUE_EMPTY) {
228  update_baro(&baro);
229  }
230 
231  if (good_runs > REQUIRED_GOOD_CYCLES)
232  AlarmsClear(SYSTEMALARMS_ALARM_SENSORS);
233  else
234  good_runs++;
235  PIOS_WDG_UpdateFlag(PIOS_WDG_SENSORS);
236 
237  // Check total time to get the sensors wasn't over the limit
238  uint32_t dT_us = PIOS_DELAY_DiffuS(timeval);
239  if (dT_us > (SENSOR_PERIOD * 1000))
240  good_runs = 0;
241 
242  }
243 }

249 static void update_accels(struct pios_sensor_accel_data *accels)
250 {
251  // Average and scale the accels before rotation
252  float accels_out[3] = {
253  accels->x * accel_scale[0] - accel_bias[0],
254  accels->y * accel_scale[1] - accel_bias[1],
255  accels->z * accel_scale[2] - accel_bias[2]
256  };
257 
258  if (rotate) {
259  float accel_rotated[3];
260  rot_mult(Rbs, accels_out, accel_rotated, false);
261  accelsData.x = accel_rotated[0];
262  accelsData.y = accel_rotated[1];
263  accelsData.z = accel_rotated[2];
264  } else {
265  accelsData.x = accels_out[0];
266  accelsData.y = accels_out[1];
267  accelsData.z = accels_out[2];
268  }
269 
270  accelsData.temperature = accels->temperature;
271  AccelsSet(&accelsData);
272 }

278 static void update_gyros(struct pios_sensor_gyro_data *gyros)
279 {
280  // Scale the gyros
281  float gyros_out[3] = {
282  gyros->x * gyro_scale[0],
283  gyros->y * gyro_scale[1],
284  gyros->z * gyro_scale[2]
285  };
286 
287  GyrosData gyrosData;
288  if (rotate) {
289  float gyros[3];
290  rot_mult(Rbs, gyros_out, gyros, false);
291  gyrosData.x = gyros[0];
292  gyrosData.y = gyros[1];
293  gyrosData.z = gyros[2];
294  } else {
295  gyrosData.x = gyros_out[0];
296  gyrosData.y = gyros_out[1];
297  gyrosData.z = gyros_out[2];
298  }
299 
300  // Update the bias due to the temperature
301  updateTemperatureComp(gyrosData.temperature, gyro_temp_bias);
302 
303  if (bias_correct_gyro) {
304  // Apply bias correction to the gyros from the state estimator
305  GyrosBiasData gyrosBias;
306  GyrosBiasGet(&gyrosBias);
307  gyrosData.x -= gyrosBias.x + gyro_temp_bias[0];
308  gyrosData.y -= gyrosBias.y + gyro_temp_bias[1];
309  gyrosData.z -= gyrosBias.z + gyro_temp_bias[2];
310  }
311 
312  gyrosData.temperature = gyros->temperature;
313  GyrosSet(&gyrosData);
314 }

320 static void update_mags(struct pios_sensor_mag_data *mag)
321 {
322  float mags[3] = {
323  mag->x * mag_scale[0] - mag_bias[0],
324  mag->y * mag_scale[1] - mag_bias[1],
325  mag->z * mag_scale[2] - mag_bias[2]
326  };
327 
328  MagnetometerData magData;
329  if (rotate) {
330  float mag_out[3];
331  rot_mult(Rbs, mags, mag_out, false);
332  magData.x = mag_out[0];
333  magData.y = mag_out[1];
334  magData.z = mag_out[2];
335  } else {
336  magData.x = mags[0];
337  magData.y = mags[1];
338  magData.z = mags[2];
339  }
340 
341  // Correct for mag bias and update if the rate is non zero
342  if (insSettings.MagBiasNullingRate > 0) {
343  switch (mag_calibration_algo) {
344  case MAG_CALIBRATION_PRELEMARI:
345  mag_calibration_prelemari(&magData);
346  break;
347  case MAG_CALIBRATION_NORMALIZE_LENGTH:
348  mag_calibration_fix_length(&magData);
349  break;
350  default:
351  // No calibration
352  break;
353  }
354  }
355 
356  MagnetometerSet(&magData);
357 }

377 static void updateTemperatureComp(float temperature, float *temp_bias)
378 {
379  static int temp_counter = -1;
380  static float temp_accum = 0;
381  static const float TEMP_MIN = -10;
382  static const float TEMP_MAX = 60;
383 
384  if (temperature < TEMP_MIN)
385  temperature = TEMP_MIN;
386  if (temperature > TEMP_MAX)
387  temperature = TEMP_MAX;
388 
389  if (temp_counter < 500) {
390  temp_accum += temperature;
391  temp_counter ++;
392  } else {
393  float t = temp_accum / temp_counter;
394  temp_accum = 0;
395  temp_counter = 0;
396 
397  // Compute a third order polynomial for each chanel after each 500 samples
398  temp_bias[0] = gyro_coeff_x[0] + gyro_coeff_x[1] * t +
399  gyro_coeff_x[2] * powf(t,2) + gyro_coeff_x[3] * powf(t,3);
400  temp_bias[1] = gyro_coeff_y[0] + gyro_coeff_y[1] * t +
401  gyro_coeff_y[2] * powf(t,2) + gyro_coeff_y[3] * powf(t,3);
402  temp_bias[2] = gyro_coeff_z[0] + gyro_coeff_z[1] * t +
403  gyro_coeff_z[2] * powf(t,2) + gyro_coeff_z[3] * powf(t,3);
404  }
405 }

412 static void mag_calibration_prelemari(MagnetometerData *mag)
413 {
414  // Constants, to possibly go into a UAVO
415  static const float MIN_NORM_DIFFERENCE = 50;
416 
417  static float B2[3] = {0, 0, 0};
418 
419  MagBiasData magBias;
420  MagBiasGet(&magBias);
421 
422  // Remove the current estimate of the bias
423  mag->x -= magBias.x;
424  mag->y -= magBias.y;
425  mag->z -= magBias.z;
426 
427  // First call
428  if (B2[0] == 0 && B2[1] == 0 && B2[2] == 0) {
429  B2[0] = mag->x;
430  B2[1] = mag->y;
431  B2[2] = mag->z;
432  return;
433  }
434 
435  float B1[3] = {mag->x, mag->y, mag->z};
436  float norm_diff = sqrtf(powf(B2[0] - B1[0],2) + powf(B2[1] - B1[1],2) + powf(B2[2] - B1[2],2));
437  if (norm_diff > MIN_NORM_DIFFERENCE) {
438  float norm_b1 = sqrtf(B1[0]*B1[0] + B1[1]*B1[1] + B1[2]*B1[2]);
439  float norm_b2 = sqrtf(B2[0]*B2[0] + B2[1]*B2[1] + B2[2]*B2[2]);
440  float scale = insSettings.MagBiasNullingRate * (norm_b2 - norm_b1) / norm_diff;
441  float b_error[3] = {(B2[0] - B1[0]) * scale, (B2[1] - B1[1]) * scale, (B2[2] - B1[2]) * scale};
442 
443  magBias.x += b_error[0];
444  magBias.y += b_error[1];
445  magBias.z += b_error[2];
446 
447  MagBiasSet(&magBias);
448 
449  // Store this value to compare against next update
450  B2[0] = B1[0]; B2[1] = B1[1]; B2[2] = B1[2];
451  }
452 }

460 static void mag_calibration_fix_length(MagnetometerData *mag)
461 {
462  MagBiasData magBias;
463  MagBiasGet(&magBias);
464 
465  // Remove the current estimate of the bias
466  mag->x -= magBias.x;
467  mag->y -= magBias.y;
468  mag->z -= magBias.z;
469 
470  HomeLocationData homeLocation;
471  HomeLocationGet(&homeLocation);
472 
473  AttitudeActualData attitude;
474  AttitudeActualGet(&attitude);
475 
476  const float Rxy = sqrtf(homeLocation.Be[0]*homeLocation.Be[0] + homeLocation.Be[1]*homeLocation.Be[1]);
477  const float Rz = homeLocation.Be[2];
478 
479  const float rate = insSettings.MagBiasNullingRate;
480  float R[3][3];
481  float B_e[3];
482  float xy[2];
483  float delta[3];
484 
485  // Get the rotation matrix
486  Quaternion2R(&attitude.q1, R);
487 
488  // Rotate the mag into the NED frame
489  B_e[0] = R[0][0] * mag->x + R[1][0] * mag->y + R[2][0] * mag->z;
490  B_e[1] = R[0][1] * mag->x + R[1][1] * mag->y + R[2][1] * mag->z;
491  B_e[2] = R[0][2] * mag->x + R[1][2] * mag->y + R[2][2] * mag->z;
492 
493  float cy = cosf(attitude.Yaw * DEG2RAD);
494  float sy = sinf(attitude.Yaw * DEG2RAD);
495 
496  xy[0] = cy * B_e[0] + sy * B_e[1];
497  xy[1] = -sy * B_e[0] + cy * B_e[1];
498 
499  float xy_norm = sqrtf(xy[0]*xy[0] + xy[1]*xy[1]);
500 
501  delta[0] = -rate * (xy[0] / xy_norm * Rxy - xy[0]);
502  delta[1] = -rate * (xy[1] / xy_norm * Rxy - xy[1]);
503  delta[2] = -rate * (Rz - B_e[2]);
504 
505  if (delta[0] == delta[0] && delta[1] == delta[1] && delta[2] == delta[2]) {
506  magBias.x += delta[0];
507  magBias.y += delta[1];
508  magBias.z += delta[2];
509  MagBiasSet(&magBias);
510  }
511 }

527  mag_scale[2] = sensorSettings.MagScale[SENSORSETTINGS_MAGSCALE_Z];
528  accel_bias[0] = sensorSettings.AccelBias[SENSORSETTINGS_ACCELBIAS_X];
529  accel_bias[1] = sensorSettings.AccelBias[SENSORSETTINGS_ACCELBIAS_Y];
530  accel_bias[2] = sensorSettings.AccelBias[SENSORSETTINGS_ACCELBIAS_Z];

532  accel_scale[1] = sensorSettings.AccelScale[SENSORSETTINGS_ACCELSCALE_Y];
533  accel_scale[2] = sensorSettings.AccelScale[SENSORSETTINGS_ACCELSCALE_Z];

535  gyro_scale[1] = sensorSettings.GyroScale[SENSORSETTINGS_GYROSCALE_Y];

549 
550  // Zero out any adaptive tracking
551  MagBiasData magBias;
552  MagBiasGet(&magBias);
553  magBias.x = 0;
554  magBias.y = 0;
555  magBias.z = 0;
556  MagBiasSet(&magBias);
557 
558  uint8_t bias_correct;
559  AttitudeSettingsBiasCorrectGyroGet(&bias_correct);
560  bias_correct_gyro = (bias_correct == ATTITUDESETTINGS_BIASCORRECTGYRO_TRUE);
561 
562  AttitudeSettingsData attitudeSettings;
563  AttitudeSettingsGet(&attitudeSettings);
564  // Indicates not to expend cycles on rotation
565  if(attitudeSettings.BoardRotation[0] == 0 && attitudeSettings.BoardRotation[1] == 0 &&
566  attitudeSettings.BoardRotation[2] == 0) {
567  rotate = 0;
568  } else {
569  float rotationQuat[4];
570  const float rpy[3] = {attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_ROLL] / 100.0f,
571  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_PITCH] / 100.0f,
572  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_YAW] / 100.0f};
573  RPY2Quaternion(rpy, rotationQuat);
574  Quaternion2R(rotationQuat, Rbs);
575  rotate = 1;
576  }
