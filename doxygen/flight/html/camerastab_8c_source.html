144 static void attitudeUpdated(UAVObjEvent* ev)
145 {
146  if (ev->obj != AttitudeActualHandle())
147  return;
148 
149  AccessoryDesiredData accessory;
150 
151  CameraStabSettingsData *settings = &csd->settings;
152 
153  // Check how long since last update, time delta between calls in ms
154  portTickType thisSysTime = xTaskGetTickCount();
155  float dT_ms = (thisSysTime - csd->lastSysTime) * portTICK_RATE_MS;
156  csd->lastSysTime = thisSysTime;
157 
158  if (dT_ms <= 0)
159  return;
160 
161  float attitude;
162  float output;
163 
164  // Read any input channels and apply LPF
165  for (uint8_t i = 0; i < MAX_AXES; i++) {
166  switch (i) {
167  case ROLL:
168  AttitudeActualRollGet(&attitude);
169  break;
170  case PITCH:
171  AttitudeActualPitchGet(&attitude);
172  break;
173  case YAW:
174  AttitudeActualYawGet(&attitude);
175  break;
176  }
177  float rt_ms = (float)settings->AttitudeFilter;
178  csd->attitude_filtered[i] = (rt_ms / (rt_ms + dT_ms)) * csd->attitude_filtered[i] + (dT_ms / (rt_ms + dT_ms)) * attitude;
179  attitude = csd->attitude_filtered[i];
180 
181  if (settings->Input[i] != CAMERASTABSETTINGS_INPUT_NONE) {
182  if (AccessoryDesiredInstGet(settings->Input[i] - CAMERASTABSETTINGS_INPUT_ACCESSORY0, &accessory) == 0) {
183  float input;
184  float input_rate;
185  rt_ms = (float) settings->InputFilter;
186  switch (settings->StabilizationMode[i]) {
187  case CAMERASTABSETTINGS_STABILIZATIONMODE_ATTITUDE:
188  input = accessory.AccessoryVal * settings->InputRange[i];
189  csd->inputs[i] = (rt_ms / (rt_ms + dT_ms)) * csd->inputs[i] + (dT_ms / (rt_ms + dT_ms)) * input;
190  break;
191  case CAMERASTABSETTINGS_STABILIZATIONMODE_AXISLOCK:
192  input_rate = accessory.AccessoryVal * settings->InputRate[i];
193  if (fabs(input_rate) > settings->MaxAxisLockRate)
194  csd->inputs[i] = bound_sym(csd->inputs[i] + input_rate * dT_ms / 1000.0f, settings->InputRange[i]);
195  break;
196  default:
197  input = 0;
198  }
199  }
200  }
201 
202  // Add Servo FeedForward
203  applyFF(i, dT_ms, &attitude, settings);
204 
205  // Set output channels
206  output = bound_sym((attitude + csd->inputs[i]) / settings->OutputRange[i], 1.0f);
207  if (thisSysTime / portTICK_RATE_MS > LOAD_DELAY) {
208  switch (i) {
209  case ROLL:
210  CameraDesiredRollSet(&output);
211  break;
212  case PITCH:
213  CameraDesiredPitchSet(&output);
214  break;
215  case YAW:
216  CameraDesiredYawSet(&output);
217  break;
218  }
219  }
220  }
221 }

230 static void applyFF(uint8_t index, float dT_ms, float *attitude, CameraStabSettingsData* cameraStab)
231 {
232  float accumulator = csd->FFfilterAccumulator[index];
233 
234  accumulator += (*attitude - csd->FFlastAttitude[index]) * cameraStab->FeedForward[index];
235  csd->FFlastAttitude[index] = *attitude;
236  *attitude += accumulator;
237 
238  float filter = (float) cameraStab->FeedForwardTime / dT_ms;
239  if (filter > 1)
240  accumulator -= accumulator / filter;
241  else
242  accumulator = 0;
243 
244  // For non-zero filter times store accumulation
245  csd->FFfilterAccumulator[index] = accumulator;
246  *attitude += accumulator;
247 
248  //acceleration and deceleration limit
249  float delta = *attitude - csd->FFlastFilteredAttitude[index];
250  float maxDelta = cameraStab->MaxAccel * dT_ms / 1000.0f;
251  if(fabs(delta) > maxDelta) //we are accelerating too hard
252  {
253  *attitude = csd->FFlastFilteredAttitude[index] + (delta > 0 ? maxDelta : - maxDelta);
254  }
255  csd->FFlastFilteredAttitude[index] = *attitude;
256 }
