200 static uint8_t CTRLB = 0x00;
201 static int32_t PIOS_HMC5883_Config(const struct pios_hmc5883_cfg * cfg)
202 {
203  uint8_t CTRLA = 0x00;
204  uint8_t MODE = 0x00;
205  CTRLB = 0;
206 
207  CTRLA |= (uint8_t) (cfg->M_ODR | cfg->Meas_Conf);
208  CTRLB |= (uint8_t) (cfg->Gain);
209  MODE |= (uint8_t) (cfg->Mode);
210 
211  // CRTL_REGA
212  if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_A, CTRLA) != 0)
213  return -1;
214 
215  // CRTL_REGB
216  if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_B, CTRLB) != 0)
217  return -1;
218 
219  // Mode register
220  if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, MODE) != 0)
221  return -1;
222 
223  return 0;
224 }

231 static int32_t PIOS_HMC5883_ReadMag(struct pios_sensor_mag_data *mag_data)
232 {
233  if(PIOS_HMC5883_Validate(dev) != 0)
234  return -1;
235 
236  uint8_t buffer[6];
237  int32_t sensitivity;
238 
239  if (PIOS_HMC5883_Read(PIOS_HMC5883_DATAOUT_XMSB_REG, buffer, 6) != 0) {
240  return -1;
241  }
242 
243  switch (CTRLB & 0xE0) {
244  case 0x00:
245  sensitivity = PIOS_HMC5883_Sensitivity_0_88Ga;
246  break;
247  case 0x20:
248  sensitivity = PIOS_HMC5883_Sensitivity_1_3Ga;
249  break;
250  case 0x40:
251  sensitivity = PIOS_HMC5883_Sensitivity_1_9Ga;
252  break;
253  case 0x60:
254  sensitivity = PIOS_HMC5883_Sensitivity_2_5Ga;
255  break;
256  case 0x80:
257  sensitivity = PIOS_HMC5883_Sensitivity_4_0Ga;
258  break;
259  case 0xA0:
260  sensitivity = PIOS_HMC5883_Sensitivity_4_7Ga;
261  break;
262  case 0xC0:
263  sensitivity = PIOS_HMC5883_Sensitivity_5_6Ga;
264  break;
265  case 0xE0:
266  sensitivity = PIOS_HMC5883_Sensitivity_8_1Ga;
267  break;
268  default:
269  PIOS_Assert(0);
270  }
271 
272  int16_t mag_x, mag_y, mag_z;
273  mag_x = ((int16_t) ((uint16_t) buffer[0] << 8) + buffer[1]) * 1000 / sensitivity;
274  mag_z = ((int16_t) ((uint16_t) buffer[2] << 8) + buffer[3]) * 1000 / sensitivity;
275  mag_y = ((int16_t) ((uint16_t) buffer[4] << 8) + buffer[5]) * 1000 / sensitivity;
276 
277  // Define "0" when the fiducial is in the front left of the board
278  switch (dev->cfg->orientation) {
279  case PIOS_HMC5883_TOP_0DEG:
280  mag_data->x = -mag_x;
281  mag_data->y = mag_y;
282  mag_data->z = -mag_z;
283  break;
284  case PIOS_HMC5883_TOP_90DEG:
285  mag_data->x = -mag_y;
286  mag_data->y = -mag_x;
287  mag_data->z = -mag_z;
288  break;
289  case PIOS_HMC5883_TOP_180DEG:
290  mag_data->x = mag_x;
291  mag_data->y = -mag_y;
292  mag_data->z = -mag_z;
293  break;
294  case PIOS_HMC5883_TOP_270DEG:
295  mag_data->x = mag_y;
296  mag_data->y = mag_x;
297  mag_data->z = -mag_z;
298  break;
299  }
300 
301  // This should not be necessary but for some reason it is coming out of continuous conversion mode
302  PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, PIOS_HMC5883_MODE_CONTINUOUS);
303 
304  return 0;
305 }

396 int32_t PIOS_HMC5883_Test(void)
397 {
398  int32_t failed = 0;
399  uint8_t registers[3] = {0,0,0};
400  uint8_t status;
401  uint8_t ctrl_a_read;
402  uint8_t ctrl_b_read;
403  uint8_t mode_read;
404  struct pios_sensor_mag_data values;
405 
406  /* Verify that ID matches (HMC5883 ID is null-terminated ASCII string "H43") */
407  char id[4];
408  PIOS_HMC5883_ReadID((uint8_t *)id);
409  if((id[0] != 'H') || (id[1] != '4') || (id[2] != '3')) // Expect H43
410  return -1;
411 
412  /* Backup existing configuration */
413  if (PIOS_HMC5883_Read(PIOS_HMC5883_CONFIG_REG_A,registers,3) != 0)
414  return -1;
415 
416  /* Stop the device and read out last value */
417  PIOS_DELAY_WaitmS(10);
418  if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, PIOS_HMC5883_MODE_IDLE) != 0)
419  return -1;
420  if( PIOS_HMC5883_Read(PIOS_HMC5883_DATAOUT_STATUS_REG, &status,1) != 0)
421  return -1;
422  if (PIOS_HMC5883_ReadMag(&values) != 0)
423  return -1;
424 
425  /*
426  * Put HMC5883 into self test mode
427  * This is done by placing measurement config into positive (0x01) or negative (0x10) bias
428  * and then placing the mode register into single-measurement mode. This causes the HMC5883
429  * to create an artificial magnetic field of ~1.1 Gauss.
430  *
431  * If gain were PIOS_HMC5883_GAIN_2_5, for example, X and Y will read around +766 LSB
432  * (1.16 Ga * 660 LSB/Ga) and Z would read around +713 LSB (1.08 Ga * 660 LSB/Ga)
433  *
434  * Changing measurement config back to PIOS_HMC5883_MEASCONF_NORMAL will leave self-test mode.
435  */
436  PIOS_DELAY_WaitmS(10);
437  if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_A, PIOS_HMC5883_MEASCONF_BIAS_POS | PIOS_HMC5883_ODR_15) != 0)
438  return -1;
439  PIOS_DELAY_WaitmS(10);
440  if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_B, PIOS_HMC5883_GAIN_8_1) != 0)
441  return -1;
442  PIOS_DELAY_WaitmS(10);
443  if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, PIOS_HMC5883_MODE_SINGLE) != 0)
444  return -1;
445 
446  /* Must wait for value to be updated */
447  PIOS_DELAY_WaitmS(200);
448 
449  if (PIOS_HMC5883_ReadMag(&values) != 0)
450  return -1;
451 
452  PIOS_HMC5883_Read(PIOS_HMC5883_CONFIG_REG_A, &ctrl_a_read,1);
453  PIOS_HMC5883_Read(PIOS_HMC5883_CONFIG_REG_B, &ctrl_b_read,1);
454  PIOS_HMC5883_Read(PIOS_HMC5883_MODE_REG, &mode_read,1);
455  PIOS_HMC5883_Read(PIOS_HMC5883_DATAOUT_STATUS_REG, &status,1);
456 
457  /* Restore backup configuration */
458  PIOS_DELAY_WaitmS(10);
459  if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_A, registers[0]) != 0)
460  return -1;
461  PIOS_DELAY_WaitmS(10);
462  if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_B, registers[1]) != 0)
463  return -1;
464  PIOS_DELAY_WaitmS(10);
465  if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, registers[2]) != 0)
466  return -1;
467 
468  return failed;
469 }

488 static void PIOS_HMC5883_Task(void *parameters)
489 {
490  while(1) {
491  if(PIOS_HMC5883_Validate(dev) != 0) {
492  vTaskDelay(100 * portTICK_RATE_MS);
493  continue;
494  }
495 
496  if (xSemaphoreTake(dev->data_ready_sema, portMAX_DELAY) != pdTRUE) {
497  vTaskDelay(100 * portTICK_RATE_MS);
498  continue;
499  }
500 
501  struct pios_sensor_mag_data mag_data;
502  if (PIOS_HMC5883_ReadMag(&mag_data) == 0)
503  xQueueSend(dev->queue, (void *) &mag_data, 0);
504  }
505 }
