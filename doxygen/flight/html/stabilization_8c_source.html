99 {
100  // Initialize variables
101  // Create object queue

103 
104  // Listen for updates.
105  // AttitudeActualConnectQueue(queue);

107 
108  StabilizationSettingsConnectCallback(SettingsUpdatedCb);

110 
111  // Start main task

113  TaskMonitorAdd(TASKINFO_RUNNING_STABILIZATION, taskHandle);
114  PIOS_WDG_RegisterFlag(PIOS_WDG_STABILIZATION);

141 static void stabilizationTask(void* parameters)
142 {
143  UAVObjEvent ev;
144 
145  uint32_t timeval = PIOS_DELAY_GetRaw();
146 
147  ActuatorDesiredData actuatorDesired;
148  StabilizationDesiredData stabDesired;
149  RateDesiredData rateDesired;
150  AttitudeActualData attitudeActual;
151  GyrosData gyrosData;
152  FlightStatusData flightStatus;
153 
154  float *stabDesiredAxis = &stabDesired.Roll;
155  float *actuatorDesiredAxis = &actuatorDesired.Roll;
156  float *rateDesiredAxis = &rateDesired.Roll;
157 
158  SettingsUpdatedCb((UAVObjEvent *) NULL);
159 
160  // Main task loop
161  ZeroPids();
162  while(1) {
163  float dT;
164 
165  PIOS_WDG_UpdateFlag(PIOS_WDG_STABILIZATION);
166 
167  // Wait until the AttitudeRaw object is updated, if a timeout then go to failsafe
168  if ( xQueueReceive(queue, &ev, FAILSAFE_TIMEOUT_MS / portTICK_RATE_MS) != pdTRUE )
169  {
170  AlarmsSet(SYSTEMALARMS_ALARM_STABILIZATION,SYSTEMALARMS_ALARM_WARNING);
171  continue;
172  }
173 
174  dT = PIOS_DELAY_DiffuS(timeval) * 1.0e-6f;
175  timeval = PIOS_DELAY_GetRaw();
176 
177  FlightStatusGet(&flightStatus);
178  StabilizationDesiredGet(&stabDesired);
179  AttitudeActualGet(&attitudeActual);
180  GyrosGet(&gyrosData);
181  ActuatorDesiredGet(&actuatorDesired);
182 #if defined(RATEDESIRED_DIAGNOSTICS)
183  RateDesiredGet(&rateDesired);
184 #endif
185 
186 #if defined(PIOS_QUATERNION_STABILIZATION)
187  // Quaternion calculation of error in each axis. Uses more memory.
188  float rpy_desired[3];
189  float q_desired[4];
190  float q_error[4];
191  float local_error[3];
192 
193  // Essentially zero errors for anything in rate or none
194  if(stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] == STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
195  rpy_desired[0] = stabDesired.Roll;
196  else
197  rpy_desired[0] = attitudeActual.Roll;
198 
199  if(stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] == STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
200  rpy_desired[1] = stabDesired.Pitch;
201  else
202  rpy_desired[1] = attitudeActual.Pitch;
203 
204  if(stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] == STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
205  rpy_desired[2] = stabDesired.Yaw;
206  else
207  rpy_desired[2] = attitudeActual.Yaw;
208 
209  RPY2Quaternion(rpy_desired, q_desired);
210  quat_inverse(q_desired);
211  quat_mult(q_desired, &attitudeActual.q1, q_error);
212  quat_inverse(q_error);
213  Quaternion2RPY(q_error, local_error);
214 
215 #else
216  // Simpler algorithm for CC, less memory
217  float local_error[3] = {stabDesired.Roll - attitudeActual.Roll,
218  stabDesired.Pitch - attitudeActual.Pitch,
219  stabDesired.Yaw - attitudeActual.Yaw};
220  local_error[2] = circular_modulus_deg(local_error[2]);
221 #endif
222 
223  float gyro_filtered[3];
224  gyro_filtered[0] = gyro_filtered[0] * gyro_alpha + gyrosData.x * (1 - gyro_alpha);
225  gyro_filtered[1] = gyro_filtered[1] * gyro_alpha + gyrosData.y * (1 - gyro_alpha);
226  gyro_filtered[2] = gyro_filtered[2] * gyro_alpha + gyrosData.z * (1 - gyro_alpha);
227 
228  // A flag to track which stabilization mode each axis is in
229  static uint8_t previous_mode[MAX_AXES] = {255,255,255};
230  bool error = false;
231 
232  //Run the selected stabilization algorithm on each axis:
233  for(uint8_t i=0; i< MAX_AXES; i++)
234  {
235  // Check whether this axis mode needs to be reinitialized
236  bool reinit = (stabDesired.StabilizationMode[i] != previous_mode[i]);
237  previous_mode[i] = stabDesired.StabilizationMode[i];
238 
239  // Apply the selected control law
240  switch(stabDesired.StabilizationMode[i])
241  {
242  case STABILIZATIONDESIRED_STABILIZATIONMODE_RATE:
243  if(reinit)
244  pids[PID_RATE_ROLL + i].iAccumulator = 0;
245 
246  // Store to rate desired variable for storing to UAVO
247  rateDesiredAxis[i] = bound(stabDesiredAxis[i], settings.ManualRate[i]);
248 
249  // Compute the inner loop
250  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
251  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
252 
253  break;
254 
255  case STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE:
256  if(reinit) {
257  pids[PID_ATT_ROLL + i].iAccumulator = 0;
258  pids[PID_RATE_ROLL + i].iAccumulator = 0;
259  }
260 
261  // Compute the outer loop
262  rateDesiredAxis[i] = pid_apply(&pids[PID_ATT_ROLL + i], local_error[i], dT);
263  rateDesiredAxis[i] = bound(rateDesiredAxis[i], settings.MaximumRate[i]);
264 
265  // Compute the inner loop
266  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
267  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
268 
269  break;
270 
271  case STABILIZATIONDESIRED_STABILIZATIONMODE_VIRTUALBAR:
272  // Store for debugging output
273  rateDesiredAxis[i] = stabDesiredAxis[i];
274 
275  // Run a virtual flybar stabilization algorithm on this axis
276  stabilization_virtual_flybar(gyro_filtered[i], rateDesiredAxis[i], &actuatorDesiredAxis[i], dT, reinit, i, &settings);
277 
278  break;
279  case STABILIZATIONDESIRED_STABILIZATIONMODE_WEAKLEVELING:
280  {
281  if (reinit)
282  pids[PID_RATE_ROLL + i].iAccumulator = 0;
283 
284  float weak_leveling = local_error[i] * weak_leveling_kp;
285  weak_leveling = bound(weak_leveling, weak_leveling_max);
286 
287  // Compute desired rate as input biased towards leveling
288  rateDesiredAxis[i] = stabDesiredAxis[i] + weak_leveling;
289  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
290  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
291 
292  break;
293  }
294  case STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK:
295  if (reinit)
296  pids[PID_RATE_ROLL + i].iAccumulator = 0;
297 
298  if(fabs(stabDesiredAxis[i]) > max_axislock_rate) {
299  // While getting strong commands act like rate mode
300  rateDesiredAxis[i] = stabDesiredAxis[i];
301  axis_lock_accum[i] = 0;
302  } else {
303  // For weaker commands or no command simply attitude lock (almost) on no gyro change
304  axis_lock_accum[i] += (stabDesiredAxis[i] - gyro_filtered[i]) * dT;
305  axis_lock_accum[i] = bound(axis_lock_accum[i], max_axis_lock);
306  rateDesiredAxis[i] = pid_apply(&pids[PID_ATT_ROLL + i], axis_lock_accum[i], dT);
307  }
308 
309  rateDesiredAxis[i] = bound(rateDesiredAxis[i], settings.MaximumRate[i]);
310 
311  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
312  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
313 
314  break;
315 
316  case STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYRATE:
317  // Store to rate desired variable for storing to UAVO
318  rateDesiredAxis[i] = bound(stabDesiredAxis[i], settings.ManualRate[i]);
319 
320  // Run the relay controller which also estimates the oscillation parameters
321  stabilization_relay_rate(rateDesiredAxis[i] - gyro_filtered[i], &actuatorDesiredAxis[i], i, reinit);
322  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0);
323 
324  break;
325 
326  case STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYATTITUDE:
327  if(reinit)
328  pids[PID_ATT_ROLL + i].iAccumulator = 0;
329 
330  // Compute the outer loop like attitude mode
331  rateDesiredAxis[i] = pid_apply(&pids[PID_ATT_ROLL + i], local_error[i], dT);
332  rateDesiredAxis[i] = bound(rateDesiredAxis[i], settings.MaximumRate[i]);
333 
334  // Run the relay controller which also estimates the oscillation parameters
335  stabilization_relay_rate(rateDesiredAxis[i] - gyro_filtered[i], &actuatorDesiredAxis[i], i, reinit);
336  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0);
337 
338  break;
339 
340  case STABILIZATIONDESIRED_STABILIZATIONMODE_COORDINATEDFLIGHT:
341  switch (i) {
342  case YAW:
343  if ( stabDesired.StabilizationMode[ROLL]==STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE && //If we are in roll attitude mode..
344  fabs(stabDesired.Roll) > 3.0f && //...and we've requested more than 3 degrees of roll...
345  fabs(stabDesired.Yaw) < 0.02) { //...and we currently have no yaw input within a 2% deadband
346  float accelsDataY;
347  AccelsyGet(&accelsDataY);
348 
349  //Reset integral if we have changed roll to opposite direction from rudder. This implies that we have changed desired turning direction.
350  if ((stabDesired.Roll > 0 && actuatorDesiredAxis[i] < 0) || (stabDesired.Roll < 0 && actuatorDesiredAxis[i] > 0)){
351  pids[PID_RATE_YAW].iAccumulator = 0;
352  }
353 
354  //Coordinate flight can simply be seen as ensuring that there is no lateral acceleration in the
355  // body frame. As such, we use the (noisy) accelerometer data as our measurement. Ideally, at
356  // some point in the future we will estimate acceleration and then we can use the estimated value
357  // instead of the measured value.
358  float errorSlip = -accelsDataY;
359 
360  //Apply a 1 second rise time low-pass filter to the accelerometer driven output.
361  // This reduces jitter in the tail, and helps dampen aileron oscillations
362  // due to the decreased turning radius as rudder input is applied.
363  float alpha = 1.0f-dT/(dT+1.0f);
364 
365  float command=(1-alpha)*pid_apply(&pids[PID_RATE_YAW], errorSlip, dT) + alpha*actuatorDesiredAxis[i];
366  actuatorDesiredAxis[i] = bound(command,1.0);
367  }
368  else{ //Else, yaw input is either passed through manually or we're not requesting
369  actuatorDesiredAxis[i] = bound(stabDesiredAxis[i], 1.0);
370  pids[PID_RATE_YAW].iAccumulator = 0;
371  pids[PID_ATT_YAW].iAccumulator = 0;
372  }
373  break;
374  case ROLL:
375  case PITCH:
376  default:
377  //Coordinated Flight has no effect in these modes. Trigger a configuration error.
378  error = true;
379  break;
380  }
381 
382  break;
383 
384  case STABILIZATIONDESIRED_STABILIZATIONMODE_NONE:
385  actuatorDesiredAxis[i] = bound(stabDesiredAxis[i],1.0f);
386  break;
387  default:
388  error = true;
389  break;
390  }
391  }
392 
393  if (settings.VbarPiroComp == STABILIZATIONSETTINGS_VBARPIROCOMP_TRUE)
394  stabilization_virtual_flybar_pirocomp(gyro_filtered[2], dT);
395 
396 #if defined(RATEDESIRED_DIAGNOSTICS)
397  RateDesiredSet(&rateDesired);
398 #endif
399 
400  // Save dT
401  actuatorDesired.UpdateTime = dT * 1000;
402  actuatorDesired.Throttle = stabDesired.Throttle;
403 
404  if(flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_MANUAL) {
405  ActuatorDesiredSet(&actuatorDesired);
406  } else {
407  // Force all axes to reinitialize when engaged
408  for(uint8_t i=0; i< MAX_AXES; i++)
409  previous_mode[i] = 255;
410  }
411 
412  if(flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED ||
413  (lowThrottleZeroIntegral && stabDesired.Throttle < 0))
414  {
415  // Force all axes to reinitialize when engaged
416  for(uint8_t i=0; i< MAX_AXES; i++)
417  previous_mode[i] = 255;
418  }
419 
420  // Clear or set alarms. Done like this to prevent toggling each cycle
421  // and hammering system alarms
422  if (error)
423  AlarmsSet(SYSTEMALARMS_ALARM_STABILIZATION,SYSTEMALARMS_ALARM_ERROR);
424  else
425  AlarmsClear(SYSTEMALARMS_ALARM_STABILIZATION);
426  }
427 }

447 static float bound(float val, float range)
448 {
449  if(val < -range) {
450  val = -range;

452  val = range;
453  }

457 static void SettingsUpdatedCb(UAVObjEvent * ev)
458 {
459  StabilizationSettingsGet(&settings);
460 
461  // Set the roll rate PID constants
462  pid_configure(&pids[PID_RATE_ROLL], settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP],
463  settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI],
464  settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KD],
465  settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_ILIMIT]);
466 
467  // Set the pitch rate PID constants
468  pid_configure(&pids[PID_RATE_PITCH], settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KP],
469  settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KI],
470  settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KD],
471  settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_ILIMIT]);
472 
473  // Set the yaw rate PID constants
474  pid_configure(&pids[PID_RATE_YAW], settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KP],
475  settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KI],
476  settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KD],
477  settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_ILIMIT]);
478 
479  // Set the roll attitude PI constants
480  pid_configure(&pids[PID_ATT_ROLL], settings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KP],
481  settings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KI], 0,
482  settings.RollPI[STABILIZATIONSETTINGS_ROLLPI_ILIMIT]);
483 
484  // Set the pitch attitude PI constants
485  pid_configure(&pids[PID_ATT_PITCH], settings.PitchPI[STABILIZATIONSETTINGS_PITCHPI_KP],
486  settings.PitchPI[STABILIZATIONSETTINGS_PITCHPI_KI], 0,
487  settings.PitchPI[STABILIZATIONSETTINGS_PITCHPI_ILIMIT]);
488 
489  // Set the yaw attitude PI constants
490  pid_configure(&pids[PID_ATT_YAW], settings.YawPI[STABILIZATIONSETTINGS_YAWPI_KP],
491  settings.YawPI[STABILIZATIONSETTINGS_YAWPI_KI], 0,
492  settings.YawPI[STABILIZATIONSETTINGS_YAWPI_ILIMIT]);
493 
494  // Set up the derivative term
495  pid_configure_derivative(settings.DerivativeCutoff, settings.DerivativeGamma);
496 
497  // Maximum deviation to accumulate for axis lock
498  max_axis_lock = settings.MaxAxisLock;
499  max_axislock_rate = settings.MaxAxisLockRate;
500 
501  // Settings for weak leveling
502  weak_leveling_kp = settings.WeakLevelingKp;
503  weak_leveling_max = settings.MaxWeakLevelingRate;
504 
505  // Whether to zero the PID integrals while throttle is low
506  lowThrottleZeroIntegral = settings.LowThrottleZeroIntegral == STABILIZATIONSETTINGS_LOWTHROTTLEZEROINTEGRAL_TRUE;
507 
508  // The dT has some jitter iteration to iteration that we don't want to
509  // make thie result unpredictable. Still, it's nicer to specify the constant
510  // based on a time (in ms) rather than a fixed multiplier. The error between
511  // update rates on OP (~300 Hz) and CC (~475 Hz) is negligible for this
512  // calculation
513  const float fakeDt = 0.0025;
514  if(settings.GyroTau < 0.0001)
515  gyro_alpha = 0; // not trusting this to resolve to 0
516  else
517  gyro_alpha = expf(-fakeDt / settings.GyroTau);
518 
519  // Compute time constant for vbar decay term based on a tau
520  vbar_decay = expf(-fakeDt / settings.VbarTau);
521 }
