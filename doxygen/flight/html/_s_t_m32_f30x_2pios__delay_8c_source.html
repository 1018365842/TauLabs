54 int32_t PIOS_DELAY_Init(void)
55 {

83 int32_t PIOS_DELAY_WaituS(uint32_t uS)
84 {
85  uint32_t elapsed = 0;
86  uint32_t last_count = DWT_CYCCNT;
87 
88  /* turn on access to the DWT registers */
89  CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
90 
91  for (;;) {
92  uint32_t current_count = DWT_CYCCNT;
93  uint32_t elapsed_uS;
94 
95  /* measure the time elapsed since the last time we checked */
96  elapsed += current_count - last_count;
97  last_count = current_count;
98 
99  /* convert to microseconds */
100  elapsed_uS = elapsed / us_ticks;
101  if (elapsed_uS >= uS)
102  break;
103 
104  /* reduce the delay by the elapsed time */
105  uS -= elapsed_uS;
106 
107  /* keep fractional microseconds for the next iteration */
108  elapsed %= us_ticks;
109  }
110 
111  /* No error */
112  return 0;
113 }
