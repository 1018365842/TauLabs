130 int32_t PIOS_MS5611_Init(const struct pios_ms5611_cfg *cfg, int32_t i2c_device)
131 {
132  dev = (struct ms5611_dev *) PIOS_MS5611_alloc();
133  if (dev == NULL)
134  return -1;
135 
136  dev->i2c_id = i2c_device;
137 
138  dev->oversampling = cfg->oversampling;
139  dev->temperature_interleaving = (cfg->temperature_interleaving) == 0 ? 1 : cfg->temperature_interleaving;
140  dev->cfg = cfg;
141 
142  if (PIOS_MS5611_WriteCommand(MS5611_RESET) != 0)
143  return -2;
144 
145  PIOS_DELAY_WaitmS(20);
146 
147  uint8_t data[2];
148 
149  /* Calibration parameters */
150  for (int i = 0; i < NELEMENTS(dev->calibration); i++) {
151  PIOS_MS5611_Read(MS5611_CALIB_ADDR + i * 2, data, 2);
152  dev->calibration[i] = (data[0] << 8) | data[1];
153  }
154 
155  portBASE_TYPE result = xTaskCreate(PIOS_MS5611_Task, (const signed char *)"pios_ms5611",
156  MS5611_TASK_STACK, NULL, MS5611_TASK_PRIORITY,
157  &dev->task);
158  PIOS_Assert(result == pdPASS);
159 
160  PIOS_SENSORS_Register(PIOS_SENSOR_BARO, dev->queue);
161 
162  return 0;
163 }

170 static int32_t PIOS_MS5611_StartADC(enum conversion_type type)
171 {
172  if (PIOS_MS5611_Validate(dev) != 0)
173  return -1;
174 
175  /* Start the conversion */
176  switch(type) {
177  case TEMPERATURE_CONV:
178  while (PIOS_MS5611_WriteCommand(MS5611_TEMP_ADDR + dev->oversampling) != 0)
179  continue;
180  break;
181  case PRESSURE_CONV:
182  while (PIOS_MS5611_WriteCommand(MS5611_PRES_ADDR + dev->oversampling) != 0)
183  continue;
184  break;
185  default:
186  return -1;
187  }
188 
189  dev->current_conversion_type = type;
190 
191  return 0;
192 }

222 static int32_t PIOS_MS5611_ReadADC(void)
223 {
224  if (PIOS_MS5611_Validate(dev) != 0)
225  return -1;
226 
227  uint8_t data[3];
228  data[0] = 0;
229  data[1] = 0;
230  data[2] = 0;
231 
232  static int64_t delta_temp;
233 
234  /* Read and store the 16bit result */
235  if (dev->current_conversion_type == TEMPERATURE_CONV) {
236  uint32_t raw_temperature;
237  /* Read the temperature conversion */
238  if (PIOS_MS5611_Read(MS5611_ADC_READ, data, 3) != 0)
239  return -1;
240 
241  raw_temperature = (data[0] << 16) | (data[1] << 8) | data[2];
242 
243  delta_temp = ((int32_t) raw_temperature) - (dev->calibration[4] << 8);
244  dev->temperature_unscaled = 2000l + ((delta_temp * dev->calibration[5]) >> 23);
245 
246  } else {
247  int64_t offset;
248  int64_t sens;
249  uint32_t raw_pressure;
250 
251  /* Read the pressure conversion */
252  if (PIOS_MS5611_Read(MS5611_ADC_READ, data, 3) != 0)
253  return -1;
254  raw_pressure = ((data[0] << 16) | (data[1] << 8) | data[2]);
255 
256  offset = (((int64_t) dev->calibration[1]) << 16) + ((((int64_t) dev->calibration[3]) * delta_temp) >> 7);
257  sens = ((int64_t) dev->calibration[0]) << 15;
258  sens = sens + ((((int64_t) dev->calibration[2]) * delta_temp) >> 8);
259 
260  dev->pressure_unscaled = (((((int64_t) raw_pressure) * sens) >> 21) - offset) >> 15;
261  }
262  return 0;
263 }

355 static void PIOS_MS5611_Task(void *parameters)
356 {
357  int32_t temp_press_interleave_count;
358  if (PIOS_MS5611_Validate(dev) != 0)
359  temp_press_interleave_count = 1;
360  else
361  temp_press_interleave_count = dev->temperature_interleaving;
362 
363  // If device handle isn't validate pause
364  while (PIOS_MS5611_Validate(dev) != 0) {
365  vTaskDelay(1000);
366  }
367 
368  while (1) {
369 
370  temp_press_interleave_count --;
371  if(temp_press_interleave_count <= 0)
372  {
373  // Update the temperature data
374  PIOS_MS5611_StartADC(TEMPERATURE_CONV);
375  vTaskDelay(PIOS_MS5611_GetDelay());
376  PIOS_MS5611_ReadADC();
377 
378  temp_press_interleave_count = dev->temperature_interleaving;
379  }
380 
381  // Update the pressure data
382  PIOS_MS5611_StartADC(PRESSURE_CONV);
383  vTaskDelay(PIOS_MS5611_GetDelay());
384  PIOS_MS5611_ReadADC();
385 
386  // Compute the altitude from the pressure and temperature and send it out
387  struct pios_sensor_baro_data data;
388  data.temperature = ((float) dev->temperature_unscaled) / 100.0f;
389  data.pressure = ((float) dev->pressure_unscaled) / 1000.0f;
390  data.altitude = 44330.0f * (1.0f - powf(data.pressure / MS5611_P0, (1.0f / 5.255f)));
391 
392  xQueueSend(dev->queue, (void*)&data, 0);
393  }
