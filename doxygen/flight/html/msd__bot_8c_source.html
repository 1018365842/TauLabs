44 /*******************************************************************************
45 * Function Name : MSD_Mass_Storage_In
46 * Description : Mass Storage IN transfer.
47 * Input : None.
48 * Output : None.
49 * Return : None.
50 *******************************************************************************/
51 void MSD_Mass_Storage_In (void)
52 {
53  switch (MSD_Bot_State)
54  {
55  case BOT_CSW_Send:
56  case BOT_ERROR:
57  MSD_Bot_State = BOT_IDLE;
58  SetEPRxStatus(ENDP2, EP_RX_VALID);/* enable the Endpoint to recive the next cmd*/
59  break;
60  case BOT_DATA_IN:
61  switch (MSD_CBW.CB[0])
62  {
63  case SCSI_READ10:
64  MSD_SCSI_Read10_Cmd(MSD_CBW.bLUN , SCSI_LBA , SCSI_BlkLen);
65  break;
66  }
67  break;
68  case BOT_DATA_IN_LAST:
69  MSD_Set_CSW (CSW_CMD_PASSED, SEND_CSW_ENABLE);
70  SetEPRxStatus(ENDP2, EP_RX_VALID);
71  break;
72 
73  default:
74  break;
75  }
76 }

78 /*******************************************************************************
79 * Function Name : MSD_Mass_Storage_Out
80 * Description : Mass Storage OUT transfer.
81 * Input : None.
82 * Output : None.
83 * Return : None.
84 *******************************************************************************/
85 void MSD_Mass_Storage_Out (void)
86 {
87  uint8_t CMD;
88  CMD = MSD_CBW.CB[0];
89  MSD_Data_Len = GetEPRxCount(ENDP2);
90 
91  PMAToUserBufferCopy(MSD_Bulk_Data_Buff, MSD_ENDP2_RXADDR, MSD_Data_Len);
92 
93  switch (MSD_Bot_State)
94  {
95  case BOT_IDLE:
96  MSD_CBW_Decode();
97  break;
98  case BOT_DATA_OUT:
99  if (CMD == SCSI_WRITE10)
100  {
101  MSD_SCSI_Write10_Cmd(MSD_CBW.bLUN , SCSI_LBA , SCSI_BlkLen);
102  break;
103  }
104  MSD_Bot_Abort(DIR_OUT);
105  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_FIELED_IN_COMMAND);
106  MSD_Set_CSW (CSW_PHASE_ERROR, SEND_CSW_DISABLE);
107  break;
108  default:
109  MSD_Bot_Abort(BOTH_DIR);
110  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_FIELED_IN_COMMAND);
111  MSD_Set_CSW (CSW_PHASE_ERROR, SEND_CSW_DISABLE);
112  break;
113  }
114 }

116 /*******************************************************************************
117 * Function Name : MSD_CBW_Decode
118 * Description : Decode the received CBW and call the related SCSI command
119 * routine.
120 * Input : None.
121 * Output : None.
122 * Return : None.
123 *******************************************************************************/
124 void MSD_CBW_Decode(void)
125 {
126  uint32_t Counter;
127 
128  for (Counter = 0; Counter < MSD_Data_Len; Counter++)
129  {
130  *((uint8_t *)&MSD_CBW + Counter) = MSD_Bulk_Data_Buff[Counter];
131  }
132  MSD_CSW.dTag = MSD_CBW.dTag;
133  MSD_CSW.dDataResidue = MSD_CBW.dDataLength;
134  if (MSD_Data_Len != BOT_CBW_PACKET_LENGTH)
135  {
136  MSD_Bot_Abort(BOTH_DIR);
137  /* reset the MSD_CBW.dSignature to desible the clear feature until receiving a Mass storage reset*/
138  MSD_CBW.dSignature = 0;
139  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, PARAMETER_LIST_LENGTH_ERROR);
140  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_DISABLE);
141  return;
142  }
143 
144  if ((MSD_CBW.CB[0] == SCSI_READ10 ) || (MSD_CBW.CB[0] == SCSI_WRITE10 ))
145  {
146  /* Calculate Logical Block Address */
147  SCSI_LBA = (MSD_CBW.CB[2] << 24) | (MSD_CBW.CB[3] << 16) | (MSD_CBW.CB[4] << 8) | MSD_CBW.CB[5];
148  /* Calculate the Number of Blocks to transfer */
149  SCSI_BlkLen = (MSD_CBW.CB[7] << 8) | MSD_CBW.CB[8];
150  }
151 
152  if (MSD_CBW.dSignature == BOT_CBW_SIGNATURE)
153  {
154  /* Valid CBW */
155  if ((MSD_CBW.bLUN >= MSD_NUM_LUN) || (MSD_CBW.bCBLength < 1) || (MSD_CBW.bCBLength > 16))
156  {
157  MSD_Bot_Abort(BOTH_DIR);
158  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_FIELED_IN_COMMAND);
159  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_DISABLE);
160  }
161  else
162  {
163  switch (MSD_CBW.CB[0])
164  {
165  case SCSI_REQUEST_SENSE:
166  MSD_SCSI_RequestSense_Cmd (MSD_CBW.bLUN);
167  break;
168  case SCSI_INQUIRY:
169  MSD_SCSI_Inquiry_Cmd(MSD_CBW.bLUN);
170  break;
171  case SCSI_START_STOP_UNIT:
172  MSD_SCSI_Start_Stop_Unit_Cmd(MSD_CBW.bLUN);
173  MSD_LUN_AvailableSet(MSD_CBW.bLUN, 0);
174  break;
175  case SCSI_ALLOW_MEDIUM_REMOVAL:
176  MSD_SCSI_Start_Stop_Unit_Cmd(MSD_CBW.bLUN);
177  break;
178  case SCSI_MODE_SENSE6:
179  MSD_SCSI_ModeSense6_Cmd (MSD_CBW.bLUN);
180  break;
181  case SCSI_MODE_SENSE10:
182  MSD_SCSI_ModeSense10_Cmd (MSD_CBW.bLUN);
183  break;
184  case SCSI_READ_FORMAT_CAPACITIES:
185  MSD_SCSI_ReadFormatCapacity_Cmd(MSD_CBW.bLUN);
186  break;
187  case SCSI_READ_CAPACITY10:
188  MSD_SCSI_ReadCapacity10_Cmd(MSD_CBW.bLUN);
189  break;
190  case SCSI_TEST_UNIT_READY:
191  MSD_SCSI_TestUnitReady_Cmd(MSD_CBW.bLUN);
192  break;
193  case SCSI_READ10:
194  MSD_SCSI_Read10_Cmd(MSD_CBW.bLUN, SCSI_LBA , SCSI_BlkLen);
195  break;
196  case SCSI_WRITE10:
197  MSD_SCSI_Write10_Cmd(MSD_CBW.bLUN, SCSI_LBA , SCSI_BlkLen);
198  break;
199  case SCSI_VERIFY10:
200  MSD_SCSI_Verify10_Cmd(MSD_CBW.bLUN);
201  break;
202  case SCSI_FORMAT_UNIT:
203  MSD_SCSI_Format_Cmd(MSD_CBW.bLUN);
204  break;
205  /*Unsupported command*/
206 
207  case SCSI_MODE_SELECT10:
208  MSD_SCSI_Mode_Select10_Cmd(MSD_CBW.bLUN);
209  break;
210  case SCSI_MODE_SELECT6:
211  MSD_SCSI_Mode_Select6_Cmd(MSD_CBW.bLUN);
212  break;
213 
214  case SCSI_SEND_DIAGNOSTIC:
215  MSD_SCSI_Send_Diagnostic_Cmd(MSD_CBW.bLUN);
216  break;
217  case SCSI_READ6:
218  MSD_SCSI_Read6_Cmd(MSD_CBW.bLUN);
219  break;
220  case SCSI_READ12:
221  MSD_SCSI_Read12_Cmd(MSD_CBW.bLUN);
222  break;
223  case SCSI_READ16:
224  MSD_SCSI_Read16_Cmd(MSD_CBW.bLUN);
225  break;
226  case SCSI_READ_CAPACITY16:
227  MSD_SCSI_READ_CAPACITY16_Cmd(MSD_CBW.bLUN);
228  break;
229  case SCSI_WRITE6:
230  MSD_SCSI_Write6_Cmd(MSD_CBW.bLUN);
231  break;
232  case SCSI_WRITE12:
233  MSD_SCSI_Write12_Cmd(MSD_CBW.bLUN);
234  break;
235  case SCSI_WRITE16:
236  MSD_SCSI_Write16_Cmd(MSD_CBW.bLUN);
237  break;
238  case SCSI_VERIFY12:
239  MSD_SCSI_Verify12_Cmd(MSD_CBW.bLUN);
240  break;
241  case SCSI_VERIFY16:
242  MSD_SCSI_Verify16_Cmd(MSD_CBW.bLUN);
243  break;
244 
245  default:
246  {
247  MSD_Bot_Abort(BOTH_DIR);
248  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_COMMAND);
249  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_DISABLE);
250  }
251  }
252  }
253  }
254  else
255  {
256  /* Invalid CBW */
257  MSD_Bot_Abort(BOTH_DIR);
258  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_COMMAND);
259  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_DISABLE);
260  }
261 }

263 /*******************************************************************************
264 * Function Name : MSD_Transfer_Data_Request
265 * Description : Send the request response to the PC HOST.
266 * Input : uint8_t* Data_Address : point to the data to transfer.
267 * uint16_t Data_Length : the nember of Bytes to transfer.
268 * Output : None.
269 * Return : None.
270 *******************************************************************************/
271 void MSD_Transfer_Data_Request(uint8_t* Data_Pointer, uint16_t Data_Len)
272 {
273  UserToPMABufferCopy(Data_Pointer, MSD_ENDP1_TXADDR, Data_Len);
274 
275  SetEPTxCount(ENDP1, Data_Len);
276  SetEPTxStatus(ENDP1, EP_TX_VALID);
277  MSD_Bot_State = BOT_DATA_IN_LAST;
278  MSD_CSW.dDataResidue -= Data_Len;
279  MSD_CSW.bStatus = CSW_CMD_PASSED;
280 }

282 /*******************************************************************************
283 * Function Name : MSD_Set_CSW
284 * Description : Set the SCW with the needed fields.
285 * Input : uint8_t CSW_Status this filed can be CSW_CMD_PASSED,CSW_CMD_FAILED,
286 * or CSW_PHASE_ERROR.
287 * Output : None.
288 * Return : None.
289 *******************************************************************************/
290 void MSD_Set_CSW (uint8_t CSW_Status, uint8_t Send_Permission)
291 {
292  MSD_CSW.dSignature = BOT_CSW_SIGNATURE;
293  MSD_CSW.bStatus = CSW_Status;
294 
295  UserToPMABufferCopy(((uint8_t *)& MSD_CSW), MSD_ENDP1_TXADDR, CSW_DATA_LENGTH);
296 
297  SetEPTxCount(ENDP1, CSW_DATA_LENGTH);
298  MSD_Bot_State = BOT_ERROR;
299  if (Send_Permission)
300  {
301  MSD_Bot_State = BOT_CSW_Send;
302  SetEPTxStatus(ENDP1, EP_TX_VALID);
303  }
304 
305 }

307 /*******************************************************************************
308 * Function Name : MSD_Bot_Abort
309 * Description : Stall the needed Endpoint according to the selected direction.
310 * Input : Endpoint direction IN, OUT or both directions
311 * Output : None.
312 * Return : None.
313 *******************************************************************************/
314 void MSD_Bot_Abort(uint8_t Direction)
315 {
316  switch (Direction)
317  {
318  case DIR_IN :
319  SetEPTxStatus(ENDP1, EP_TX_STALL);
320  break;
321  case DIR_OUT :
322  SetEPRxStatus(ENDP2, EP_RX_STALL);
323  break;
324  case BOTH_DIR :
325  SetEPTxStatus(ENDP1, EP_TX_STALL);
326  SetEPRxStatus(ENDP2, EP_RX_STALL);
327  break;
328  default:
329  break;
330  }
331 }
