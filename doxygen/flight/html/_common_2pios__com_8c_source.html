113 int32_t PIOS_COM_Init(uintptr_t * com_id, const struct pios_com_driver * driver, uint32_t lower_id, uint8_t * rx_buffer, uint16_t rx_buffer_len, uint8_t * tx_buffer, uint16_t tx_buffer_len)
114 {
115  PIOS_Assert(com_id);
116  PIOS_Assert(driver);
117 
118  bool has_rx = (rx_buffer && rx_buffer_len > 0);
119  bool has_tx = (tx_buffer && tx_buffer_len > 0);
120  PIOS_Assert(has_rx || has_tx);
121  PIOS_Assert(driver->bind_tx_cb || !has_tx);
122  PIOS_Assert(driver->bind_rx_cb || !has_rx);
123 
124  struct pios_com_dev * com_dev;
125 
126  com_dev = (struct pios_com_dev *) PIOS_COM_alloc();
127  if (!com_dev) goto out_fail;
128 
129  com_dev->driver = driver;
130  com_dev->lower_id = lower_id;
131 
132  com_dev->has_rx = has_rx;
133  com_dev->has_tx = has_tx;
134 
135  if (has_rx) {
136  fifoBuf_init(&com_dev->rx, rx_buffer, rx_buffer_len);
137 #if defined(PIOS_INCLUDE_FREERTOS)
138  vSemaphoreCreateBinary(com_dev->rx_sem);
139 #endif /* PIOS_INCLUDE_FREERTOS */
140  (com_dev->driver->bind_rx_cb)(lower_id, PIOS_COM_RxInCallback, (uint32_t)com_dev);
141  if (com_dev->driver->rx_start) {
142  /* Start the receiver */
143  (com_dev->driver->rx_start)(com_dev->lower_id,
144  fifoBuf_getFree(&com_dev->rx));
145  }
146  }
147 
148  if (has_tx) {
149  fifoBuf_init(&com_dev->tx, tx_buffer, tx_buffer_len);
150 #if defined(PIOS_INCLUDE_FREERTOS)
151  vSemaphoreCreateBinary(com_dev->tx_sem);
152 #endif /* PIOS_INCLUDE_FREERTOS */
153  (com_dev->driver->bind_tx_cb)(lower_id, PIOS_COM_TxOutCallback, (uint32_t)com_dev);
154  }
155 
156  *com_id = (uintptr_t)com_dev;
157  return(0);
158 
159 out_fail:
160  return(-1);
161 }

195 static uint16_t PIOS_COM_RxInCallback(uint32_t context, uint8_t * buf, uint16_t buf_len, uint16_t * headroom, bool * need_yield)
196 {
197  struct pios_com_dev * com_dev = (struct pios_com_dev *)context;
198 
199  bool valid = PIOS_COM_validate(com_dev);
200  PIOS_Assert(valid);
201  PIOS_Assert(com_dev->has_rx);
202 
203  uint16_t bytes_into_fifo = fifoBuf_putData(&com_dev->rx, buf, buf_len);
204 
205  if (bytes_into_fifo > 0) {
206  /* Data has been added to the buffer */
207  PIOS_COM_UnblockRx(com_dev, need_yield);
208  }
209 
210  if (headroom) {
211  *headroom = fifoBuf_getFree(&com_dev->rx);
212  }
213 
214  return (bytes_into_fifo);
215 }

275 int32_t PIOS_COM_SendBufferNonBlocking(uintptr_t com_id, const uint8_t *buffer, uint16_t len)
276 {
277  struct pios_com_dev * com_dev = (struct pios_com_dev *)com_id;
278 
279  if (!PIOS_COM_validate(com_dev)) {
280  /* Undefined COM port for this board (see pios_board.c) */
281  return -1;
282  }
283 
284  PIOS_Assert(com_dev->has_tx);
285 
286  if (com_dev->driver->available && !com_dev->driver->available(com_dev->lower_id)) {
287  /*
288  * Underlying device is down/unconnected.
289  * Dump our fifo contents and act like an infinite data sink.
290  * Failure to do this results in stale data in the fifo as well as
291  * possibly having the caller block trying to send to a device that's
292  * no longer accepting data.
293  */
294  fifoBuf_clearData(&com_dev->tx);
295  return len;
296  }
297 
298  if (len > fifoBuf_getFree(&com_dev->tx)) {
299  /* Buffer cannot accept all requested bytes (retry) */
300  return -2;
301  }
302 
303  uint16_t bytes_into_fifo = fifoBuf_putData(&com_dev->tx, buffer, len);
304 
305  if (bytes_into_fifo > 0) {
306  /* More data has been put in the tx buffer, make sure the tx is started */
307  if (com_dev->driver->tx_start) {
308  com_dev->driver->tx_start(com_dev->lower_id,
309  fifoBuf_getUsed(&com_dev->tx));
310  }
311  }
312 
313  return (bytes_into_fifo);
314 }

325 int32_t PIOS_COM_SendBuffer(uintptr_t com_id, const uint8_t *buffer, uint16_t len)
326 {
327  struct pios_com_dev * com_dev = (struct pios_com_dev *)com_id;
328 
329  if (!PIOS_COM_validate(com_dev)) {
330  /* Undefined COM port for this board (see pios_board.c) */
331  return -1;
332  }
333 
334  PIOS_Assert(com_dev->has_tx);
335 
336  uint32_t max_frag_len = fifoBuf_getSize(&com_dev->tx);
337  uint32_t bytes_to_send = len;
338  while (bytes_to_send) {
339  uint32_t frag_size;
340 
341  if (bytes_to_send > max_frag_len) {
342  frag_size = max_frag_len;
343  } else {
344  frag_size = bytes_to_send;
345  }
346  int32_t rc = PIOS_COM_SendBufferNonBlocking(com_id, buffer, frag_size);
347  if (rc >= 0) {
348  bytes_to_send -= rc;
349  buffer += rc;
350  } else {
351  switch (rc) {
352  case -1:
353  /* Device is invalid, this will never work */
354  return -1;
355  case -2:
356  /* Device is busy, wait for the underlying device to free some space and retry */
357  /* Make sure the transmitter is running while we wait */
358  if (com_dev->driver->tx_start) {
359  (com_dev->driver->tx_start)(com_dev->lower_id,
360  fifoBuf_getUsed(&com_dev->tx));
361  }
362 #if defined(PIOS_INCLUDE_FREERTOS)
363  if (xSemaphoreTake(com_dev->tx_sem, 5000) != pdTRUE) {
364  return -3;
365  }
366 #endif
367  continue;
368  default:
369  /* Unhandled return code */
370  return rc;
371  }
372  }
373  }
374 
375  return len;
376 }

477 uint16_t PIOS_COM_ReceiveBuffer(uintptr_t com_id, uint8_t * buf, uint16_t buf_len, uint32_t timeout_ms)
478 {
479  PIOS_Assert(buf);
480  PIOS_Assert(buf_len);
481  uint16_t bytes_from_fifo;
482 
483  struct pios_com_dev * com_dev = (struct pios_com_dev *)com_id;
484 
485  if (!PIOS_COM_validate(com_dev)) {
486  /* Undefined COM port for this board (see pios_board.c) */
487  PIOS_Assert(0);
488  }
489  PIOS_Assert(com_dev->has_rx);
490 
491  check_again:
492  bytes_from_fifo = fifoBuf_getData(&com_dev->rx, buf, buf_len);
493 
494  if (bytes_from_fifo == 0) {
495  /* No more bytes in receive buffer */
496  /* Make sure the receiver is running while we wait */
497  if (com_dev->driver->rx_start) {
498  /* Notify the lower layer that there is now room in the rx buffer */
499  (com_dev->driver->rx_start)(com_dev->lower_id,
500  fifoBuf_getFree(&com_dev->rx));
501  }
502  if (timeout_ms > 0) {
503 #if defined(PIOS_INCLUDE_FREERTOS)
504  if (xSemaphoreTake(com_dev->rx_sem, timeout_ms / portTICK_RATE_MS) == pdTRUE) {
505  /* Make sure we don't come back here again */
506  timeout_ms = 0;
507  goto check_again;
508  }
509 #else
510  PIOS_DELAY_WaitmS(1);
511  timeout_ms--;
512  goto check_again;
513 #endif
514  }
515  }
516 
517  /* Return received byte */
518  return (bytes_from_fifo);
519 }
