48 int32_t HwQuantonInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(HWQUANTON_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(HWQUANTON_OBJID,
56  HWQUANTON_ISSINGLEINST, HWQUANTON_ISSETTINGS, HWQUANTON_NUMBYTES, &HwQuantonSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

74 void HwQuantonSetDefaults(UAVObjHandle obj, uint16_t instId)
75 {
76  HwQuantonData data;
77  UAVObjMetadata metadata;
78 
79  // Initialize object fields to their default values
80  UAVObjGetInstanceData(obj, instId, &data);
81  memset(&data, 0, sizeof(HwQuantonData));
82  data.RcvrPort = 1;
83  data.Uart1 = 0;
84  data.Uart2 = 0;
85  data.Uart3 = 0;
86  data.Uart4 = 1;
87  data.Uart5 = 2;
88  data.USB_HIDPort = 0;
89  data.USB_VCPPort = 3;
90  data.DSMxBind = 0;
91  data.GyroRange = 1;
92  data.AccelRange = 2;
93  data.MPU6000Rate = 3;
94  data.MPU6000DLPF = 0;
95 
96  UAVObjSetInstanceData(obj, instId, &data);
97 
98  // Initialize object metadata to their default values
99  metadata.flags =
100  ACCESS_READWRITE << UAVOBJ_ACCESS_SHIFT |
101  ACCESS_READWRITE << UAVOBJ_GCS_ACCESS_SHIFT |
102  1 << UAVOBJ_TELEMETRY_ACKED_SHIFT |
103  1 << UAVOBJ_GCS_TELEMETRY_ACKED_SHIFT |
104  UPDATEMODE_ONCHANGE << UAVOBJ_TELEMETRY_UPDATE_MODE_SHIFT |
105  UPDATEMODE_ONCHANGE << UAVOBJ_GCS_TELEMETRY_UPDATE_MODE_SHIFT;
106  metadata.telemetryUpdatePeriod = 0;
107  metadata.gcsTelemetryUpdatePeriod = 0;
108  metadata.loggingUpdatePeriod = 0;
109  UAVObjSetMetadata(obj, &metadata);
110 }
