70 int32_t BatteryInitialize(void)
71 {
72 #ifdef MODULE_Battery_BUILTIN
73  module_enabled = true;
74 #else
75  uint8_t module_state[MODULESETTINGS_ADMINSTATE_NUMELEM];
76  ModuleSettingsAdminStateGet(module_state);
77  if (module_state[MODULESETTINGS_ADMINSTATE_BATTERY] == MODULESETTINGS_ADMINSTATE_ENABLED) {
78  module_enabled = true;
79  } else {
80  module_enabled = false;
81  return 0;
82  }
83 #endif
84  ADCRoutingInitialize();
85  uint8_t adc_channel_map[ADCROUTING_CHANNELMAP_NUMELEM];
86  ADCRoutingChannelMapGet(adc_channel_map);
87 
88  //Determine if the battery sensors are routed to ADC pins
89  for (int i = 0; i < ADCROUTING_CHANNELMAP_NUMELEM; i++) {
90  if (adc_channel_map[i] == ADCROUTING_CHANNELMAP_BATTERYVOLTAGE) {
91  voltageADCPin = i;
92  }
93  if (adc_channel_map[i] == ADCROUTING_CHANNELMAP_BATTERYCURRENT) {
94  currentADCPin = i;
95  }
96  }
97 
98  //Don't enable module if no ADC pins are routed to the sensors
99  if (voltageADCPin < 0 && currentADCPin < 0)
100  module_enabled = false;
101 
102  //Start module
103  if (module_enabled) {
104  FlightBatteryStateInitialize();
105  FlightBatterySettingsInitialize();
106  }
107 
108  return 0;
109 }

116 static void batteryTask(void * parameters)
117 {
118  static FlightBatteryStateData flightBatteryData;
119  FlightBatterySettingsData batterySettings;
120 
121  FlightBatterySettingsGet(&batterySettings);
122 
123  static float dT = SAMPLE_PERIOD_MS / 1000.0f;
124  float energyRemaining;
125  // Main task loop
126  portTickType lastSysTime;
127  lastSysTime = xTaskGetTickCount();
128  while (true) {
129  vTaskDelayUntil(&lastSysTime, SAMPLE_PERIOD_MS / portTICK_RATE_MS);
130 
131  //calculate the battery parameters
132  if (voltageADCPin >= 0) {
133  flightBatteryData.Voltage = ((float) PIOS_ADC_GetChannel(voltageADCPin)) * batterySettings.SensorCalibrations[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONS_VOLTAGEFACTOR]; //in Volts
134  } else {
135  flightBatteryData.Voltage = 0; //Dummy placeholder value. This is in case we get another source of battery current which is not from the ADC
136  }
137 
138  if (currentADCPin >= 0) {
139  flightBatteryData.Current = ((float) PIOS_ADC_GetChannel(currentADCPin)) * batterySettings.SensorCalibrations[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONS_CURRENTFACTOR]; //in Amps
140  if (flightBatteryData.Current > flightBatteryData.PeakCurrent)
141  flightBatteryData.PeakCurrent = flightBatteryData.Current; //in Amps
142  } else { //If there's no current measurement, we still need to assign one. Make it negative, so it can never trigger an alarm
143  flightBatteryData.Current = 0; //Dummy placeholder value. This is in case we get another source of battery current which is not from the ADC
144  }
145 
146  flightBatteryData.ConsumedEnergy += (flightBatteryData.Current * dT * 1000.0f / 3600.0f); //in mAh
147 
148  //Apply a 2 second rise time low-pass filter to average the current
149  float alpha = 1.0f - dT / (dT + 2.0f);
150  flightBatteryData.AvgCurrent = alpha * flightBatteryData.AvgCurrent + (1 - alpha) * flightBatteryData.Current; //in Amps
151 
152  energyRemaining = batterySettings.Capacity - flightBatteryData.ConsumedEnergy; // in mAh
153  if (flightBatteryData.AvgCurrent > 0)
154  flightBatteryData.EstimatedFlightTime = (energyRemaining / (flightBatteryData.AvgCurrent * 1000.0f)) * 3600.0f; //in Sec
155  else
156  flightBatteryData.EstimatedFlightTime = 9999;
157 
158  //generate alarms where needed...
159  if ((flightBatteryData.Voltage <= 0) && (flightBatteryData.Current <= 0)) {
160  //FIXME: There's no guarantee that a floating ADC will give 0. So this
161  // check might fail, even when there's nothing attached.
162  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_ERROR);
163  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_ERROR);
164  } else {
165  // FIXME: should make the timer alarms user configurable
166  if (flightBatteryData.EstimatedFlightTime < 30)
167  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_CRITICAL);
168  else if (flightBatteryData.EstimatedFlightTime < 120)
169  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_WARNING);
170  else
171  AlarmsClear(SYSTEMALARMS_ALARM_FLIGHTTIME);
172 
173  // FIXME: should make the battery voltage detection dependent on battery type.
174  /*Not so sure. Some users will want to run their batteries harder than others, so it should be the user's choice. [KDS]*/
175  if (flightBatteryData.Voltage < batterySettings.VoltageThresholds[FLIGHTBATTERYSETTINGS_VOLTAGETHRESHOLDS_ALARM])
176  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_CRITICAL);
177  else if (flightBatteryData.Voltage < batterySettings.VoltageThresholds[FLIGHTBATTERYSETTINGS_VOLTAGETHRESHOLDS_WARNING])
178  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_WARNING);
179  else
180  AlarmsClear(SYSTEMALARMS_ALARM_BATTERY);
181  }
182 
183  FlightBatteryStateSet(&flightBatteryData);
184  }
185 }
