72 void arm_biquad_cascade_df1_fast_q15(
73  const arm_biquad_casd_df1_inst_q15 * S,
74  q15_t * pSrc,
75  q15_t * pDst,
76  uint32_t blockSize)
77 {
78  q15_t *pIn = pSrc; /* Source pointer */
79  q15_t *pOut = pDst; /* Destination pointer */
80  q31_t in; /* Temporary variable to hold input value */
81  q31_t out; /* Temporary variable to hold output value */
82  q31_t b0; /* Temporary variable to hold bo value */
83  q31_t b1, a1; /* Filter coefficients */
84  q31_t state_in, state_out; /* Filter state variables */
85  q31_t acc; /* Accumulator */
86  int32_t shift = (int32_t) (15 - S->postShift); /* Post shift */
87  q15_t *pState = S->pState; /* State pointer */
88  q15_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
89  uint32_t sample, stage = S->numStages; /* Stage loop counter */
90 
91 
92 
93  do
94  {
95 
96  /* Read the b0 and 0 coefficients using SIMD */
97  b0 = *__SIMD32(pCoeffs)++;
98 
99  /* Read the b1 and b2 coefficients using SIMD */
100  b1 = *__SIMD32(pCoeffs)++;
101 
102  /* Read the a1 and a2 coefficients using SIMD */
103  a1 = *__SIMD32(pCoeffs)++;
104 
105  /* Read the input state values from the state buffer: x[n-1], x[n-2] */
106  state_in = *__SIMD32(pState)++;
107 
108  /* Read the output state values from the state buffer: y[n-1], y[n-2] */
109  state_out = *__SIMD32(pState)--;
110 
111  /* Apply loop unrolling and compute 2 output values simultaneously. */
112  /* The variable acc hold output values that are being computed:
113  *
114  * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
115  * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
116  */
117  sample = blockSize >> 1u;
118 
119  /* First part of the processing with loop unrolling. Compute 2 outputs at a time.
120  ** a second loop below computes the remaining 1 sample. */
121  while(sample > 0u)
122  {
123 
124  /* Read the input */
125  in = *__SIMD32(pIn)++;
126 
127  /* out = b0 * x[n] + 0 * 0 */
128  out = __SMUAD(b0, in);
129  /* acc = b1 * x[n-1] + acc += b2 * x[n-2] + out */
130  acc = __SMLAD(b1, state_in, out);
131  /* acc += a1 * y[n-1] + acc += a2 * y[n-2] */
132  acc = __SMLAD(a1, state_out, acc);
133 
134  /* The result is converted from 3.29 to 1.31 and then saturation is applied */
135  out = __SSAT((acc >> shift), 16);
136 
137  /* Every time after the output is computed state should be updated. */
138  /* The states should be updated as: */
139  /* Xn2 = Xn1 */
140  /* Xn1 = Xn */
141  /* Yn2 = Yn1 */
142  /* Yn1 = acc */
143  /* x[n-N], x[n-N-1] are packed together to make state_in of type q31 */
144  /* y[n-N], y[n-N-1] are packed together to make state_out of type q31 */
145 
146 #ifndef ARM_MATH_BIG_ENDIAN
147 
148  state_in = __PKHBT(in, state_in, 16);
149  state_out = __PKHBT(out, state_out, 16);
150 
151 #else
152 
153  state_in = __PKHBT(state_in >> 16, (in >> 16), 16);
154  state_out = __PKHBT(state_out >> 16, (out), 16);
155 
156 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
157 
158  /* out = b0 * x[n] + 0 * 0 */
159  out = __SMUADX(b0, in);
160  /* acc0 = b1 * x[n-1] , acc0 += b2 * x[n-2] + out */
161  acc = __SMLAD(b1, state_in, out);
162  /* acc += a1 * y[n-1] + acc += a2 * y[n-2] */
163  acc = __SMLAD(a1, state_out, acc);
164 
165  /* The result is converted from 3.29 to 1.31 and then saturation is applied */
166  out = __SSAT((acc >> shift), 16);
167 
168 
169  /* Store the output in the destination buffer. */
170 
171 #ifndef ARM_MATH_BIG_ENDIAN
172 
173  *__SIMD32(pOut)++ = __PKHBT(state_out, out, 16);
174 
175 #else
176 
177  *__SIMD32(pOut)++ = __PKHBT(out, state_out >> 16, 16);
178 
179 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
180 
181  /* Every time after the output is computed state should be updated. */
182  /* The states should be updated as: */
183  /* Xn2 = Xn1 */
184  /* Xn1 = Xn */
185  /* Yn2 = Yn1 */
186  /* Yn1 = acc */
187  /* x[n-N], x[n-N-1] are packed together to make state_in of type q31 */
188  /* y[n-N], y[n-N-1] are packed together to make state_out of type q31 */
189 
190 #ifndef ARM_MATH_BIG_ENDIAN
191 
192  state_in = __PKHBT(in >> 16, state_in, 16);
193  state_out = __PKHBT(out, state_out, 16);
194 
195 #else
196 
197  state_in = __PKHBT(state_in >> 16, in, 16);
198  state_out = __PKHBT(state_out >> 16, out, 16);
199 
200 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
201 
202 
203  /* Decrement the loop counter */
204  sample--;
205 
206  }
207 
208  /* If the blockSize is not a multiple of 2, compute any remaining output samples here.
209  ** No loop unrolling is used. */
210 
211  if((blockSize & 0x1u) != 0u)
212  {
213  /* Read the input */
214  in = *pIn++;
215 
216  /* out = b0 * x[n] + 0 * 0 */
217 
218 #ifndef ARM_MATH_BIG_ENDIAN
219 
220  out = __SMUAD(b0, in);
221 
222 #else
223 
224  out = __SMUADX(b0, in);
225 
226 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
227 
228  /* acc = b1 * x[n-1], acc += b2 * x[n-2] + out */
229  acc = __SMLAD(b1, state_in, out);
230  /* acc += a1 * y[n-1] + acc += a2 * y[n-2] */
231  acc = __SMLAD(a1, state_out, acc);
232 
233  /* The result is converted from 3.29 to 1.31 and then saturation is applied */
234  out = __SSAT((acc >> shift), 16);
235 
236  /* Store the output in the destination buffer. */
237  *pOut++ = (q15_t) out;
238 
239  /* Every time after the output is computed state should be updated. */
240  /* The states should be updated as: */
241  /* Xn2 = Xn1 */
242  /* Xn1 = Xn */
243  /* Yn2 = Yn1 */
244  /* Yn1 = acc */
245  /* x[n-N], x[n-N-1] are packed together to make state_in of type q31 */
246  /* y[n-N], y[n-N-1] are packed together to make state_out of type q31 */
247 
248 #ifndef ARM_MATH_BIG_ENDIAN
249 
250  state_in = __PKHBT(in, state_in, 16);
251  state_out = __PKHBT(out, state_out, 16);
252 
253 #else
254 
255  state_in = __PKHBT(state_in >> 16, in, 16);
256  state_out = __PKHBT(state_out >> 16, out, 16);
257 
258 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
259 
260  }
261 
262  /* The first stage goes from the input buffer to the output buffer. */
263  /* Subsequent (numStages - 1) occur in-place in the output buffer */
264  pIn = pDst;
265 
266  /* Reset the output pointer */
267  pOut = pDst;
268 
269  /* Store the updated state variables back into the state array */
270  *__SIMD32(pState)++ = state_in;
271  *__SIMD32(pState)++ = state_out;
272 
273 
274  /* Decrement the loop counter */
275  stage--;
276 
277  } while(stage > 0u);
278 }
