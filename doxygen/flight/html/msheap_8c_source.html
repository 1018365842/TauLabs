39 /*
40  * Integrity checking.
41  *
42  * The HEAP_CHECK_LEVEL define determines the level of heap and
43  * argument integrity checking that will be performed.
44  *
45  * Higher checking levels include all the checks from lower levels.
46  *
47  * 0 : Pointers passed to msheap_free are checked; this will
48  * detect overflowing of the freed region and overflows from
49  * adjacent regions into the freed region.
50  * msheap_panic() is called if an overflow is detected.
51  *
52  * 1 : Heap elements are checked during msheap_alloc(); this will
53  * detect overflowing of any region that is seen during the
54  * allocation.
55  * msheap_panic() is called if an overflow is detected.
56  *
57  * 2 : msheap_panic() is called with more informative diagnostics
58  * including the test that failed.
59  *
60  * 3 : The entire heap is checked on every call to msheap_alloc()
61  * and msheap_free(). This will detect overflowing of any
62  * region.
63  * Every argument is checked before use.
64  * Regions are checked after operations.
65  *
66  * -1: No integrity checking of any sort is performed.
67  *
68  * HEAP_CHECK_LEVEL defaults to 2 if DEBUG is defined, or 0
69  * otherwise.
70  */
71 #ifndef HEAP_CHECK_LEVEL
72 # ifdef DEBUG
73 # define HEAP_CHECK_LEVEL 2
74 # else
75 # define HEAP_CHECK_LEVEL 0
76 # endif
77 #endif

127 /*
128  * The heap consists of ranges (free or allocated) separated and
129  * bounded by markers.
130  *
131  * For maximum space efficiency, the default is to use 4-byte
132  * 'compact' markers, which limits the heap to a maxium of 128KiB.
133  * For larger heaps define HEAP_SUPPORT_LARGE, which doubles markers
134  * to 8 bytes each, but allows heaps up to 2^^33 bytes in size.
135  *
136  * Each marker contains two structures, one describing the previous
137  * region and one describing the next. Thus, markers form a
138  * doubly-linked list chaining each region together.
139  *
140  * Each region is described by two identical structures, providing
141  * a measure of referential integrity that can be used to detect
142  * overflows out of the region without the use of separate magic
143  * numbers.
144  *
145  * The region descriptor size includes the size of the marker at its
146  * head. This means that zero is not a legal marker value.
147  *
148  * Free regions are always coalesced, and a pointer is kept to the
149  * most recently-created free region to accelerate allocation in the
150  * common case where a large number of free objects are allocated
151  * early.
152  *
153  * The heap is bounded by markers pointing to zero-sized allocated
154  * ranges, so they can never be merged.
155  */
156 #ifdef HEAP_SUPPORT_LARGE
157

210 void
211 msheap_init(void *base, void *limit)
212 {
213  heap_base = (marker_t)round_up((uintptr_t)base, marker_size);
214  heap_limit = (marker_t)round_down((uintptr_t)limit, marker_size) - 1;
215 
216  ASSERT(3, heap_base); /* must not be NULL */
217  ASSERT(3, heap_limit); /* must not be NULL */
218  ASSERT(3, heap_limit > heap_base); /* limit must be above base */
219 
220  /* Initial size of the free region (includes the heap_base marker) */
221  heap_free = heap_limit - heap_base;
222  ASSERT(0, heap_free <= max_free); /* heap must not be too large */
223  ASSERT(3, heap_free > 1); /* heap must be at least 1 marker in size */
224 
225  /*
226  * Initialise the base and limit markers.
227  */
228  heap_base->prev.size = 0;
229  heap_base->prev.free = 0;
230  heap_base->next.size = heap_free;
231  heap_base->next.free = 1;
232  heap_limit->prev.size = heap_free;
233  heap_limit->prev.free = 1;
234  heap_limit->next.size = 0;
235  heap_limit->next.free = 0;
236 
237  free_hint = heap_base; /* a good place to start ... */
238 
239  region_check(heap_base);
240  region_check(heap_limit);
241 }

243 void *
244 msheap_alloc(uint32_t size)
245 {
246  marker_t cursor;
247  marker_t best;
248 
249  ASSERT(3, msheap_check());
250 
251  /* convert the passed-in size to the number of marker-size units we need to allocate */
252  size += marker_size;
253  size = round_up(size, marker_size);
254  size /= marker_size;
255 
256  /* cannot possibly satisfy this allocation */
257  if (size > heap_free)
258  return 0;
259 
260  /* simple single-pass best-fit search */
261 restart:
262  cursor = free_hint;
263  best = 0;
264  while (cursor != heap_limit) {
265 
266  ASSERT(1, region_check(cursor));
267 
268  /* if the region is free and large enough */
269  if ((cursor->next.free) && (cursor->next.size >= size)) {
270 
271  /* if we have no candidate, or the new one is smaller, take it */
272  if (!best || (cursor->next.size < best->next.size))
273  best = cursor;
274  }
275 
276  cursor += cursor->next.size;
277  }
278 
279  if (!best) {
280  /*
281  * If we were working from the hint and found nothing, reset
282  * the hint and try again
283  */
284  if (free_hint != heap_base) {
285  free_hint = heap_base;
286  goto restart;
287  }
288 
289  /* no space */
290  return 0;
291  }
292 
293  /* split the free region to make space */
294  split_region(best, size);
295 
296  /* update free space counter */
297  heap_free -= size;
298 
299  /* and return a pointer to the allocated region */
300  return (void *)(best + 1);
301 }

303 void
304 msheap_free(void *ptr)
305 {
306  marker_t marker;
307 
308  marker = (marker_t)ptr - 1;
309 
310  ASSERT(0, region_check(marker));
311  ASSERT(3, msheap_check());
312 
313  /* this region is free, mark it accordingly */
314  marker->next.free = 1;
315  (marker + marker->next.size)->prev.free = 1;
316 
317  /* account for space we are freeing */
318  heap_free += marker->next.size;
319 
320  /* possibly merge this region and the following */
321  merge_region(marker);
322 
323  /* possibly merge this region and the preceeding */
324  if (marker->prev.free) {
325  marker -= marker->prev.size;
326  merge_region(marker);
327  }
328 
329  /*
330  * Marker now points to the new free region, so update
331  * the free hint if this has opened space earlier in the heap.
332  */
333  if (marker < free_hint)
334  free_hint = marker;
335 }

337 int
338 msheap_check(void)
339 {
340  marker_t cursor;
341  uint32_t free_space = 0;
342 
343  cursor = heap_base; /* start at the base of the heap */
344 
345  for (;;) {
346  if (ASSERT_TEST(2, region_check(cursor))) /* check the current region */
347  return 0;
348  if (cursor->next.free) /* if the region is free */
349  free_space += cursor->next.size; /* count it as free space */
350  if (cursor == heap_limit) /* if this was the last region, stop */
351  break;
352  cursor += cursor->next.size; /* next region */
353  }
354 
355  if (ASSERT_TEST(2, region_check(free_hint)))
356  return 0;
357  if (ASSERT_TEST(2, free_space == heap_free))
358  return 0;
359 
360  return 1;
361 }

383 void
384 msheap_extend(uint32_t size)
385 {
386  marker_t new_free;
387 
388  /* convert to marker-sized units (and implicitly round down) */
389  size /= marker_size;
390  if (size < 1)
391  return;
392 
393  /*
394  * We can either extend a free region immediately prior to
395  * the heap limit, or we can turn the heap limit marker
396  * into the marker for a free region.
397  */
398  if (heap_limit->prev.free) {
399  new_free = heap_limit - heap_limit->prev.size;
400  } else {
401  new_free = heap_limit;
402  }
403 
404  /* update new free region */
405  new_free->next.size += size;
406  new_free->next.free = 1;
407 
408  /* new end marker */
409  heap_limit = new_free + new_free->next.size;
410  heap_limit->prev.size = new_free->next.size;
411  heap_limit->prev.free = 1;
412  heap_limit->next.size = 0;
413  heap_limit->next.free = 0;
414 
415  ASSERT(3, msheap_check());
416 }

442 static int
443 region_check(marker_t marker)
444 {
445  marker_t other;
446 
447  if (ASSERT_TEST(2, marker) | /* not NULL */
448  ASSERT_TEST(2, !((uintptr_t)marker % marker_size)) | /* properly aligned */
449  ASSERT_TEST(2, marker >= heap_base) | /* within the heap */
450  ASSERT_TEST(2, marker <= heap_limit))
451  return 0;
452 
453  /* validate link to next marker & return link from that marker */
454  if (marker->next.size > 0) {
455  other = marker + marker->next.size;
456 
457  if (ASSERT_TEST(2, other > marker) | /* must be after */
458  ASSERT_TEST(2, other <= heap_limit) | /* must be inside the heap */
459  ASSERT_TEST(2, marker->next.size == other->prev.size) | /* sizes must match */
460  ASSERT_TEST(2, marker->next.free == other->prev.free)) /* free state must match */
461  return 0;
462  } else {
463  if (ASSERT_TEST(2, marker == heap_limit)) /* or it's the end of the heap */
464  return 0;
465  }
466 
467  /* validate link to previous marker & return link from that marker */
468  if (marker->prev.size > 0) {
469  other = marker - marker->prev.size;
470 
471  if (ASSERT_TEST(2, other < marker) | /* must be before */
472  ASSERT_TEST(2, other >= heap_base) | /* must be inside the heap */
473  ASSERT_TEST(2, marker->prev.size == other->next.size) | /* sizes must match */
474  ASSERT_TEST(2, marker->prev.free == other->next.free)) /* free state must match */
475  return 0;
476  } else {
477  if (ASSERT_TEST(2, marker == heap_base)) /* or it's the end of the heap */
478  return 0;
479  }
480 
481  /* must never be two free regions adjacent */
482  if (ASSERT_TEST(2, !(marker->prev.free && marker->next.free)))
483  return 0;
484 
485  return 1;
486 }

494 static void
495 split_region(marker_t marker, uint32_t size)
496 {
497  marker_t split, tail;
498 
499  ASSERT(1, marker->next.free); /* must be splitting a free region */
500  ASSERT(1, !marker->prev.free); /* free region must never follow a free region */
501  ASSERT(1, marker->next.size >= size); /* size must fit in region */
502 
503  ASSERT(3, size); /* split result must be at least one marker in size */
504 
505  tail = marker + marker->next.size;
506  ASSERT(1, region_check(tail)); /* validate the following region */
507 
508  /*
509  * The split marker is at the end of the allocated region; it may actually
510  * be at the end of the previous free region as well.
511  */
512  split = marker + size;
513 
514  /* describe the now-allocated region */
515  split->prev.size = size;
516  split->prev.free = 0;
517 
518  /* if there is a real split, then describe the free region */
519  if (split != tail) {
520  split->next.size = marker->next.size - size;
521  split->next.free = 1;
522  tail->prev.size = split->next.size;
523  tail->prev.free = 1;
524 
525  /*
526  * Update the allocation speedup hint to
527  * point to the new free region if we just used it.
528  */
529  if (free_hint == marker)
530  free_hint = split;
531  } else {
532 
533  /*
534  * If we just allocated all of what the free hint
535  * pointed to, reset it to the base of the heap.
536  */
537  if (free_hint == marker)
538  free_hint = heap_base;
539  }
540 
541  /* and update the allocated region */
542  marker->next.size = size;
543  marker->next.free = 0;
544 
545  ASSERT(3, region_check(marker));
546  ASSERT(3, region_check(split));
547  ASSERT(3, region_check(tail));
548 }

555 static void
556 merge_region(marker_t marker)
557 {
558  marker_t other;
559 
560  /*
561  * note - cannot region_check(marker) here as we are
562  * actively fixing adjacent free regions.
563  */
564 
565  other = marker + marker->next.size;
566 
567  /* if this region and the next region are both free, merge */
568  if (marker->next.free && other->next.free) {
569 
570  /* update region size */
571  marker->next.size += other->next.size;
572 
573  /* update the marker following the end of the merged regions */
574  other = marker + marker->next.size;
575  other->prev.size = marker->next.size;
576  }
577 }
