44 int32_t PIOS_Servo_Init(const struct pios_servo_cfg * cfg)
45 {
46  uint32_t tim_id;
47  if (PIOS_TIM_InitChannels(&tim_id, cfg->channels, cfg->num_channels, NULL, 0)) {
48  return -1;
49  }
50 
51  /* Store away the requested configuration */
52  servo_cfg = cfg;
53 
54  /* Configure the channels to be in output compare mode */
55  for (uint8_t i = 0; i < cfg->num_channels; i++) {
56  const struct pios_tim_channel * chan = &cfg->channels[i];
57 
58  /* Set up for output compare function */

61  TIM_OC1Init(chan->timer, (TIM_OCInitTypeDef*)&cfg->tim_oc_init);

65  TIM_OC2Init(chan->timer, (TIM_OCInitTypeDef*)&cfg->tim_oc_init);

67  break;

69  TIM_OC3Init(chan->timer, (TIM_OCInitTypeDef*)&cfg->tim_oc_init);
70  TIM_OC3PreloadConfig(chan->timer, TIM_OCPreload_Enable);
71  break;
72  case TIM_Channel_4:
73  TIM_OC4Init(chan->timer, (TIM_OCInitTypeDef*)&cfg->tim_oc_init);
74  TIM_OC4PreloadConfig(chan->timer, TIM_OCPreload_Enable);
75  break;
76  }

91 void PIOS_Servo_SetHz(const uint16_t * speeds, uint8_t banks)
92 {
93  if (!servo_cfg) {
94  return;
95  }
96 
97  TIM_TimeBaseInitTypeDef TIM_TimeBaseStructure = servo_cfg->tim_base_init;
98  TIM_TimeBaseStructure.TIM_ClockDivision = TIM_CKD_DIV1;
99  TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
100  //
101 
102  uint8_t set = 0;
103 
104  for(uint8_t i = 0; (i < servo_cfg->num_channels) && (set < banks); i++) {
105  bool new = true;
106  const struct pios_tim_channel * chan = &servo_cfg->channels[i];
107 
108  /* See if any previous channels use that same timer */
109  for(uint8_t j = 0; (j < i) && new; j++)
110  new &= chan->timer != servo_cfg->channels[j].timer;
111 
112  if(new) {
113  // Choose the correct prescaler value for the APB the timer is attached
114  if (chan->timer==TIM2 || chan->timer==TIM3 || chan->timer==TIM4 || chan->timer==TIM6 || chan->timer==TIM7 ){
115  //those timers run at double APB1 speed if APB1 prescaler is != 1 which is usually the case
116  TIM_TimeBaseStructure.TIM_Prescaler = (PIOS_PERIPHERAL_APB1_CLOCK / 1000000 * 2) - 1;
117  }
118  else {
119  TIM_TimeBaseStructure.TIM_Prescaler = (PIOS_PERIPHERAL_APB2_CLOCK / 1000000) - 1;
120  }
121 
122  TIM_TimeBaseStructure.TIM_Period = ((1000000 / speeds[set]) - 1);
123  TIM_TimeBaseInit(chan->timer, &TIM_TimeBaseStructure);
124  set++;
125  }
126  }
127 }
