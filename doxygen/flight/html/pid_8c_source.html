50 float pid_apply(struct pid *pid, const float err, float dT)
51 {
52  if (pid->i == 0) {
53  // If Ki is zero, reset the integrator
54  pid->iAccumulator = 0;
55  } else {
56  // Scale up accumulator by 1000 while computing to avoid losing precision
57  pid->iAccumulator += err * (pid->i * dT * 1000.0f);
58  pid->iAccumulator = bound_sym(pid->iAccumulator, pid->iLim * 1000.0f);
59  }
60 
61  // Calculate DT1 term
62  float diff = (err - pid->lastErr);
63  float dterm = 0;
64  pid->lastErr = err;
65  if(pid->d && dT)
66  {
67  dterm = pid->lastDer + dT / ( dT + deriv_tau) * ((diff * pid->d / dT) - pid->lastDer);

69  } // 7.9577e-3 means 20 Hz f_cutoff
