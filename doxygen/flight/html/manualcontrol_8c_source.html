104 static void manualControlTask(void *parameters)
105 {
106  /* Make sure disarmed on power up */
107  FlightStatusData flightStatus;
108  FlightStatusGet(&flightStatus);
109  flightStatus.Armed = FLIGHTSTATUS_ARMED_DISARMED;
110 
111  // Main task loop
112  lastSysTime = xTaskGetTickCount();
113 
114  // Select failsafe before run
115  failsafe_control_select(true);
116 
117  while (1) {
118 
119  // Process periodic data for each of the controllers, including reading
120  // all available inputs
121  failsafe_control_update();
122  transmitter_control_update();
123  tablet_control_update();
124 
125  static enum control_selection last_control_selection = CONTROL_SELECTION_FAILSAFE;
126  enum control_events control_events = CONTROL_EVENTS_NONE;
127 
128  // Control logic to select the valid controller
129  enum control_selection control_selection = transmitter_control_selected_controller();
130  bool reset_controller = control_selection != last_control_selection;
131 
132  switch(control_selection) {
133  case CONTROL_SELECTION_FAILSAFE:
134  failsafe_control_select(reset_controller);
135  control_events = failsafe_control_get_events();
136  break;
137  case CONTROL_SELECTION_TRANSMITTER:
138  transmitter_control_select(reset_controller);
139  control_events = transmitter_control_get_events();
140  break;
141  case CONTROL_SELECTION_TABLET:
142  if (tablet_control_select(reset_controller) == 0) {
143  control_events = tablet_control_get_events();
144  } else {
145  // Failure in tablet control. This would be better if done
146  // at the selection stage before the tablet is even used.
147  failsafe_control_select(false);
148  control_events = failsafe_control_get_events();
149  }
150  break;
151  }
152  last_control_selection = control_selection;
153 
154  // TODO: This can evolve into a full FSM like I2C possibly
155  switch(control_events) {
156  case CONTROL_EVENTS_NONE:
157  break;
158  case CONTROL_EVENTS_ARM:
159  control_event_arm();
160  break;
161  case CONTROL_EVENTS_ARMING:
162  control_event_arming();
163  break;
164  case CONTROL_EVENTS_DISARM:
165  control_event_disarm();
166  break;
167  }
168 
169  // Wait until next update
170  vTaskDelayUntil(&lastSysTime, UPDATE_PERIOD_MS / portTICK_RATE_MS);
171  PIOS_WDG_UpdateFlag(PIOS_WDG_MANUAL);
172  }
173 }

176 static int32_t control_event_arm()
177 {
178  if(ok_to_arm()) {
179  FlightStatusData flightStatus;
180  FlightStatusGet(&flightStatus);
181  if (flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED) {

218 static bool ok_to_arm(void)
219 {
220  // read alarms
221  SystemAlarmsData alarms;
222  SystemAlarmsGet(&alarms);
223 
224  // Check each alarm
225  for (int i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++)
226  {
227  if (alarms.Alarm[i] >= SYSTEMALARMS_ALARM_ERROR &&
228  i != SYSTEMALARMS_ALARM_GPS &&
229  i != SYSTEMALARMS_ALARM_TELEMETRY)
230  {
231  return false;
232  }
233  }
234 
235  return true;
236 }
