127 static void AutotuneTask(void *parameters)
128 {
129  //AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
130 
131  enum AUTOTUNE_STATE state = AT_INIT;
132 
133  portTickType lastUpdateTime = xTaskGetTickCount();
134 
135  while(1) {
136 
137  PIOS_WDG_UpdateFlag(PIOS_WDG_AUTOTUNE);
138  // TODO:
139  // 1. get from queue
140  // 2. based on whether it is flightstatus or manualcontrol
141 
142  portTickType diffTime;
143 
144  const uint32_t PREPARE_TIME = 2000;
145  const uint32_t MEAURE_TIME = 30000;
146 
147  FlightStatusData flightStatus;
148  FlightStatusGet(&flightStatus);
149 
150  // Only allow this module to run when autotuning
151  if (flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_AUTOTUNE) {
152  state = AT_INIT;
153  vTaskDelay(50);
154  continue;
155  }
156 
157  StabilizationDesiredData stabDesired;
158  StabilizationDesiredGet(&stabDesired);
159 
160  StabilizationSettingsData stabSettings;
161  StabilizationSettingsGet(&stabSettings);
162 
163  ManualControlSettingsData manualSettings;
164  ManualControlSettingsGet(&manualSettings);
165 
166  ManualControlCommandData manualControl;
167  ManualControlCommandGet(&manualControl);
168 
169  RelayTuningSettingsData relaySettings;
170  RelayTuningSettingsGet(&relaySettings);
171 
172  bool rate = relaySettings.Mode == RELAYTUNINGSETTINGS_MODE_RATE;
173 
174  if (rate) { // rate mode
175  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
176  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
177 
178  stabDesired.Roll = manualControl.Roll * stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_ROLL];
179  stabDesired.Pitch = manualControl.Pitch * stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_PITCH];
180  } else {
181  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
182  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
183 
184  stabDesired.Roll = manualControl.Roll * stabSettings.RollMax;
185  stabDesired.Pitch = manualControl.Pitch * stabSettings.PitchMax;
186  }
187 
188  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
189  stabDesired.Yaw = manualControl.Yaw * stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_YAW];
190  stabDesired.Throttle = manualControl.Throttle;
191 
192  switch(state) {
193  case AT_INIT:
194 
195  lastUpdateTime = xTaskGetTickCount();
196 
197  // Only start when armed and flying
198  if (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED && stabDesired.Throttle > 0)
199  state = AT_START;
200  break;
201 
202  case AT_START:
203 
204  diffTime = xTaskGetTickCount() - lastUpdateTime;
205 
206  // Spend the first block of time in normal rate mode to get airborne
207  if (diffTime > PREPARE_TIME) {
208  state = AT_ROLL;
209  lastUpdateTime = xTaskGetTickCount();
210  }
211  break;
212 
213  case AT_ROLL:
214 
215  diffTime = xTaskGetTickCount() - lastUpdateTime;
216 
217  // Run relay mode on the roll axis for the measurement time
218  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = rate ? STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYRATE :
219  STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYATTITUDE;
220  if (diffTime > MEAURE_TIME) { // Move on to next state
221  state = AT_PITCH;
222  lastUpdateTime = xTaskGetTickCount();
223  }
224  break;
225 
226  case AT_PITCH:
227 
228  diffTime = xTaskGetTickCount() - lastUpdateTime;
229 
230  // Run relay mode on the pitch axis for the measurement time
231  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = rate ? STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYRATE :
232  STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYATTITUDE;
233  if (diffTime > MEAURE_TIME) { // Move on to next state
234  state = AT_FINISHED;
235  lastUpdateTime = xTaskGetTickCount();
236  }
237  break;
238 
239  case AT_FINISHED:
240 
241  // Wait until disarmed and landed before updating the settings
242  if (flightStatus.Armed == FLIGHTSTATUS_ARMED_DISARMED && stabDesired.Throttle <= 0)
243  state = AT_SET;
244 
245  break;
246 
247  case AT_SET:
248  update_stabilization_settings();
249  state = AT_INIT;
250  break;
251 
252  default:
253  // Set an alarm or some shit like that
254  break;
255  }
256 
257  StabilizationDesiredSet(&stabDesired);
258 
259  vTaskDelay(10);
260  }
261 }

269 static void update_stabilization_settings()
270 {
271  RelayTuningData relayTuning;
272  RelayTuningGet(&relayTuning);
273 
274  RelayTuningSettingsData relaySettings;
275  RelayTuningSettingsGet(&relaySettings);
276 
277  StabilizationSettingsData stabSettings;
278  StabilizationSettingsGet(&stabSettings);
279 
280  // Eventually get these settings from RelayTuningSettings
281  const float gain_ratio_r = 1.0f / 3.0f;
282  const float zero_ratio_r = 1.0f / 3.0f;
283  const float gain_ratio_p = 1.0f / 5.0f;
284  const float zero_ratio_p = 1.0f / 5.0f;
285 
286  // For now just run over roll and pitch
287  for (uint32_t i = 0; i < 2; i++) {
288  float wu = 1000.0f * 2 * PI / relayTuning.Period[i]; // ultimate freq = output osc freq (rad/s)
289 
290  float wc = wu * gain_ratio_r; // target openloop crossover frequency (rad/s)
291  float zc = wc * zero_ratio_r; // controller zero location (rad/s)
292  float kpu = 4.0f / PI / relayTuning.Gain[i]; // ultimate gain, i.e. the proportional gain for instablity
293  float kp = kpu * gain_ratio_r; // proportional gain
294  float ki = zc * kp; // integral gain
295 
296  // Now calculate gains for the next loop out knowing it is the integral of
297  // the inner loop -- the plant is position/velocity = scale*1/s
298  float wc2 = wc * gain_ratio_p; // crossover of the attitude loop
299  float kp2 = wc2; // kp of attitude
300  float ki2 = wc2 * zero_ratio_p * kp2; // ki of attitude
301 
302  switch(i) {
303  case 0: // roll
304  stabSettings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP] = kp;
305  stabSettings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI] = ki;
306  stabSettings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KP] = kp2;
307  stabSettings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KI] = ki2;
308  break;
309  case 1: // Pitch
310  stabSettings.PitchRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP] = kp;
311  stabSettings.PitchRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI] = ki;
312  stabSettings.PitchPI[STABILIZATIONSETTINGS_ROLLPI_KP] = kp2;
313  stabSettings.PitchPI[STABILIZATIONSETTINGS_ROLLPI_KI] = ki2;
314  break;
315  default:
316  // Oh shit oh shit oh shit
317  break;
318  }
319  }
320  switch(relaySettings.Behavior) {
321  case RELAYTUNINGSETTINGS_BEHAVIOR_MEASURE:
322  // Just measure, don't update the stab settings
323  break;
324  case RELAYTUNINGSETTINGS_BEHAVIOR_COMPUTE:
325  StabilizationSettingsSet(&stabSettings);
326  break;
327  case RELAYTUNINGSETTINGS_BEHAVIOR_SAVE:
328  StabilizationSettingsSet(&stabSettings);
329  UAVObjSave(StabilizationSettingsHandle(), 0);
330  break;
331  }
332 
333 }
