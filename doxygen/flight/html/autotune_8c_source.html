108 static void AutotuneTask(void *parameters)
109 {
110  //AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
111 
112  enum AUTOTUNE_STATE state = AT_INIT;
113 
114  portTickType lastUpdateTime = xTaskGetTickCount();
115 
116  while(1) {
117 
118  PIOS_WDG_UpdateFlag(PIOS_WDG_AUTOTUNE);
119  // TODO:
120  // 1. get from queue
121  // 2. based on whether it is flightstatus or manualcontrol
122 
123  portTickType diffTime;
124 
125  const uint32_t PREPARE_TIME = 2000;
126  const uint32_t MEAURE_TIME = 30000;
127 
128  FlightStatusData flightStatus;
129  FlightStatusGet(&flightStatus);
130 
131  // Only allow this module to run when autotuning
132  if (flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_AUTOTUNE) {
133  state = AT_INIT;
134  vTaskDelay(50);
135  continue;
136  }
137 
138  StabilizationDesiredData stabDesired;
139  StabilizationDesiredGet(&stabDesired);
140 
141  StabilizationSettingsData stabSettings;
142  StabilizationSettingsGet(&stabSettings);
143 
144  ManualControlSettingsData manualSettings;
145  ManualControlSettingsGet(&manualSettings);
146 
147  ManualControlCommandData manualControl;
148  ManualControlCommandGet(&manualControl);
149 
150  RelayTuningSettingsData relaySettings;
151  RelayTuningSettingsGet(&relaySettings);
152 
153  bool rate = relaySettings.Mode == RELAYTUNINGSETTINGS_MODE_RATE;
154 
155  if (rate) { // rate mode
156  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
157  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
158 
159  stabDesired.Roll = manualControl.Roll * stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_ROLL];
160  stabDesired.Pitch = manualControl.Pitch * stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_PITCH];
161  } else {
162  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
163  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
164 
165  stabDesired.Roll = manualControl.Roll * stabSettings.RollMax;
166  stabDesired.Pitch = manualControl.Pitch * stabSettings.PitchMax;
167  }
168 
169  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
170  stabDesired.Yaw = manualControl.Yaw * stabSettings.ManualRate[STABILIZATIONSETTINGS_MANUALRATE_YAW];
171  stabDesired.Throttle = manualControl.Throttle;
172 
173  switch(state) {
174  case AT_INIT:
175 
176  lastUpdateTime = xTaskGetTickCount();
177 
178  // Only start when armed and flying
179  if (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED && stabDesired.Throttle > 0)
180  state = AT_START;
181  break;
182 
183  case AT_START:
184 
185  diffTime = xTaskGetTickCount() - lastUpdateTime;
186 
187  // Spend the first block of time in normal rate mode to get airborne
188  if (diffTime > PREPARE_TIME) {
189  state = AT_ROLL;
190  lastUpdateTime = xTaskGetTickCount();
191  }
192  break;
193 
194  case AT_ROLL:
195 
196  diffTime = xTaskGetTickCount() - lastUpdateTime;
197 
198  // Run relay mode on the roll axis for the measurement time
199  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = rate ? STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYRATE :
200  STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYATTITUDE;
201  if (diffTime > MEAURE_TIME) { // Move on to next state
202  state = AT_PITCH;
203  lastUpdateTime = xTaskGetTickCount();
204  }
205  break;
206 
207  case AT_PITCH:
208 
209  diffTime = xTaskGetTickCount() - lastUpdateTime;
210 
211  // Run relay mode on the pitch axis for the measurement time
212  stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = rate ? STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYRATE :
213  STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYATTITUDE;
214  if (diffTime > MEAURE_TIME) { // Move on to next state
215  state = AT_FINISHED;
216  lastUpdateTime = xTaskGetTickCount();
217  }
218  break;
219 
220  case AT_FINISHED:
221 
222  // Wait until disarmed and landed before updating the settings
223  if (flightStatus.Armed == FLIGHTSTATUS_ARMED_DISARMED && stabDesired.Throttle <= 0)
224  state = AT_SET;
225 
226  break;
227 
228  case AT_SET:
229  update_stabilization_settings();
230  state = AT_INIT;
231  break;
232 
233  default:
234  // Set an alarm or some shit like that
235  break;
236  }
237 
238  StabilizationDesiredSet(&stabDesired);
239 
240  vTaskDelay(10);
241  }
242 }

250 static void update_stabilization_settings()
251 {
252  RelayTuningData relayTuning;
253  RelayTuningGet(&relayTuning);
254 
255  RelayTuningSettingsData relaySettings;
256  RelayTuningSettingsGet(&relaySettings);
257 
258  StabilizationSettingsData stabSettings;
259  StabilizationSettingsGet(&stabSettings);
260 
261  // Eventually get these settings from RelayTuningSettings
262  const float gain_ratio_r = 1.0f / 3.0f;
263  const float zero_ratio_r = 1.0f / 3.0f;
264  const float gain_ratio_p = 1.0f / 5.0f;
265  const float zero_ratio_p = 1.0f / 5.0f;
266 
267  // For now just run over roll and pitch
268  for (uint32_t i = 0; i < 2; i++) {
269  float wu = 1000.0f * 2 * PI / relayTuning.Period[i]; // ultimate freq = output osc freq (rad/s)
270 
271  float wc = wu * gain_ratio_r; // target openloop crossover frequency (rad/s)
272  float zc = wc * zero_ratio_r; // controller zero location (rad/s)
273  float kpu = 4.0f / PI / relayTuning.Gain[i]; // ultimate gain, i.e. the proportional gain for instablity
274  float kp = kpu * gain_ratio_r; // proportional gain
275  float ki = zc * kp; // integral gain
276 
277  // Now calculate gains for the next loop out knowing it is the integral of
278  // the inner loop -- the plant is position/velocity = scale*1/s
279  float wc2 = wc * gain_ratio_p; // crossover of the attitude loop
280  float kp2 = wc2; // kp of attitude
281  float ki2 = wc2 * zero_ratio_p * kp2; // ki of attitude
282 
283  switch(i) {
284  case 0: // roll
285  stabSettings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP] = kp;
286  stabSettings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI] = ki;
287  stabSettings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KP] = kp2;
288  stabSettings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KI] = ki2;
289  break;
290  case 1: // Pitch
291  stabSettings.PitchRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP] = kp;
292  stabSettings.PitchRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI] = ki;
293  stabSettings.PitchPI[STABILIZATIONSETTINGS_ROLLPI_KP] = kp2;
294  stabSettings.PitchPI[STABILIZATIONSETTINGS_ROLLPI_KI] = ki2;
295  break;
296  default:
297  // Oh shit oh shit oh shit
298  break;
299  }
300  }
301  switch(relaySettings.Behavior) {
302  case RELAYTUNINGSETTINGS_BEHAVIOR_MEASURE:
303  // Just measure, don't update the stab settings
304  break;
305  case RELAYTUNINGSETTINGS_BEHAVIOR_COMPUTE:
306  StabilizationSettingsSet(&stabSettings);
307  break;
308  case RELAYTUNINGSETTINGS_BEHAVIOR_SAVE:
309  StabilizationSettingsSet(&stabSettings);
310  UAVObjSave(StabilizationSettingsHandle(), 0);
311  break;
312  }
