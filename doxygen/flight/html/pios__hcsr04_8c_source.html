158 void TIM3_IRQHandler(void)
159 {
160  /* Zero value always will be changed but this prevents compiler warning */
161  int32_t i = 0;
162 
163  /* Do this as it's more efficient */
164  if (TIM_GetITStatus(TIM3, TIM_IT_CC2) == SET) {
165  i = 7;
166  if (CaptureState == 0) {
167  RiseValue = TIM_GetCapture2(TIM3);
168  } else {
169  FallValue = TIM_GetCapture2(TIM3);
170  }
171  }
172 
173  /* Clear TIM3 Capture compare interrupt pending bit */
174  TIM_ClearITPendingBit(TIM3, TIM_IT_CC2);
175 
176  /* Simple rise or fall state machine */
177  if (CaptureState == 0) {
178  /* Switch states */
179  CaptureState = 1;
180 
181  /* Switch polarity of input capture */
182  TIM_ICInitStructure.TIM_ICPolarity = TIM_ICPolarity_Falling;
183  TIM_ICInitStructure.TIM_Channel = TIM_Channel_2;
184  TIM_ICInit(TIM3, &TIM_ICInitStructure);
185 
186  } else {
187  /* Capture computation */
188  if (FallValue > RiseValue) {
189  CaptureValue = (FallValue - RiseValue);
190  } else {
191  CaptureValue = ((0xFFFF - RiseValue) + FallValue);
192  }
193 
194  /* Switch states */
195  CaptureState = 0;
196 
197  /* Increase supervisor counter */
198  CapCounter++;
199  TIM_ITConfig(TIM3, TIM_IT_CC2, DISABLE);
200 
201  /* Switch polarity of input capture */
202  TIM_ICInitStructure.TIM_ICPolarity = TIM_ICPolarity_Rising;
203  TIM_ICInitStructure.TIM_Channel = TIM_Channel_2;
204  TIM_ICInit(TIM3, &TIM_ICInitStructure);
205 
206  }
