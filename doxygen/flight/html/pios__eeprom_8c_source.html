56 int32_t PIOS_EEPROM_Save(uint8_t *data, uint32_t len)
57 {
58 
59  // We need to write 32 bit words, so extend the length to be an even multiple of 4 bytes,
60  // and include 4 bytes for the 32 bit CRC.
61  uint32_t nwords = (len / 4) + 1 + (len % 4 ? 1 : 0);
62  uint32_t size = nwords * 4;
63 
64  // Ensure that the length is not longer than the max size.
65  if (size > config.max_size)
66  return -1;
67 
68  // Calculate a 32 bit CRC of the data.
69  uint32_t crc = PIOS_CRC32_updateCRC(0xffffffff, data, len);
70 
71  // Unlock the Flash Program Erase controller
72  FLASH_Unlock();
73 
74  // See if we have to write the data.
75  if ((memcmp(data, (uint8_t*)config.base_address, len) == 0) &&
76  (memcmp((uint8_t*)&crc, (uint8_t*)config.base_address + size - 4, 4) == 0))
77  return 0;
78 
79  // TODO: Check that the area isn't already erased
80 
81  // Erase page
82  FLASH_Status fs = FLASH_ErasePage(config.base_address);
83  if (fs != FLASH_COMPLETE)
84  { // error
85  FLASH_Lock();
86  return -2;
87  }
88 
89  // write 4 bytes at a time into program flash area (emulated EEPROM area)
90  uint8_t *p1 = data;
91  uint32_t *p3 = (uint32_t *)config.base_address;
92  for (int32_t i = 0; i < size; p3++)
93  {
94  uint32_t value = 0;
95 
96  if (i == (size - 4))
97  {
98  // write the CRC.
99  value = crc;
100  i += 4;
101  }
102  else
103  {
104  if (i < len) value |= (uint32_t)*p1++ << 0; else value |= 0x000000ff; i++;
105  if (i < len) value |= (uint32_t)*p1++ << 8; else value |= 0x0000ff00; i++;
106  if (i < len) value |= (uint32_t)*p1++ << 16; else value |= 0x00ff0000; i++;
107  if (i < len) value |= (uint32_t)*p1++ << 24; else value |= 0xff000000; i++;
108  }
109 
110  // write a 32-bit value
111  fs = FLASH_ProgramWord((uint32_t)p3, value);
112  if (fs != FLASH_COMPLETE)
113  {
114  FLASH_Lock();
115  return -3;
116  }
117  }
118 
119  // Lock the Flash Program Erase controller
120  FLASH_Lock();
121 
122  return 0;
123 }
