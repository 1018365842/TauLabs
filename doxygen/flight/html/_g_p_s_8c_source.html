105 int32_t GPSStart(void)
106 {
107  if (module_enabled) {
108  if (gpsPort) {
109  // Start gps task
110  xTaskCreate(gpsTask, (signed char *)"GPS", STACK_SIZE_BYTES/4, NULL, TASK_PRIORITY, &gpsTaskHandle);
111  TaskMonitorAdd(TASKINFO_RUNNING_GPS, gpsTaskHandle);
112  return 0;
113  }
114 
115  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_CRITICAL);
116  }
117  return -1;
118 }

125 int32_t GPSInitialize(void)
126 {
127  gpsPort = PIOS_COM_GPS;
128  uint8_t gpsProtocol;
129 
130 #ifdef MODULE_GPS_BUILTIN
131  module_enabled = true;
132 #else
133  uint8_t module_state[MODULESETTINGS_ADMINSTATE_NUMELEM];
134  ModuleSettingsAdminStateGet(module_state);
135  if (module_state[MODULESETTINGS_ADMINSTATE_GPS] == MODULESETTINGS_ADMINSTATE_ENABLED) {
136  module_enabled = true;
137  } else {
138  module_enabled = false;
139  }
140 #endif
141 
142 #if defined(REVOLUTION)
143  // These objects MUST be initialized for Revolution
144  // because the rest of the system expects to just
145  // attach to their queues
146  GPSPositionInitialize();
147  GPSVelocityInitialize();
148  GPSTimeInitialize();
149  GPSSatellitesInitialize();
150  HomeLocationInitialize();
151  updateSettings();
152 
153 #else
154  if (gpsPort && module_enabled) {
155  GPSPositionInitialize();
156  GPSVelocityInitialize();
157 #if !defined(PIOS_GPS_MINIMAL)
158  GPSTimeInitialize();
159  GPSSatellitesInitialize();
160 #endif
161 #ifdef PIOS_GPS_SETS_HOMELOCATION
162  HomeLocationInitialize();
163 #endif
164 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
165  AirspeedActualInitialize();
166 #endif
167  updateSettings();
168  }
169 #endif
170 
171  if (gpsPort && module_enabled) {
172  ModuleSettingsGPSDataProtocolGet(&gpsProtocol);
173  switch (gpsProtocol) {
174  case MODULESETTINGS_GPSDATAPROTOCOL_NMEA:
175  gps_rx_buffer = pvPortMalloc(NMEA_MAX_PACKET_LENGTH);
176  break;
177  case MODULESETTINGS_GPSDATAPROTOCOL_UBX:
178  gps_rx_buffer = pvPortMalloc(sizeof(struct UBXPacket));
179  break;
180  default:
181  gps_rx_buffer = NULL;
182  }

186  }

188  return -1;

198 static void gpsTask(void *parameters)
199 {
200  portTickType xDelay = 100 / portTICK_RATE_MS;
201  uint32_t timeNowMs = xTaskGetTickCount() * portTICK_RATE_MS;
202 
203  GPSPositionData gpsposition;
204  uint8_t gpsProtocol;
205 
206  ModuleSettingsGPSDataProtocolGet(&gpsProtocol);
207 
208 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
209  gps_airspeed_initialize();
210 #endif
211 
212  timeOfLastUpdateMs = timeNowMs;
213  timeOfLastCommandMs = timeNowMs;
214 
215  GPSPositionGet(&gpsposition);
216  // Loop forever
217  while (1)
218  {
219  uint8_t c;
220 
221  // This blocks the task until there is something on the buffer
222  while (PIOS_COM_ReceiveBuffer(gpsPort, &c, 1, xDelay) > 0)
223  {
224  int res;
225  switch (gpsProtocol) {
226 #if defined(PIOS_INCLUDE_GPS_NMEA_PARSER)
227  case MODULESETTINGS_GPSDATAPROTOCOL_NMEA:
228  res = parse_nmea_stream (c,gps_rx_buffer, &gpsposition, &gpsRxStats);
229  break;
230 #endif
231 #if defined(PIOS_INCLUDE_GPS_UBX_PARSER)
232  case MODULESETTINGS_GPSDATAPROTOCOL_UBX:
233  res = parse_ubx_stream (c,gps_rx_buffer, &gpsposition, &gpsRxStats);
234  break;
235 #endif
236  default:
237  res = NO_PARSER; // this should not happen
238  break;
239  }
240 
241  if (res == PARSER_COMPLETE) {
242  timeNowMs = xTaskGetTickCount() * portTICK_RATE_MS;
243  timeOfLastUpdateMs = timeNowMs;
244  timeOfLastCommandMs = timeNowMs;
245  }
246  }
247 
248  // Check for GPS timeout
249  timeNowMs = xTaskGetTickCount() * portTICK_RATE_MS;
250  if ((timeNowMs - timeOfLastUpdateMs) >= GPS_TIMEOUT_MS) {
251  // we have not received any valid GPS sentences for a while.
252  // either the GPS is not plugged in or a hardware problem or the GPS has locked up.
253  uint8_t status = GPSPOSITION_STATUS_NOGPS;
254  GPSPositionStatusSet(&status);
255  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_ERROR);
256  } else {
257  // we appear to be receiving GPS sentences OK, we've had an update
258  //criteria for GPS-OK taken from this post...
259  //http://forums.openpilot.org/topic/1523-professors-insgps-in-svn/page__view__findpost__p__5220
260  if ((gpsposition.PDOP < 3.5) && (gpsposition.Satellites >= 7) &&
261  (gpsposition.Status == GPSPOSITION_STATUS_FIX3D)) {
262  AlarmsClear(SYSTEMALARMS_ALARM_GPS);
263 #ifdef PIOS_GPS_SETS_HOMELOCATION
264  HomeLocationData home;
265  HomeLocationGet(&home);
266 
267  if (home.Set == HOMELOCATION_SET_FALSE)
268  setHomeLocation(&gpsposition);
269 #endif
270  } else if (gpsposition.Status == GPSPOSITION_STATUS_FIX3D)
271  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_WARNING);
272  else
273  AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_CRITICAL);
274  }
275 
276  }
277 }

283 static void setHomeLocation(GPSPositionData * gpsData)
284 {
285  HomeLocationData home;
286  HomeLocationGet(&home);
287  GPSTimeData gps;
288  GPSTimeGet(&gps);
289 
290  if (gps.Year >= 2000)
291  {
292  // Store LLA
293  home.Latitude = gpsData->Latitude;
294  home.Longitude = gpsData->Longitude;
295  home.Altitude = gpsData->Altitude + gpsData->GeoidSeparation;
296 
297  // Compute home ECEF coordinates and the rotation matrix into NED
298  double LLA[3] = { ((double)home.Latitude) / 10e6, ((double)home.Longitude) / 10e6, ((double)home.Altitude) };
299 
300  // Compute magnetic flux direction at home location
301  if (WMM_GetMagVector(LLA[0], LLA[1], LLA[2], gps.Month, gps.Day, gps.Year, &home.Be[0]) >= 0)
302  { // calculations appeared to go OK
303 
304  // Compute local acceleration due to gravity. Vehicles that span a very large
305  // range of altitude (say, weather balloons) may need to update this during the
306  // flight.
307  home.Set = HOMELOCATION_SET_TRUE;
308  HomeLocationSet(&home);
309  }
310  }
311 }
