266 static void AttitudeTask(void *parameters)
267 {
268  bool first_run = true;
269  uint32_t last_algorithm;
270  bool last_complementary;
271  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
272 
273  // Force settings update to make sure rotation loaded
274  settingsUpdatedCb(NULL);
275 
276  // Wait for all the sensors be to read
277  vTaskDelay(100);
278 
279  // Invalidate previous algorithm to trigger a first run
280  last_algorithm = 0xfffffff;
281  last_complementary = false;
282 
283  // Main task loop
284  while (1) {
285 
286  int32_t ret_val = -1;
287 
288  // When changing the attitude filter reinitialize
289  if (last_algorithm != stateEstimation.AttitudeFilter) {
290  last_algorithm = stateEstimation.AttitudeFilter;
291  first_run = true;
292  }
293 
294  // There are two options to select:
295  // Attitude filter - what sets the attitude
296  // Navigation filter - what sets the position and velocity
297  // If the INS is used for either then it should run
298  bool ins = (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSOUTDOOR) ||
299  (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSINDOOR) ||
300  (stateEstimation.NavigationFilter == STATEESTIMATION_NAVIGATIONFILTER_INS);
301 
302  // INS outdoor mode when used for navigation OR explicit outdoor mode
303  bool outdoor = (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSOUTDOOR) ||
304  ((stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_COMPLEMENTARY) &&
305  (stateEstimation.NavigationFilter == STATEESTIMATION_NAVIGATIONFILTER_INS));
306 
307  // Complementary filter only needed when used for attitude
308  bool complementary = stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_COMPLEMENTARY;
309 
310  // Update one or both filters
311  if (ins) {
312  ret_val = updateAttitudeINSGPS(first_run, outdoor);
313  if (complementary)
314  updateAttitudeComplementary(first_run || complementary != last_complementary, true);
315  } else {
316  ret_val = updateAttitudeComplementary(first_run, false);
317  }
318 
319  last_complementary = complementary;
320 
321  // Get the requested data
322  // This function blocks on data queue
323  switch (stateEstimation.AttitudeFilter ) {
324  case STATEESTIMATION_ATTITUDEFILTER_COMPLEMENTARY:
325  setAttitudeComplementary();
326  break;
327  case STATEESTIMATION_ATTITUDEFILTER_INSOUTDOOR:
328  case STATEESTIMATION_ATTITUDEFILTER_INSINDOOR:
329  setAttitudeINSGPS();
330  break;
331  }
332 
333  // Use the selected source for position and velocity
334  switch (stateEstimation.NavigationFilter) {
335  case STATEESTIMATION_NAVIGATIONFILTER_INS:
336  // TODO: When running in dual mode and the INS is not initialized set
337  // an error here
338  setNavigationINSGPS();
339  break;
340  default:
341  setNavigationRaw();
342  }
343 
344  if(ret_val == 0)
345  first_run = false;
346 
347  PIOS_WDG_UpdateFlag(PIOS_WDG_ATTITUDE);
348  }
349 }

359 static int32_t updateAttitudeComplementary(bool first_run, bool secondary)
360 {
361  UAVObjEvent ev;
362  GyrosData gyrosData;
363  AccelsData accelsData;
364  static int32_t timeval;
365  float dT;
366 
367  // Wait until the accel and gyro object is updated, if a timeout then go to failsafe
368  if (!secondary && (
369  xQueueReceive(gyroQueue, &ev, FAILSAFE_TIMEOUT_MS / portTICK_RATE_MS) != pdTRUE ||
370  xQueueReceive(accelQueue, &ev, 1 / portTICK_RATE_MS) != pdTRUE ) )
371  {
372  // When one of these is updated so should the other
373  // Do not set attitude timeout warnings in simulation mode
374  if (!AttitudeActualReadOnly()){
375  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE,SYSTEMALARMS_ALARM_WARNING);
376  return -1;
377  }
378  }
379 
380  AccelsGet(&accelsData);
381 
382  // When this algorithm is first run force it to a known condition
383  if(first_run) {
384  MagnetometerData magData;
385  magData.x = 100;
386  magData.y = 0;
387  magData.z = 0;
388 
389  // Wait for a mag reading if a magnetometer was registered
390  if (PIOS_SENSORS_GetQueue(PIOS_SENSOR_MAG) != NULL) {
391  if ( !secondary && xQueueReceive(magQueue, &ev, 20 / portTICK_RATE_MS) != pdTRUE ) {
392  return -1;
393  }
394  MagnetometerGet(&magData);
395  }
396 
397  float RPY[3];
398  float theta = atan2f(accelsData.x, -accelsData.z);
399  RPY[1] = theta * RAD2DEG;
400  RPY[0] = atan2f(-accelsData.y, -accelsData.z / cosf(theta)) * RAD2DEG;
401  RPY[2] = atan2f(-magData.y, magData.x) * RAD2DEG;
402  RPY2Quaternion(RPY, cf_q);
403 
404  complementary_filter_state.initialization = CF_POWERON;
405  complementary_filter_state.reset_timeval = PIOS_DELAY_GetRaw();
406  timeval = PIOS_DELAY_GetRaw();
407 
408  complementary_filter_state.arming_count = 0;
409 
410  return 0;
411  }
412 
413  FlightStatusData flightStatus;
414  FlightStatusGet(&flightStatus);
415 
416  uint32_t ms_since_reset = PIOS_DELAY_DiffuS(complementary_filter_state.reset_timeval) / 1000;
417  if (complementary_filter_state.initialization == CF_POWERON) {
418  // Wait one second before starting to initialize
419  complementary_filter_state.initialization =
420  (ms_since_reset > 1000) ?
421  CF_INITIALIZING :
422  CF_POWERON;
423  } else if(complementary_filter_state.initialization == CF_INITIALIZING &&
424  (ms_since_reset < 7000) &&
425  (ms_since_reset > 1000)) {
426 
427  // For first 7 seconds use accels to get gyro bias
428  attitudeSettings.AccelKp = 0.1f + 0.1f * (xTaskGetTickCount() < 4000);
429  attitudeSettings.AccelKi = 0.1f;
430  attitudeSettings.YawBiasRate = 0.1f;
431  attitudeSettings.MagKp = 0.1f;
432  } else if ((attitudeSettings.ZeroDuringArming == ATTITUDESETTINGS_ZERODURINGARMING_TRUE) &&
433  (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMING)) {
434 
435  // Use a rapidly decrease accelKp to force the attitude to snap back
436  // to level and then converge more smoothly
437  if (complementary_filter_state.arming_count < 20)
438  attitudeSettings.AccelKp = 1.0f;
439  else if (attitudeSettings.AccelKp > 0.1f)
440  attitudeSettings.AccelKp -= 0.01f;
441  complementary_filter_state.arming_count++;
442 
443  // Set the other parameters to drive faster convergence
444  attitudeSettings.AccelKi = 0.1f;
445  attitudeSettings.YawBiasRate = 0.1f;
446  attitudeSettings.MagKp = 0.1f;
447 
448  // Don't apply LPF to the accels during arming
449  complementary_filter_state.accel_filter_enabled = false;
450 
451  // Indicate arming so that after arming it reloads
452  // the normal settings
453  if (complementary_filter_state.initialization != CF_ARMING) {
454  accumulate_gyro_zero();
455  complementary_filter_state.initialization = CF_ARMING;
456  complementary_filter_state.accumulating_gyro = true;
457  }
458 
459  } else if (complementary_filter_state.initialization == CF_ARMING ||
460  complementary_filter_state.initialization == CF_INITIALIZING) {
461 
462  AttitudeSettingsGet(&attitudeSettings);
463  if(complementary_filter_state.accel_alpha > 0.0f)
464  complementary_filter_state.accel_filter_enabled = true;
465 
466  // If arming that means we were accumulating gyro
467  // samples. Compute new bias.
468  if (complementary_filter_state.initialization == CF_ARMING) {
469  accumulate_gyro_compute();
470  complementary_filter_state.accumulating_gyro = false;
471  complementary_filter_state.arming_count = 0;
472  }
473 
474  // Indicate normal mode to prevent rerunning this code
475  complementary_filter_state.initialization = CF_NORMAL;
476  }
477 
478  GyrosGet(&gyrosData);
479  accumulate_gyro(&gyrosData);
480 
481  // Compute the dT using the cpu clock
482  dT = PIOS_DELAY_DiffuS(timeval) / 1000000.0f;
483  timeval = PIOS_DELAY_GetRaw();
484 
485  float grot[3];
486  float accel_err[3];
487  float *grot_filtered = complementary_filter_state.grot_filtered;
488  float *accels_filtered = complementary_filter_state.accels_filtered;
489 
490  // Apply smoothing to accel values, to reduce vibration noise before main calculations.
491  apply_accel_filter(&accelsData.x,accels_filtered);
492 
493  // Rotate gravity to body frame and cross with accels
494  grot[0] = -(2 * (cf_q[1] * cf_q[3] - cf_q[0] * cf_q[2]));
495  grot[1] = -(2 * (cf_q[2] * cf_q[3] + cf_q[0] * cf_q[1]));
496  grot[2] = -(cf_q[0]*cf_q[0] - cf_q[1]*cf_q[1] - cf_q[2]*cf_q[2] + cf_q[3]*cf_q[3]);
497  CrossProduct((const float *) &accelsData.x, (const float *) grot, accel_err);
498 
499  // Apply same filtering to the rotated attitude to match delays
500  apply_accel_filter(grot,grot_filtered);
501 
502  // Compute the error between the predicted direction of gravity and smoothed acceleration
503  CrossProduct((const float *) accels_filtered, (const float *) grot_filtered, accel_err);
504 
505  float grot_mag;
506  if (complementary_filter_state.accel_filter_enabled)
507  grot_mag = sqrtf(grot_filtered[0]*grot_filtered[0] + grot_filtered[1]*grot_filtered[1] + grot_filtered[2]*grot_filtered[2]);
508  else
509  grot_mag = 1.0f;
510 
511  // Account for accel magnitude
512  float accel_mag;
513  accel_mag = accels_filtered[0]*accels_filtered[0] + accels_filtered[1]*accels_filtered[1] + accels_filtered[2]*accels_filtered[2];
514  accel_mag = sqrtf(accel_mag);
515  if (grot_mag > 1.0e-3f && accel_mag > 1.0e-3f) {
516  accel_err[0] /= (accel_mag * grot_mag);
517  accel_err[1] /= (accel_mag * grot_mag);
518  accel_err[2] /= (accel_mag * grot_mag);
519  } else {
520  accel_err[0] = 0;
521  accel_err[1] = 0;
522  accel_err[2] = 0;
523  }
524 
525  float mag_err[3];
526  if ( secondary || xQueueReceive(magQueue, &ev, 0) == pdTRUE )
527  {
528  // Rotate gravity to body frame and cross with accels
529  float brot[3];
530  float Rbe[3][3];
531  MagnetometerData mag;
532 
533  Quaternion2R(cf_q, Rbe);
534  MagnetometerGet(&mag);
535 
536  // If the mag is producing bad data don't use it (normally bad calibration)
537  if (mag.x == mag.x && mag.y == mag.y && mag.z == mag.z &&
538  homeLocation.Set == HOMELOCATION_SET_TRUE) {
539  rot_mult(Rbe, homeLocation.Be, brot, false);
540 
541  float mag_len = sqrtf(mag.x * mag.x + mag.y * mag.y + mag.z * mag.z);
542  mag.x /= mag_len;
543  mag.y /= mag_len;
544  mag.z /= mag_len;
545 
546  float bmag = sqrtf(brot[0] * brot[0] + brot[1] * brot[1] + brot[2] * brot[2]);
547  brot[0] /= bmag;
548  brot[1] /= bmag;
549  brot[2] /= bmag;
550 
551  // Only compute if neither vector is null
552  if (bmag < 1 || mag_len < 1)
553  mag_err[0] = mag_err[1] = mag_err[2] = 0;
554  else
555  CrossProduct((const float *) &mag.x, (const float *) brot, mag_err);
556 
557  if (mag_err[2] != mag_err[2])
558  mag_err[2] = 0;
559  } else
560  mag_err[2] = 0;
561  } else {
562  mag_err[2] = 0;
563  }
564 
565  // Accumulate integral of error. Scale here so that units are (deg/s) but Ki has units of s
566  GyrosBiasData gyrosBias;
567  GyrosBiasGet(&gyrosBias);
568  gyrosBias.x -= accel_err[0] * attitudeSettings.AccelKi;
569  gyrosBias.y -= accel_err[1] * attitudeSettings.AccelKi;
570  gyrosBias.z -= mag_err[2] * attitudeSettings.MagKi;
571  GyrosBiasSet(&gyrosBias);
572 
573  // Correct rates based on error, integral component dealt with in updateSensors
574  gyrosData.x += accel_err[0] * attitudeSettings.AccelKp / dT;
575  gyrosData.y += accel_err[1] * attitudeSettings.AccelKp / dT;
576  gyrosData.z += mag_err[2] * attitudeSettings.MagKp / dT;
577 
578  // Work out time derivative from INSAlgo writeup
579  // Also accounts for the fact that gyros are in deg/s
580  float qdot[4];
581  qdot[0] = (-cf_q[1] * gyrosData.x - cf_q[2] * gyrosData.y - cf_q[3] * gyrosData.z) * dT * DEG2RAD / 2;
582  qdot[1] = (cf_q[0] * gyrosData.x - cf_q[3] * gyrosData.y + cf_q[2] * gyrosData.z) * dT * DEG2RAD / 2;
583  qdot[2] = (cf_q[3] * gyrosData.x + cf_q[0] * gyrosData.y - cf_q[1] * gyrosData.z) * dT * DEG2RAD / 2;
584  qdot[3] = (-cf_q[2] * gyrosData.x + cf_q[1] * gyrosData.y + cf_q[0] * gyrosData.z) * dT * DEG2RAD / 2;
585 
586  // Take a time step
587  cf_q[0] = cf_q[0] + qdot[0];
588  cf_q[1] = cf_q[1] + qdot[1];
589  cf_q[2] = cf_q[2] + qdot[2];
590  cf_q[3] = cf_q[3] + qdot[3];
591 
592  if(cf_q[0] < 0) {
593  cf_q[0] = -cf_q[0];
594  cf_q[1] = -cf_q[1];
595  cf_q[2] = -cf_q[2];
596  cf_q[3] = -cf_q[3];
597  }
598 
599  // Renomalize
600  float qmag;
601  qmag = sqrtf(cf_q[0]*cf_q[0] + cf_q[1]*cf_q[1] + cf_q[2]*cf_q[2] + cf_q[3]*cf_q[3]);
602  cf_q[0] = cf_q[0] / qmag;
603  cf_q[1] = cf_q[1] / qmag;
604  cf_q[2] = cf_q[2] / qmag;
605  cf_q[3] = cf_q[3] / qmag;
606 
607  // If quaternion has become inappropriately short or is nan reinit.
608  // THIS SHOULD NEVER ACTUALLY HAPPEN
609  if((fabs(qmag) < 1.0e-3f) || (qmag != qmag)) {
610  cf_q[0] = 1;
611  cf_q[1] = 0;
612  cf_q[2] = 0;
613  cf_q[3] = 0;
614  }
615 
616  if (!secondary)
617  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
618 
619  return 0;
620 }
