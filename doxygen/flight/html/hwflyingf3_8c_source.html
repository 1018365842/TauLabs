48 int32_t HwFlyingF3Initialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(HWFLYINGF3_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(HWFLYINGF3_OBJID,
56  HWFLYINGF3_ISSINGLEINST, HWFLYINGF3_ISSETTINGS, HWFLYINGF3_NUMBYTES, &HwFlyingF3SetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

75 {
