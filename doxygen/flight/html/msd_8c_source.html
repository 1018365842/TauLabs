106 int32_t MSD_Init(uint32_t mode)
107 {
108  /* Update the serial number string descriptor with the data from the unique ID*/
109  uint8_t serial_number_str[40];
110  int i, len;
111  PIOS_SYS_SerialNumberGet((char *) serial_number_str);
112  for(i = 0, len = 0; serial_number_str[i] != '\0' && len < 25; ++i) {
113  MSD_MASS_StringSerial[len++] = serial_number_str[i];
114  MSD_MASS_StringSerial[len++] = 0;
115  }
116 
117  lun_available = 0;
118 
119  /* All LUNs available after USB init */
120  for(i = 0; i < MSD_NUM_LUN; ++i) {
121  MSD_LUN_AvailableSet(i, 1);
122  }
123 
124  PIOS_IRQ_Disable();
125 
126  /* Clear all USB interrupt requests */
127  _SetCNTR(0); /* Interrupt Mask */
128  _SetISTR(0); /* clear pending requests */
129 
130  /* Switch to MSD driver hooks */
131  memcpy(&Device_Table, (DEVICE *) &My_Device_Table, sizeof(Device_Table));
132  pProperty = (DEVICE_PROP *) &My_Device_Property;
133  pUser_Standard_Requests = (USER_STANDARD_REQUESTS *) &My_User_Standard_Requests;
134 
135  /* Change endpoints */
136  pEpInt_IN[0] = MSD_Mass_Storage_In;
137  pEpInt_OUT[1] = MSD_Mass_Storage_Out;
138 
139  /* Force re-enumeration w/o overwriting PIOS hooks */
140  PIOS_USB_Init(2);
141 
142  /* Clear pending interrupts (again) */
143  _SetISTR(0);
144 
145  /* Set interrupts mask */
146  _SetCNTR(MSD_IMR_MSK);
147 
148  PIOS_IRQ_Enable();
149 
150  /* No error */
151  return 0;
152 }

198 int32_t MSD_LUN_AvailableSet(uint8_t lun, uint8_t available)
199 {
200  if(lun >= MSD_NUM_LUN) {
201  return -1;
202  }
203 
204  if(available) {
205  lun_available |= (1 << lun);
206  } else {
207  lun_available &= ~(1 << lun);
208  }
209 
210  /* No error */
211  return 0;
212 }

221  return 0;
222

229 static void MSD_MASS_Reset()
230 {
231  /* Set the device as not configured */
232  pInformation->Current_Configuration = 0;
233 
234  /* Current Feature initialization */
235  pInformation->Current_Feature = MSD_MASS_ConfigDescriptor[7];
236 
237  SetBTABLE(MSD_BTABLE_ADDRESS);
238 
239  /* Initialize Endpoint 0 */
240  SetEPType(ENDP0, EP_CONTROL);
241  SetEPTxStatus(ENDP0, EP_TX_NAK);
242  SetEPRxAddr(ENDP0, MSD_ENDP0_RXADDR);
243  SetEPRxCount(ENDP0, pProperty->MaxPacketSize);
244  SetEPTxAddr(ENDP0, MSD_ENDP0_TXADDR);
245  Clear_Status_Out(ENDP0);
246  SetEPRxValid(ENDP0);
247 
248  /* Initialize Endpoint 1 */
249  SetEPType(ENDP1, EP_BULK);
250  SetEPTxAddr(ENDP1, MSD_ENDP1_TXADDR);
251  SetEPTxStatus(ENDP1, EP_TX_NAK);
252  SetEPRxStatus(ENDP1, EP_RX_DIS);
253 
254  /* Initialize Endpoint 2 */
255  SetEPType(ENDP2, EP_BULK);
256  SetEPRxAddr(ENDP2, MSD_ENDP2_RXADDR);
257  SetEPRxCount(ENDP2, pProperty->MaxPacketSize);
258  SetEPRxStatus(ENDP2, EP_RX_VALID);
259  SetEPTxStatus(ENDP2, EP_TX_DIS);
260 
261  SetEPRxCount(ENDP0, pProperty->MaxPacketSize);
262  SetEPRxValid(ENDP0);
263 
264  /* Set the device to response on default address */
265  SetDeviceAddress(0);
266 
267  MSD_CBW.dSignature = BOT_CBW_SIGNATURE;
268  MSD_Bot_State = BOT_IDLE;
269 }

329 static RESULT MSD_MASS_Data_Setup(uint8_t RequestNo)
330 {
331  uint8_t *(*CopyRoutine)( uint16_t);
332 
333  CopyRoutine = NULL;
334  if((Type_Recipient == (CLASS_REQUEST | INTERFACE_RECIPIENT)) && (RequestNo == GET_MAX_LUN) && (pInformation->USBwValue == 0)
335  && (pInformation->USBwIndex == 0) && (pInformation->USBwLength == 0x01)) {
336  CopyRoutine = Get_Max_Lun;
337  } else {
338  return USB_UNSUPPORT;
339  }
340 
341  if(CopyRoutine == NULL) {
342  return USB_UNSUPPORT;
343  }
344 
345  pInformation->Ctrl_Info.CopyData = CopyRoutine;
346  pInformation->Ctrl_Info.Usb_wOffset = 0;
347  (*CopyRoutine)(0);
348 
349  return USB_SUCCESS;
350 }

357 static RESULT MSD_MASS_NoData_Setup(uint8_t RequestNo)
358 {
359  if((Type_Recipient == (CLASS_REQUEST | INTERFACE_RECIPIENT)) && (RequestNo == MASS_STORAGE_RESET) && (pInformation->USBwValue == 0)
360  && (pInformation->USBwIndex == 0) && (pInformation->USBwLength == 0x00)) {
361  /* Initialise Endpoint 1 */
362  ClearDTOG_TX(ENDP1);
363 
364  /* Initialise Endpoint 2 */
365  ClearDTOG_RX(ENDP2);
366 
367  /* Initialise the CBW signature to enable the clear feature*/
368  MSD_CBW.dSignature = BOT_CBW_SIGNATURE;
369  MSD_Bot_State = BOT_IDLE;
370 
371  return USB_SUCCESS;
372  }
373  return USB_UNSUPPORT;
374 }
