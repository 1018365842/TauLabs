397  //Check to see if we've gone past our destination. Since the path follower
398  //is simply following a vector, it has no concept of where the vector ends.
399  //It will simply keep following it to infinity if we don't stop it. So while
400  //we don't know why the commutation to the next point failed, we don't know
401  //we don't want the plane flying off.

405  //IT WOULD BE NICE TO ONLY DO THIS ONCE PER WAYPOINT UPDATE, INSTEAD OF

409  //Perform a quick vector math operation, |a| < a.b/|a| = |b|cos(theta),

411  //of flight time, for good measure to make sure we don't add jitter.
412  if (((p[0] - r[0]) * q[0] + (p[1] - r[1]) * q[1]) >
413  pathLength2 + 5.0f * pathDesired.EndingVelocity) {

415  //received any instructions. Start circling

417  //no danger of it getting stuck in orbit mode.

513 float followStraightLine(float r[3], float q[3], float p[3], float psi,
514  float chi_inf, float k_path, float k_psi_int,
515  float delT)
516 {
517  float chi_q = atan2f(q[1], q[0]);
518  while (chi_q - psi < -PI) {
519  chi_q += 2.0f * PI;
520  }
521  while (chi_q - psi > PI) {
522  chi_q -= 2.0f * PI;
523  }
524 
525  float err_p = -sinf(chi_q) * (p[0] - r[0]) + cosf(chi_q) * (p[1] - r[1]);
526  integral->lineError += delT * err_p;
527  float psi_command = chi_q - chi_inf * 2.0f / PI * atanf(k_path * err_p) -
528  k_psi_int * integral->lineError;
529 
530  return psi_command;
531 }

537 float followOrbit(float c[3], float rho, bool direction, float p[3], float psi,
538  float k_orbit, float k_psi_int, float delT)
539 {
540  float pncn = p[0] - c[0];
541  float pece = p[1] - c[1];
542  float d = sqrtf(pncn * pncn + pece * pece);
543 
544  float err_orbit = d - rho;
545  integral->circleError += err_orbit * delT;
546 
547  float phi = atan2f(pece, pncn);
548  while (phi - psi < -PI) {
549  phi = phi + 2.0f * PI;
550  }
551  while (phi - psi > PI) {
552  phi = phi - 2.0f * PI;
553  }
554 
555  float psi_command = (direction == true) ?
556  phi + (PI / 2.0f + atanf(k_orbit * err_orbit) +
557  k_psi_int * integral->circleError) : phi - (PI / 2.0f +
558  atanf(k_orbit *
559  err_orbit) +
560  k_psi_int *
561  integral->
562  circleError);
563 
564 #ifdef SIM_OSX
565  fprintf(stderr,
566  "actual heading: %f, circle error: %f, circl integral: %f, heading command: %f",
567  psi, err_orbit, integral->circleError, psi_command);
568 #endif
569  return psi_command;
570 }
