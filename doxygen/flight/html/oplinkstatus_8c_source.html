48 int32_t OPLinkStatusInitialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(OPLINKSTATUS_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(OPLINKSTATUS_OBJID,
56  OPLINKSTATUS_ISSINGLEINST, OPLINKSTATUS_ISSETTINGS, OPLINKSTATUS_NUMBYTES, &OPLinkStatusSetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }
