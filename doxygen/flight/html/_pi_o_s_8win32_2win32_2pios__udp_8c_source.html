76

81  udp_dev->server.sin_family = AF_INET;
82  udp_dev->server.sin_addr.s_addr = inet_addr(udp_dev->cfg->ip);
83  udp_dev->server.sin_port = htons(udp_dev->cfg->port);
84  BOOL tmp = TRUE;
85  setsockopt(udp_dev->socket, SOL_SOCKET, SO_BROADCAST, (char *)&tmp, sizeof(tmp));
86  int res= bind(udp_dev->socket, (struct sockaddr *)&udp_dev->server,sizeof(udp_dev->server));
87  /* use nonblocking IO */
88  u_long argp = 1;
89  ioctlsocket(udp_dev->socket, FIONBIO, &argp);
90  printf("udp dev %i - socket %i opened - result %i\n",i,udp_dev->socket,res);

93  }

123 int32_t PIOS_UDP_RxBufferPut(uint8_t udp, uint8_t b)
124 {
125  struct pios_udp_dev * udp_dev;
126 
127  /* Get a handle for the device configuration */
128  udp_dev = find_udp_dev_by_id(udp);
129 
130  if (!udp_dev) {
131  /* Undefined UDP port for this board (see pios_board.c) */
132  return -1;
133  }
134 
135  if (udp_dev->rx.size >= sizeof(udp_dev->rx.buf)) {
136  /* Buffer full (retry) */
137  return -2;
138  }
139 
140  /* Copy received byte into receive buffer */
141  udp_dev->rx.buf[udp_dev->rx.head++] = b;
142  if (udp_dev->rx.head >= sizeof(udp_dev->rx.buf)) {
143  udp_dev->rx.head = 0;
144  }
145  udp_dev->rx.size++;
146 
147  /* No error */
148  return 0;
149 }

154 void PIOS_UDP_RECV(uint8_t udp) {
155 
156  struct pios_udp_dev * udp_dev;
157  char localbuffer[PIOS_UDP_RX_BUFFER_SIZE];
158 
159  /* Get a handle for the device configuration */
160  udp_dev = find_udp_dev_by_id(udp);
161 
162  if (!udp_dev) {
163  /* Undefined UDP port for this board (see pios_board.c) */
164  return;
165  }
166 
167  /* use nonblocking IO */
168  u_long argp = 1;
169  ioctlsocket(udp_dev->socket, FIONBIO, &argp);
170 
171  /* receive data */
172  int received;
173  udp_dev->clientLength=sizeof(udp_dev->client);
174  if ((received = recvfrom(udp_dev->socket,
175  localbuffer,
176  (PIOS_UDP_RX_BUFFER_SIZE - udp_dev->rx.size),
177  0,
178  (struct sockaddr *) &udp_dev->client,
179  (int*)&udp_dev->clientLength)) == SOCKET_ERROR) {
180 
181  return;
182  }
183  /* copy received data to buffer */
184  int t;
185  for (t=0;t<received;t++) {
186  PIOS_UDP_RxBufferPut(udp,localbuffer[t]);
187  }
188

190 }
