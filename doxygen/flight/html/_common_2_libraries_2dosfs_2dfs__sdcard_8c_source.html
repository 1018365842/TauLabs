83 uint32_t DFS_ReadSector(uint8_t unit, uint8_t *buffer, uint32_t sector, uint32_t count)
84 {
85  /* Only allow access to single unit */
86  if(unit != 0) {
87  return 1;
88  }
89 
90  /* According to README.txt, count is always 1 - check this! */
91  if(count != 1) {
92  return 2;
93  }
94 
95  /* Cache: */
96  if(caching_enabled && sector == last_sector) {
97  /* we assume that sector is already in *buffer */
98  /* since the user has to take care that the same buffer is used for file reads, this */
99  /* feature has to be explicitly enabled with DFS_CachingEnabledSet(uint8_t enable) */
100  return 0;
101  }
102 
103  last_sector = sector;
104 
105  /* Forward to PIOS */
106  int32_t status;
107  if((status = PIOS_SDCARD_SectorRead(sector, buffer)) != 0) {
108  /* Cannot access SD Card */
109  return 3;
110  }
111 
112  /* Success */
113  return 0;
114 }

120 uint32_t DFS_WriteSector(uint8_t unit, uint8_t *buffer, uint32_t sector, uint32_t count)
