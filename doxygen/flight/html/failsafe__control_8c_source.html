55 int32_t failsafe_control_select(bool reset_controller)
56 {
57  uint8_t flight_status;
58  FlightStatusFlightModeGet(&flight_status);
59  if (flight_status != FLIGHTSTATUS_FLIGHTMODE_STABILIZED1 || reset_controller) {
60  flight_status = FLIGHTSTATUS_FLIGHTMODE_STABILIZED1;
61  FlightStatusFlightModeSet(&flight_status);
62  }
63 
64  // Pick default values that will roughly cause a plane to circle down
65  // and a quad to fall straight down
66  StabilizationDesiredData stabilization_desired;
67  StabilizationDesiredGet(&stabilization_desired);
68  stabilization_desired.Throttle = -1;
69  stabilization_desired.Roll = -10;
70  stabilization_desired.Pitch = 0;
71  stabilization_desired.Yaw = -5;
72  stabilization_desired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
73  stabilization_desired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
74  stabilization_desired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK;
75  StabilizationDesiredSet(&stabilization_desired);
76 
77  return 0;
78 }
