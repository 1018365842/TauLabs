56 int8_t getSensorsCC(float *prelim_accels, float *prelim_gyros, xQueueHandle * gyro_queue, GlobalAttitudeVariables *glblAtt, GyrosBiasData *gyrosBias, SensorSettingsData *inertialSensorSettings)
57 {
58  struct pios_adxl345_data accel_data;
59  float gyro[4];
60 
61  // Only wait the time for two nominal updates before setting an alarm
62  if (xQueueReceive(*gyro_queue, (void *const)gyro, LOOP_RATE_MS * 2) ==
63  errQUEUE_EMPTY) {
64  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE,
65  SYSTEMALARMS_ALARM_ERROR);
66  return -1;
67  }
68  // Do not read raw sensor data in simulation mode
69  if (GyrosReadOnly() || AccelsReadOnly())
70  return 0;
71 
72  // No accel data available
73  if (PIOS_ADXL345_FifoElements() == 0)
74  return -1;
75 
76  // Scale ADC data into deg/s. First sample is temperature, so ignore.
77  // Rotate data from internal gryoscope sensor frame into board sensor frame
78  prelim_gyros[0] = -(gyro[1] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_X];
79  prelim_gyros[1] = (gyro[2] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_Y];
80  prelim_gyros[2] = -(gyro[3] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_Z];
81 
82  // When this is enabled remove estimate of bias
83  if (glblAtt->bias_correct_gyro) {
84  prelim_gyros[0] -= gyrosBias->x;
85  prelim_gyros[1] -= gyrosBias->y;
86  prelim_gyros[2] -= gyrosBias->z;
87  }
88 
89  // Process accelerometer sensor data. In this case, average the data
90  int32_t x = 0;
91  int32_t y = 0;
92  int32_t z = 0;
93  uint8_t i = 0;
94  uint8_t samples_remaining;
95  do {
96  i++;
97  samples_remaining = PIOS_ADXL345_Read(&accel_data);
98 
99  //Assign data, rotating from internal accelerometer frame into board sensor frame
100  x += accel_data.x;
101  y += -accel_data.y;
102  z += -accel_data.z;
103  } while ((i < 32) && (samples_remaining > 0)); //<-- i=32 being hardcoded means that if the accelerometer ADC sample rate is increased, we could wind up never being able to empty the buffer
104 
105  // Apply scaling and bias correction in sensor frame
106  prelim_accels[0] = (float)x / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[0] - inertialSensorSettings->AccelBias[0];
107  prelim_accels[1] = (float)y / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[1] - inertialSensorSettings->AccelBias[1];
108  prelim_accels[2] = (float)z / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[2] - inertialSensorSettings->AccelBias[2];
109 
110  return 0;
111 }
