69 int8_t getSensorsCC(float *prelim_accels, float *prelim_gyros, xQueueHandle * gyro_queue, GlobalAttitudeVariables *glblAtt, GyrosBiasData *gyrosBias, SensorSettingsData *inertialSensorSettings)
70 {
71  struct pios_adxl345_data accel_data;
72  float gyro[4];
73 
74  // Only wait the time for two nominal updates before setting an alarm
75  if (xQueueReceive(*gyro_queue, (void *const)gyro, LOOP_RATE_MS * 2) ==
76  errQUEUE_EMPTY) {
77  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE,
78  SYSTEMALARMS_ALARM_ERROR);
79  return -1;
80  }
81  // Do not read raw sensor data in simulation mode
82  if (GyrosReadOnly() || AccelsReadOnly())
83  return 0;
84 
85  // No accel data available
86  if (PIOS_ADXL345_FifoElements() == 0)
87  return -1;
88 
89  // Scale ADC data into deg/s. First sample is temperature, so ignore.
90  // Rotate data from internal gryoscope sensor frame into board sensor frame
91  prelim_gyros[0] = -(gyro[1] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_X];
92  prelim_gyros[1] = (gyro[2] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_Y];
93  prelim_gyros[2] = -(gyro[3] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_Z];
94 
95  // When this is enabled remove estimate of bias
96  if (glblAtt->bias_correct_gyro) {
97  prelim_gyros[0] -= gyrosBias->x;
98  prelim_gyros[1] -= gyrosBias->y;
99  prelim_gyros[2] -= gyrosBias->z;
100  }
101 
102  // Process accelerometer sensor data. In this case, average the data
103  int32_t x = 0;
104  int32_t y = 0;
105  int32_t z = 0;
106  uint8_t i = 0;
107  uint8_t samples_remaining;
108  do {
109  i++;
110  samples_remaining = PIOS_ADXL345_Read(&accel_data);
111 
112  //Assign data, rotating from internal accelerometer frame into board sensor frame
113  x += accel_data.x;
114  y += -accel_data.y;
115  z += -accel_data.z;
116  } while ((i < 32) && (samples_remaining > 0)); //<-- i=32 being hardcoded means that if the accelerometer ADC sample rate is increased, we could wind up never being able to empty the buffer
117 
118  // Apply scaling and bias correction in sensor frame
119  prelim_accels[0] = (float)x / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[0] - inertialSensorSettings->AccelBias[0];
120  prelim_accels[1] = (float)y / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[1] - inertialSensorSettings->AccelBias[1];
121  prelim_accels[2] = (float)z / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[2] - inertialSensorSettings->AccelBias[2];
122 
123  return 0;
124 }
