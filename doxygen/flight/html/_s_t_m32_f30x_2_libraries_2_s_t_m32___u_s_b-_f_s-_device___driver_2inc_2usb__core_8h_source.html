56 typedef struct _ENDPOINT_INFO
57 {
58  /* When send data out of the device,
59  CopyData() is used to get data buffer 'Length' bytes data
60  if Length is 0,
61  CopyData() returns the total length of the data
62  if the request is not supported, returns 0
63  (NEW Feature )
64  if CopyData() returns -1, the calling routine should not proceed
65  further and will resume the SETUP process by the class device
66  if Length is not 0,
67  CopyData() returns a pointer to indicate the data location
68  Usb_wLength is the data remain to be sent,
69  Usb_wOffset is the Offset of original data
70  When receive data from the host,
71  CopyData() is used to get user data buffer which is capable
72  of Length bytes data to copy data from the endpoint buffer.
73  if Length is 0,
74  CopyData() returns the available data length,
75  if Length is not 0,
76  CopyData() returns user buffer address
77  Usb_rLength is the data remain to be received,
78  Usb_rPointer is the Offset of data buffer
79  */
80  uint16_t Usb_wLength;
81  uint16_t Usb_wOffset;
82  uint16_t PacketSize;
83  const uint8_t *(*CopyDataIn)(uint16_t Length);
84  uint8_t *(*CopyDataOut)(uint16_t Length);
85 }ENDPOINT_INFO;
