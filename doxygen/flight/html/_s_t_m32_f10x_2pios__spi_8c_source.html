71 int32_t PIOS_SPI_Init(uint32_t *spi_id, const struct pios_spi_cfg *cfg)
72 {
73  uint32_t init_ssel = 0;
74 
75  PIOS_Assert(spi_id);
76  PIOS_Assert(cfg);
77 
78  struct pios_spi_dev *spi_dev;
79 
80  spi_dev = (struct pios_spi_dev *) PIOS_SPI_alloc();
81  if (!spi_dev) goto out_fail;
82 
83  /* Bind the configuration to the device instance */
84  spi_dev->cfg = cfg;
85 
86 #if defined(PIOS_INCLUDE_FREERTOS)
87  vSemaphoreCreateBinary(spi_dev->busy);
88  xSemaphoreGive(spi_dev->busy);
89 #else
90  spi_dev->busy = 0;
91 #endif
92 
93  /* Disable callback function */
94  spi_dev->callback = NULL;
95 
96  /* Set rx/tx dummy bytes to a known value */
97  spi_dev->rx_dummy_byte = 0xFF;
98  spi_dev->tx_dummy_byte = 0xFF;
99 
100  switch (spi_dev->cfg->init.SPI_NSS) {
101  case SPI_NSS_Soft:
102  if (spi_dev->cfg->init.SPI_Mode == SPI_Mode_Master) {
103  /* We're a master in soft NSS mode, make sure we see NSS high at all times. */
104  SPI_NSSInternalSoftwareConfig(spi_dev->cfg->regs, SPI_NSSInternalSoft_Set);
105  /* Init as many slave selects as the config advertises. */
106  init_ssel = spi_dev->cfg->slave_count;
107  } else {
108  /* We're a slave in soft NSS mode, make sure we see NSS low at all times. */
109  SPI_NSSInternalSoftwareConfig(spi_dev->cfg->regs, SPI_NSSInternalSoft_Reset);
110  }
111  break;
112  case SPI_NSS_Hard:
113  /* only legal for single-slave config */
114  PIOS_Assert(spi_dev->cfg->slave_count == 1);
115  init_ssel = 1;
116  /* FIXME: Should this also call SPI_SSOutputCmd()? */
117  break;
118 
119  default:
120  PIOS_Assert(0);
121  }
122 
123  /* Initialize the GPIO pins */
124  GPIO_Init(spi_dev->cfg->sclk.gpio, &(spi_dev->cfg->sclk.init));
125  GPIO_Init(spi_dev->cfg->mosi.gpio, &(spi_dev->cfg->mosi.init));
126  GPIO_Init(spi_dev->cfg->miso.gpio, &(spi_dev->cfg->miso.init));
127  for (uint32_t i = 0; i < init_ssel; i++) {
128  /* Since we're driving the SSEL pin in software, ensure that the slave is deselected */
129  /* XXX multi-slave support - maybe have another SPI_NSS_ mode? */
130  GPIO_SetBits(spi_dev->cfg->ssel[i].gpio, spi_dev->cfg->ssel[i].init.GPIO_Pin);
131  GPIO_Init(spi_dev->cfg->ssel[i].gpio, (GPIO_InitTypeDef *) & (spi_dev->cfg->ssel[i].init));
132  }
133 
134  /* Enable the associated peripheral clock */
135  switch ((uint32_t) spi_dev->cfg->regs) {
136  case (uint32_t) SPI1:
137  /* Enable SPI peripheral clock (APB2 == high speed) */
138  RCC_APB2PeriphClockCmd(RCC_APB2Periph_SPI1, ENABLE);
139  break;
140  case (uint32_t) SPI2:
141  /* Enable SPI peripheral clock (APB1 == slow speed) */
142  RCC_APB1PeriphClockCmd(RCC_APB1Periph_SPI2, ENABLE);
143  break;
144  case (uint32_t) SPI3:
145  /* Enable SPI peripheral clock (APB1 == slow speed) */
146  RCC_APB1PeriphClockCmd(RCC_APB1Periph_SPI3, ENABLE);
147  break;
148  }
149 
150  /* Enable DMA clock */
151  RCC_AHBPeriphClockCmd(spi_dev->cfg->dma.ahb_clk, ENABLE);
152 
153  /* Configure DMA for SPI Rx */
154  DMA_Cmd(spi_dev->cfg->dma.rx.channel, DISABLE);
155  DMA_Init(spi_dev->cfg->dma.rx.channel, &(spi_dev->cfg->dma.rx.init));
156 
157  /* Configure DMA for SPI Tx */
158  DMA_Cmd(spi_dev->cfg->dma.tx.channel, DISABLE);
159  DMA_Init(spi_dev->cfg->dma.tx.channel, &(spi_dev->cfg->dma.tx.init));
160 
161  /* Initialize the SPI block */
162  SPI_Init(spi_dev->cfg->regs, &(spi_dev->cfg->init));
163 
164  /* Configure CRC calculation */
165  if (spi_dev->cfg->use_crc) {
166  SPI_CalculateCRC(spi_dev->cfg->regs, ENABLE);
167  } else {
168  SPI_CalculateCRC(spi_dev->cfg->regs, DISABLE);
169  }
170 
171  /* Enable SPI */
172  SPI_Cmd(spi_dev->cfg->regs, ENABLE);
173 
174  /* Enable SPI interrupts to DMA */
175  SPI_I2S_DMACmd(spi_dev->cfg->regs, SPI_I2S_DMAReq_Tx | SPI_I2S_DMAReq_Rx, ENABLE);
176 
177  /* Configure DMA interrupt */
178  NVIC_Init(&(spi_dev->cfg->dma.irq.init));
179 
180  *spi_id = (uint32_t)spi_dev;
181  return (0);
182 
183 out_fail:
184  return (-1);
185 }

385 int32_t PIOS_SPI_TransferByte(uint32_t spi_id, uint8_t b)
386 {
387  struct pios_spi_dev *spi_dev = (struct pios_spi_dev *)spi_id;
388 
389  bool valid = PIOS_SPI_validate(spi_dev);
390  PIOS_Assert(valid)
391 
392  uint8_t rx_byte;
393 
394  /*
395  * Procedure taken from STM32F10xxx Reference Manual section 23.3.5
396  */
397 
398  /* Make sure the RXNE flag is cleared by reading the DR register */
399  dummy = spi_dev->cfg->regs->DR;
400 
401  /* Start the transfer */
402  spi_dev->cfg->regs->DR = b;
403 
404  /* Wait until there is a byte to read */
405  while (!(spi_dev->cfg->regs->SR & SPI_I2S_FLAG_RXNE)) ;
406 
407  /* Read the rx'd byte */
408  rx_byte = spi_dev->cfg->regs->DR;
409 
410  /* Wait until the TXE goes high */
411  while (!(spi_dev->cfg->regs->SR & SPI_I2S_FLAG_TXE)) ;
412 
413  /* Wait for SPI transfer to have fully completed */
414  while (spi_dev->cfg->regs->SR & SPI_I2S_FLAG_BSY) ;
415 
416  /* Return received byte */
417  return rx_byte;
418 }

436 int32_t PIOS_SPI_TransferBlock(uint32_t spi_id, const uint8_t *send_buffer, uint8_t *receive_buffer, uint16_t len, void *callback)
437 {
438  struct pios_spi_dev *spi_dev = (struct pios_spi_dev *)spi_id;
439 
440  bool valid = PIOS_SPI_validate(spi_dev);
441  PIOS_Assert(valid)
442 
443  DMA_InitTypeDef dma_init;
444 
445  /* Exit if ongoing transfer */
446  if (DMA_GetCurrDataCounter(spi_dev->cfg->dma.rx.channel)) {
447  return -3;
448  }
449 
450  /* Disable the SPI peripheral */
451  SPI_Cmd(spi_dev->cfg->regs, DISABLE);
452 
453  /* Disable the DMA channels */
454  DMA_Cmd(spi_dev->cfg->dma.rx.channel, DISABLE);
455  DMA_Cmd(spi_dev->cfg->dma.tx.channel, DISABLE);
456 
457  /* Set callback function */
458  spi_dev->callback = callback;
459 
460  /*
461  * Configure Rx channel
462  */
463 
464  /* Start with the default configuration for this peripheral */
465  dma_init = spi_dev->cfg->dma.rx.init;
466 
467  if (receive_buffer != NULL) {
468  /* Enable memory addr. increment - bytes written into receive buffer */
469  dma_init.DMA_MemoryBaseAddr = (uint32_t) receive_buffer;
470  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
471  } else {
472  /* Disable memory addr. increment - bytes written into dummy buffer */
473  spi_dev->rx_dummy_byte = 0xFF;
474  dma_init.DMA_MemoryBaseAddr = (uint32_t) &spi_dev->rx_dummy_byte;
475  dma_init.DMA_MemoryInc = DMA_MemoryInc_Disable;
476  }
477  if (spi_dev->cfg->use_crc) {
478  /* Make sure the CRC error flag is cleared before we start */
479  SPI_I2S_ClearFlag(spi_dev->cfg->regs, SPI_FLAG_CRCERR);
480  }
481 
482  dma_init.DMA_BufferSize = len;
483  DMA_Init(spi_dev->cfg->dma.rx.channel, &(dma_init));
484 
485  /*
486  * Configure Tx channel
487  */
488 
489  /* Start with the default configuration for this peripheral */
490  dma_init = spi_dev->cfg->dma.tx.init;
491 
492  if (send_buffer != NULL) {
493  /* Enable memory addr. increment - bytes written into receive buffer */
494  dma_init.DMA_MemoryBaseAddr = (uint32_t) send_buffer;
495  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
496  } else {
497  /* Disable memory addr. increment - bytes written into dummy buffer */
498  spi_dev->tx_dummy_byte = 0xFF;
499  dma_init.DMA_MemoryBaseAddr = (uint32_t) &spi_dev->tx_dummy_byte;
500  dma_init.DMA_MemoryInc = DMA_MemoryInc_Disable;
501  }
502 
503  if (spi_dev->cfg->use_crc) {
504  /* The last byte of the payload will be replaced with the CRC8 */
505  dma_init.DMA_BufferSize = len - 1;
506  } else {
507  dma_init.DMA_BufferSize = len;
508  }
509 
510  DMA_Init(spi_dev->cfg->dma.tx.channel, &(dma_init));
511 
512  /* Enable DMA interrupt if callback function active */
513  DMA_ITConfig(spi_dev->cfg->dma.rx.channel, DMA_IT_TC, (callback != NULL) ? ENABLE : DISABLE);
514 
515  /* Flush out the CRC registers */
516  SPI_CalculateCRC(spi_dev->cfg->regs, DISABLE);
517  (void)SPI_GetCRC(spi_dev->cfg->regs, SPI_CRC_Rx);
518  SPI_I2S_ClearFlag(spi_dev->cfg->regs, SPI_FLAG_CRCERR);
519 
520  /* Make sure to flush out the receive buffer */
521  (void)SPI_I2S_ReceiveData(spi_dev->cfg->regs);
522 
523  if (spi_dev->cfg->use_crc) {
524  /* Need a 0->1 transition to reset the CRC logic */
525  SPI_CalculateCRC(spi_dev->cfg->regs, ENABLE);
526  }
527 
528  /* Start DMA transfers */
529  DMA_Cmd(spi_dev->cfg->dma.rx.channel, ENABLE);
530  DMA_Cmd(spi_dev->cfg->dma.tx.channel, ENABLE);
531 
532  /* Reenable the SPI device */
533  SPI_Cmd(spi_dev->cfg->regs, ENABLE);
534 
535  if (callback) {
536  /* User has requested a callback, don't wait for the transfer to complete. */
537  return 0;
538  }
539 
540  /* Wait until all bytes have been transmitted/received */
541  while (DMA_GetCurrDataCounter(spi_dev->cfg->dma.rx.channel));
542 
543  /* Wait for the final bytes of the transfer to complete, including CRC byte(s). */
544  while (!(SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_I2S_FLAG_TXE)));
545 
546  /* Wait for the final bytes of the transfer to complete, including CRC byte(s). */
547  while (SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_I2S_FLAG_BSY));
548 
549  /* Check the CRC on the transfer if enabled. */
550  if (spi_dev->cfg->use_crc) {
551  /* Check the SPI CRC error flag */
552  if (SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_FLAG_CRCERR)) {
553  return -4;
554  }
555  }
556 
557  /* No error */
558  return 0;
559 }

597 void PIOS_SPI_IRQ_Handler(uint32_t spi_id)
598 {
599  struct pios_spi_dev *spi_dev = (struct pios_spi_dev *)spi_id;
600 
601  bool valid = PIOS_SPI_validate(spi_dev);
602  PIOS_Assert(valid)
603 
604  DMA_ClearFlag(spi_dev->cfg->dma.irq.flags);
605 
606  /* Wait for the final bytes of the transfer to complete, including CRC byte(s). */
607  while (!(SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_I2S_FLAG_TXE))) ;
608 
609  /* Wait for the final bytes of the transfer to complete, including CRC byte(s). */
610  while (SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_I2S_FLAG_BSY)) ;
611 
612  if (spi_dev->callback != NULL) {
613  bool crc_ok = TRUE;
614  uint8_t crc_val;
615 
616  if (SPI_I2S_GetFlagStatus(spi_dev->cfg->regs, SPI_FLAG_CRCERR)) {
617  crc_ok = FALSE;
618  SPI_I2S_ClearFlag(spi_dev->cfg->regs, SPI_FLAG_CRCERR);
619  }
620  crc_val = SPI_GetCRC(spi_dev->cfg->regs, SPI_CRC_Rx);
621  spi_dev->callback(crc_ok, crc_val);
622  }
623 }
