273 static void PIOS_Board_configure_com (const struct pios_usart_cfg *usart_port_cfg, size_t rx_buf_len, size_t tx_buf_len,
274  const struct pios_com_driver *com_driver, uintptr_t *pios_com_id)
275 {
276  uint32_t pios_usart_id;
277  if (PIOS_USART_Init(&pios_usart_id, usart_port_cfg)) {
278  PIOS_Assert(0);
279  }
280 
281  uint8_t * rx_buffer;
282  if (rx_buf_len > 0) {
283  rx_buffer = (uint8_t *) pvPortMalloc(rx_buf_len);
284  PIOS_Assert(rx_buffer);
285  } else {
286  rx_buffer = NULL;
287  }
288 
289  uint8_t * tx_buffer;
290  if (tx_buf_len > 0) {
291  tx_buffer = (uint8_t *) pvPortMalloc(tx_buf_len);
292  PIOS_Assert(tx_buffer);
293  } else {
294  tx_buffer = NULL;
295  }
296 
297  if (PIOS_COM_Init(pios_com_id, com_driver, pios_usart_id,
298  rx_buffer, rx_buf_len,
299  tx_buffer, tx_buf_len)) {
300  PIOS_Assert(0);
301  }
302 }
