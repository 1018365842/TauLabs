{
    if (OBJ_GET_TYPE(pi) != OBJ_TYPE_INT)
    {
        PM_RAISE(retval, PM_RET_EX_TYPE);
        return retval;
    }
    i = (uint16_t)((pPmInt_t)pi)->val;

{
    i = -1;

/* Return the object to the caller */
