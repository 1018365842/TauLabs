114 void PIOS_BMP085_Init(void)
115 {
116 #ifdef PIOS_BMP085_HAS_GPIOS
117 
118 #if defined(PIOS_INCLUDE_FREERTOS)
119  /* Semaphore used by ISR to signal End-Of-Conversion */
120  vSemaphoreCreateBinary(PIOS_BMP085_EOC);
121  /* Must start off empty so that first transfer waits for EOC */
122  xSemaphoreTake(PIOS_BMP085_EOC, portMAX_DELAY);
123 #else
124  PIOS_BMP085_EOC = 0;
125 #endif
126 
127  /* Enable EOC GPIO clock */
128  RCC_APB2PeriphClockCmd(PIOS_BMP085_EOC_CLK | RCC_APB2Periph_AFIO, ENABLE);
129 
130  if (PIOS_EXTI_Init(&pios_exti_bmp085_cfg)) {
131  PIOS_Assert(0);
132  }
133 
134  /* Configure XCLR pin as push/pull alternate funtion output */
135  GPIO_InitTypeDef GPIO_InitStructure;
136  GPIO_InitStructure.GPIO_Pin = PIOS_BMP085_XCLR_GPIO_PIN;
137  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF_PP;
138  GPIO_Init(PIOS_BMP085_XCLR_GPIO_PORT, &GPIO_InitStructure);
139 
140 #endif /* PIOS_BMP085_HAS_GPIOS */
141 
142  /* Read all 22 bytes of calibration data in one transfer, this is a very optimized way of doing things */
143  uint8_t Data[BMP085_CALIB_LEN];
144  while (PIOS_BMP085_Read(BMP085_CALIB_ADDR, Data, BMP085_CALIB_LEN) != 0)
145  continue;
146 
147  /* Parameters AC1-AC6 */
148  CalibData.AC1 = (Data[0] << 8) | Data[1];
149  CalibData.AC2 = (Data[2] << 8) | Data[3];
150  CalibData.AC3 = (Data[4] << 8) | Data[5];
151  CalibData.AC4 = (Data[6] << 8) | Data[7];
152  CalibData.AC5 = (Data[8] << 8) | Data[9];
153  CalibData.AC6 = (Data[10] << 8) | Data[11];
154 
155  /* Parameters B1, B2 */
156  CalibData.B1 = (Data[12] << 8) | Data[13];
157  CalibData.B2 = (Data[14] << 8) | Data[15];
158 
159  /* Parameters MB, MC, MD */
160  CalibData.MB = (Data[16] << 8) | Data[17];
161  CalibData.MC = (Data[18] << 8) | Data[19];
162  CalibData.MD = (Data[20] << 8) | Data[21];
163 }

170 void PIOS_BMP085_StartADC(ConversionTypeTypeDef Type)
171 {
172  /* Start the conversion */
173  if (Type == TemperatureConv) {
174  while (PIOS_BMP085_Write(BMP085_CTRL_ADDR, BMP085_TEMP_ADDR) != 0)
175  continue;
176  } else if (Type == PressureConv) {
177  while (PIOS_BMP085_Write(BMP085_CTRL_ADDR, BMP085_PRES_ADDR) != 0)
178  continue;
179  }
180 
181  CurrentRead = Type;
182 }

189 void PIOS_BMP085_ReadADC(void)
190 {
191  uint8_t Data[3];
192  Data[0] = 0;
193  Data[1] = 0;
194  Data[2] = 0;
195 
196  /* Read and store the 16bit result */
197  if (CurrentRead == TemperatureConv) {
198  /* Read the temperature conversion */
199  while (PIOS_BMP085_Read(BMP085_ADC_MSB, Data, 2) != 0)
200  continue;
201  RawTemperature = ((Data[0] << 8) | Data[1]);
202 
203  X1 = (RawTemperature - CalibData.AC6) * CalibData.AC5 >> 15;
204  X2 = ((int32_t) CalibData.MC << 11) / (X1 + CalibData.MD);
205  B5 = X1 + X2;
206  Temperature = (B5 + 8) >> 4;
207  } else {
208  /* Read the pressure conversion */
209  while (PIOS_BMP085_Read(BMP085_ADC_MSB, Data, 3) != 0)
210  continue;
211  RawPressure = ((Data[0] << 16) | (Data[1] << 8) | Data[2]) >> (8 - BMP085_OVERSAMPLING);
212 
213  B6 = B5 - 4000;
214  X1 = (CalibData.B2 * (B6 * B6 >> 12)) >> 11;
215  X2 = CalibData.AC2 * B6 >> 11;
216  X3 = X1 + X2;
217  B3 = ((((int32_t) CalibData.AC1 * 4 + X3) << BMP085_OVERSAMPLING) + 2) >> 2;
218  X1 = CalibData.AC3 * B6 >> 13;
219  X2 = (CalibData.B1 * (B6 * B6 >> 12)) >> 16;
220  X3 = ((X1 + X2) + 2) >> 2;
221  B4 = (CalibData.AC4 * (uint32_t) (X3 + 32768)) >> 15;
222  B7 = ((uint32_t) RawPressure - B3) * (50000 >> BMP085_OVERSAMPLING);
223  P = B7 < 0x80000000 ? (B7 * 2) / B4 : (B7 / B4) * 2;
224 
225  X1 = (P >> 8) * (P >> 8);
226  X1 = (X1 * 3038) >> 16;
227  X2 = (-7357 * P) >> 16;
228  Pressure = P + ((X1 + X2 + 3791) >> 4);
229  }
230 }
