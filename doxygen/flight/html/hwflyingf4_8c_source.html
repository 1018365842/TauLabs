48 int32_t HwFlyingF4Initialize(void)
49 {
50  // Don't set the handle to null if already registered
51  if(UAVObjGetByID(HWFLYINGF4_OBJID) != NULL)
52  return -2;
53 
54  // Register object with the object manager
55  handle = UAVObjRegister(HWFLYINGF4_OBJID,
56  HWFLYINGF4_ISSINGLEINST, HWFLYINGF4_ISSETTINGS, HWFLYINGF4_NUMBYTES, &HwFlyingF4SetDefaults);
57 
58  // Done
59  if (handle != 0)
60  {
61  return 0;
62  }
63  else
64  {
65  return -1;
66  }
67 }

74 void HwFlyingF4SetDefaults(UAVObjHandle obj, uint16_t instId)
75 {
76  HwFlyingF4Data data;
77  UAVObjMetadata metadata;
78 
79  // Initialize object fields to their default values
80  UAVObjGetInstanceData(obj, instId, &data);
81  memset(&data, 0, sizeof(HwFlyingF4Data));
82  data.RcvrPort = 1;
83  data.Uart1 = 0;
84  data.Uart2 = 0;
85  data.Uart3 = 0;
86  data.USB_HIDPort = 0;
87  data.USB_VCPPort = 3;
88  data.DSMxBind = 0;
89  data.GyroRange = 1;
90  data.AccelRange = 2;
91  data.MPU6050Rate = 3;
92  data.MPU6050DLPF = 0;
93 
94  UAVObjSetInstanceData(obj, instId, &data);
95 
96  // Initialize object metadata to their default values
97  metadata.flags =
98  ACCESS_READWRITE << UAVOBJ_ACCESS_SHIFT |
99  ACCESS_READWRITE << UAVOBJ_GCS_ACCESS_SHIFT |
100  1 << UAVOBJ_TELEMETRY_ACKED_SHIFT |
101  1 << UAVOBJ_GCS_TELEMETRY_ACKED_SHIFT |
102  UPDATEMODE_ONCHANGE << UAVOBJ_TELEMETRY_UPDATE_MODE_SHIFT |
103  UPDATEMODE_ONCHANGE << UAVOBJ_GCS_TELEMETRY_UPDATE_MODE_SHIFT;
104  metadata.telemetryUpdatePeriod = 0;
105  metadata.gcsTelemetryUpdatePeriod = 0;
106  metadata.loggingUpdatePeriod = 0;
107  UAVObjSetMetadata(obj, &metadata);
108 }
