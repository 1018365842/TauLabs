267 void arm_sin_cos_q31(
268  q31_t theta,
269  q31_t * pSinVal,
270  q31_t * pCosVal)
271 {
272  q31_t x0; /* Nearest input value */
273  q31_t y0, y1; /* Nearest output values */
274  q31_t xSpacing = INPUT_SPACING; /* Spaing between inputs */
275  int32_t i; /* Index */
276  q31_t oneByXSpacing; /* 1/ xSpacing value */
277  q31_t out; /* temporary variable */
278  uint32_t sign_bits; /* No.of sign bits */
279  uint32_t firstX = 0x80000000; /* First X value */
280 
281  /* Calculation of index */
282  i = ((uint32_t) theta - firstX) / (uint32_t) xSpacing;
283 
284  /* Checking min and max index of table */
285  if(i < 0)
286  {
287  i = 0;
288  }
289  else if(i >= 359)
290  {
291  i = 358;
292  }
293 
294  /* Calculation of first nearest input value */
295  x0 = (q31_t) firstX + ((q31_t) i * xSpacing);
296 
297  /* Reading nearest sine output values from table */
298  y0 = sinTableQ31[i];
299  y1 = sinTableQ31[i + 1u];
300 
301  /* Calculation of 1/(x1-x0) */
302  /* (x1-x0) is xSpacing which is fixed value */
303  sign_bits = 8u;
304  oneByXSpacing = 0x5A000000;
305 
306  /* Calculation of (theta - x0)/(x1-x0) */
307  out =
308  (((q31_t) (((q63_t) (theta - x0) * oneByXSpacing) >> 32)) << sign_bits);
309 
310  /* Calculation of y0 + (y1 - y0) * ((theta - x0)/(x1-x0)) */
311  *pSinVal = __QADD(y0, ((q31_t) (((q63_t) (y1 - y0) * out) >> 30)));
312 
313  /* Reading nearest cosine output values from table */
314  y0 = cosTableQ31[i];
315  y1 = cosTableQ31[i + 1u];
316 
317  /* Calculation of y0 + (y1 - y0) * ((theta - x0)/(x1-x0)) */
318  *pCosVal = __QADD(y0, ((q31_t) (((q63_t) (y1 - y0) * out) >> 30)));
319 
320 }
