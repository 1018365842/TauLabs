29 /*
30  * @note This is a stripped-down ADC driver intended primarily for sampling
31  * voltage and current values. Samples are averaged over the period between
32  * fetches so that relatively accurate measurements can be obtained without
33  * forcing higher-level logic to poll aggressively.
34  *
35  * @todo This module needs more work to be more generally useful. It should
36  * almost certainly grow callback support so that e.g. voltage and current readings
37  * can be shipped out for coulomb counting purposes. The F1xx interface presumes
38  * use with analog sensors, but that implementation largely dominates the ADC
39  * resources. Rather than commit to a new API without a defined use case, we
40  * should stick to our lightweight subset until we have a better idea of what's needed.
41  */

151 static void PIOS_INTERNAL_DMAConfig(uint32_t internal_adc_id)
152 {
153  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *) internal_adc_id;
154  if (!PIOS_INTERNAL_ADC_validate(adc_dev)) {
155  return;
156  }
157  /* Disable interrupts */
158  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, adc_dev->cfg->dma.irq.flags, DISABLE);
159 
160  switch (current_instances) {
161  case 1:
162  PIOS_DMA_Install_Interrupt_handler(adc_dev->cfg->dma.rx.channel, &PIOS_INTERNAL_ADC_DMA_Handler1);
163  break;
164  case 2:
165  PIOS_DMA_Install_Interrupt_handler(adc_dev->cfg->dma.rx.channel, &PIOS_INTERNAL_ADC_DMA_Handler2);
166  break;
167  case 3:
168  PIOS_DMA_Install_Interrupt_handler(adc_dev->cfg->dma.rx.channel, &PIOS_INTERNAL_ADC_DMA_Handler3);
169  break;
170  case 4:
171  PIOS_DMA_Install_Interrupt_handler(adc_dev->cfg->dma.rx.channel, &PIOS_INTERNAL_ADC_DMA_Handler4);
172  break;
173  default:
174  break;
175  }
176 
177  /* Configure DMA channel */
178  DMA_DeInit(adc_dev->cfg->dma.rx.channel);
179  DMA_InitTypeDef DMAInit = adc_dev->cfg->dma.rx.init;
180  if (adc_dev->cfg->adc_dev_slave) {
181  if (adc_dev->cfg->adc_dev_master == ADC1 )
182  DMAInit.DMA_PeripheralBaseAddr = (uint32_t) &ADC1_2->CDR;
183  else
184  DMAInit.DMA_PeripheralBaseAddr = (uint32_t) &ADC3_4->CDR;
185  }
186  else
187  DMAInit.DMA_PeripheralBaseAddr = (uint32_t) &adc_dev->cfg->adc_dev_master->DR;
188  DMAInit.DMA_MemoryBaseAddr = (uint32_t) adc_dev->raw_data_buffer;
189  DMAInit.DMA_BufferSize = adc_dev->dma_half_buffer_size;
190  DMAInit.DMA_DIR = DMA_DIR_PeripheralSRC;
191  DMAInit.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
192  DMAInit.DMA_MemoryInc = DMA_MemoryInc_Enable;
193  if(adc_dev->cfg->adc_dev_slave)
194  {
195  DMAInit.DMA_PeripheralDataSize = DMA_PeripheralDataSize_Word;
196  DMAInit.DMA_MemoryDataSize = DMA_MemoryDataSize_Word;
197  }
198  else
199  {
200  DMAInit.DMA_PeripheralDataSize = DMA_PeripheralDataSize_HalfWord;
201  DMAInit.DMA_MemoryDataSize = DMA_MemoryDataSize_HalfWord;
202  }
203  DMAInit.DMA_Mode = DMA_Mode_Circular;
204  DMAInit.DMA_M2M = DMA_M2M_Disable;
205 
206 
207  DMA_Init(adc_dev->cfg->dma.rx.channel, &DMAInit); /* channel is actually stream ... */
208 
209  /* enable DMA */
210  DMA_Cmd(adc_dev->cfg->dma.rx.channel, ENABLE);
211 
212  /* configure for double-buffered mode and interrupt on every buffer flip */
213  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, DMA_IT_TC, ENABLE);
214  DMA_ITConfig(adc_dev->cfg->dma.rx.channel, DMA_IT_HT, ENABLE);
215 
216  /* Configure DMA interrupt */
217  NVIC_InitTypeDef NVICInit = adc_dev->cfg->dma.irq.init;
218  NVIC_Init(&NVICInit);

226  struct pios_internal_adc_dev * adc_dev = (struct pios_internal_adc_dev *) internal_adc_id;
227 
228  ADC_DeInit(adc_dev->cfg->adc_dev_master);
229  if (adc_dev->cfg->adc_dev_slave)
230  ADC_DeInit(adc_dev->cfg->adc_dev_slave);

232  RCC_ADCCLKConfig(RCC_ADC12PLLCLK_Div2);//TODO
233  ADC_VoltageRegulatorCmd(adc_dev->cfg->adc_dev_master, ENABLE);

236  ADC_StartCalibration(adc_dev->cfg->adc_dev_master);

244  ADC_StartCalibration(adc_dev->cfg->adc_dev_slave);

246  }
247 
248  /* Do common ADC init */
249  ADC_CommonInitTypeDef ADC_CommonInitStructure;
250  ADC_CommonStructInit(&ADC_CommonInitStructure);
251  if (adc_dev->cfg->adc_dev_slave) {
252  ADC_CommonInitStructure.ADC_Mode = ADC_Mode_RegSimul;
253  ADC_CommonInitStructure.ADC_DMAAccessMode = ADC_DMAAccessMode_1;
254  }
255  else {
256  ADC_CommonInitStructure.ADC_Mode = ADC_Mode_Independent;
257  ADC_CommonInitStructure.ADC_DMAAccessMode = ADC_DMAAccessMode_Disabled;
258  }
259

273  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Right;

277  ADC_Init(adc_dev->cfg->adc_dev_master, &ADC_InitStructure);
278

280  ADC_Init(adc_dev->cfg->adc_dev_slave, &ADC_InitStructure);
281 
282  /* Enable DMA request */
283  ADC_DMACmd(adc_dev->cfg->adc_dev_master, ENABLE);

285  /* Configure input scan
286  * channel_map indexing corresponds to each conversion slot in order
287  * in single channel mode this corresponds to 0,1,2,3...
288  * in dual channel mode this corresponds to 0,2,4...
289  * 1,3,5...
290  * channel_map value is a pointer to an accumulator, if the same channel is used multiple times the same accumulator is used
291  *
292  * Input scan is setup to repeat channels if needed, i.e if a channel has more conversions to make the other will repeat conversions
293  * example:
294  * 2 ADC1 pins to convert pinA1, pinA2
295  * 3 ADC2 pins to convert pinB1, pinB2, pinB3
296  * input scan becomes:
297  * pinA1, pinA2, pinA1
298  * pinB1, pinB2, pinB3
299  */
300  uint32_t current_index = 0;
301  if (!adc_dev->cfg->adc_dev_slave) {
302  for (uint32_t i = 0; i < adc_dev->cfg->number_of_used_pins; i++) {
303  ADC_RegularChannelConfig(adc_dev->cfg->adc_dev_master, adc_dev->cfg->adc_pins[i].adc_channel, current_index + 1, ADC_SampleTime_61Cycles5); /* XXX this is totally arbitrary... */
304  adc_dev->channel_map[current_index] = &adc_dev->accumulator[current_index];
305  ++current_index;
306  }
307  }
308  else {
309  bool again = true;
310  current_index = 0;
311  uint32_t acc_index = 0;
312  while (again) {
313  for (uint32_t i = 0; i < adc_dev->cfg->number_of_used_pins; i++) {
314  if (adc_dev->cfg->adc_pins[i].is_master_channel) {
315  ADC_RegularChannelConfig(adc_dev->cfg->adc_dev_master, adc_dev->cfg->adc_pins[i].adc_channel, current_index + 1, ADC_SampleTime_61Cycles5); /* XXX this is totally arbitrary... */
316  adc_dev->channel_map[acc_index] = &adc_dev->accumulator[i];
317  ++current_index;
318  acc_index += 2;
319  if (current_index == (adc_dev->number_used_master_channels > adc_dev->number_used_slave_channels ?
320  adc_dev->number_used_master_channels : adc_dev->number_used_slave_channels)) {
321  again = false;
322  break;
323  }
324  }
325  }
326  }
327  again = true;
328  current_index = 0;
329  acc_index = 1;
330  while (again) {
331  for (uint32_t i = 0; i < adc_dev->cfg->number_of_used_pins; i++) {
332  if (!adc_dev->cfg->adc_pins[i].is_master_channel) {
333  ADC_RegularChannelConfig(adc_dev->cfg->adc_dev_slave, adc_dev->cfg->adc_pins[i].adc_channel, current_index + 1, ADC_SampleTime_61Cycles5); /* XXX this is totally arbitrary... */
334  adc_dev->channel_map[acc_index] = &adc_dev->accumulator[i];
335  ++current_index;
336  acc_index += 2;
337  if (current_index == (adc_dev->number_used_master_channels > adc_dev->number_used_slave_channels ?
338  adc_dev->number_used_master_channels : adc_dev->number_used_slave_channels)) {
339  again = false;
340  break;
341  }
342  }
343  }
344  }
345  }
346  ADC_Cmd(adc_dev->cfg->adc_dev_master, ENABLE);
347  if (adc_dev->cfg->adc_dev_slave)
348  ADC_Cmd(adc_dev->cfg->adc_dev_slave, ENABLE);
349  while (!ADC_GetFlagStatus(ADC1, ADC_FLAG_RDY));
350  ADC_StartConversion(adc_dev->cfg->adc_dev_master);
351 }

471 static void accumulate(struct pios_internal_adc_dev *adc_dev, uint16_t *buffer)
472 {
473  uint16_t *sp;
474  /*
475  * Accumulate sampled values.
476  */
477  uint32_t increment;
478  uint32_t scan_size;
479  if (adc_dev->cfg->adc_dev_slave) {
480  increment = 2;
481  scan_size = adc_dev->regular_group_size * 2;
482  }
483  else {
484  increment = 1;
485  scan_size = adc_dev->regular_group_size;
486  }
487  for (uint8_t i = 0; i < adc_dev->cfg->oversampling; ++i) {
488  sp = buffer + adc_dev->regular_group_size * i * increment;
489  for (uint8_t scan_index = 0; scan_index < scan_size; scan_index++) {
490  adc_dev->channel_map[scan_index]->accumulator += *sp;
491  adc_dev->channel_map[scan_index]->count++;
492  sp++;
493  /*
494  * If the accumulator reaches half-full, rescale in order to
495  * make more space.
496  */
497  if (adc_dev->channel_map[scan_index]->accumulator >= (1 << 31)) {
498  adc_dev->channel_map[scan_index]->accumulator /= 2;
499  adc_dev->channel_map[scan_index]->count /= 2;
500  }
501 
502  }
503  }
504 }

534 static void PIOS_ADC_DMA_Handler(struct pios_internal_adc_dev *adc_dev)
535 {
536  if(!PIOS_INTERNAL_ADC_validate(adc_dev))
537  PIOS_Assert(0);
538  /* terminal count, buffer has flipped */
539  if (DMA_GetFlagStatus(adc_dev->cfg->full_flag /*DMA1_IT_TC1*/)) { // whole double buffer filled
540  if (adc_dev->cfg->adc_dev_slave)
541  {
542  DMA_ClearFlag(adc_dev->cfg->full_flag);
543  accumulate(adc_dev, adc_dev->raw_data_buffer + adc_dev->dma_half_buffer_size);
544  }
545  else {
546  DMA_ClearFlag(adc_dev->cfg->full_flag);
547  accumulate(adc_dev, adc_dev->raw_data_buffer+adc_dev->cfg->oversampling * adc_dev->cfg->number_of_used_pins);
548  }
549  }
550  else if (DMA_GetFlagStatus(adc_dev->cfg->half_flag /*DMA1_IT_HT1*/)) {
551  DMA_ClearFlag(adc_dev->cfg->half_flag);
