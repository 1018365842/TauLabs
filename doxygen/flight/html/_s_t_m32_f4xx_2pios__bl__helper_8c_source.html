120 static bool PIOS_BL_HELPER_FLASH_GetSectorInfo(uint32_t address, uint8_t * sector_number, uint32_t *sector_start, uint32_t *sector_size)
121 {
122  for (uint8_t i = 0; i < NELEMENTS(flash_sectors); i++) {
123  struct device_flash_sector * sector = &flash_sectors[i];
124  if ((address >= sector->start) &&
125  (address < (sector->start + sector->size))) {
126  /* address lies within this sector */
127  *sector_number = sector->st_sector;
128  *sector_start = sector->start;
129  *sector_size = sector->size;
130  return (true);
131  }
132  }
133 
134  return (false);
135 }

137 uint8_t PIOS_BL_HELPER_FLASH_Start()
138 {
139  const struct pios_board_info * bdinfo = &pios_board_info_blob;
140  uint32_t pageAddress = bdinfo->fw_base;
141  bool fail = false;
142  while ((pageAddress < (bdinfo->fw_base + bdinfo->fw_size + bdinfo->desc_size))
143  && (fail == false)) {
144  uint8_t sector_number;
145  uint32_t sector_start;
146  uint32_t sector_size;
147  if (!PIOS_BL_HELPER_FLASH_GetSectorInfo(pageAddress,
148  &sector_number,
149  &sector_start,
150  &sector_size)) {
151  /* We're asking for an invalid flash address */
152  PIOS_Assert(0);
153  }
154  for (int retry = 0; retry < MAX_DEL_RETRYS; ++retry) {
155  if (FLASH_EraseSector(sector_number, VoltageRange_3) == FLASH_COMPLETE) {
156  fail = false;
157  break;
158  } else {
159  fail = true;
160  }
161 
162  }
163  /* Move to the next sector */
164  pageAddress += sector_size;
165  }
166 
167  return (fail == true) ? 0 : 1;
168 }
