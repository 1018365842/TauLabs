97 float sin_lookup_deg(float angle)
98 {
99  if (sin_table == NULL)
100  return 0;
101 
102  int i_ang = ((int32_t) angle) % 360;
103  if (i_ang >= 180) // for 180 to 360 deg
104  return -sin_table[i_ang - 180];
105  else // for 0 to 179 deg
106  return sin_table[i_ang];
107 
108  return 0;
109 }
