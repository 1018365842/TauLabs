28 // forward substitution, row-major
29 template<typename Lhs, typename Rhs>
30 struct ei_solve_triangular_selector<Lhs,Rhs,LowerTriangular,RowMajor|IsSparse>
31 {
32  typedef typename Rhs::Scalar Scalar;
33  static void run(const Lhs& lhs, Rhs& other)
34  {
35  for(int col=0 ; col<other.cols() ; ++col)
36  {
37  for(int i=0; i<lhs.rows(); ++i)
38  {
39  Scalar tmp = other.coeff(i,col);
40  Scalar lastVal = 0;
41  int lastIndex = 0;
42  for(typename Lhs::InnerIterator it(lhs, i); it; ++it)
43  {
44  lastVal = it.value();
45  lastIndex = it.index();
46  if(lastIndex == i)
47  break;
48  tmp -= lastVal * other.coeff(lastIndex,col);
49  }
50 
51  if (Lhs::Flags & UnitDiagBit)
52  other.coeffRef(i,col) = tmp;
53  else
54  {
55  ei_assert(lastIndex==i);
56  other.coeffRef(i,col) = tmp/lastVal;
57  }
58  }
59  }
60  }
61 };

63 // backward substitution, row-major
64 template<typename Lhs, typename Rhs>
65 struct ei_solve_triangular_selector<Lhs,Rhs,UpperTriangular,RowMajor|IsSparse>
66 {
67  typedef typename Rhs::Scalar Scalar;
68  static void run(const Lhs& lhs, Rhs& other)
69  {
70  for(int col=0 ; col<other.cols() ; ++col)
71  {
72  for(int i=lhs.rows()-1 ; i>=0 ; --i)
73  {
74  Scalar tmp = other.coeff(i,col);
75  typename Lhs::InnerIterator it(lhs, i);
76  if (it.index() == i)
77  ++it;
78  for(; it; ++it)
79  {
80  tmp -= it.value() * other.coeff(it.index(),col);
81  }
82 
83  if (Lhs::Flags & UnitDiagBit)
84  other.coeffRef(i,col) = tmp;
85  else
86  {
87  typename Lhs::InnerIterator it(lhs, i);
88  ei_assert(it.index() == i);
89  other.coeffRef(i,col) = tmp/it.value();
90  }
91  }
92  }
93  }
94 };

96 // forward substitution, col-major
97 template<typename Lhs, typename Rhs>
98 struct ei_solve_triangular_selector<Lhs,Rhs,LowerTriangular,ColMajor|IsSparse>
99 {
100  typedef typename Rhs::Scalar Scalar;
101  static void run(const Lhs& lhs, Rhs& other)
102  {
103  for(int col=0 ; col<other.cols() ; ++col)
104  {
105  for(int i=0; i<lhs.cols(); ++i)
106  {
107  typename Lhs::InnerIterator it(lhs, i);
108  if(!(Lhs::Flags & UnitDiagBit))
109  {
110  // std::cerr << it.value() << " ; " << it.index() << " == " << i << "\n";
111  ei_assert(it.index()==i);
112  other.coeffRef(i,col) /= it.value();
113  }
114  Scalar tmp = other.coeffRef(i,col);
115  if (it.index()==i)
116  ++it;
117  for(; it; ++it)
118  other.coeffRef(it.index(), col) -= tmp * it.value();
119  }
120  }
121  }
122 };

124 // backward substitution, col-major
125 template<typename Lhs, typename Rhs>
126 struct ei_solve_triangular_selector<Lhs,Rhs,UpperTriangular,ColMajor|IsSparse>
127 {
128  typedef typename Rhs::Scalar Scalar;
129  static void run(const Lhs& lhs, Rhs& other)
130  {
131  for(int col=0 ; col<other.cols() ; ++col)
132  {
133  for(int i=lhs.cols()-1; i>=0; --i)
134  {
135  if(!(Lhs::Flags & UnitDiagBit))
136  {
137  // FIXME lhs.coeff(i,i) might not be always efficient while it must simply be the
138  // last element of the column !
139  other.coeffRef(i,col) /= lhs.coeff(i,i);
140  }
141  Scalar tmp = other.coeffRef(i,col);
142  typename Lhs::InnerIterator it(lhs, i);
143  for(; it && it.index()<i; ++it)
144  other.coeffRef(it.index(), col) -= tmp * it.value();
145  }
146  }
147  }
148 };

150 template<typename Derived>
151 template<typename OtherDerived>
152 void SparseMatrixBase<Derived>::solveTriangularInPlace(MatrixBase<OtherDerived>& other) const
153 {
154  ei_assert(derived().cols() == derived().rows());
155  ei_assert(derived().cols() == other.rows());
156  ei_assert(!(Flags & ZeroDiagBit));
157  ei_assert(Flags & (UpperTriangularBit|LowerTriangularBit));
158 
159  enum { copy = ei_traits<OtherDerived>::Flags & RowMajorBit };
160 
161  typedef typename ei_meta_if<copy,
162  typename ei_plain_matrix_type_column_major<OtherDerived>::type, OtherDerived&>::ret OtherCopy;
163  OtherCopy otherCopy(other.derived());
164 
165  ei_solve_triangular_selector<Derived, typename ei_unref<OtherCopy>::type>::run(derived(), otherCopy);
166 
167  if (copy)
168  other = otherCopy;
169 }

171 template<typename Derived>
172 template<typename OtherDerived>
173 typename ei_plain_matrix_type_column_major<OtherDerived>::type
174 SparseMatrixBase<Derived>::solveTriangular(const MatrixBase<OtherDerived>& other) const
