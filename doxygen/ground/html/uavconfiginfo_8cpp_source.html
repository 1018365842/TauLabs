177 bool UAVConfigInfo::askToAbort(int compat, QString message)
178 {
179  QMessageBox msgBox;
180  msgBox.setInformativeText(tr("Do you want to continue the import?"));
181  msgBox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
182 
183  int result;
184 
185  switch(compat){
186 
187  case FullyCompatible:
188  return false;
189 
190  case MinorLossOfConfiguration:
191  msgBox.setText(tr("INFO: ") + message + TEXT_MINOR_LOSS_OF_CONFIGURATION);
192  msgBox.setDefaultButton(QMessageBox::Ok);
193  result = msgBox.exec();
194  break;
195 
196  case MissingConfiguration:
197  msgBox.setText(tr("WARNING: ") + message + TEXT_MISSING_CONFIGURATION);
198  msgBox.setDefaultButton(QMessageBox::Ok);
199  result = msgBox.exec();
200  break;
201 
202  case MajorLossOfConfiguration:
203  msgBox.setText(tr("ERROR: ") + message + TEXT_MAJOR_LOSS_OF_CONFIGURATION);
204  msgBox.setDefaultButton(QMessageBox::Cancel);
205  result = msgBox.exec();
206  break;
207 
208  case NotCompatible:
209  msgBox.setText("ERROR: " + message + TEXT_NOT_COMPATIBLE);
210  msgBox.setInformativeText(tr(""));
211  msgBox.setStandardButtons(QMessageBox::Ok);
212  msgBox.exec();
213  return true;
214 
215  default:
216  msgBox.setText("INTERNAL ERROR: " + message + tr("Unknown compatibility level: " + compat));
217  }
218  if ( result == QMessageBox::Ok )
219  return false;
220  else
221  return true;
222 
223 }

232 int UAVConfigInfo::checkCompatibilityWith(UAVConfigVersion programVersion)
233 {
234  if ( m_version.majorNr != programVersion.majorNr )
235  return NotCompatible;
236  if ( m_version.minorNr < programVersion.minorNr )
237  return MissingConfiguration;
238  if ( m_version.minorNr > programVersion.minorNr )
239  return MajorLossOfConfiguration;
240  if ( m_version.patchNr > programVersion.patchNr )
241  return MinorLossOfConfiguration;
242 
243  return FullyCompatible;
244 }

260 UAVConfigVersion::UAVConfigVersion(QString versionString)
261 {
262  int begin;
263  int end = 0;
264 
265  begin = end;
266  end = versionString.indexOf(".", begin);
267  majorNr = versionString.mid(begin, end-begin).toInt();
268 
269  begin = end+1;
270  end = versionString.indexOf(".", begin);
271  minorNr = versionString.mid(begin, end-begin).toInt();
272 
273  begin = end+1;
274  patchNr = versionString.mid(begin).toInt();
275 }
