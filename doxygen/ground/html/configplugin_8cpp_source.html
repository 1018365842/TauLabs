45 bool ConfigPlugin::initialize(const QStringList& args, QString *errMsg)
46 {
47  Q_UNUSED(args);
48  Q_UNUSED(errMsg);
49  cf = new ConfigGadgetFactory(this);
50  addAutoReleasedObject(cf);
51 
52  // Add Menu entry to erase all settings
53  Core::ActionManager* am = Core::ICore::instance()->actionManager();
54  Core::ActionContainer* ac = am->actionContainer(Core::Constants::M_TOOLS);
55 
56  // Command to erase all settings from the board
57  cmd = am->registerAction(new QAction(this),
58  "ConfigPlugin.EraseAll",
59  QList<int>() <<
60  Core::Constants::C_GLOBAL_ID);
61  cmd->action()->setText(tr("Erase all settings from board..."));
62 
63  ac->menu()->addSeparator();
64  ac->appendGroup("Utilities");
65  ac->addAction(cmd, "Utilities");
66 
67  connect(cmd->action(), SIGNAL(triggered(bool)), this, SLOT(eraseAllSettings()));
68 
69  // *********************
70  // Listen to autopilot connection events
71  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
72  TelemetryManager* telMngr = pm->getObject<TelemetryManager>();
73  connect(telMngr, SIGNAL(connected()), this, SLOT(onAutopilotConnect()));
74  connect(telMngr, SIGNAL(disconnected()), this, SLOT(onAutopilotDisconnect()));
75 
76  // And check whether by any chance we are not already connected
77  if (telMngr->isConnected())
78  onAutopilotConnect();
79 
80  return true;
81 }

125 void ConfigPlugin::eraseAllSettings()
126 {
127  QMessageBox msgBox;
128  msgBox.setText(tr("Are you sure you want to erase all board settings?."));
129  msgBox.setInformativeText(tr("All settings stored in your board flash will be deleted."));
130  msgBox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
131  msgBox.setDefaultButton(QMessageBox::Ok);
132  if (msgBox.exec() != QMessageBox::Ok)
133  return;
134 
135  settingsErased = false;
136 
137  //TODO: Replace the second and third [in eraseDone()] pop-up dialogs with a progress indicator,
138  // counter, or infinite chain of `......` tied to the original dialog box
139  msgBox.setText(tr("Settings will now erase."));
140  msgBox.setInformativeText(tr("Press <OK> and then please wait until a completion box appears. This can take up to %1 seconds.").arg(FLASH_ERASE_TIMEOUT_MS/1000));
141  msgBox.setStandardButtons(QMessageBox::Ok);
142  msgBox.exec();
143 
144 
145  ObjectPersistence* objper = ObjectPersistence::GetInstance(getObjectManager());
146  Q_ASSERT(objper);
147 
148  connect(objper, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(eraseDone(UAVObject *)));
149 
150  ObjectPersistence::DataFields data = objper->getData();

152 
153  // No need for manual updated event, this is triggered by setData

161 void ConfigPlugin::eraseFailed()
162 {
163  if (settingsErased)
164  return;
165 
166  ObjectPersistence* objper = ObjectPersistence::GetInstance(getObjectManager());
167 
168  ObjectPersistence::DataFields data = objper->getData();
169  if(data.Operation == ObjectPersistence::OPERATION_FULLERASE) {
170  // First attempt via flash erase failed. Fall back on erase all settings
171  data.Operation = ObjectPersistence::OPERATION_DELETE;
172  data.Selection = ObjectPersistence::SELECTION_ALLSETTINGS;
173  objper->setData(data);
174  objper->updated();
175  QTimer::singleShot(FLASH_ERASE_TIMEOUT_MS,this,SLOT(eraseFailed()));
176  } else {
177  disconnect(objper, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(eraseDone(UAVObject *)));
178  QMessageBox msgBox;
179  msgBox.setText(tr("Error trying to erase settings."));
180  msgBox.setInformativeText(tr("Power-cycle your board after removing all blades. Settings might be inconsistent."));
181  msgBox.setStandardButtons(QMessageBox::Ok);
182  msgBox.setDefaultButton(QMessageBox::Ok);
183  msgBox.exec();
184  }
185 }

187 void ConfigPlugin::eraseDone(UAVObject * obj)
188 {
189  QMessageBox msgBox;
190  ObjectPersistence* objper = ObjectPersistence::GetInstance(getObjectManager());
191  ObjectPersistence::DataFields data = objper->getData();
192  Q_ASSERT(obj->getInstID() == objper->getInstID());
193 
194  if(data.Operation != ObjectPersistence::OPERATION_COMPLETED) {
195  return;
196  }
197 
198  disconnect(objper, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(eraseDone(UAVObject *)));
199  if (data.Operation == ObjectPersistence::OPERATION_COMPLETED) {
200  settingsErased = true;
201  msgBox.setText(tr("Settings are now erased."));
202  msgBox.setInformativeText(tr("Please ensure that the status LED is flashing regularly and then power-cycle your board."));
203  } else {
204  msgBox.setText(tr("Error trying to erase settings."));
205  msgBox.setInformativeText(tr("Power-cycle your board after removing all blades. Settings might be inconsistent."));
206  }
207  msgBox.setStandardButtons(QMessageBox::Ok);
208  msgBox.setDefaultButton(QMessageBox::Ok);
209  msgBox.exec();
210 }
