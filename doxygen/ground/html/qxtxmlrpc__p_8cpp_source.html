44 QString QxtXmlRpc::serialize(QVariant data)
45 {
46  if (data.isNull())
47  {
48  return "<nil/>";
49  }
50  int t = data.type();
51  if (t == QVariant::String)
52  {
53  return "<string>" + xmlEncode(data.toString()) + "</string>";
54  }
55  else if (t == QVariant::Bool)
56  {
57  return "<boolean>" + (data.toBool() ? QString("1") : QString("0")) + "</boolean>";
58  }
59  else if (t == QVariant::Int)
60  {
61  return "<int>" + QString::number(data.toInt()) + "</int>";
62  }
63  else if (t == QVariant::Double)
64  {
65  return "<double>" + QString::number(data.toDouble()) + "</double>";
66  }
67  else if (t == QVariant::DateTime)
68  {
69  return "<dateTime.iso8601>" + data.toDateTime().toString(Qt::ISODate) + "</dateTime.iso8601>";
70  }
71  else if (t == QVariant::ByteArray)
72  {
73  return "<base64>" + data.toByteArray().toBase64() + "</base64>";
74  }
75  else if (t == QVariant::Map)
76  {
77  QString ret = "<struct>";
78  QMap<QString, QVariant> map = data.toMap();
79  QMapIterator<QString, QVariant> i(map);
80  while (i.hasNext())
81  {
82  i.next();
83  ret += "<member><name>" + i.key() + "</name><value>" + serialize(i.value()) + "</value></member>";
84  }
85  ret += "</struct>";
86  return ret;
87  }
88 #if QT_VERSION >= 0x040500
89  else if (t == QVariant::Hash)
90  {
91  QString ret = "<struct>";
92  QHash<QString, QVariant> map = data.toHash();
93  QHashIterator<QString, QVariant> i(map);
94  while (i.hasNext())
95  {
96  i.next();
97  ret += "<member><name>" + i.key() + "</name><value>" + serialize(i.value()) + "</value></member>";
98  }
99  ret += "</struct>";
100  return ret;
101  }
102 #endif
103  else if (t == QVariant::StringList)
104  {
105  QString ret = "<array><data>";
106  QStringList l = data.toStringList();
107  foreach(QString i, l)
108  {
109  ret += "<value>" + xmlEncode(i) + "</value>";
110  }
111  ret += "</data></array>";
112  return ret;
113  }
114  else if (t == QVariant::List)
115  {
116  QString ret = "<array><data>";
117  QVariantList l = data.toList();
118  foreach(QVariant i, l)
119  {
120  ret += "<value>" + serialize(i) + "</value>";
121  }
122  ret += "</data></array>";
123  return ret;
124  }
125  else
126  {
127  return "";
128  }
129 }

137 QVariant deserializeArray(QXmlStreamReader & xml)
138 {
139  QVariantList l;
140  int s = 0;
141  while (!xml.atEnd())
142  {
143  xml.readNext();
144  if (xml.isStartElement())
145  {
146  if (s == 0)
147  {
148  if (xml.name().toString() == "data")
149  {
150  s = 1;
151  }
152  else
153  {
154  xml.raiseError("expected <data>. got : <" + xml.name().toString() + ">");
155  }
156  }
157  else if (s == 1)
158  {
159  if (xml.name().toString() == "value")
160  {
161  l += deserialize(xml);
162  s = 2;
163  if (xml.isEndElement() && xml.name().toString() == "value"){
164  s=1;
165  }
166  }
167  else
168  {
169  xml.raiseError("expected <value>. got : <" + xml.name().toString() + ">");
170  }
171  }
172  else if (s == 2)
173  {
174  xml.raiseError("expected </value>. got : <" + xml.name().toString() + ">");
175  }
176  }
177  else if (xml.isEndElement())
178  {
179  if (s == 2)
180  {
181  if (xml.name().toString() == "value")
182  {
183  s = 1;
184  }
185  else
186  {
187  xml.raiseError("expected </value>. got : </" + xml.name().toString() + ">");
188  }
189  }
190  else if (s == 1)
191  {
192  if (xml.name().toString() == "data")
193  {
194  s = -1;
195  }
196  else
197  {
198  xml.raiseError("expected </data>. got : </" + xml.name().toString() + ">");
199  }
200  }
201  else if (s == -1)
202  {
203  if (xml.name().toString() == "array")
204  {
205  return l;
206  }
207  else
208  {
209  xml.raiseError("expected </array>. got : </" + xml.name().toString() + ">");
210  }
211  }
212  else if (s == 0)
213  {
214  xml.raiseError("expected <data>. got : </" + xml.name().toString() + ">");
215  }
216  }
217  }
218  return QVariant();
219 }

221 QVariant deserializeStruct(QXmlStreamReader & xml)
222 {
223 
224  QVariantMap l;
225  QString key;
226  QVariant value;
227  int s = 0;
228 
229  while (!xml.atEnd())
230  {
231  xml.readNext();
232  if (xml.isStartElement())
233  {
234  if (s == 0)
235  {
236  if (xml.name().toString() == "member")
237  {
238  s = 1;
239  }
240  else
241  {
242  xml.raiseError("expected <member>. got : <" + xml.name().toString() + ">");
243  }
244  }
245  else if (s == 1)
246  {
247  if (xml.name().toString() == "name")
248  {
249  key = xml.readElementText();
250  }
251  else if (xml.name().toString() == "value")
252  {
253  value = deserialize(xml);
254  s = 2;
255  if (xml.isEndElement() && xml.name().toString() == "value"){
256  s=1;
257  }
258  }
259  else
260  {
261  xml.raiseError("expected <name> or <value>. got : <" + xml.name().toString() + ">");
262  }
263  }
264  else if (s == 2)
265  {
266  xml.raiseError("expected </name>. got : <" + xml.name().toString() + ">");
267  }
268  else if (s == 3)
269  {
270  xml.raiseError("expected </value>. got : <" + xml.name().toString() + ">");
271  }
272 
273  }
274  else if (xml.isEndElement())
275  {
276  if (s == 2)
277  {
278  if (xml.name().toString() == "value")
279  {
280  s = 1;
281  }
282  else
283  {
284  xml.raiseError("expected </value>. got : </" + xml.name().toString() + ">");
285  }
286  }
287  else if (s == 1)
288  {
289  if (xml.name().toString() == "member")
290  {
291  l[key] = value;
292  s = 0;
293  }
294  else
295  {
296  xml.raiseError("expected </member>. got : </" + xml.name().toString() + ">");
297  }
298  }
299  else if (s == 0)
300  {
301  if (xml.name().toString() == "struct")
302  {
303  return l;
304  }
305  else
306  {
307  xml.raiseError("expected </struct>. got : </" + xml.name().toString() + ">");
308  }
309  }
310  }
311  }
312  return QVariant();
313 }

315 QVariant QxtXmlRpc::deserialize(QXmlStreamReader & xml)
316 {
317  QVariant ret;
318  while (!xml.atEnd())
319  {
320  xml.readNext();
321  if (xml.isStartElement())
322  {
323  if (xml.name().toString() == "array")
324  {
325  return deserializeArray(xml);
326  }
327  else if (xml.name().toString() == "base64")
328  {
329  return QByteArray::fromBase64(xml.readElementText().toAscii());
330  }
331  else if (xml.name().toString() == "boolean")
332  {
333  return (xml.readElementText().toInt() == 1);
334  }
335  else if (xml.name().toString() == "dateTime.iso8601")
336  {
337  return QDateTime::fromString(xml.readElementText(), Qt::ISODate);
338  }
339  else if (xml.name().toString() == "double")
340  {
341  return xml.readElementText().toDouble();
342  }
343  else if (xml.name().toString() == "integer" || xml.name().toString() == "i4")
344  {
345  return xml.readElementText().toInt();
346  }
347  else if (xml.name().toString() == "string")
348  {
349  return xml.readElementText();
350  }
351  else if (xml.name().toString() == "struct")
352  {
353  return deserializeStruct(xml);
354  }
355  else
356  {
357  ret = QVariant();
358  }
359  }
360  }
361  return QVariant();
362 }
