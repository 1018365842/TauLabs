110 bool UAVObjectGeneratorWireshark::process_object(ObjectInfo* info, QDir outputpath)
111 {
112  if (info == NULL)
113  return false;
114 
115  // Prepare output strings
116  QString outCode = wiresharkCodeTemplate;
117 
118  // Replace common tags
119  replaceCommonTags(outCode, info);
120 
121  // Replace the $(SUBTREES) and $(SUBTREESTATICS) tags
122  QString subtrees;
123  QString subtreestatics;
124  subtreestatics.append( QString("static gint ett_uavo = -1;\r\n") );
125  subtrees.append( QString("&ett_uavo,\r\n") );
126  for (int n = 0; n < info->fields.length(); ++n) {
127  if (info->fields[n]->numElements > 1) {
128  /* Reserve a subtree for each array */
129  subtreestatics.append( QString("static gint ett_%1_%2 = -1;\r\n")
130  .arg(info->namelc)
131  .arg(info->fields[n]->name) );
132  subtrees.append( QString("&ett_%1_%2,\r\n")
133  .arg(info->namelc)
134  .arg(info->fields[n]->name) );
135  }
136  }
137  outCode.replace(QString("$(SUBTREES)"), subtrees);
138  outCode.replace(QString("$(SUBTREESTATICS)"), subtreestatics);
139 
140  // Replace the $(FIELDHANDLES) tag
141  QString type;
142  QString fields;
143  for (int n = 0; n < info->fields.length(); ++n) {
144  fields.append( QString("static int hf_op_uavobjects_%1_%2 = -1;\r\n")
145  .arg(info->namelc)
146  .arg(info->fields[n]->name));
147  if (info->fields[n]->numElements > 1) {
148  QStringList elemNames = info->fields[n]->elementNames;
149  for (int m = 0; m < elemNames.length(); ++m) {
150  fields.append( QString("static int hf_op_uavobjects_%1_%2_%3 = -1;\r\n")
151  .arg(info->namelc)
152  .arg(info->fields[n]->name)
153  .arg(elemNames[m]) );
154  }
155  }
156  }
157  outCode.replace(QString("$(FIELDHANDLES)"), fields);
158 
159  // Replace the $(ENUMFIELDNAMES) tag
160  QString enums;
161  for (int n = 0; n < info->fields.length(); ++n) {
162  // Only for enum types
163  if (info->fields[n]->type == FIELDTYPE_ENUM) {
164  enums.append(QString("/* Field %1 information */\r\n").arg(info->fields[n]->name) );
165  enums.append(QString("/* Enumeration options for field %1 */\r\n").arg(info->fields[n]->name));
166  enums.append( QString("static const value_string uavobjects_%1_%2[]= {\r\n")
167  .arg(info->namelc)
168  .arg(info->fields[n]->name) );
169  // Go through each option
170  QStringList options = info->fields[n]->options;
171  for (int m = 0; m < options.length(); ++m) {
172  enums.append ( QString("\t{ %1, \"%2\" },\r\n")
173  .arg(m)
174  .arg(options[m].replace(QRegExp(ENUM_SPECIAL_CHARS), "") ) );
175  }
176  enums.append( QString("\t{ 0, NULL }\r\n") );
177  enums.append( QString("};\r\n") );
178  }
179  }
180  outCode.replace(QString("$(ENUMFIELDNAMES)"), enums);
181 
182  // Replace the $(POPULATETREE) tag
183  QString treefields;
184  for (int n = 0; n < info->fields.length(); ++n) {
185  if ( info->fields[n]->numElements == 1 ) {
186  treefields.append( QString(" ptvcursor_add(cursor, hf_op_uavobjects_%1_%2, sizeof(%3), ENC_LITTLE_ENDIAN);\r\n")
187  .arg(info->namelc)
188  .arg(info->fields[n]->name)
189  .arg(fieldTypeStrGlib[info->fields[n]->type]) );
190  } else {
191  treefields.append( QString(" {\r\n") );
192  treefields.append( QString(" proto_item * it = NULL;\r\n") );
193  treefields.append( QString(" it = ptvcursor_add_no_advance(cursor, hf_op_uavobjects_%1_%2, 1, ENC_NA);\r\n")
194  .arg(info->namelc)
195  .arg(info->fields[n]->name) );
196  treefields.append( QString(" ptvcursor_push_subtree(cursor, it, ett_%1_%2);\r\n")
197  .arg(info->namelc)
198  .arg(info->fields[n]->name) );
199  /* Populate each array element into the table */
200  QStringList elemNames = info->fields[n]->elementNames;
201  for (int m = 0; m < elemNames.length(); ++m) {
202  treefields.append( QString(" ptvcursor_add(cursor, hf_op_uavobjects_%1_%2_%3, sizeof(%4), ENC_LITTLE_ENDIAN);\r\n")
203  .arg(info->namelc)
204  .arg(info->fields[n]->name)
205  .arg(elemNames[m])
206  .arg(fieldTypeStrGlib[info->fields[n]->type]) );
207  }
208  treefields.append( QString(" ptvcursor_pop_subtree(cursor);\r\n") );
209  treefields.append( QString(" }\r\n") );
210  }
211  }
212  outCode.replace(QString("$(POPULATETREE)"), treefields);
213 
214  // Replace the $(HEADERFIELDS) tag
215  QString headerfields;
216  headerfields.append( QString(" static hf_register_info hf[] = {\r\n") );
217  for (int n = 0; n < info->fields.length(); ++n) {
218  // For non-array fields
219  if ( info->fields[n]->numElements == 1) {
220  headerfields.append( QString("\t { &hf_op_uavobjects_%1_%2,\r\n")
221  .arg( info->namelc )
222  .arg( info->fields[n]->name ) );
223  headerfields.append( QString("\t { \"%1\", \"%2.%3\", %4,\r\n")
224  .arg( info->fields[n]->name )
225  .arg( info->namelc )
226  .arg( info->fields[n]->name )
227  .arg( fieldTypeStrHf[info->fields[n]->type] ) );
228  if ( info->fields[n]->type == FIELDTYPE_ENUM ) {
229  headerfields.append( QString("\t BASE_DEC, VALS(uavobjects_%1_%2), 0x0, NULL, HFILL \r\n")
230  .arg( info->namelc )
231  .arg( info->fields[n]->name ) );
232  } else if ( info->fields[n]->type == FIELDTYPE_FLOAT32 ) {
233  headerfields.append( QString("\t BASE_NONE, NULL, 0x0, NULL, HFILL \r\n") );
234  } else {
235  headerfields.append( QString("\t BASE_DEC_HEX, NULL, 0x0, NULL, HFILL\r\n") );
236  }
237  headerfields.append( QString("\t },\r\n") );
238  headerfields.append( QString("\t },\r\n") );
239  } else {
240  headerfields.append( QString("\t { &hf_op_uavobjects_%1_%2,\r\n")
241  .arg( info->namelc )
242  .arg( info->fields[n]->name ) );
243  headerfields.append( QString("\t { \"%1\", \"%2.%3\", FT_NONE,\r\n")
244  .arg( info->fields[n]->name )
245  .arg( info->namelc )
246  .arg( info->fields[n]->name ) );
247  headerfields.append( QString("\t BASE_NONE, NULL, 0x0, NULL, HFILL\r\n") );
248  headerfields.append( QString("\t },\r\n") );
249  headerfields.append( QString("\t },\r\n") );
250 
251  QStringList elemNames = info->fields[n]->elementNames;
252  for (int m = 0; m < elemNames.length(); ++m) {
253  headerfields.append( QString("\t { &hf_op_uavobjects_%1_%2_%3,\r\n")
254  .arg( info->namelc )
255  .arg( info->fields[n]->name )
256  .arg( elemNames[m]) );
257  headerfields.append( QString("\t { \"%1\", \"%2.%3.%4\", %5,\r\n")
258  .arg( elemNames[m] )
259  .arg( info->namelc )
260  .arg( info->fields[n]->name )
261  .arg( elemNames[m] )
262  .arg( fieldTypeStrHf[info->fields[n]->type] ) );
263  if ( info->fields[n]->type == FIELDTYPE_ENUM ) {
264  headerfields.append( QString("\t BASE_DEC, VALS(uavobjects_%1_%2), 0x0, NULL, HFILL \r\n")
265  .arg( info->namelc )
266  .arg( info->fields[n]->name ) );
267  } else if ( info->fields[n]->type == FIELDTYPE_FLOAT32 ) {
268  headerfields.append( QString("\t BASE_NONE, NULL, 0x0, NULL, HFILL \r\n") );
269  } else {
270  headerfields.append( QString("\t BASE_DEC_HEX, NULL, 0x0, NULL, HFILL\r\n") );
271  }
272  headerfields.append( QString("\t },\r\n") );
273  headerfields.append( QString("\t },\r\n") );
274  }
275  }
276  }
277  headerfields.append( QString(" };\r\n") );
278  outCode.replace(QString("$(HEADERFIELDS)"), headerfields);
279 
280  // Write the flight code
281  bool res = writeFileIfDiffrent( outputpath.absolutePath() + "/packet-op-uavobjects-" + info->namelc + ".c", outCode );
282  if (!res) {
283  cout << "Error: Could not write wireshark code files" << endl;
284  return false;
285  }
286 
287  return true;
288 }
