278 QAction *ActionContainerPrivate::beforeAction(int pos, int *prevKey) const
279 {
280  ActionManagerPrivate *am = ActionManagerPrivate::instance();
281 
282  int baId = -1;
283 
284  (*prevKey) = -1;
285 
286  QMap<int, int>::const_iterator i = m_posmap.constBegin();
287  while (i != m_posmap.constEnd()) {
288  if (i.key() > pos) {
289  baId = i.value();
290  break;
291  }
292  (*prevKey) = i.key();
293  ++i;
294  }
295 
296  if (baId == -1)
297  return 0;
298 
299  if (Command *cmd = am->command(baId))
300  return cmd->action();
301  if (ActionContainer *container = am->actionContainer(baId))
302  if (QMenu *menu = container->menu())
303  return menu->menuAction();
304 
305  return 0;
306 }

308 int ActionContainerPrivate::calcPosition(int pos, int prevKey) const
309 {
310  int grp = (pos & 0xFFFF0000);
311  if (prevKey == -1)
312  return grp;
313 
314  int prevgrp = (prevKey & 0xFFFF0000);
315 
316  if (grp != prevgrp)
317  return grp;
318 
319  return grp + (prevKey & 0xFFFF) + 10;
320 }

370 bool MenuActionContainer::update()
371 {
372  if (hasEmptyAction(EA_None))
373  return true;
374 
375  bool hasitems = false;
376 
377  foreach (ActionContainer *container, subContainers()) {
378  if (container == this) {
379  qWarning() << Q_FUNC_INFO << "container" << (this->menu() ? this->menu()->title() : "") << "contains itself as subcontainer";
380  continue;
381  }
382  if (container->update()) {
383  hasitems = true;
384  break;
385  }
386  }
387  if (!hasitems) {
388  foreach (Command *command, commands()) {
389  if (command->isActive()) {
390  hasitems = true;
391  break;
392  }
393  }
394  }
395 
396  if (hasEmptyAction(EA_Hide))
397  m_menu->setVisible(hasitems);
398  else if (hasEmptyAction(EA_Disable))
399  m_menu->setEnabled(hasitems);
400 
401  return hasitems;
402 }
