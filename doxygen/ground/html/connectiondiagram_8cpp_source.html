154 void ConnectionDiagram::setupGraphicsSceneItems(QList<QString> elementsToShow)
155 {
156  qreal z = 0;
157  QRectF backgBounds = m_renderer->boundsOnElement("background");
158 
159  foreach(QString elementId, elementsToShow) {
160  if(m_renderer->elementExists(elementId)) {
161  QGraphicsSvgItem* element = new QGraphicsSvgItem();
162  element->setSharedRenderer(m_renderer);
163  element->setElementId(elementId);
164  element->setZValue(z++);
165  element->setOpacity(1.0);
166 
167  QMatrix matrix = m_renderer->matrixForElement(elementId);
168  QRectF orig = matrix.mapRect(m_renderer->boundsOnElement(elementId));
169  element->setPos(orig.x(),orig.y());
170 
171  //QRectF orig = m_renderer->boundsOnElement(elementId);
172  //element->setPos(orig.x() - backgBounds.x(), orig.y() - backgBounds.y());
173 
174  m_scene->addItem(element);
175  qDebug() << "Adding " << elementId << " to scene at " << element->pos();
176  }
177  else {
178  qDebug() << "Element with id: " << elementId << " not found.";
179  }
180  }
181 }
