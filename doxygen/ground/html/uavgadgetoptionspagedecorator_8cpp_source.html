51 QWidget *UAVGadgetOptionsPageDecorator::createPage(QWidget *parent)
52 {
53  m_page = new Ui_TopOptionsPage();
54  QWidget *w = new QWidget(parent);
55  m_page->setupUi(w);
56  if (m_config->locked()) {
57  m_page->deleteButton->hide();
58  m_page->lockCheckBox->hide();
59  m_page->nameLineEdit->setDisabled(true);
60  }
61  if (!m_instanceManager->canDeleteConfiguration(m_config))
62  m_page->deleteButton->setDisabled(true);
63  m_page->lockCheckBox->hide(); //
64  m_page->nameLineEdit->setText(m_id);
65 
66  QWidget *wi = m_optionsPage->createPage(w);
67  wi->setSizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::MinimumExpanding);
68  m_page->verticalLayout_4->addWidget(wi);
69 
70  // For some gadgets it might not make sense to have multiple configurations
71  if (m_isSingleConfigurationGadget) {
72  m_page->configurationBox->hide();
73  }
74 
75  connect(m_page->cloneButton, SIGNAL(clicked()), this, SLOT(cloneConfiguration()));
76  connect(m_page->deleteButton, SIGNAL(clicked()), this, SLOT(deleteConfiguration()));
77  connect(m_page->nameLineEdit, SIGNAL(textEdited(QString)), this, SLOT(textEdited(QString)));
