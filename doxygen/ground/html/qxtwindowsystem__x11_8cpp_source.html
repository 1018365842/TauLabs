119 QRect QxtWindowSystem::windowGeometry(WId window)
120 {
121  QRect rect = qxt_getWindowRect(window);
122 
123  Window root = 0;
124  Window parent = 0;
125  Window* children = 0;
126  unsigned int count = 0;
127  Display* display = QX11Info::display();
128  if (XQueryTree(display, window, &root, &parent, &children, &count))
129  {
130  window = parent; // exclude decoration
131  XFree(children);
132  while (window != 0 && XQueryTree(display, window, &root, &parent, &children, &count))
133  {
134  XWindowAttributes attr;
135  if (parent != 0 && XGetWindowAttributes(display, parent, &attr))
136  rect.translate(QRect(attr.x, attr.y, attr.width, attr.height).topLeft());
137  window = parent;
138  XFree(children);
139  }
140  }
141  return rect;
142 }

159 uint QxtWindowSystem::idleTime()
160 {
161  static bool xssResolved = false;
162  if (!xssResolved) {
163  QLibrary xssLib(QLatin1String("Xss"), 1);
164  if (xssLib.load()) {
165  _xScreenSaverAllocInfo = (XScreenSaverAllocInfo) xssLib.resolve("XScreenSaverAllocInfo");
166  _xScreenSaverQueryInfo = (XScreenSaverQueryInfo) xssLib.resolve("XScreenSaverQueryInfo");
167  xssResolved = true;
168  }
169  }
170 
171  uint idle = 0;
172  if (xssResolved)
173  {
174  XScreenSaverInfo* info = _xScreenSaverAllocInfo();
175  const int screen = QX11Info::appScreen();
176  Qt::HANDLE rootWindow = QX11Info::appRootWindow(screen);
177  _xScreenSaverQueryInfo(QX11Info::display(), (Drawable*) rootWindow, info);
178  idle = info->idle;
179  XFree(info);
180  }
181  return idle;
182 }
