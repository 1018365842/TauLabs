61 void UdpSender::sendDatagram(const simToPlugin *stp)
62 {
63  QByteArray data;
64  data.resize(188);
65  QDataStream out(&data, QIODevice::WriteOnly);
66  out.setFloatingPointPrecision(QDataStream::SinglePrecision);
67 
68  // magic header, "AERO"
69  out << quint32(0x4153494D);
70  // simulation step
71  out << stp->simTimeStep;
72  // home location
73  out << stp->initPosX << stp->initPosY << stp->initPosZ;
74  out << stp->wpHomeX << stp->wpHomeY << stp->wpHomeLat << stp->wpHomeLong;
75  // position
76  out << stp->posX << stp->posY << stp->posZ;
77  // velocity (world)
78  out << stp->velX << stp->velY << stp->velZ;
79  // angular velocity (model)
80  out << stp->angVelXm << stp->angVelYm << stp->angVelZm;
81  // acceleration (model)
82  out << stp->accelXm << stp->accelYm << stp->accelZm;
83  // coordinates
84  out << stp->latitude << stp->longitude;
85  // sonar
86  out << stp->AGL;
87  // attitude
88  out << stp->heading << stp->pitch << stp->roll;
89  // electric
90  out << stp->voltage << stp->current;
91  // matrix
92  out << stp->axisXx << stp->axisXy << stp->axisXz;
93  out << stp->axisYx << stp->axisYy << stp->axisYz;
94  out << stp->axisZx << stp->axisZy << stp->axisZz;
95  // channels
96  for (int i = 0; i < 8; ++i) {
97  quint8 mapTo = channelsMap.at(i);
98  if (mapTo == 255) // unused channel
99  out << 0.0;
100  else if (takeFromTX) // use values from simulators transmitter
101  out << stp->chSimTX[mapTo];
102  else // direct use values from ESC/motors/ailerons/etc
103  out << stp->chSimRX[mapTo];
104  }
105 
106  // packet counter
107  out << packetsSended;
108 
109  outSocket->writeDatagram(data, outHost, outPort);
110  ++packetsSended;
111 }

195 void UdpReceiver::onReadyRead()
196 {
197  if (!inSocket->waitForReadyRead(8)) // 1/60fps ~= 16.7ms, 1/120fps ~= 8.3ms
198  return;
199  // TODO: add failsafe
200  // if a command not recieved then slowly return all channel to neutral
201  //
202  while (inSocket->hasPendingDatagrams()) {
203  QByteArray datagram;
204  datagram.resize(inSocket->pendingDatagramSize());
205  quint64 datagramSize;
206  datagramSize = inSocket->readDatagram(datagram.data(), datagram.size());
207 
208  processDatagram(datagram);
209  }
210 }

212 void UdpReceiver::processDatagram(QByteArray &datagram)
213 {
214  QDataStream stream(datagram);
215  stream.setFloatingPointPrecision(QDataStream::SinglePrecision);
216  // check magic header
217  quint32 magic;
218  stream >> magic;
219  if (magic != 0x52434D44) // "RCMD"
220  return;
221  // read channels
222  for (int i = 0; i < 10; ++i)
223  stream >> channels[i];
224  // read flight mode
225  stream >> armed >> mode;
226  // read counter
227  stream >> packetsRecived;
228 }
