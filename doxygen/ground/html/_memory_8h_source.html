62 inline void* ei_handmade_aligned_malloc(std::size_t size)
63 {
64  void *original = std::malloc(size+16);
65  void *aligned = reinterpret_cast<void*>((reinterpret_cast<std::size_t>(original) & ~(std::size_t(15))) + 16);
66  *(reinterpret_cast<void**>(aligned) - 1) = original;
67  return aligned;
68 }

80 inline void* ei_aligned_malloc(std::size_t size)
81 {
82  #ifdef EIGEN_NO_MALLOC
83  ei_assert(false && "heap allocation is forbidden (EIGEN_NO_MALLOC is defined)");
84  #endif
85 
86  void *result;
87  #if !EIGEN_ALIGN
88  result = std::malloc(size);
89  #elif EIGEN_MALLOC_ALREADY_ALIGNED
90  result = std::malloc(size);
91  #elif EIGEN_HAS_POSIX_MEMALIGN
92  if(posix_memalign(&result, 16, size)) result = 0;
93  #elif EIGEN_HAS_MM_MALLOC
94  result = _mm_malloc(size, 16);
95  #elif (defined _MSC_VER)
96  result = _aligned_malloc(size, 16);
97  #else
98  result = ei_handmade_aligned_malloc(size);
99  #endif
100 
101  #ifdef EIGEN_EXCEPTIONS
102  if(result == 0)
103  throw std::bad_alloc();
104  #endif
105  return result;
106 }

229 template<typename Scalar, typename Integer>
230 inline static Integer ei_alignmentOffset(const Scalar* array, Integer size)
231 {
232  typedef typename ei_packet_traits<Scalar>::type Packet;
233  enum { PacketSize = ei_packet_traits<Scalar>::size,
234  PacketAlignedMask = PacketSize-1
235  };
236 
237  if(PacketSize==1)
238  {
239  // Either there is no vectorization, or a packet consists of exactly 1 scalar so that all elements
240  // of the array have the same aligment.
241  return 0;
242  }
243  else if(std::size_t(array) & (sizeof(Scalar)-1))
244  {
245  // There is vectorization for this scalar type, but the array is not aligned to the size of a single scalar.
246  // Consequently, no element of the array is well aligned.
247  return size;
248  }
249  else
250  {
251  return std::min<Integer>( (PacketSize - (Integer((std::size_t(array)/sizeof(Scalar))) & PacketAlignedMask))
252  & PacketAlignedMask, size);
253  }
254 }
