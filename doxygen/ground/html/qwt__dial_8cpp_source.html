492 void QwtDial::drawContents( QPainter *painter ) const
493 {
494  if ( testAttribute( Qt::WA_NoSystemBackground ) ||
495  palette().brush( QPalette::Base ) !=
496  palette().brush( QPalette::Window ) )
497  {
498  const QRectF br = boundingRect();
499 
500  painter->save();
501  painter->setPen( Qt::NoPen );
502  painter->setBrush( palette().brush( QPalette::Base ) );
503  painter->drawEllipse( br );
504  painter->restore();
505  }
506 
507 
508  const QRectF insideScaleRect = scaleInnerRect();
509  if ( palette().brush( QPalette::WindowText ) !=
510  palette().brush( QPalette::Base ) )
511  {
512  painter->save();
513  painter->setPen( Qt::NoPen );
514  painter->setBrush( palette().brush( QPalette::WindowText ) );
515  painter->drawEllipse( insideScaleRect );
516  painter->restore();
517  }
518 
519  const QPointF center = insideScaleRect.center();
520  const double radius = 0.5 * insideScaleRect.width();
521 
522  double direction = d_data->origin;
523 
524  if ( isValid() )
525  {
526  direction = d_data->minScaleArc;
527  if ( maxValue() > minValue() &&
528  d_data->maxScaleArc > d_data->minScaleArc )
529  {
530  const double ratio =
531  ( value() - minValue() ) / ( maxValue() - minValue() );
532  direction += ratio * ( d_data->maxScaleArc - d_data->minScaleArc );
533  }
534 
535  if ( d_data->direction == QwtDial::CounterClockwise )
536  direction = d_data->maxScaleArc - ( direction - d_data->minScaleArc );
537 
538  direction += d_data->origin;
539  if ( direction >= 360.0 )
540  direction -= 360.0;
541  else if ( direction < 0.0 )
542  direction += 360.0;
543  }
544 
545  double origin = d_data->origin;
546  if ( mode() == RotateScale )
547  {
548  origin -= direction - d_data->origin;
549  direction = d_data->origin;
550  }
551 
552  painter->save();
553  drawScale( painter, center, radius, origin,
554  d_data->minScaleArc, d_data->maxScaleArc );
555  painter->restore();
556 
557  painter->save();
558  drawScaleContents( painter, center, radius );
559  painter->restore();
560 
561  if ( isValid() )
562  {
563  QPalette::ColorGroup cg;
564  if ( isEnabled() )
565  cg = hasFocus() ? QPalette::Active : QPalette::Inactive;
566  else
567  cg = QPalette::Disabled;
568 
569  painter->save();
570  drawNeedle( painter, center, radius, direction, cg );
571  painter->restore();
572  }
573 }

606 void QwtDial::drawScale( QPainter *painter, const QPointF &center,
607  double radius, double origin, double minArc, double maxArc ) const
608 {
609  if ( d_data->scaleDraw == NULL )
610  return;
611 
612  origin -= 270.0; // hardcoded origin of QwtScaleDraw
613 
614  double angle = maxArc - minArc;
615  if ( angle > 360.0 )
616  angle = ::fmod( angle, 360.0 );
617 
618  minArc += origin;
619  if ( minArc < -360.0 )
620  minArc = ::fmod( minArc, 360.0 );
621 
622  maxArc = minArc + angle;
623  if ( maxArc > 360.0 )
624  {
625  // QwtRoundScaleDraw::setAngleRange accepts only values
626  // in the range [-360.0..360.0]
627  minArc -= 360.0;
628  maxArc -= 360.0;
629  }
630 
631  if ( d_data->direction == QwtDial::CounterClockwise )
632  qSwap( minArc, maxArc );
633 
634  painter->setFont( font() );
635 
636  d_data->scaleDraw->setAngleRange( minArc, maxArc );
637  d_data->scaleDraw->setRadius( radius );
638  d_data->scaleDraw->moveCenter( center );
639 
640  QPalette pal = palette();
641 
642  const QColor textColor = pal.color( QPalette::Text );
643  pal.setColor( QPalette::WindowText, textColor ); //ticks, backbone
644 
645  painter->setPen( QPen( textColor, d_data->scaleDraw->penWidth() ) );
646 
647  painter->setBrush( Qt::red );
648  d_data->scaleDraw->draw( painter, pal );
649 }

947 static double line2Radians( const QPointF &p1, const QPointF &p2 )
948 {
949  const QPointF p = p2 - p1;
950 
951  double angle;
952  if ( p.x() == 0 )
953  angle = ( p.y() <= 0.0 ) ? M_PI_2 : 3 * M_PI_2;
954  else
955  {
956  angle = qAtan( double( -p.y() ) / double( p.x() ) );
957  if ( p.x() < 0.0 )
958  angle += M_PI;
959  if ( angle < 0.0 )
960  angle += 2 * M_PI;
961  }
962  return 360.0 - angle * 180.0 / M_PI;
963 }

971 double QwtDial::getValue( const QPoint &pos )
972 {
973  if ( d_data->maxScaleArc == d_data->minScaleArc || maxValue() == minValue() )
974  return minValue();
975 
976  double dir = line2Radians( innerRect().center(), pos ) - d_data->origin;
977  if ( dir < 0.0 )
978  dir += 360.0;
979 
980  if ( mode() == RotateScale )
981  dir = 360.0 - dir;
982 
983  // The position might be in the area that is outside the scale arc.
984  // We need the range of the scale if it was a complete circle.
985 
986  const double completeCircle = 360.0 / ( d_data->maxScaleArc - d_data->minScaleArc )
987  * ( maxValue() - minValue() );
988 
989  double posValue = minValue() + completeCircle * dir / 360.0;
990 
991  if ( scrollMode() == ScrMouse )
992  {
993  if ( d_data->previousDir >= 0.0 ) // valid direction
994  {
995  // We have to find out whether the mouse is moving
996  // clock or counter clockwise
997 
998  bool clockWise = false;
999 
1000  const double angle = dir - d_data->previousDir;
1001  if ( ( angle >= 0.0 && angle <= 180.0 ) || angle < -180.0 )
1002  clockWise = true;
1003 
1004  if ( clockWise )
1005  {
1006  if ( dir < d_data->previousDir && mouseOffset() > 0.0 )
1007  {
1008  // We passed 360 -> 0
1009  setMouseOffset( mouseOffset() - completeCircle );
1010  }
1011 
1012  if ( wrapping() )
1013  {
1014  if ( posValue - mouseOffset() > maxValue() )
1015  {
1016  // We passed maxValue and the value will be set
1017  // to minValue. We have to adjust the mouseOffset.
1018 
1019  setMouseOffset( posValue - minValue() );
1020  }
1021  }
1022  else
1023  {
1024  if ( posValue - mouseOffset() > maxValue() ||
1025  value() == maxValue() )
1026  {
1027  // We fix the value at maxValue by adjusting
1028  // the mouse offset.
1029 
1030  setMouseOffset( posValue - maxValue() );
1031  }
1032  }
1033  }
1034  else
1035  {
1036  if ( dir > d_data->previousDir && mouseOffset() < 0.0 )
1037  {
1038  // We passed 0 -> 360
1039  setMouseOffset( mouseOffset() + completeCircle );
1040  }
1041 
1042  if ( wrapping() )
1043  {
1044  if ( posValue - mouseOffset() < minValue() )
1045  {
1046  // We passed minValue and the value will be set
1047  // to maxValue. We have to adjust the mouseOffset.
1048 
1049  setMouseOffset( posValue - maxValue() );
1050  }
1051  }
1052  else
1053  {
1054  if ( posValue - mouseOffset() < minValue() ||
1055  value() == minValue() )
1056  {
1057  // We fix the value at minValue by adjusting
1058  // the mouse offset.
1059 
1060  setMouseOffset( posValue - minValue() );
1061  }
1062  }
1063  }
1064  }
1065  d_data->previousDir = dir;
1066  }
1067 
1068  return posValue;
