47 bool QxtGlobalShortcutPrivate::setShortcut(const QKeySequence& shortcut)
48 {
49  Qt::KeyboardModifiers allMods = Qt::ShiftModifier | Qt::ControlModifier | Qt::AltModifier | Qt::MetaModifier;
50  key = shortcut.isEmpty() ? Qt::Key(0) : Qt::Key((shortcut[0] ^ allMods) & shortcut[0]);
51  mods = shortcut.isEmpty() ? Qt::KeyboardModifiers(0) : Qt::KeyboardModifiers(shortcut[0] & allMods);
52  const quint32 nativeKey = nativeKeycode(key);
53  const quint32 nativeMods = nativeModifiers(mods);
54  const bool res = registerShortcut(nativeKey, nativeMods);
55  shortcuts.insert(qMakePair(nativeKey, nativeMods), &qxt_p());
56  if (!res)
57  qWarning() << "QxtGlobalShortcut failed to register:" << QKeySequence(key + mods).toString();
58  return res;
59 }

61 bool QxtGlobalShortcutPrivate::unsetShortcut()
62 {
63  const quint32 nativeKey = nativeKeycode(key);
64  const quint32 nativeMods = nativeModifiers(mods);
65  const bool res = unregisterShortcut(nativeKey, nativeMods);
66  shortcuts.remove(qMakePair(nativeKey, nativeMods));
67  if (!res)
68  qWarning() << "QxtGlobalShortcut failed to unregister:" << QKeySequence(key + mods).toString();
69  key = Qt::Key(0);
70  mods = Qt::KeyboardModifiers(0);
71  return res;
72 }
