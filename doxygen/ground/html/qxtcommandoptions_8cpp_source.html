140  * This function is used to check to see if a parameter

143 static int isQtOption(const QString& param)
144 {
145  // Qt options all start with a single dash regardless of platform
146  if (param.length() < 2) return 0;
147  if (param[0] != '-') return 0;
148  if (param[1] == '-') return 0;
149 #ifdef Q_OS_MAC
150  if (param.left(5) == "-psn_") return 1;
151 #endif
152  QString name = param.mid(1), value;
153  bool hasEquals;
154 
155  // Separate the option and the value, if present
156  if (name.indexOf('=') != -1)
157  {
158  value = param.section('=', 1);
159  name = param.section('=', 0, 0);
160  hasEquals = true;
161  }
162  else
163  {
164  value = "";
165  hasEquals = false;
166  }
167 
168  const char* option;
169  bool optionHasValue;
170  for (int i = 0; qxt_qt_options[i]; i += 2)
171  {
172  option = qxt_qt_options[i];
173  // In the table above, options that require parameters start with =
174  if (option[0] == '=')
175  {
176  optionHasValue = true;
177  option = option + 1; // pointer math to skip =
178  }
179  else
180  {
181  optionHasValue = false;
182  }
183 
184  // The return value indicates how many parameters to skip
185  if (name == option)
186  {
187  if (optionHasValue) return 2;
188  return 1;
189  }
190  }
191 
192  return 0;
193 }

536 /* Do the work of parsing the command line */
537 void QxtCommandOptionsPrivate::parse(const QStringList& params)
538 {
539  int pos = 1; // 0 is the application name
540  int ct = params.count();
541  int skip = 0;
542  bool endFlags = false;
543  bool notUnrecognized, hasEquals;
544  QString name, param, value;
545 
546  while (pos < ct)
547  {
548  // Ignore Qt built-in options
549  while ((skip = isQtOption(params[pos])))
550  {
551  pos += skip;
552  if (pos >= ct)
553  return;
554  }
555 
556  param = params[pos];
557  pos++;
558 
559  if (!endFlags && ((flagStyle == QxtCommandOptions::Slash && param[0] == '/') ||
560  (flagStyle != QxtCommandOptions::Slash && param[0] == '-')))
561  {
562  // tagged argument
563  if (param.length() == 1)
564  {
565  // "-" or "/" alone can't possibly match a flag, so use positional.
566  positional.append(param);
567  continue;
568  }
569  notUnrecognized = false;
570 
571  if (flagStyle != QxtCommandOptions::Slash && param == "--")
572  {
573  // End of parameters flag
574  endFlags = true;
575  }
576  else if (flagStyle == QxtCommandOptions::DoubleDash && param[1] != '-')
577  {
578  // Handle short-form options
579  int len = param.length();
580  QxtCommandOption* option;
581  for (int i = 1; i < len; i++)
582  {
583  QString ch(param[i]);
584  if (ch == "-")
585  {
586  endFlags = true;
587  }
588  else
589  {
590  option = lookup.value(ch, 0);
591  if (!option)
592  {
593  // single-letter flag has no known equivalent
594  unrecognized.append(QString("-") + param[i]);
595  }
596  else
597  {
598  if (option->paramType & QxtCommandOptions::ValueRequired)
599  {
600  // Check for required parameters
601  // Short options can't have optional parameters
602  if (pos >= params.count())
603  {
604  missingParams.append(param);
605  break;
606  }
607  value = params[pos];
608  }
609  else
610  {
611  value = "";
612  }
613  setOption(option, value);
614  }
615  }
616  }
617  }
618  else
619  {
620  // Break apart a value
621  if (param.indexOf('=') != -1)
622  {
623  value = param.section('=', 1);
624  param = param.section('=', 0, 0);
625  hasEquals = true;
626  }
627  else
628  {
629  value = "";
630  hasEquals = false;
631  }
632 
633  if (flagStyle == QxtCommandOptions::DoubleDash)
634  name = param.mid(2);
635  else
636  name = param.mid(1);
637 
638  QxtCommandOption* option = lookup.value(name, 0);
639  if (!option)
640  {
641  unrecognized.append(param);
642  }
643  else
644  {
645  if (option->paramType & QxtCommandOptions::ValueRequired && !hasEquals)
646  {
647  // Check for parameters
648  if (pos >= params.count())
649  {
650  missingParams.append(param);
651  break;
652  }
653  value = params[pos];
654  pos++;
655  }
656  else if ((paramStyle & QxtCommandOptions::Space)
657  && (option->paramType & QxtCommandOptions::ValueOptional)
658  && !hasEquals)
659  {
660  if (pos < params.count())
661  {
662  if (!((flagStyle == QxtCommandOptions::Slash && params.at(pos)[0] == '/') ||
663  (flagStyle != QxtCommandOptions::Slash && params.at(pos)[0] == '-')))
664  {
665  value = params[pos];
666  pos++;
667  }
668  }
669  }
670 
671  setOption(option, value);
672  }
673  }
674  }
675  else
676  {
677  // positional argument
678  positional.append(param);
679  }
680  }
681 }

807 void QxtCommandOptions::showUsage(bool showQtOptions, QTextStream& stream) const
808 {
809  QStringList names;
810  QStringList descs;
811  int maxNameLength = 0;
812  QString name;
813 
814  foreach(const QxtCommandOption& option, qxt_d().options)
815  {
816  // Don't generate usage for undocumented parameters
817  if (option.paramType & Undocumented) continue;
818 
819  foreach(const QString& n, option.names)
820  {
821  if (name.length()) name += ", ";
822  if (qxt_d().flagStyle == Slash)
823  name += '/';
824  else if (qxt_d().flagStyle == DoubleDash && n.length() > 1)
825  name += "--";
826  else
827  name += '-';
828  name += n;
829  if (option.paramType & (ValueOptional | ValueRequired))
830  {
831  if (option.paramType & ValueOptional)
832  name += "[=x]";
833  else if (qxt_d().paramStyle == SpaceAndEquals)
834  name += "[=]x";
835  else if (qxt_d().paramStyle == Equals)
836  name += "=x";
837  else
838  name += " x";
839  }
840  }
841 
842  // The maximum name length is used for formatting the output columns
843  if (name.length() > maxNameLength)
844  maxNameLength = name.length();
845  names.append(name);
846  descs.append(option.desc);
847  name = "";
848  }
849 
850  if (showQtOptions)
851  {
852  // Add a section header
853  names.append(QString());
854  descs.append("Common Qt Options");
855 
856  // Parse through qxt_qt_options
857  const char* option;
858  bool optionHasValue;
859  for (int i = 0; qxt_qt_options[i]; i += 2)
860  {
861  option = qxt_qt_options[i];
862 
863  if (option[0] == '=')
864  {
865  // The option takes a parameter
866  optionHasValue = true;
867  option = option + 1; // pointer math to skip the =
868  }
869  else
870  {
871  optionHasValue = false;
872  }
873 
874  // Concatenate on the option alias
875  if (!name.isEmpty()) name += ", ";
876  name += '-';
877  name += option;
878  if (optionHasValue) name += "[=]x";
879 
880  if (qxt_qt_options[i+1][0] != 0)
881  {
882  // The last alias for the option has the description
883  if (name.length() > maxNameLength)
884  maxNameLength = name.length();
885  names.append(name);
886  descs.append(qxt_qt_options[i+1]);
887  name = "";
888  }
889  }
890  }
891 
892  int ct = names.count();
893  QString line, wrap(maxNameLength + 3, ' ');
894  for (int i = 0; i < ct; i++)
895  {
896  if (names[i].isEmpty())
897  {
898  // Section headers have no name entry
899  stream << endl << descs[i] << ":" << endl;
900  continue;
901  }
902  line = ' ' + names[i] + QString(maxNameLength - names[i].length() + 2, ' ');
903  foreach(const QString& word, descs[i].split(' ', QString::SkipEmptyParts))
904  {
905  if (qxt_d().screenWidth > 0 && line.length() + word.length() >= qxt_d().screenWidth)
906  {
907  stream << line << endl;
908  line = wrap;
909  }
910  line += word + ' ';
911  }
912  stream << line << endl;
913  }
