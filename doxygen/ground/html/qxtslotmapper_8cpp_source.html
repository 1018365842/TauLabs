76 int QxtSlotMapperPrivate::checkSignal(const QObject* sender, const char* signal)
77 {
78  if (!sender || !signal)
79  {
80  qWarning("QxtSlotMapper: Cannot connect %s::%s",
81  sender ? sender->metaObject()->className() : "(null)",
82  (signal && *signal) ? signal+1 : "(null)");
83  return false;
84  }
85 
86  const QByteArray signalName = QMetaObject::normalizedSignature(signal+1);
87  int signalId = sender->metaObject()->indexOfSignal(signalName);
88  if (signalId < 0)
89  {
90  qWarning("QxtSlotMapper: No such signal %s::%s",
91  sender->metaObject()->className(), signal+1);
92  return false;
93  }
94  return signalId;
95 }

127 void QxtSlotMapper::removeMapping(const QVariant& parameter, QObject* receiver, const char* member)
128 {
129  QMutableListIterator<QxtSlotMapInfo> it(qxt_d().mappings);
130  while (it.hasNext())
131  {
132  const QxtSlotMapInfo& info = it.next();
133  if (info.parameter == parameter)
134  {
135  if ((!receiver || receiver == info.receiver) &&
136  (!member || QxtMetaObject::methodName(member) == QxtMetaObject::methodName(info.member)))
137  {
138  it.remove();
139  }
140  }
141  }
142 }

148 bool QxtSlotMapper::connect(const QObject* sender, const char* signal)
149 {
150  QByteArray signalName = QMetaObject::normalizedSignature(signal);
151  int signalId = QxtSlotMapperPrivate::checkSignal(sender, signal);
152  if (signalId < 0)
153  return false;
154 
155  int index = signalName.indexOf('(');
156  if (index != -1)
157  {
158  QByteArray typeName = signalName.mid(index + 1);
159  int index = typeName.indexOf(')');
160  if (index != -1)
161  typeName.truncate(index);
162  typeName = QMetaObject::normalizedType(typeName);
163 
164  int type = QMetaType::type(typeName);
165  int methods = metaObject()->methodCount();
166  return QMetaObject::connect(sender, signalId, this, methods + type);
167  }
168  return false;
169 }

174 int QxtSlotMapper::qt_metacall(QMetaObject::Call call, int id, void** arguments)
175 {
176  id = QObject::qt_metacall(call, id, arguments);
177  if (id < 0 || call != QMetaObject::InvokeMetaMethod)
178  return id;
179 
180  QVariant param(id, arguments[1]);
181  foreach (const QxtSlotMapInfo& info, qxt_d().mappings)
182  {
183  if (info.receiver && info.parameter == param)
184  {
185  const QMetaObject* metaObject = info.receiver->metaObject();
186  int index = metaObject->indexOfMethod(QxtMetaObject::methodSignature(info.member));
187  if (index != -1)
188  {
189  QMetaMethod method = metaObject->method(index);
190  switch (method.parameterTypes().count())
191  {
192  case 0:
193  QMetaObject::invokeMethod(info.receiver,
194  QxtMetaObject::methodName(info.member));
195  break;
196  case 1:
197  QMetaObject::invokeMethod(info.receiver,
198  QxtMetaObject::methodName(info.member),
199  QGenericArgument(QMetaType::typeName(param.type()), arguments[1]));
200  break;
201  default:
202  qWarning("QxtSlotMapper does not support slots with more than one parameter");
203  break;
204  }
205  }
206  }
207  }
208  return -1;
