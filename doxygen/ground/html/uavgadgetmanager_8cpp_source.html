81 UAVGadgetManager::UAVGadgetManager(ICore *core, QString name, QIcon icon, int priority, QString uniqueName, QWidget *parent) :
82  m_showToolbars(true),
83  m_splitterOrView(0),
84  m_currentGadget(0),

88  m_priority(priority),

91 
92  // checking that the mode name is unique gives harmless

95  if (!modeManager->mode(uniqueName)) {
96  m_uniqueName = uniqueName;
97  } else {
98  // this shouldn't happen
99  m_uniqueName = uniqueName + QString::number(quint64(this));
100  }
101  m_uniqueNameBA = m_uniqueName.toLatin1();
102  m_uniqueModeName = m_uniqueNameBA.data();
103 
104  connect(m_core, SIGNAL(contextAboutToChange(Core::IContext *)),
105  this, SLOT(handleContextChange(Core::IContext *)));
106 
107  connect(modeManager, SIGNAL(currentModeChanged(Core::IMode*)),
108  this, SLOT(modeChanged(Core::IMode*)));
109 
110  // other setup
111  m_splitterOrView = new SplitterOrView(this, 0);
112 
113  // SplitterOrView with 0 as gadget calls our setCurrentGadget, which relies on currentSplitterOrView(),
114  // which needs our m_splitterorView to be set, which isn't set yet at that time.
115  // So directly set our currentGadget to 0, and do it again.
116  m_currentGadget = 0;
117  setCurrentGadget(m_splitterOrView->view()->gadget());
118 
119  QHBoxLayout *layout = new QHBoxLayout(m_widget);
120  layout->setMargin(0);
121  layout->setSpacing(0);
122  layout->addWidget(m_splitterOrView);

162 void UAVGadgetManager::setCurrentGadget(IUAVGadget *uavGadget)
163 {
164  if (m_currentGadget == uavGadget)
165  return;
166 
167  SplitterOrView *oldView = currentSplitterOrView();
168  m_currentGadget = uavGadget;
169  SplitterOrView *view = currentSplitterOrView();
170  if (oldView != view) {
171  if (oldView)
172  oldView->update();

174  view->update();
175  }
176  uavGadget->widget()->setFocus();
177  emit currentGadgetChanged(uavGadget);

211 void UAVGadgetManager::closeView(Core::Internal::UAVGadgetView *view)
212 {
213  if (!view)
214  return;
215  SplitterOrView *splitterOrView = m_splitterOrView->findView(view);
216  Q_ASSERT(splitterOrView);
217  Q_ASSERT(splitterOrView->view() == view);
218  if (splitterOrView == m_splitterOrView)
219  return;
220 
221  IUAVGadget *gadget = view->gadget();
222  emptyView(view);
223  UAVGadgetInstanceManager *im = ICore::instance()->uavGadgetInstanceManager();
224  im->removeGadget(gadget);
225 
226  SplitterOrView *splitter = m_splitterOrView->findSplitter(splitterOrView);
227  Q_ASSERT(splitterOrView->hasGadget() == false);
228  Q_ASSERT(splitter->isSplitter() == true);
229  splitterOrView->hide();
230  delete splitterOrView;
231 
232  splitter->unsplit();
233 
234  SplitterOrView *newCurrent = splitter->findFirstView();
235  Q_ASSERT(newCurrent);
236  if (newCurrent)
237  setCurrentGadget(newCurrent->gadget());
238 }

399 // Removes all gadgets and splits in the workspace, except the current/active gadget.
400 void UAVGadgetManager::removeAllSplits()
401 {
402  if (m_core->modeManager()->currentMode() != this)
403  return;
404 
405  if (!m_splitterOrView->isSplitter())
406  return;
407 
408  // Use a QPointer, just in case we accidently delete the gadget we want to keep.
409  QPointer<IUAVGadget> currentGadget = m_currentGadget;
410 
411  Q_ASSERT(currentGadget);
412  QList<IUAVGadget*> gadgets = m_splitterOrView->gadgets();
413  Q_ASSERT(gadgets.count(currentGadget) == 1);
414  gadgets.removeOne(currentGadget);
415 
416  // Remove all splits and their gadgets, then create a new view with the current gadget.
417  m_splitterOrView->unsplitAll(currentGadget);
418 
419  // Zeroing the current gadget means setCurrentGadget will do something when we call it.
420  m_currentGadget = 0;
421  setCurrentGadget(currentGadget);
422 
423  // Remove all other gadgets from the instance manager.
424  UAVGadgetInstanceManager *im = ICore::instance()->uavGadgetInstanceManager();
425  foreach (IUAVGadget *g, gadgets) {
426  im->removeGadget(g);
427  }
428  updateUavGadgetMenus();
429 }
