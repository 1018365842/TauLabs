127 void QxtXmlRpcCallPrivate::d_finished()
128 {
129 
130  if (!reply->error())
131  {
132  int s = 0;
133 
134  QXmlStreamReader xml(reply->readAll());
135  while (!xml.atEnd())
136  {
137  xml.readNext();
138  if (xml.isStartElement())
139  {
140  if (s == 0)
141  {
142  if (xml.name().toString() == "methodResponse")
143  {
144  s = 1;
145  }
146  else
147  {
148  xml.raiseError("expected <methodResponse>, got:<" + xml.name().toString() + ">");
149  }
150  }
151  else if (s == 1)
152  {
153  if (xml.name().toString() == "params")
154  {
155  s = 2;
156  }
157  else if (xml.name().toString() == "fault")
158  {
159  isFault = true;
160  s = 3;
161  }
162  else
163  {
164  xml.raiseError("expected <params> or <fault>, got:<" + xml.name().toString() + ">");
165  }
166  }
167  else if (s == 2)
168  {
169  if (xml.name().toString() == "param")
170  {
171  s = 3;
172  }
173  else
174  {
175  xml.raiseError("expected <param>, got:<" + xml.name().toString() + ">");
176  }
177  }
178  else if (s == 3)
179  {
180  if (xml.name().toString() == "value")
181  {
182  result = QxtXmlRpc::deserialize(xml);
183  s = 4;
184  }
185  else
186  {
187  xml.raiseError("expected <value>, got:<" + xml.name().toString() + ">");
188  }
189  }
190 
191  }
192  }
193  if (xml.hasError())
194  {
195  qWarning("QxtXmlRpcCall: %s at line %lld column %lld", xml.errorString().toLocal8Bit().data(),
196  xml.lineNumber(),
197  xml.columnNumber());
198  }
199  }
200  emit pub->finished();
201 }
