98 void UAVObjectBrowserWidget::on_TreeItemExpanded(QModelIndex currentIndex)
99 {
100  TreeItem *item = static_cast<TreeItem*>(currentIndex.internalPointer());
101  TopTreeItem *top = dynamic_cast<TopTreeItem*>(item->parent());
102 
103  //Check if current tree index is the child of the top tree item
104  if (top)
105  {
106  ObjectTreeItem *objItem = dynamic_cast<ObjectTreeItem*>(item);
107  //If the cast succeeds, then this is a UAVO
108  if (objItem)
109  {
110  UAVObject *obj = objItem->object();
111  // Check for multiple instance UAVO
112  if(!obj){
113  objItem = dynamic_cast<ObjectTreeItem*>(item->getChild(0));
114  obj = objItem->object();
115  }
116  Q_ASSERT(obj);
117  UAVObject::Metadata mdata = obj->getMetadata();
118 
119  // Determine fastest update
120  quint16 tmpUpdatePeriod = MAXIMUM_UPDATE_PERIOD;
121  int accessType = UAVObject::GetGcsTelemetryUpdateMode(mdata);
122  if (accessType != UAVObject::UPDATEMODE_MANUAL){
123  switch(accessType){
124  case UAVObject::UPDATEMODE_ONCHANGE:
125  tmpUpdatePeriod = 0;
126  break;
127  case UAVObject::UPDATEMODE_PERIODIC:
128  case UAVObject::UPDATEMODE_THROTTLED:
129  tmpUpdatePeriod = std::min(mdata.gcsTelemetryUpdatePeriod, tmpUpdatePeriod);
130  break;
131  }
132  }
133 
134  accessType = UAVObject::GetFlightTelemetryUpdateMode(mdata);
135  if (accessType != UAVObject::UPDATEMODE_MANUAL){
136  switch(accessType){
137  case UAVObject::UPDATEMODE_ONCHANGE:
138  tmpUpdatePeriod = 0;
139  break;
140  case UAVObject::UPDATEMODE_PERIODIC:
141  case UAVObject::UPDATEMODE_THROTTLED:
142  tmpUpdatePeriod = std::min(mdata.flightTelemetryUpdatePeriod, tmpUpdatePeriod);
143  break;
144  }
145  }
146 
147  expandedUavoItems.insert(obj->getName(), tmpUpdatePeriod);
148 
149  if (tmpUpdatePeriod < updatePeriod){
150  updatePeriod = tmpUpdatePeriod;
151  treeView->updateTimerPeriod(updatePeriod);
152  }
153  }
154  }

377 ObjectTreeItem *UAVObjectBrowserWidget::findCurrentObjectTreeItem()
378 {
379  QModelIndex current = treeView->currentIndex();
380  TreeItem *item = static_cast<TreeItem*>(current.internalPointer());
381  ObjectTreeItem *objItem = 0;
382 
383  // Recursively iterate over child branches until the parent UAVO branch is found
384  while (item) {
385  //Attempt a dynamic cast
386  objItem = dynamic_cast<ObjectTreeItem*>(item);
387 
388  //If the cast succeeds, then this is a UAVO or UAVO metada. Stop the while loop.
389  if (objItem)
390  break;
391 
392  //If it fails, then set item equal to the parent branch, and try again.
393  item = item->parent();
394  }
395 
396  return objItem;
397 }

564 void UAVOBrowserTreeView::updateTimerPeriod(unsigned int val)
565 {
566  if (val == 0){
567  // If val == 0, disable throttling by stopping the timer.
568  m_updateViewTimer.stop();
569  }
570  else
571  {
572  // If the UAVO has a very fast data rate, then don't go the full speed.
573  if (val < 100)
574  {
575  val = 100- powf((100-val),0.914); //This drives the throttled speed exponentially toward 30Hz.
576  }
577  m_updateViewTimer.start(val);
578  }
579 }

615 void UAVOBrowserTreeView::updateView(QModelIndex topLeft, QModelIndex bottomRight)
616 {
617  TopTreeItem *treeItemPtr = static_cast<TopTreeItem*>(topLeft.internalPointer());
618 
619  // Determine if the new indices lie outside of the set of indices queued for update
620  if (treeItemPtr->parent() == m_model->getNonSettingsTree()){
621  if (topmostData < 0 || topmostData > topLeft.row())
622  topmostData = topLeft.row();
623  if (bottommostData < 0 || bottommostData < bottomRight.row())
624  bottommostData = bottomRight.row();
625  }
626  else if(treeItemPtr->parent() == m_model->getSettingsTree()){
627  if (topmostSettings < 0 || topmostSettings > topLeft.row())
628  topmostSettings = topLeft.row();
629  if (bottommostSettings < 0 || bottommostSettings < bottomRight.row())
630  bottommostSettings = bottomRight.row();
631  }
632  else{
633  // Do nothing. These QModelIndices are generated by the highlight manager or for individual
634  // UAVO fields, which are both updated when updating that UAVO's branch of the settings or
635  // dynamic data tree.
636  }
637 }
