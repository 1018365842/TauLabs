86

88  return QVariant::Invalid;

90  pathPlanData * row=dataStorage.at(index.row());
91

94  // Qt::UserRole this should fall through and return the numerical value of
95  // the enum

97  return modeNames.value(row->mode);

190 bool FlightDataModel::setData(const QModelIndex &index, const QVariant &value, int role)
191 {
192  if (index.isValid() && role == Qt::EditRole)
193  {
194  pathPlanData *row = dataStorage.at(index.row());
195 
196  // Do not allow changing any values except locked when the column is locked
197  if (row->locked && index.column() != (int) FlightDataModel::LOCKED)
198  return false;
199 
200  struct FlightDataModel::NED NED;
201  QModelIndex otherIndex;
202  switch(index.column())
203  {
204  case WPDESCRITPTION:
205  row->wpDescritption=value.toString();
206  break;
207  case LATPOSITION:
208  row->latPosition=value.toDouble();
209  // Indicate this also changed the north
210  otherIndex = this->index(index.row(), FlightDataModel::NED_NORTH);
211  emit dataChanged(otherIndex,otherIndex);
212  break;
213  case LNGPOSITION:
214  row->lngPosition=value.toDouble();
215  // Indicate this also changed the east
216  otherIndex = this->index(index.row(), FlightDataModel::NED_EAST);
217  emit dataChanged(otherIndex,otherIndex);
218  break;
219  case ALTITUDE:
220  row->altitude=value.toDouble();
221  // Indicate this also changed the NED down
222  otherIndex = this->index(index.row(), FlightDataModel::NED_DOWN);
223  emit dataChanged(otherIndex,otherIndex);
224  break;
225  case NED_NORTH:
226  NED = getNED(index.row());
227  NED.North = value.toDouble();
228  setNED(index.row(), NED);
229  // Indicate this also changed the latitude
230  otherIndex = this->index(index.row(), FlightDataModel::LATPOSITION);
231  emit dataChanged(otherIndex,otherIndex);
232  break;
233  case NED_EAST:
234  NED = getNED(index.row());
235  NED.East = value.toDouble();
236  setNED(index.row(), NED);
237  // Indicate this also changed the longitude
238  otherIndex = this->index(index.row(), FlightDataModel::LNGPOSITION);
239  emit dataChanged(otherIndex,otherIndex);
240  break;
241  case NED_DOWN:
242  NED = getNED(index.row());
243  NED.Down = value.toDouble();
244  setNED(index.row(), NED);
245  // Indicate this also changed the altitude
246  otherIndex = this->index(index.row(), FlightDataModel::ALTITUDE);
247  emit dataChanged(otherIndex,otherIndex);
248  break;
249  case VELOCITY:
250  row->velocity=value.toFloat();
251  break;
252  case MODE:
253  row->mode=value.toInt();
254  break;
255  case MODE_PARAMS:
256  row->mode_params=value.toFloat();
257  break;
258  case LOCKED:
259  row->locked = value.toBool();
260  break;
261  default:
262  return false;
263  }
264 
265  emit dataChanged(index,index);
266  return true;
267  }
268  return false;
269 }

296 bool FlightDataModel::insertRows(int row, int count, const QModelIndex &/*parent*/)
297 {
298  pathPlanData * data;
299  beginInsertRows(QModelIndex(),row,row+count-1);
300  for(int x=0; x<count;++x)
301  {
302  // Initialize new internal representation
303  data=new pathPlanData;
304  data->latPosition=0;
305  data->lngPosition=0;
306 
307  // If there is a previous waypoint, initialize some of the fields to that value
308  if(rowCount() > 0)
309  {
310  pathPlanData * prevRow = dataStorage.at(rowCount()-1);
311  data->altitude = prevRow->altitude;
312  data->velocity = prevRow->velocity;
313  data->mode = prevRow->mode;
314  data->mode_params = prevRow->mode_params;
315  data->locked = prevRow->locked;
316  } else {
317  data->altitude = 0;
318  data->velocity = 0;
319  data->mode = Waypoint::MODE_FLYVECTOR;
320  data->mode_params = 0;
321  data->locked = false;
322  }
323  dataStorage.insert(row,data);
324  }
325 
326  endInsertRows();
327 
328  return true;
329 }

426 void FlightDataModel::readFromFile(QString fileName)
427 {
428  removeRows(0,rowCount());
429  QFile file(fileName);
430  file.open(QIODevice::ReadOnly);
431  QDomDocument doc("PathPlan");
432  QByteArray array=file.readAll();
433  QString error;
434  if (!doc.setContent(array,&error)) {
435  QMessageBox msgBox;
436  msgBox.setText(tr("File Parsing Failed."));
437  msgBox.setInformativeText(QString(tr("This file is not a correct XML file:%0")).arg(error));
438  msgBox.setStandardButtons(QMessageBox::Ok);
439  msgBox.exec();
440  return;
441  }
442  file.close();
443 
444  QDomElement root = doc.documentElement();
445 
446  if (root.isNull() || (root.tagName() != "waypoints")) {
447  QMessageBox msgBox;
448  msgBox.setText(tr("Wrong file contents"));
449  msgBox.setInformativeText(tr("This file does not contain correct UAVSettings"));
450  msgBox.setStandardButtons(QMessageBox::Ok);
451  msgBox.exec();
452  return;
453  }
454 
455  pathPlanData * data=NULL;
456  QDomNode node = root.firstChild();
457  while (!node.isNull()) {
458  QDomElement e = node.toElement();
459  if (e.tagName() == "waypoint") {
460  QDomNode fieldNode=e.firstChild();
461  data=new pathPlanData;
462  while (!fieldNode.isNull()) {
463  QDomElement field = fieldNode.toElement();
464  if (field.tagName() == "field") {
465  if(field.attribute("name")=="altitude")
466  data->altitude=field.attribute("value").toDouble();
467  else if(field.attribute("name")=="description")
468  data->wpDescritption=field.attribute("value");
469  else if(field.attribute("name")=="latitude")
470  data->latPosition=field.attribute("value").toDouble();
471  else if(field.attribute("name")=="longitude")
472  data->lngPosition=field.attribute("value").toDouble();
473  else if(field.attribute("name")=="altitude")
474  data->altitude=field.attribute("value").toDouble();
475  else if(field.attribute("name")=="velocity")
476  data->velocity=field.attribute("value").toFloat();
477  else if(field.attribute("name")=="mode")
478  data->mode=field.attribute("value").toInt();
479  else if(field.attribute("name")=="mode_params")
480  data->mode_params=field.attribute("value").toFloat();
481  else if(field.attribute("name")=="is_locked")
482  data->locked=field.attribute("value").toInt();
483  }
484  fieldNode=fieldNode.nextSibling();
485  }
486  beginInsertRows(QModelIndex(),dataStorage.length(),dataStorage.length());
487  dataStorage.append(data);
488  endInsertRows();
489  }
490  node=node.nextSibling();
491  }
492 }
