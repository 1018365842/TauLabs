181 template<typename MatrixType>
182 int QR<MatrixType>::rank() const
183 {
184  ei_assert(m_isInitialized && "QR is not initialized.");
185  if (!m_rankIsUptodate)
186  {
187  RealScalar maxCoeff = m_qr.diagonal().cwise().abs().maxCoeff();
188  int n = m_qr.cols();
189  m_rank = 0;
190  while(m_rank<n && !ei_isMuchSmallerThan(m_qr.diagonal().coeff(m_rank), maxCoeff))
191  ++m_rank;
192  m_rankIsUptodate = true;
193  }
194  return m_rank;
195 }

199 template<typename MatrixType>
200 void QR<MatrixType>::compute(const MatrixType& matrix)
201 {
202  m_rankIsUptodate = false;
203  m_qr = matrix;
204  m_hCoeffs.resize(matrix.cols());
205 
206  int rows = matrix.rows();
207  int cols = matrix.cols();
208  RealScalar eps2 = precision<RealScalar>()*precision<RealScalar>();
209 
210  for (int k = 0; k < cols; ++k)
211  {
212  int remainingSize = rows-k;
213 
214  RealScalar beta;
215  Scalar v0 = m_qr.col(k).coeff(k);
216 
217  if (remainingSize==1)
218  {
219  if (NumTraits<Scalar>::IsComplex)
220  {
221  // Householder transformation on the remaining single scalar
222  beta = ei_abs(v0);
223  if (ei_real(v0)>0)
224  beta = -beta;
225  m_qr.coeffRef(k,k) = beta;
226  m_hCoeffs.coeffRef(k) = (beta - v0) / beta;
227  }
228  else
229  {
230  m_hCoeffs.coeffRef(k) = 0;
231  }
232  }
233  else if ((beta=m_qr.col(k).end(remainingSize-1).squaredNorm())>eps2)
234  // FIXME what about ei_imag(v0) ??
235  {
236  // form k-th Householder vector
237  beta = ei_sqrt(ei_abs2(v0)+beta);
238  if (ei_real(v0)>=0.)
239  beta = -beta;
240  m_qr.col(k).end(remainingSize-1) /= v0-beta;
241  m_qr.coeffRef(k,k) = beta;
242  Scalar h = m_hCoeffs.coeffRef(k) = (beta - v0) / beta;
243 
244  // apply the Householder transformation (I - h v v') to remaining columns, i.e.,
245  // R <- (I - h v v') * R where v = [1,m_qr(k+1,k), m_qr(k+2,k), ...]
246  int remainingCols = cols - k -1;
247  if (remainingCols>0)
248  {
249  m_qr.coeffRef(k,k) = Scalar(1);
250  m_qr.corner(BottomRight, remainingSize, remainingCols) -= ei_conj(h) * m_qr.col(k).end(remainingSize)
251  * (m_qr.col(k).end(remainingSize).adjoint() * m_qr.corner(BottomRight, remainingSize, remainingCols));
252  m_qr.coeffRef(k,k) = beta;
253  }
254  }
255  else
256  {
257  m_hCoeffs.coeffRef(k) = 0;
258  }
259  }
260  m_isInitialized = true;
261 }

263 template<typename MatrixType>
264 template<typename OtherDerived, typename ResultType>
265 bool QR<MatrixType>::solve(
266  const MatrixBase<OtherDerived>& b,
267  ResultType *result
268 ) const
269 {
270  ei_assert(m_isInitialized && "QR is not initialized.");
271  const int rows = m_qr.rows();
272  ei_assert(b.rows() == rows);
273  // enforce the computation of the rank
274  rank();
275 
276  result->resize(m_qr.cols(), b.cols());
277 
278  // TODO(keir): There is almost certainly a faster way to multiply by
279  // Q^T without explicitly forming matrixQ(). Investigate.
280  *result = matrixQ().transpose()*b;
281 
282  if(m_rank==0)
283  return result->isZero();
284 
285  if(!isSurjective())
286  {
287  // is result is in the image of R ?
288  RealScalar biggest_in_res = result->corner(TopLeft, m_rank, result->cols()).cwise().abs().maxCoeff();
289  for(int col = 0; col < result->cols(); ++col)
290  for(int row = m_rank; row < result->rows(); ++row)
291  if(!ei_isMuchSmallerThan(result->coeff(row,col), biggest_in_res))
292  return false;
293  }
294  m_qr.corner(TopLeft, m_rank, m_rank)
295  .template marked<UpperTriangular>()
296  .solveTriangularInPlace(result->corner(TopLeft, m_rank, result->cols()));
297 
298  return true;
299 }

302 template<typename MatrixType>
303 MatrixType QR<MatrixType>::matrixQ() const
304 {
305  ei_assert(m_isInitialized && "QR is not initialized.");
306  // compute the product Q_0 Q_1 ... Q_n-1,
307  // where Q_k is the k-th Householder transformation I - h_k v_k v_k'
308  // and v_k is the k-th Householder vector [1,m_qr(k+1,k), m_qr(k+2,k), ...]
309  int rows = m_qr.rows();
310  int cols = m_qr.cols();
311  MatrixType res = MatrixType::Identity(rows, cols);
312  for (int k = cols-1; k >= 0; k--)
313  {
314  // to make easier the computation of the transformation, let's temporarily
315  // overwrite m_qr(k,k) such that the end of m_qr.col(k) is exactly our Householder vector.
316  Scalar beta = m_qr.coeff(k,k);
317  m_qr.const_cast_derived().coeffRef(k,k) = 1;
318  int endLength = rows-k;
319  res.corner(BottomRight,endLength, cols-k) -= ((m_hCoeffs.coeff(k) * m_qr.col(k).end(endLength))
320  * (m_qr.col(k).end(endLength).adjoint() * res.corner(BottomRight,endLength, cols-k)).lazy()).lazy();
321  m_qr.const_cast_derived().coeffRef(k,k) = beta;
322  }
323  return res;
324 }
