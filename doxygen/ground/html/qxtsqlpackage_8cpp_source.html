211 void QxtSqlPackage::insert(QSqlQuery query)
212 {
213  map.clear();
214  record = -1;
215 
216  /*query will be invalid if next is not called first*/
217  if (!query.isValid())
218  query.next();
219 
220  QSqlRecord infoRecord = query.record();
221  int iNumCols = infoRecord.count();
222  QVector<QString> tableMap = QVector<QString>(iNumCols);
223 
224  /*first create a map of index->colname pairs*/
225  for (int iLoop = 0; iLoop < iNumCols; iLoop++)
226  {
227  tableMap[iLoop] = infoRecord.fieldName(iLoop);
228  }
229 
230  /*now use this created map to get column names
231  *this should be faster than querying the QSqlRecord every time
232  *but that depends on the databasetype and size of the table (number of rows and cols)
233  */
234  do
235  {
236  QHash<QString, QString> hash;
237  for (int iColLoop = 0; iColLoop < iNumCols; iColLoop++)
238  {
239  hash[tableMap[iColLoop]] = query.value(iColLoop).toString();
240  }
241  map.append(hash);
242 
243  }
244  while (query.next());
245 }
