1 /***********************************************************************************************************
2 *
3 * NAME: ssp.c
4 * DESCRIPTION: simple serial protocol - packet based serial transport layer.
5 * AUTHOR: Joe Hlebasko
6 * HISTORY: Created 1/1/2010
7 *
8 * Packet Formats
9 * Format:
10 * +------+----+------+---------------------------+--------+
11 * | 225 | L1 | S# | App Data (0-254 bytes) | CRC 16 |
12 * +------+----+------+---------------------------+--------+
13 *
14 * 225 = sync byte, indicates start of a packet
15 * L1 = 1 byte for size of data payload. (sequence number is part of data payload.)
16 * S# = 1 byte for sequence number.
17 * Seq of 0 = seq # synchronise request, forces other end to reset receive sequence number to 1.
18 * sender of synchronise request will reset the tx seq number to 1
19 * Seq # of 1..127 = normal data packets. Sequence number is incremented by for each transmitted
20 * packet. Rolls over from 127 to 1.
21 * if most sig. bit is set then the packet is an ACK packet of data packet sequence number of the
22 * lower 7 bits (1..127)
23 * App Data may contain 0..254 bytes. The sequence number is consider part of the payload.
24 * CRC 16 - 16 bits of CRC values of Sequence # and data bytes.
25 *
26 * Protocol has two types of packets: data and ack packets. ACK packets have the most sig. bit set in the
27 * sequence number, this implies that valid sequence numbers are 1..127
28 *
29 * This protocol uses the concept of sequences numbers to determine if a given packet has been received. This
30 * requires both devices to be able to synchronize sequence numbers. This is accomplished by sending a packet
31 * length 1 and sequence number = 0. The receive then resets it's transmit sequence number to 1.
32 *
33 * ACTIVE_SYNCH is a version that will automatically send a synch request if it receives a synch packet. Only
34 * one device in the communication should do otherwise you end up with an endless loops of synchronization.
35 * Right now each side needs to manually issues a synch request.
36 *
37 * This protocol is best used in cases where one device is the master and the other is the slave, or a don't
38 * speak unless spoken to type of approach.
39 *
40 * The following are items are required to initialize a port for communications:
41 * 1. The number attempts for each packet
42 * 2. time to wait for an ack.
43 * 3. pointer to buffer to be used for receiving.
44 * 4. pointer to a buffer to be used for transmission
45 * 5. length of each buffer (rx and tx)
46 * 6. Four functions:
47 * 1. write byte = writes a byte out the serial port (or other comm device)
48 * 2. read byte = retrieves a byte from the serial port. Returns -1 if a byte is not available
49 * 3. callback = function to call when a valid data packet has been received. This function is responsible
50 * to do what needs to be done with the data when it is received. The primary mission of this function
51 * should be to copy the data to a private buffer out of the working receive buffer to prevent overrun.
52 * processing should be kept to a minimum.
53 * 4. get time = function should return the current time. Note that time units are not specified it just
54 * needs to be some measure of time that increments as time passes by. The timeout values for a given
55 * port should the units used/returned by the get time function.
56 *
57 * All of the state information of a communication port is contained in a Port_t structure. This allows this
58 * module to operature on multiple communication ports with a single code base.
59 *
60 * The ssp_ReceiveProcess and ssp_SendProcess functions need to be called to process data through the
61 * respective state machines. Typical implementation would have a serial ISR to pull bytes out of the UART
62 * and place into a circular buffer. The serial read function would then pull bytes out this buffer
63 * processing. The TX side has the write function placing bytes into a circular buffer with the TX ISR
64 * pulling bytes out of the buffer and putting into the UART. It is possible to run the receive process from
65 * the receive ISR but care must be taken on processing data when it is received to avoid holding up the ISR
66 * and sending ACK packets from the receive ISR.
67 *
68 ***********************************************************************************************************/

222 int16_t ssp_SendProcess( Port_t *thisport )
223 {
224  int16_t value = SSP_TX_WAITING;
225 
226  if (thisport->SendState == SSP_AWAITING_ACK ) {
227  if (sf_CheckTimeout(thisport) == TRUE) {
228  if (thisport->retryCount < thisport->maxRetryCount) {
229  // Try again
230  sf_SendPacket(thisport);
231  sf_SetSendTimeout(thisport);
232  value = SSP_TX_WAITING;
233  } else {
234  // Give up, # of trys has exceded the limit
235  value = SSP_TX_TIMEOUT;
236  CLEARBIT( thisport->flags, ACK_RECEIVED);
237  thisport->SendState = SSP_IDLE;
238  }
239  } else {
240  value = SSP_TX_WAITING;
241  }
242  } else if( thisport->SendState == SSP_ACKED ) {
243  SETBIT( thisport->flags, ACK_RECEIVED);
244  value = SSP_TX_ACKED;
245  thisport->SendState = SSP_IDLE;
246  } else {
247  thisport->SendState = SSP_IDLE;
248  value = SSP_TX_IDLE;
249  }
250  return value;
251 }

338 int16_t ssp_SendData( Port_t *thisport, const uint8_t *data, const uint16_t length )
339 {
340 
341  int16_t value = SSP_TX_WAITING;
342 
343  if( (length + 2) > thisport->txBufSize ) {
344  // TRYING to send too much data.
345  value = SSP_TX_BUFOVERRUN;
346  } else if( thisport->SendState == SSP_IDLE ) {
347 #ifdef ACTIVE_SYNCH
348  if( thisport->sendSynch == TRUE ) {
349  sf_SendSynchPacket(thisport);
350  }
351 #endif
352 
353 #ifdef SYNCH_SEND
354  if( length == 0 ) {
355  // TODO this method could allow a task/user to start a synchronisation step if a zero is mistakenly passed to this function.
356  // could add a check for a NULL data pointer, or use some sort of static flag that can only be accessed by a static function
357  // that must be called before calling this function.
358  // we are attempting to send a synch packet
359  thisport->txSeqNo = 0; // make this zero to cause the other end to re-synch with us
360  SETBIT(thisport->flags, SENT_SYNCH);
361  } else {
362  // we are sending a data packet
363  CLEARBIT( thisport->txSeqNo, ACK_BIT ); // make sure we are not sending a ACK packet
364  thisport->txSeqNo++; // update the sequence number.
365  if( thisport->txSeqNo > 0x7F) { // check for sequence number rollover
366  thisport->txSeqNo = 1; // if we do have rollover then reset to 1 not zero,
367  // zero is reserviced for synchronization requests
368  }
369  }
370 
371 #else
372  CLEARBIT( thisport->txSeqNo, ACK_BIT ); // make sure we are not sending a ACK packet
373  thisport->txSeqNo++; // update the sequence number.
374  if( thisport->txSeqNo > 0x7F) { // check for sequence number rollover
375  thisport->txSeqNo = 1; // if we do have rollover then reset to 1 not zero,
376  // zero is reserved for synchronization requests
377  }
378 #endif
379  CLEARBIT( thisport->flags, ACK_RECEIVED);
380  thisport->SendState = SSP_AWAITING_ACK;
381  value = SSP_TX_WAITING;
382  thisport->retryCount = 0; // zero out the retry counter for this transmission
383  sf_MakePacket( thisport->txBuf, data, length, thisport->txSeqNo );
384  sf_SendPacket( thisport ); // punch out the packet to the serial port
385  sf_SetSendTimeout( thisport ); // do the timeout values
386  } else {
387  // error we are already sending a packet. Need to wait for the current packet to be acked or timeout.
388  value = SSP_TX_BUSY;
389  }
390  return value;
391 }

484 void sf_MakePacket( uint8_t *txBuf, const uint8_t * pdata, uint16_t length, uint8_t seqNo )
485 {
486  uint16_t crc = 0xffff;
487  uint16_t bufPos = 0;
488  uint8_t b;
489 
490  // add 1 for the seq. number
491  txBuf[LENGTH] = length + 1;
492  txBuf[SEQNUM] = seqNo;
493  crc = sf_crc16( crc, seqNo );
494 
495  length = length + 2; // add two for the length and seqno bytes which are added before the loop.
496  for( bufPos = 2; bufPos < length; bufPos++ ) {
497  b = *pdata++;
498  txBuf[bufPos] = b;
499  crc = sf_crc16( crc, b ); // update CRC value
500  }
501  txBuf[bufPos++] = LOWERBYTE(crc);
502  txBuf[bufPos] = UPPERBYTE(crc);

682 static int16_t sf_DecodeState( Port_t *thisport, uint8_t c )
683 {
684  int16_t retval;
685  switch( thisport->DecodeState ) {
686  case decode_idle_e:
687  // 'c' is ignored in this state as the only way to leave the idle state is
688  // recognition of the SYNC byte in the sf_ReceiveState function.
689  retval = SSP_RX_IDLE;
690  break;
691  case decode_len1_e:
692  thisport->rxBuf[LENGTH]= c;
693  thisport->rxBufLen = c;
694  if( thisport->rxBufLen <= thisport->rxBufSize ) {
695  thisport->DecodeState = decode_seqNo_e;
696  retval = SSP_RX_RECEIVING;
697  } else {
698  thisport->DecodeState = decode_idle_e;
699  retval = SSP_RX_IDLE;
700  }
701  break;
702  case decode_seqNo_e:
703  thisport->rxBuf[SEQNUM] = c;
704  thisport->crc = 0xffff;
705  thisport->rxBufLen--; // subtract 1 for the seq. no.
706  thisport->rxBufPos = 2;
707 
708  thisport->crc = sf_crc16( thisport->crc, c );
709  if( thisport->rxBufLen > 0 ) {
710  thisport->DecodeState = decode_data_e;
711  } else {
712  thisport->DecodeState = decode_crc1_e;
713  }
714  retval = SSP_RX_RECEIVING;
715  break;
716  case decode_data_e:
717  thisport->rxBuf[ (thisport->rxBufPos)++] = c;
718  thisport->crc = sf_crc16( thisport->crc, c );
719  if( thisport->rxBufPos == (thisport->rxBufLen+2) ) {
720  thisport->DecodeState = decode_crc1_e;
721  }
722  retval = SSP_RX_RECEIVING;
723  break;
724  case decode_crc1_e:
725  thisport->crc = sf_crc16( thisport->crc, c );
726  thisport->DecodeState = decode_crc2_e;
727  retval = SSP_RX_RECEIVING;
728  break;
729  case decode_crc2_e:
730  thisport->DecodeState = decode_idle_e;
731  // verify the CRC value for the packet
732  if( sf_crc16( thisport->crc, c) == 0) {
733  // TODO shouldn't the return value of sf_ReceivePacket() be checked?
734  sf_ReceivePacket( thisport );
735  retval = SSP_RX_COMPLETE;
736  } else {
737  thisport->RxError++;
738  retval = SSP_RX_IDLE;
739  }
740  break;
741  default:
742  thisport->DecodeState = decode_idle_e; // unknown state so reset to idle state and wait for the next start of a packet.
743  retval = SSP_RX_IDLE;
744  break;
745  }
746  return retval;
747 }

749 /************************************************************************************************************
750 *
751 * NAME: int16_t sf_ReceivePacket( )
752 * DESCRIPTION: Receive one packet, assumed that data is in rec.buff[]
753 * ARGUMENTS:
754 * RETURN: 0 . no new packet was received, could be ack or same packet
755 * 1 . new packet received
756 * SSP_PACKET_?
757 * SSP_PACKET_COMPLETE
758 * SSP_PACKET_ACK
759 * CREATED: 5/8/02
760 *
761 *************************************************************************************************************/
773 static int16_t sf_ReceivePacket(Port_t *thisport)
774 {
775  int16_t value = FALSE;
776 
777  if( ISBITSET(thisport->rxBuf[SEQNUM], ACK_BIT ) ) {
778  // Received an ACK packet, need to check if it matches the previous sent packet
779  if( ( thisport->rxBuf[SEQNUM] & 0x7F) == (thisport->txSeqNo & 0x7f)) {
780  // It matches the last packet sent by us
781  SETBIT( thisport->txSeqNo, ACK_BIT );
782  thisport->SendState = SSP_ACKED;
783  value = FALSE;
784  }
785  // else ignore the ACK packet
786  } else {
787  // Received a 'data' packet, figure out what type of packet we received...
788  if( thisport->rxBuf[SEQNUM] == 0 ) {
789  // Synchronize sequence number with host
790 #ifdef ACTIVE_SYNCH
791  thisport->sendSynch = TRUE;
792 #endif
793  sf_SendAckPacket( thisport, thisport->rxBuf[SEQNUM] );
794  thisport->rxSeqNo = 0;
795  value = FALSE;
796  } else if( thisport->rxBuf[SEQNUM] == thisport->rxSeqNo ) {
797  // Already seen this packet, just ack it, don't act on the packet.
798  sf_SendAckPacket( thisport, thisport->rxBuf[SEQNUM] );
799  value = FALSE;
800  } else {
801  //New Packet
802  thisport->rxSeqNo = thisport->rxBuf[SEQNUM];
803  // Let the application do something with the data/packet.
804  if( thisport->pfCallBack != NULL ) {
805  // skip the first two bytes (length and seq. no.) in the buffer.
806  thisport->pfCallBack( &(thisport->rxBuf[2]), thisport->rxBufLen);
807  }
808  // after we send the ACK, it is possible for the host to send a new packet.
809  // Thus the application needs to copy the data and reset the receive buffer
810  // inside of thisport->pfCallBack()
811  sf_SendAckPacket( thisport, thisport->rxBuf[SEQNUM] );
812  value = TRUE;
813  }
814  }
815  return value;
816 }
