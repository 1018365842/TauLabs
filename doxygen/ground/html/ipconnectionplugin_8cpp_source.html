72 void IPConnection::onOpenDevice(QString HostName, int Port, bool UseTCP)
73 {
74  QAbstractSocket *ipSocket;
75  const int Timeout = 5 * 1000;
76 
77  ipConMutex.lock();
78  if (UseTCP) {
79  ipSocket = new QTcpSocket(this);
80  } else {
81  ipSocket = new QUdpSocket(this);
82  }
83 
84  //do sanity check on hostname and port...
85  if((HostName.length()==0)||(Port<1)){
86  errorMsg = "Please configure Host and Port options before opening the connection";
87 
88  }
89  else {
90  //try to connect...
91  ipSocket->connectToHost(HostName, Port);
92 
93  //in blocking mode so we wait for the connection to succeed
94  if (ipSocket->waitForConnected(Timeout)) {
95  ret = ipSocket;
96  openDeviceWait.wakeAll();
97  ipConMutex.unlock();
98  return;
99  }
100  //tell user something went wrong
101  errorMsg = ipSocket->errorString ();
102  }
103  /* BUGBUG TODO - returning null here leads to segfault because some caller still calls disconnect without checking our return value properly
104  * someone needs to debug this, I got lost in the calling chain.*/
105  ret = NULL;
106  openDeviceWait.wakeAll();
107  ipConMutex.unlock();
108 }

176 QIODevice *IPconnectionConnection::openDevice(Core::IDevice *deviceName)
177 {
178  Q_UNUSED(deviceName)
179 
180  QString HostName;
181  int Port;
182  bool UseTCP;
183  QMessageBox msgBox;
184 
185  //get the configuration info
186  HostName = m_config->HostName();
187  Port = m_config->Port();
188  UseTCP = m_config->UseTCP();
189 
190  if (ipSocket){
191  //Andrew: close any existing socket... this should never occur
192  ipConMutex.lock();
193  emit CloseSocket(ipSocket);
194  closeDeviceWait.wait(&ipConMutex);
195  ipConMutex.unlock();
196  ipSocket = NULL;
197  }
198 
199  ipConMutex.lock();
200  emit CreateSocket(HostName, Port, UseTCP);
201  openDeviceWait.wait(&ipConMutex);
202  ipConMutex.unlock();
203  ipSocket = ret;
204  if(ipSocket == NULL)
205  {
206  msgBox.setText((const QString )errorMsg);
207  msgBox.exec();
208  }
209  return ipSocket;
210 }
