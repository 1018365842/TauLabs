32 /*
33 \class QxtHmac
34 
35 \inmodule QxtCore
36 
37 \brief The QxtHmac class calculates keyed-Hash Message Authentication Codes
38 
39 HMAC is a well-known algorithm for generating a message authentication code (MAC) that can be used to verify the
40 integrity and authenticity of a message.
41 
42 This class requires Qt 4.3.0 or greater.
43 
44 To verify a message, the sender creates a MAC using a key, which is a secret known only to the sender and recipient,
45 and the content of the message. This MAC is then sent along with the message. The recipient then creates another MAC
46 using the shared key and the content of the message. If the two codes match, the message is verified.
47 
48 HMAC has been used as a password encryption scheme. The final output of the HMAC algorithm depends on the shared key
49 and an inner hash. This inner hash is generated from the message content and the key. To use HMAC as a password
50 scheme, the key should be the username; the message should be the user's password. The authenticating party (for
51 instance, a login server) only needs to store this inner hash generated by the innerHash() function. When requesting
52 authentication, the user calculates a HMAC using this key and message and sends his username and this HMAC to the
53 authenticator. The authenticator can then use verify() using the provided HMAC and the stored inner hash. When using
54 this scheme, the password is never stored or transmitted in plain text.
55 */

92 void QxtHmac::setKey(QByteArray key)
93 {
94  // We make the assumption that all hashes use a 512-bit block size; as of Qt 4.4.0 this is true of all supported hash functions
95  QxtHmacPrivate* d = &qxt_d();
96  d->opad = QByteArray(64, 0x5c);
97  d->ipad = QByteArray(64, 0x36);
98  if (key.size() > 64)
99  {
100  key = QCryptographicHash::hash(key, d->algorithm);
101  }
102  for (int i = key.size() - 1; i >= 0; --i)
103  {
104  d->opad[i] = d->opad[i] ^ key[i];
105  d->ipad[i] = d->ipad[i] ^ key[i];
106  }
107  reset();
108 }

139 QByteArray QxtHmac::result()
140 {
141  QxtHmacPrivate* d = &qxt_d();
142  Q_ASSERT(d->opad.size());
143  if (d->result.size())
144  return d->result;
145  d->ohash->reset();
146  d->ohash->addData(d->opad);
147  d->ohash->addData(innerHash());
148  d->result = d->ohash->result();
149  return d->result;
150 }

199 bool QxtHmac::verify(const QByteArray& key, const QByteArray& hmac, const QByteArray& inner, Algorithm algorithm)
200 {
201  QxtHmac calc(algorithm);
202  calc.setKey(key);
203 
204  QxtHmacPrivate* d = &calc.qxt_d();
205  d->ohash->reset();
