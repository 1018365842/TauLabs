186 void ConfigGadgetWidget::onAutopilotConnect() {
187 
188  QIcon* icon;
189  QWidget* qwd;
190 
191  qDebug()<<"ConfigGadgetWidget onAutopilotConnect";
192  // First of all, check what Board type we are talking to, and
193  // if necessary, remove/add tabs in the config gadget:
194  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
195  UAVObjectUtilManager* utilMngr = pm->getObject<UAVObjectUtilManager>();
196  if (utilMngr) {
197  int board = utilMngr->getBoardModel();
198  if ((board & 0xff00) == 1024) {
199  // CopterControl family
200  ftw->setCurrentIndex(ConfigGadgetWidget::hardware);
201 
202  // Delete the current hardware panel, replace with CopterControl/CC3D hardware configuration
203  ftw->removeTab(ConfigGadgetWidget::hardware);
204  icon = new QIcon();
205  icon->addFile(":/configgadget/images/hardware_normal.png", QSize(), QIcon::Normal, QIcon::Off);
206  icon->addFile(":/configgadget/images/hardware_selected.png", QSize(), QIcon::Selected, QIcon::Off);
207  qwd = new ConfigCCHWWidget(this);
208  ftw->insertTab(ConfigGadgetWidget::hardware, qwd, *icon, QString("Hardware"));
209  ftw->setCurrentIndex(ConfigGadgetWidget::hardware);

211  // Non-CopterControl family
212  ftw->setCurrentIndex(ConfigGadgetWidget::hardware);
213  ftw->removeTab(ConfigGadgetWidget::hardware);

234 void ConfigGadgetWidget::tabAboutToChange(int i, bool * proceed)
235 {
236  Q_UNUSED(i);
237  *proceed = true;
238  ConfigTaskWidget * wid = qobject_cast<ConfigTaskWidget *>(ftw->currentWidget());
239  if(!wid) {
240  return;
241  }
242 
243  // Check if widget is dirty (i.e. has unsaved changes), and if it does, then check if
244  // either an autopilot or a PipX/OPLink telemetry unit is connected.
245  if(wid->isDirty() &&
246  (wid->isAutopilotConnected()||
247  QString(wid->metaObject()->className()) == "ConfigPipXtremeWidget"))
248  {
249  int ans=QMessageBox::warning(this,tr("Unsaved changes"),tr("The tab you are leaving has unsaved changes,"
250  "if you proceed they may be lost."
251  "Do you still want to proceed?"),QMessageBox::Yes,QMessageBox::No);
252  if(ans==QMessageBox::No) {
253  *proceed=false;
254  } else {
255  wid->setDirty(false);
256  }
257  }
258 }
