93 lib3ds_chunk_read(Lib3dsChunk *c, Lib3dsIo *io)
94 {
95  ASSERT(c);

98  c->chunk=lib3ds_io_read_word(io);
99  c->size=lib3ds_io_read_dword(io);

227 Lib3dsBool
228 lib3ds_chunk_write_end(Lib3dsChunk *c, Lib3dsIo *io)
229 {
230  ASSERT(c);
231  c->size=lib3ds_io_tell(io) - c->cur;
232  lib3ds_io_seek(io, c->cur+2, LIB3DS_SEEK_SET);
233  if (!lib3ds_io_write_dword(io, c->size)) {
234  LIB3DS_ERROR_LOG;
235  return(LIB3DS_FALSE);
236  }
237 
238  c->cur+=c->size;
239  lib3ds_io_seek(io, c->cur, LIB3DS_SEEK_SET);
240  if (lib3ds_io_error(io)) {
241  LIB3DS_ERROR_LOG;
242  return(LIB3DS_FALSE);
243  }
244  return(LIB3DS_TRUE);
245 }
