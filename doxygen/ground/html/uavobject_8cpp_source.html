332 bool UAVObject::save(QFile& file)
333 {
334  QMutexLocker locker(mutex);
335  quint8 buffer[numBytes];
336  quint8 tmpId[4];
337 
338  // Write the object ID
339  qToLittleEndian<quint32>(objID, tmpId);
340  if ( file.write((const char*)tmpId, 4) == -1 )
341  {
342  return false;
343  }
344 
345  // Write the instance ID
346  if (!isSingleInst)
347  {
348  qToLittleEndian<quint16>(instID, tmpId);
349  if ( file.write((const char*)tmpId, 2) == -1 )
350  {
351  return false;
352  }
353  }
354 
355  // Write the data
356  pack(buffer);
357  if ( file.write((const char*)buffer, numBytes) == -1 )
358  {
359  return false;
360  }
361 
362  // Done
363  return true;
364 }

401 bool UAVObject::load(QFile& file)
402 {
403  QMutexLocker locker(mutex);
404  quint8 buffer[numBytes];
405  quint8 tmpId[4];
406 
407  // Read the object ID
408  if ( file.read((char*)tmpId, 4) != 4 )
409  {
410  return false;
411  }
412 
413  // Check that the IDs match
414  if (qFromLittleEndian<quint32>(tmpId) != objID)
415  {
416  return false;
417  }
418 
419  // Read the instance ID
420  if ( file.read((char*)tmpId, 2) != 2 )
421  {
422  return false;
423  }
424 
425  // Check that the IDs match
426  if (qFromLittleEndian<quint16>(tmpId) != instID)
427  {
428  return false;
429  }
430 
431  // Read and unpack the data
432  if ( file.read((char*)buffer, numBytes) != numBytes )
433  {
434  return false;
435  }
436  unpack(buffer);
437 
438  // Done
439  return true;
440 }
