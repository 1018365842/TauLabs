318 void QwtDialSimpleNeedle::drawNeedle( QPainter *painter,
319  double length, QPalette::ColorGroup colorGroup ) const
320 {
321  double knobWidth = 0.0;
322  double width = d_width;
323 
324  if ( d_style == Arrow )
325  {
