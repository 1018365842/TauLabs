266 int QwtNullPaintDevice::metric( PaintDeviceMetric metric ) const
267 {
268  static QPixmap pm;
269 
270  int value;
271 
272  switch ( metric )
273  {
274  case PdmWidth:
275  value = qMax( d_data->size.width(), 0 );
276  break;
277  case PdmHeight:
278  value = qMax( d_data->size.height(), 0 );
279  break;
280  case PdmNumColors:
281  value = 16777216;
282  break;
283  case PdmDepth:
284  value = 24;
285  break;
286  case PdmPhysicalDpiX:
287  case PdmDpiY:
288  case PdmPhysicalDpiY:
289  case PdmWidthMM:
290  case PdmHeightMM:
291  case PdmDpiX:
292  default:
293  value = 0;
294  }
295  return value;
296 
297 }
