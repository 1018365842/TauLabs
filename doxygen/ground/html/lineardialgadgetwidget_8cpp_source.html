119 void LineardialGadgetWidget::updateIndex(UAVObject *object1) {
120  // Double check that the field exists:
121  UAVObjectField* field = object1->getField(field1);
122  if (field) {
123  QString s;
124  if (field->isNumeric()) {
125  double v;
126  if(haveSubField1){
127  int indexOfSubField = field->getElementNames().indexOf(QRegExp(subfield1, Qt::CaseSensitive, QRegExp::FixedString));
128  v = field->getDouble(indexOfSubField)*factor;
129  }else
130  v = field->getDouble()*factor;
131  setIndex(v);
132  s.sprintf("%.*f",places,v);
133  }
134  if (field->isText()) {
135  s = field->getValue().toString();
136  if (fieldSymbol) {
137  // If we defined a symbol, we will look for a matching
138  // SVG element to display:
139  if (m_renderer->elementExists("symbol-" + s)) {
140  fieldSymbol->setElementId("symbol-" + s);
141  } else {
142  fieldSymbol->setElementId("symbol");
143  }
144  }
145  }
146 
147  if (fieldValue)
148  fieldValue->setPlainText(s);
149 
150  if (index && !dialTimer.isActive())
151  dialTimer.start();
152  } else {
153  qDebug() << "Wrong field, maybe an issue with object disconnection ?";
154  }
155 }

163 void LineardialGadgetWidget::setDialFile(QString dfn)
164 {
165  QGraphicsScene *l_scene = scene();
166  setBackgroundBrush(QBrush(Utils::StyleHelper::baseColor()));
167  if (QFile::exists(dfn) && m_renderer->load(dfn) && m_renderer->isValid() )
168  {
169  l_scene->clear(); // Beware: clear also deletes all objects
170  // which are currently in the scene
171  background = new QGraphicsSvgItem();
172  background->setSharedRenderer(m_renderer);
173  background->setElementId("background");
174  background->setFlags(QGraphicsItem::ItemClipsChildrenToShape|
175  QGraphicsItem::ItemClipsToShape);
176  l_scene->addItem(background);
177 
178  // The red/yellow/green zones are optional, we just
179  // test on the presence of "red"
180  if (m_renderer->elementExists("red")) {
181  // Order is important: red, then yellow then green
182  // overlayed on top of each other
183  red = new QGraphicsSvgItem();
184  red->setSharedRenderer(m_renderer);
185  red->setElementId("red");
186  red->setParentItem(background);
187  yellow = new QGraphicsSvgItem();
188  yellow->setSharedRenderer(m_renderer);
189  yellow->setElementId("yellow");
190  yellow->setParentItem(background);
191  green = new QGraphicsSvgItem();
192  green->setSharedRenderer(m_renderer);
193  green->setElementId("green");
194  green->setParentItem(background);
195  // In order to properly render the Green/Yellow/Red graphs, we need to find out
196  // the starting location of the bargraph rendering area:
197  QMatrix textMatrix = m_renderer->matrixForElement("bargraph");
198  qreal bgX = textMatrix.mapRect(m_renderer->boundsOnElement("bargraph")).x();
199  qreal bgY = textMatrix.mapRect(m_renderer->boundsOnElement("bargraph")).y();
200  bargraphSize = textMatrix.mapRect(m_renderer->boundsOnElement("bargraph")).width();
201  // Detect if the bargraph is vertical or horizontal.
202  qreal bargraphHeight = textMatrix.mapRect(m_renderer->boundsOnElement("bargraph")).height();
203  if (bargraphHeight > bargraphSize) {
204  verticalDial = true;
205  bargraphSize = bargraphHeight;
206  } else {
207  verticalDial = false;
208  }
209  // Now adjust the red/yellow/green zones:
210  double range = maxValue-minValue;
211 
212  green->resetTransform();
213  double greenScale = (greenMax-greenMin)/range;
214  double greenStart = verticalDial ? (maxValue-greenMax)/range*green->boundingRect().height() :
215  (greenMin-minValue)/range*green->boundingRect().width();
216  QTransform matrix;
217  matrix.reset();
218  if (verticalDial) {
219  matrix.scale(1,greenScale);
220  matrix.translate(bgX,(greenStart+bgY)/greenScale);
221  } else {
222  matrix.scale(greenScale,1);
223  matrix.translate((greenStart+bgX)/greenScale,bgY);
224  }
225  green->setTransform(matrix,false);
226 
227  yellow->resetTransform();
228  double yellowScale = (yellowMax-yellowMin)/range;
229  double yellowStart = verticalDial ? (maxValue-yellowMax)/range*yellow->boundingRect().height() :
230  (yellowMin-minValue)/range*yellow->boundingRect().width();
231  matrix.reset();
232  if (verticalDial) {
233  matrix.scale(1,yellowScale);
234  matrix.translate(bgX,(yellowStart+bgY)/yellowScale);
235  } else {
236  matrix.scale(yellowScale,1);
237  matrix.translate((yellowStart+bgX)/yellowScale,bgY);
238  }
239  yellow->setTransform(matrix,false);
240 
241  red->resetTransform();
242  double redScale = (redMax-redMin)/range;
243  double redStart = verticalDial ? (maxValue-redMax)/range*red->boundingRect().height() :
244  (redMin-minValue)/range*red->boundingRect().width();
245  matrix.reset();
246  if (verticalDial) {
247  matrix.scale(1,redScale);
248  matrix.translate(bgX,(redStart+bgY)/redScale);
249  } else {
250  matrix.scale(redScale,1);
251  matrix.translate((redStart+bgX)/redScale,bgY);
252  }
253  red->setTransform(matrix,false);
254 
255  } else {
256  red = NULL;
257  yellow = NULL;
258  green = NULL;
259  }
260 
261  // Check whether the dial wants to display a moving index:
262  if (m_renderer->elementExists("needle")) {
263  QMatrix textMatrix = m_renderer->matrixForElement("needle");
264  QRectF nRect = textMatrix.mapRect(m_renderer->boundsOnElement("needle"));
265  startX = nRect.x();
266  startY = nRect.y();
267  QTransform matrix;

272  index->setTransform(matrix,false);
273  index->setParentItem(background);
274  } else {
275  index = NULL;
276  }
277 
278  // Check whether the dial wants display its field name:

281  QRectF rect = textMatrix.mapRect(m_renderer->boundsOnElement("field"));
282  qreal startX = rect.x();

288  fieldName->setFont(QFont("Arial",(int)elHeight));

290  fieldName->setTransform(matrix,false);
291  fieldName->setParentItem(background);
292  } else {
293  fieldName = NULL;
294  }
295 
296  // Check whether the dial wants display the numeric value:
297  if (m_renderer->elementExists("value")) {

299  QRectF nRect = textMatrix.mapRect(m_renderer->boundsOnElement("value"));
300  qreal startX = nRect.x();

311  fieldValue = NULL;
312  }

314  // Check whether the dial wants to display the value as a

316  if (m_renderer->elementExists("symbol")) {
317  QMatrix textMatrix = m_renderer->matrixForElement("symbol");
318  qreal startX = textMatrix.mapRect(m_renderer->boundsOnElement("symbol")).x();
319  qreal startY = textMatrix.mapRect(m_renderer->boundsOnElement("symbol")).y();

417 // Take an input value and move the index accordingly
418 // Move is smooth, starts fast and slows down when
419 // approaching the target.
420 void LineardialGadgetWidget::moveIndex()
421 {
422  if (!index) { // Safeguard
423  dialTimer.stop();
424  return;
425  }
426  if ((abs((indexValue-indexTarget)*10) > 3)) {
427  indexValue += (indexTarget - indexValue)/5;
428  } else {
429  indexValue = indexTarget;
430  dialTimer.stop();
431  }
432  QTransform matrix;
433  index->resetTransform();
434  qreal trans = indexValue*bargraphSize/100;
435  if (verticalDial) {
436  matrix.translate(startX,trans+startY);
437  } else {
438  matrix.translate(trans+startX,startY);
439  }
440  index->setTransform(matrix,false);
441 
442  update();
443 }
