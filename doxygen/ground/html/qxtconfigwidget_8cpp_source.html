73 void QxtConfigDelegate::paint(QPainter* painter, const QStyleOptionViewItem& option, const QModelIndex& index) const
74 {
75  QStyleOptionViewItem opt = option;
76  if (hover)
77  {
78  QPalette::ColorGroup cg = (option.state & QStyle::State_Enabled) ? QPalette::Normal : QPalette::Disabled;
79  if (cg == QPalette::Normal && !(option.state & QStyle::State_Active))
80  cg = QPalette::Inactive;
81 
82  if (option.state & QStyle::State_Selected)
83  painter->fillRect(opt.rect, option.palette.brush(cg, QPalette::Highlight));
84  else if ((option.state & QStyle::State_MouseOver) && (option.state & QStyle::State_Enabled))
85  {
86  QColor color = option.palette.color(cg, QPalette::Highlight).light();
87  if (color == option.palette.color(cg, QPalette::Base))
88  color = option.palette.color(cg, QPalette::AlternateBase);
89  painter->fillRect(opt.rect, color);
90  }
91  else
92  painter->fillRect(opt.rect, option.palette.brush(cg, QPalette::Base));
93 
94  opt.showDecorationSelected = false;
95  opt.state &= ~QStyle::State_HasFocus;
96  opt.state &= ~QStyle::State_Selected;
97  }
98  QItemDelegate::paint(painter, opt, index);
99 }

137 void QxtConfigWidgetPrivate::relayout()
138 {
139  if (pos == QxtConfigWidget::North)
140  {
141  splitter->setOrientation(Qt::Vertical);
142  table->setRowCount(1);
143  table->setColumnCount(0);
144  table->horizontalHeader()->setResizeMode(QHeaderView::ResizeToContents);
145  table->verticalHeader()->setResizeMode(QHeaderView::Stretch);
146  table->setHorizontalScrollBarPolicy(Qt::ScrollBarAsNeeded);
147  table->setVerticalScrollBarPolicy(Qt::ScrollBarAlwaysOff);
148  }
149  else
150  {
151  splitter->setOrientation(Qt::Horizontal);
152  table->setRowCount(0);
153  table->setColumnCount(1);
154  table->horizontalHeader()->setResizeMode(QHeaderView::Stretch);
155  table->verticalHeader()->setResizeMode(QHeaderView::ResizeToContents);
156  table->setHorizontalScrollBarPolicy(Qt::ScrollBarAlwaysOff);
157  table->setVerticalScrollBarPolicy(Qt::ScrollBarAsNeeded);
158  }
159 
160  // clear
161  for (int i = splitter->count() - 1; i >= 0; --i)
162  {
163  splitter->widget(i)->setParent(0);
164  }
165 
166  // relayout
167  switch (pos)
168  {
169  case QxtConfigWidget::North:
170  // +-----------+
171  // | Icons |
172  // +-----------|
173  // | Stack |
174  // +-----------|
175  splitter->addWidget(table);
176  splitter->addWidget(stack);
177  break;
178 
179  case QxtConfigWidget::West:
180  // +---+-------+
181  // | I | |
182  // | c | |
183  // | o | Stack |
184  // | n | |
185  // | s | |
186  // +---+-------+
187  splitter->addWidget(table);
188  splitter->addWidget(stack);
189  break;
190 
191  case QxtConfigWidget::East:
192  // +-------+---+
193  // | | I |
194  // | | c |
195  // | Stack | o |
196  // | | n |
197  // | | s |
198  // +-------+---+
199  splitter->addWidget(stack);
200  splitter->addWidget(table);
201  break;
202 
203  default:
204  qWarning("QxtConfigWidgetPrivate::relayout(): unknown position");
205  break;
206  }
207 
208  if (pos == QxtConfigWidget::East)
209  {
210  splitter->setStretchFactor(0, 10);
211  splitter->setStretchFactor(1, 1);
212  }
213  else
214  {
215  splitter->setStretchFactor(0, 1);
216  splitter->setStretchFactor(1, 10);
217  }
218 }

384 int QxtConfigWidget::insertPage(int index, QWidget* page, const QIcon& icon, const QString& title)
385 {
386  if (!page)
387  {
388  qWarning("QxtConfigWidget::insertPage(): Attempt to insert null page");
389  return -1;
390  }
391 
392  index = qxt_d().stack->insertWidget(index, page);
393  const QString label = !title.isEmpty() ? title : page->windowTitle();
394  if (label.isEmpty())
395  qWarning("QxtConfigWidget::insertPage(): Inserting a page with an empty title");
396  QTableWidgetItem* item = new QTableWidgetItem(icon, label);
397  item->setToolTip(label);
398  if (qxt_d().pos == QxtConfigWidget::North)
399  {
400  qxt_d().table->model()->insertColumn(index);
401  qxt_d().table->setItem(0, index, item);
402  qxt_d().table->resizeRowToContents(0);
403  }
404  else
405  {
406  qxt_d().table->model()->insertRow(index);
407  qxt_d().table->setItem(index, 0, item);
408  qxt_d().table->resizeColumnToContents(0);
409  }
410  qxt_d().table->updateGeometry();
411  return index;
412 }
