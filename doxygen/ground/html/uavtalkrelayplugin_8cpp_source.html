76 void UavTalkRelayPlugin::readConfig(QSettings *qSettings, UAVConfigInfo *configInfo)
77 {
78  Q_UNUSED(configInfo)
79  qSettings->beginGroup(QLatin1String("General Settings"));
80  m_IpAddress = (qSettings->value(QLatin1String("ListeningInterface"),m_IpAddress).toString());
81  m_Port = (qSettings->value(QLatin1String("ListeningPort"), m_Port).toInt());
82  m_DefaultRule=(UavTalkRelayComon::accessType)qSettings->value(QLatin1String("Defaul Rule"),m_DefaultRule).toInt();
83  qSettings->endGroup();
84  int size=qSettings->beginReadArray(QLatin1String("Rule"));
85  for(int i=0;i < size;++i)
86  {
87  qSettings->setArrayIndex(i);
88  QString host=qSettings->value(QLatin1String("Host"),tr("")).toString();
89  quint32 uavo=qSettings->value(QLatin1String("UAVO"),tr("")).toInt();
90  UavTalkRelayComon::accessType aType=(UavTalkRelayComon::accessType)qSettings->value(QLatin1String("Access Type"),tr("")).toInt();
91  if(rules.keys().contains(host))
92  {
93  QHash <quint32,UavTalkRelayComon::accessType> temp=rules.value(host);
94  temp.insert(uavo,aType);
95  }
96  else
97  {
98  QHash <quint32,UavTalkRelayComon::accessType> temp;
99  temp.insert(uavo,aType);
100  rules.insert(host,temp);
101  }
102  }
103  qSettings->endArray();
104 }

110 void UavTalkRelayPlugin::saveConfig(QSettings *qSettings, UAVConfigInfo *configInfo)
111 {
112  Q_UNUSED(configInfo)
113  qSettings->remove("Rule");
114  qSettings->beginGroup(QLatin1String("General Settings"));
115  qSettings->setValue(QLatin1String("ListeningInterface"), m_IpAddress);
116  qSettings->setValue(QLatin1String("ListeningPort"), m_Port);
117  qSettings->setValue(QLatin1String("Default Rule"),m_DefaultRule);
118  qSettings->endGroup();
119  qSettings->beginWriteArray(QLatin1String("Rule"));
120  int index=0;
121  foreach(QString host,rules.keys())
122  {
123  qSettings->setArrayIndex(index);
124  ++index;
125  foreach(quint32 uavo,rules.value(host).keys())
126  {
127  qSettings->setValue(QLatin1String("Host"),host);
128  qSettings->setValue(QLatin1String("UAVO"),uavo);
129  qSettings->setValue(QLatin1String("Access Type"),rules.value(host).value(uavo));
130  }
131  }
132  qSettings->endArray();
133  Q_ASSERT(relay);
134  relay->setIpAdress(m_IpAddress);
135  relay->setPort(m_Port);
136  relay->setRules(rules);
137  relay->restartServer();
138 }
