239 double QwtWheel::getValue( const QPoint &p )
240 {
241  const QRectF rect = wheelRect();
242 
243  // The reference position is arbitrary, but the
244  // sign of the offset is important
245  double w, dx;
246  if ( orientation() == Qt::Vertical )
247  {
248  w = rect.height();
249  dx = rect.y() - p.y();
250  }
251  else
252  {
253  w = rect.width();
254  dx = p.x() - rect.x();
255  }
256 
257  if ( w == 0.0 )
258  return 0.0;
259 
260  // w pixels is an arc of viewAngle degrees,
261  // so we convert change in pixels to change in angle
262  const double ang = dx * d_data->viewAngle / w;
263 
264  // value range maps to totalAngle degrees,
265  // so convert the change in angle to a change in value
266  const double val = ang * ( maxValue() - minValue() ) / d_data->totalAngle;
267 
268  // Note, range clamping and rasterizing to step is automatically
269  // handled by QwtAbstractSlider, so we simply return the change in value
270  return val;
271 }

313 {
314  painter->save();
315

317 
318  // draw shaded background
319  QLinearGradient gradient( rect.topLeft(),

321  gradient.setColorAt( 0.0, pal.color( QPalette::Button ) );
322  gradient.setColorAt( 0.2, pal.color( QPalette::Light ) );
323  gradient.setColorAt( 0.7, pal.color( QPalette::Mid ) );
324  gradient.setColorAt( 1.0, pal.color( QPalette::Dark ) );
325 
326  painter->fillRect( rect, gradient );
327 
328  // draw internal border
329 
330  const QPen lightPen( palette().color( QPalette::Light ),
331  d_data->wheelBorderWidth, Qt::SolidLine, Qt::FlatCap );
332  const QPen darkPen( pal.color( QPalette::Dark ),
333  d_data->wheelBorderWidth, Qt::SolidLine, Qt::FlatCap );
334 
335  const double bw2 = 0.5 * d_data->wheelBorderWidth;
336 
337  if ( orientation() == Qt::Horizontal )
338  {
339  painter->setPen( lightPen );
340  painter->drawLine( rect.left(), rect.top() + bw2,
341  rect.right(), rect.top() + bw2 );
342 
343  painter->setPen( darkPen );
344  painter->drawLine( rect.left(), rect.bottom() - bw2,
345  rect.right(), rect.bottom() - bw2 );
346  }
347  else // Qt::Vertical
348  {
349  painter->setPen( lightPen );
350  painter->drawLine( rect.left() + bw2, rect.top(),
351  rect.left() + bw2, rect.bottom() );
352 
353  painter->setPen( darkPen );
354  painter->drawLine( rect.right() - bw2, rect.top(),
355  rect.right() - bw2, rect.bottom() );
356  }

367 void QwtWheel::drawTicks( QPainter *painter, const QRectF &rect )
368 {
369  if ( maxValue() == minValue() || d_data->totalAngle == 0.0 )
370  return;
371 
372  const QPen lightPen( palette().color( QPalette::Light ),
373  0, Qt::SolidLine, Qt::FlatCap );
374  const QPen darkPen( palette().color( QPalette::Dark ),
375  0, Qt::SolidLine, Qt::FlatCap );
376 
377  const double sign = ( minValue() < maxValue() ) ? 1.0 : -1.0;
378  const double cnvFactor = qAbs( d_data->totalAngle / ( maxValue() - minValue() ) );
379  const double halfIntv = 0.5 * d_data->viewAngle / cnvFactor;
380  const double loValue = value() - halfIntv;
381  const double hiValue = value() + halfIntv;
382  const double tickWidth = 360.0 / double( d_data->tickCnt ) / cnvFactor;
383  const double sinArc = qFastSin( d_data->viewAngle * M_PI / 360.0 );
384 
385  if ( orientation() == Qt::Horizontal )
386  {
387  const double halfSize = rect.width() * 0.5;
388 
389  double l1 = rect.top() + d_data->wheelBorderWidth;
390  double l2 = rect.bottom() - d_data->wheelBorderWidth - 1;
391 
392  // draw one point over the border if border > 1
393  if ( d_data->wheelBorderWidth > 1 )
394  {
395  l1--;
396  l2++;
397  }
398 
399  const double maxpos = rect.right() - 2;
400  const double minpos = rect.left() + 2;
401 
402  // draw tick marks
403  for ( double tickValue = qwtCeilF( loValue / tickWidth ) * tickWidth;
404  tickValue < hiValue; tickValue += tickWidth )
405  {
406  const double angle = ( tickValue - value() ) * M_PI / 180.0;
407  const double s = qFastSin( angle * cnvFactor );
408 
409  const double tickPos =
410  rect.right() - halfSize * ( sinArc + sign * s ) / sinArc;
411 
412  if ( ( tickPos <= maxpos ) && ( tickPos > minpos ) )
413  {
414  painter->setPen( darkPen );
415  painter->drawLine( tickPos - 1 , l1, tickPos - 1, l2 );
416  painter->setPen( lightPen );
417  painter->drawLine( tickPos, l1, tickPos, l2 );
418  }
419  }
420  }
421  else // Qt::Vertical
422  {
423  const double halfSize = rect.height() * 0.5;
424 
425  double l1 = rect.left() + d_data->wheelBorderWidth;
426  double l2 = rect.right() - d_data->wheelBorderWidth - 1;
427 
428  if ( d_data->wheelBorderWidth > 1 )
429  {
430  l1--;
431  l2++;
432  }
433 
434  const double maxpos = rect.bottom() - 2;
435  const double minpos = rect.top() + 2;
436 
437  for ( double tickValue = qwtCeilF( loValue / tickWidth ) * tickWidth;
438  tickValue < hiValue; tickValue += tickWidth )
439  {
440  const double angle = ( tickValue - value() ) * M_PI / 180.0;
441  const double s = qFastSin( angle * cnvFactor );
442 
443  const double tickPos =
444  rect.y() + halfSize * ( sinArc + sign * s ) / sinArc;
445 
446  if ( ( tickPos <= maxpos ) && ( tickPos > minpos ) )
447  {
448  painter->setPen( darkPen );
449  painter->drawLine( l1, tickPos - 1 , l2, tickPos - 1 );
450  painter->setPen( lightPen );
451  painter->drawLine( l1, tickPos, l2, tickPos );
452  }
453  }
454  }
455 }
