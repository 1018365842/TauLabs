252 {
253  myDevice->verticalGroupBox_loaded->setVisible(false);
254  myDevice->groupCustom->setVisible(false);
255 
256  filename = setOpenFileName();
257 
258  if (filename.isEmpty()) {
259  status("Empty filename", STATUSICON_FAIL);

261  }
262 
263  QFile file(filename);
264  if (!file.open(QIODevice::ReadOnly)) {

267  }
268

272  QByteArray desc = loadedFW.right(100);

275  myDevice->lblCRCL->setText(tr("Can't calculate, file too big for device"));
276  else

278  //myDevice->lblFirmwareSizeL->setText(QString("Firmware size: ")+QVariant(loadedFW.length()).toString()+ QString(" bytes"));
279  if (populateLoadedStructuredDescription(desc))
280  {
281  myDevice->youdont->setChecked(true);
282  myDevice->verticalGroupBox_loaded->setVisible(true);
283  myDevice->groupCustom->setVisible(false);
284  if(myDevice->lblCRC->text()==myDevice->lblCRCL->text())
285  {
286  myDevice->statusLabel->setText(tr("The board has the same firmware as loaded. No need to update"));
287  px.load(QString(":/uploader/images/warning.svg"));
288  }
289  else if(myDevice->lblDevName->text()!=myDevice->lblBrdNameL->text())
290  {
291  myDevice->statusLabel->setText(tr("WARNING: the loaded firmware is for different hardware. Do not update!"));
292  px.load(QString(":/uploader/images/error.svg"));
293  }
294  else if(QDateTime::fromString(onBoardDescription.gitDate)>QDateTime::fromString(LoadedDescription.gitDate))
295  {
296  myDevice->statusLabel->setText(tr("The board has newer firmware than loaded. Are you sure you want to update?"));

300  {
301  myDevice->statusLabel->setText(tr("The loaded firmware is untagged or custom build. Update only if it was received from a trusted source (official website or your own build)"));

309  }

312  myDevice->statusLabel->setText(tr("WARNING: the loaded firmware was not packaged with the Tau Labs format. Do not update unless you know what you are doing"));
313  px.load(QString(":/uploader/images/error.svg"));
314  myDevice->youdont->setChecked(false);
315  myDevice->youdont->setVisible(true);

325 void deviceWidget::uploadFirmware()
326 {
327  freeze();
328  if (!m_dfu->devices[deviceID].Writable) {
329  status("Device not writable!", STATUSICON_FAIL);
330  unfreeze();
331  return;
332  }
333 
334  bool verify = false;
335  /* TODO: does not work properly on current Bootloader!
336  if (m_dfu->devices[deviceID].Readable)
337  verify = true;
338  */
339 
340  QByteArray desc = loadedFW.right(100);
341  if (desc.startsWith("TlFw") || desc.startsWith("OpFw")) {
342  descriptionArray = desc;
343  // Now do sanity checking:
344  // - Check whether board type matches firmware:
345  int board = m_dfu->devices[deviceID].ID;
346  int firmwareBoard = ((desc.at(12)&0xff)<<8) + (desc.at(13)&0xff);
347  if((board == 0x401 && firmwareBoard == 0x402) ||
348  (board == 0x901 && firmwareBoard == 0x902) || // L3GD20 revo supports Revolution firmware
349  (board == 0x902 && firmwareBoard == 0x903)) // RevoMini1 supporetd by RevoMini2 firmware
350  {
351  // These firmwares are designed to be backwards compatible
352  } else if (firmwareBoard != board) {
353  status("Error: firmware does not match board", STATUSICON_FAIL);
354  unfreeze();
355  return;
356  }
357  // Check the firmware embedded in the file:
358  QByteArray firmwareHash = desc.mid(40,20);
359  QByteArray fileHash = QCryptographicHash::hash(loadedFW.left(loadedFW.length()-100), QCryptographicHash::Sha1);
360  if (firmwareHash != fileHash) {
361  status("Error: firmware file corrupt", STATUSICON_FAIL);
362  unfreeze();
363  return;
364  }
365  } else {
366  // The firmware is not packaged, just upload the text in the description field
367  // if it is there.
368  descriptionArray.clear();
369  }
370 
371 
372  status("Starting firmware upload", STATUSICON_RUNNING);
373  // We don't know which device was used previously, so we
374  // are cautious and reenter DFU for this deviceID:
375  emit uploadStarted();
376  if(!m_dfu->enterDFU(deviceID))
377  {
378  status("Error:Could not enter DFU mode", STATUSICON_FAIL);
379  unfreeze();
380  emit uploadEnded(false);
381  return;
382  }
383  OP_DFU::Status ret=m_dfu->StatusRequest();
384  qDebug() << m_dfu->StatusToString(ret);
385  m_dfu->AbortOperation(); // Necessary, otherwise I get random failures.
386 
387  connect(m_dfu, SIGNAL(progressUpdated(int)), this, SLOT(setProgress(int)));
388  connect(m_dfu, SIGNAL(operationProgress(QString)), this, SLOT(dfuStatus(QString)));
389  connect(m_dfu, SIGNAL(uploadFinished(OP_DFU::Status)), this, SLOT(uploadFinished(OP_DFU::Status)));
390  bool retstatus = m_dfu->UploadFirmware(filename,verify, deviceID);
391  if(!retstatus ) {
392  status("Could not start upload", STATUSICON_FAIL);
393  unfreeze();
394  emit uploadEnded(false);
395  return;
396  }
397  status("Uploading, please wait...", STATUSICON_RUNNING);
398 }

446 void deviceWidget::uploadFinished(OP_DFU::Status retstatus)
447 {
448  unfreeze();
449  disconnect(m_dfu, SIGNAL(uploadFinished(OP_DFU::Status)), this, SLOT(uploadFinished(OP_DFU::Status)));
450  disconnect(m_dfu, SIGNAL(progressUpdated(int)), this, SLOT(setProgress(int)));
451  disconnect(m_dfu, SIGNAL(operationProgress(QString)), this, SLOT(dfuStatus(QString)));
452  if(retstatus != OP_DFU::Last_operation_Success) {
453  status(QString("Upload failed with code: ") + m_dfu->StatusToString(retstatus).toLatin1().data(), STATUSICON_FAIL);
454  emit uploadEnded(false);
455  return;
456  } else
457  if (!descriptionArray.isEmpty()) {
458  // We have a structured array to save
459  status(QString("Updating description"), STATUSICON_RUNNING);
460  repaint(); // Make sure the text above shows right away
461  retstatus = m_dfu->UploadDescription(descriptionArray);
462  if( retstatus != OP_DFU::Last_operation_Success) {
463  status(QString("Upload failed with code: ") + m_dfu->StatusToString(retstatus).toLatin1().data(), STATUSICON_FAIL);
464  emit uploadEnded(false);
465  return;
466  }
467 
468  } else if (!myDevice->description->text().isEmpty()) {
469  // Fallback: we save the description field:
470  status(QString("Updating description"), STATUSICON_RUNNING);
471  repaint(); // Make sure the text above shows right away
472  retstatus = m_dfu->UploadDescription(myDevice->description->text());
473  if( retstatus != OP_DFU::Last_operation_Success) {
474  status(QString("Upload failed with code: ") + m_dfu->StatusToString(retstatus).toLatin1().data(), STATUSICON_FAIL);
475  emit uploadEnded(false);
476  return;
477  }
478  }
479  populate();
480  emit uploadEnded(true);
481  status("Upload successful", STATUSICON_OK);
482
