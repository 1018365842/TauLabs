148 void ConfigPipXtremeWidget::updateStatus(UAVObject *object)
149 {
150 
151  // Request and update of the setting object if we haven't received it yet.
152  if (!settingsUpdated)
153  oplinkSettingsObj->requestUpdate();
154 
155  // Get the current pairID
156  OPLinkSettings *oplinkSettings = OPLinkSettings::GetInstance(getObjectManager());
157  quint32 pairID = 0;
158  if (oplinkSettings)
159  pairID = oplinkSettings->getPairID();
160 
161  // Update the detected devices.
162  UAVObjectField* pairIdField = object->getField("PairIDs");
163  if (pairIdField) {
164  quint32 pairid1 = pairIdField->getValue(0).toUInt();
165  m_oplink->PairID1->setText(QString::number(pairid1, 16).toUpper());
166  m_oplink->PairID1->setEnabled(false);
167  m_oplink->PairSelect1->setChecked(pairID && (pairID == pairid1));
168  m_oplink->PairSelect1->setEnabled(pairid1);
169  quint32 pairid2 = pairIdField->getValue(1).toUInt();
170  m_oplink->PairID2->setText(QString::number(pairIdField->getValue(1).toUInt(), 16).toUpper());
171  m_oplink->PairID2->setEnabled(false);
172  m_oplink->PairSelect2->setChecked(pairID && (pairID == pairid2));
173  m_oplink->PairSelect2->setEnabled(pairid2);
174  quint32 pairid3 = pairIdField->getValue(2).toUInt();
175  m_oplink->PairID3->setText(QString::number(pairIdField->getValue(2).toUInt(), 16).toUpper());
176  m_oplink->PairID3->setEnabled(false);
177  m_oplink->PairSelect3->setChecked(pairID && (pairID == pairid3));
178  m_oplink->PairSelect3->setEnabled(pairid3);
179  quint32 pairid4 = pairIdField->getValue(3).toUInt();
180  m_oplink->PairID4->setText(QString::number(pairIdField->getValue(3).toUInt(), 16).toUpper());
181  m_oplink->PairID4->setEnabled(false);
182  m_oplink->PairSelect4->setChecked(pairID && (pairID == pairid4));
183  m_oplink->PairSelect4->setEnabled(pairid4);
184  } else {
185  qDebug() << "PipXtremeGadgetWidget: Count not read PairID field.";
186  }
187  UAVObjectField* pairRssiField = object->getField("PairSignalStrengths");
188  if (pairRssiField) {
189  m_oplink->PairSignalStrengthBar1->setValue(pairRssiField->getValue(0).toInt());
190  m_oplink->PairSignalStrengthBar2->setValue(pairRssiField->getValue(1).toInt());
191  m_oplink->PairSignalStrengthBar3->setValue(pairRssiField->getValue(2).toInt());
192  m_oplink->PairSignalStrengthBar4->setValue(pairRssiField->getValue(3).toInt());
193  m_oplink->PairSignalStrengthLabel1->setText(QString("%1dB").arg(pairRssiField->getValue(0).toInt()));
194  m_oplink->PairSignalStrengthLabel2->setText(QString("%1dB").arg(pairRssiField->getValue(1).toInt()));
195  m_oplink->PairSignalStrengthLabel3->setText(QString("%1dB").arg(pairRssiField->getValue(2).toInt()));
196  m_oplink->PairSignalStrengthLabel4->setText(QString("%1dB").arg(pairRssiField->getValue(3).toInt()));
197  } else {
198  qDebug() << "PipXtremeGadgetWidget: Count not read PairID field.";
199  }
200 
201  // Update the Description field
202  UAVObjectField* descField = object->getField("Description");
203  if (descField) {
204  /*
205  * This looks like a binary with a description at the end
206  * 4 bytes: header: "TlFw"
207  * 4 bytes: git commit hash (short version of SHA1)
208  * 4 bytes: Unix timestamp of last git commit
209  * 2 bytes: target platform. Should follow same rule as BOARD_TYPE and BOARD_REVISION in board define files.
210  * 26 bytes: commit tag if it is there, otherwise "Unreleased". Zero-padded
211  * ---- 40 bytes limit ---
212  * 20 bytes: SHA1 sum of the firmware.
213  * 40 bytes: free for now.
214  */
215  char buf[OPLinkStatus::DESCRIPTION_NUMELEM];
216  for (unsigned int i = 0; i < 26; ++i)
217  buf[i] = descField->getValue(i + 14).toChar().toAscii();
218  buf[26] = '\0';
219  QString descstr(buf);
220  quint32 gitDate = descField->getValue(11).toChar().toAscii() & 0xFF;
221  for (int i = 1; i < 4; i++) {
222  gitDate = gitDate << 8;
223  gitDate += descField->getValue(11-i).toChar().toAscii() & 0xFF;
224  }
225  QString date = QDateTime::fromTime_t(gitDate).toUTC().toString("yyyy-MM-dd HH:mm");
226  m_oplink->FirmwareVersion->setText(descstr + " " + date);
227  } else {
228  qDebug() << "PipXtremeGadgetWidget: Count not read Description field.";
229  }
230 
231  // Update the serial number field
232  UAVObjectField* serialField = object->getField("CPUSerial");
233  if (serialField) {
234  char buf[OPLinkStatus::CPUSERIAL_NUMELEM * 2 + 1];
235  for (unsigned int i = 0; i < OPLinkStatus::CPUSERIAL_NUMELEM; ++i)
236  {
237  unsigned char val = serialField->getValue(i).toUInt() >> 4;
238  buf[i * 2] = ((val < 10) ? '0' : '7') + val;
239  val = serialField->getValue(i).toUInt() & 0xf;
240  buf[i * 2 + 1] = ((val < 10) ? '0' : '7') + val;
241  }
242  buf[OPLinkStatus::CPUSERIAL_NUMELEM * 2] = '\0';
243  m_oplink->SerialNumber->setText(buf);
244  } else {
245  qDebug() << "PipXtremeGadgetWidget: Count not read Description field.";
246  }
247 
248  // Update the DeviceID field
249  UAVObjectField* idField = object->getField("DeviceID");
250  if (idField) {
251  m_oplink->DeviceID->setText(QString::number(idField->getValue().toUInt(), 16).toUpper());
252  } else {
253  qDebug() << "PipXtremeGadgetWidget: Count not read DeviceID field.";
254  }
255 
256  // Update the PairID field
257  m_oplink->PairID->setText(QString::number(pairID, 16).toUpper());
258 
259  // Update the link state
260  UAVObjectField* linkField = object->getField("LinkState");
261  if (linkField) {
262  m_oplink->LinkState->setText(linkField->getValue().toString());
263  } else {
264  qDebug() << "PipXtremeGadgetWidget: Count not read link state field.";
265  }
266 }
