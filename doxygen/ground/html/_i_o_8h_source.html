124 template<typename Derived>
125 std::ostream & ei_print_matrix(std::ostream & s, const Derived& _m, const IOFormat& fmt)
126 {
127  const typename Derived::Nested m = _m;
128 
129  int width = 0;
130  if (fmt.flags & AlignCols)
131  {
132  // compute the largest width
133  for(int j = 1; j < m.cols(); ++j)
134  for(int i = 0; i < m.rows(); ++i)
135  {
136  std::stringstream sstr;
137  sstr.precision(fmt.precision);
138  sstr << m.coeff(i,j);
139  width = std::max<int>(width, int(sstr.str().length()));
140  }
141  }
142  s.precision(fmt.precision);
143  s << fmt.matPrefix;
144  for(int i = 0; i < m.rows(); ++i)
145  {
146  if (i)
147  s << fmt.rowSpacer;
148  s << fmt.rowPrefix;
149  if(width) s.width(width);
150  s << m.coeff(i, 0);
151  for(int j = 1; j < m.cols(); ++j)
152  {
153  s << fmt.coeffSeparator;
154  if (width) s.width(width);
155  s << m.coeff(i, j);
156  }
157  s << fmt.rowSuffix;
158  if( i < m.rows() - 1)
159  s << fmt.rowSeparator;
160  }
161  s << fmt.matSuffix;
162  return s;
163 }
