53 bool UAVObjectGeneratorPython::process_object(ObjectInfo* info)
54 {
55  if (info == NULL)
56  return false;
57 
58  // Prepare output strings
59  QString outCode = pythonCodeTemplate;
60 
61  // Replace common tags
62  replaceCommonTags(outCode, info);
63 
64  // Replace the ($DATAFIELDS) tag
65  QString datafields;
66  for (int n = 0; n < info->fields.length(); ++n)
67  {
68  // Class header
69  datafields.append(QString("# Field %1 definition\n").arg(info->fields[n]->name));
70  datafields.append(QString("class %1Field(UAVObjectField):\n").arg(info->fields[n]->name));
71  // Only for enum types
72  if (info->fields[n]->type == FIELDTYPE_ENUM)
73  {
74  datafields.append(QString("\t# Enumeration options\n"));
75  // Go through each option
76  QStringList options = info->fields[n]->options;
77  for (int m = 0; m < options.length(); ++m) {
78  QString name = options[m].toUpper().replace(QRegExp(ENUM_SPECIAL_CHARS), "");
79  if (name[0].isDigit())
80  name = QString("N%1").arg(name);
81  datafields.append(QString("\t%1 = %2\n").arg(name).arg(m));
82  }
83  }
84  // Generate element names (only if field has more than one element)
85  if (info->fields[n]->numElements > 1 && !info->fields[n]->defaultElementNames)
86  {
87  datafields.append(QString("\t# Array element names\n"));
88  // Go through the element names
89  QStringList elemNames = info->fields[n]->elementNames;
90  for (int m = 0; m < elemNames.length(); ++m)
91  {
92  QString name = elemNames[m].toUpper().replace(QRegExp(ENUM_SPECIAL_CHARS), "");
93  if (name[0].isDigit())
94  name = QString("N%1").arg(name);
95  datafields.append(QString("\t%1 = %2\n").arg(name).arg(m));
96  }
97  }
98  // Constructor
99  datafields.append(QString("\tdef __init__(self):\n"));
100  datafields.append(QString("\t\tUAVObjectField.__init__(self, %1, %2)\n\n").arg(info->fields[n]->type).arg(info->fields[n]->numElements));
101  }
102  outCode.replace(QString("$(DATAFIELDS)"), datafields);
103 
104  // Replace the $(DATAFIELDINIT) tag
105  QString fields;
106  for (int n = 0; n < info->fields.length(); ++n)
107  {
108  fields.append(QString("\t\tself.%1 = %1Field()\n").arg(info->fields[n]->name));
109  fields.append(QString("\t\tself.addField(self.%1)\n").arg(info->fields[n]->name));
110  }
111  outCode.replace(QString("$(DATAFIELDINIT)"), fields);
112 
113  // Write the Python code
114  bool res = writeFileIfDiffrent( pythonOutputPath.absolutePath() + "/" + info->namelc + ".py", outCode );
115  if (!res) {
116  cout << "Error: Could not write Python output files" << endl;
117  return false;
118  }
119 
120  return true;
121 }
