57 template<> inline int ei_random(int a, int b)
58 {
59  // We can't just do rand()%n as only the high-order bits are really random
60  return a + static_cast<int>((b-a+1) * (std::rand() / (RAND_MAX + 1.0)));
61 }

99 {
100 #ifdef EIGEN_NICE_RANDOM
101  int i;
102  do { i = ei_random<int>(256*int(a),256*int(b));
103  } while(i==0);
104  return float(i)/256.f;

285 template<typename T> inline T ei_hypot(T x, T y)
286 {
287  T _x = ei_abs(x);
288  T _y = ei_abs(y);
289  T p = std::max(_x, _y);
290  T q = std::min(_x, _y);
291  T qp = q/p;
292  return p * ei_sqrt(T(1) + qp*qp);
293 }
