56 void QxtScheduleItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
57 {
58 
59  const QxtStyleOptionScheduleViewItem *agendaOption = qstyleoption_cast<const QxtStyleOptionScheduleViewItem *>(&option);
60  if (!agendaOption)
61  return;
62 
63  QStringList rowsData = index.data(Qt::EditRole).toStringList();
64 
65  QRect currRect;
66 
67  painter->save();
68 
69  if (agendaOption->itemPaintCache->size() != agendaOption->itemGeometries.size())
70  (*agendaOption->itemPaintCache) = QVector<QPixmap>(agendaOption->itemGeometries.size(), QPixmap());
71 
72  int lastPart = agendaOption->itemGeometries.size() - 1;
73  int paintedSubItems = 0;
74 
75  for (int iLoop = 0; iLoop < agendaOption->itemGeometries.size();iLoop++)
76  {
77  if ((*agendaOption->itemPaintCache)[iLoop].width() != agendaOption->itemGeometries[iLoop].width()
78  || (*agendaOption->itemPaintCache)[iLoop].height() != agendaOption->itemGeometries[iLoop].height())
79  {
80  //If we enter this codepath we have to rebuild the pixmap cache
81  //so first we create a empty pixmap
82  (*agendaOption->itemPaintCache)[iLoop] = QPixmap(agendaOption->itemGeometries[iLoop].size());
83  (*agendaOption->itemPaintCache)[iLoop].fill(Qt::transparent);
84 
85  QPainter cachePainter(&(*agendaOption->itemPaintCache)[iLoop]);
86  QRect rect = QRect(QPoint(0, 0), agendaOption->itemGeometries[iLoop].size());
87 
88  //what kind of itempart do we need to paint?
89  ItemPart part = iLoop == 0 ? Top : (iLoop == lastPart ? Bottom : Middle);
90 
91  //if the item has only one part
92  if (lastPart == 0)
93  part = Single;
94 
95  //paint the item body
96  cachePainter.save();
97  paintItemBody(&cachePainter, rect, *agendaOption, part, index);
98  cachePainter.restore();
99 
100  int remainingHeight = rect.height();
101 
102  //paint item header
103  if (iLoop == 0 && agendaOption->itemHeaderHeight > 0 && agendaOption->itemHeaderHeight < remainingHeight)
104  {
105  QRect headerRect(0, 0, rect.width(), agendaOption->itemHeaderHeight);
106  paintItemHeader(&cachePainter, headerRect, *agendaOption, index);
107  remainingHeight -= agendaOption->itemHeaderHeight;
108  }
109 
110  //paint subitems if there are any
111  int subItems = index.model()->rowCount(index);
112  for (int items = paintedSubItems; items < subItems; items++)
113  {
114  QModelIndex currSubItem = index.model()->index(items, 0, index);
115  QSize size = sizeHint(option, currSubItem);
116 
117  if (currSubItem.isValid()){
118  paintSubItem(&cachePainter, QRect(), *agendaOption, currSubItem);
119  }
120 
121  paintedSubItems++;
122  }
123 
124  cachePainter.end();
125 
126  }
127  currRect = agendaOption->itemGeometries[iLoop];
128  currRect.translate(agendaOption->translate);
129  painter->drawPixmap(currRect, (*agendaOption->itemPaintCache)[iLoop]);
130  }
131  painter->restore();
132 }

143 {

176 void QxtScheduleItemDelegate::paintItemHeader(QPainter *painter, const QRect rect , const QxtStyleOptionScheduleViewItem & option, const QModelIndex &index) const
177 {
178  bool converted = false;
179  int startUnixTime = index.data(Qxt::ItemStartTimeRole).toInt(&converted);
180  if (!converted)
181  return;
182 
183  int duration = index.data(Qxt::ItemDurationRole).toInt(&converted);
184  if (!converted)
185  return;
186 
187  QDateTime startTime = QDateTime::fromTime_t(startUnixTime);
188  QDateTime endTime = QDateTime::fromTime_t(startUnixTime + duration);
189 
190  if (!startTime.isValid() || !endTime.isValid())
191  return;
192 
193  QFont font;
194  QVariant vfont = index.data(Qt::FontRole);
195 
196  if (vfont.isValid())
197  font = vfont.value<QFont>();
198  else
199  font = option.font;
200 
201  QString text = startTime.toString("hh:mm") + ' ' + endTime.toString("hh:mm");
202  QFontMetrics metr(font);
203  text = metr.elidedText(text, Qt::ElideRight, rect.width());
204  painter->drawText(rect, Qt::AlignCenter, text);
205 }

250 void QxtScheduleItemDelegate::createPainterPath(QPainterPath & emptyPath, const QRect & fullItemRect, const int iRoundTop, const int iRoundBottom) const
251 {
252  emptyPath = QPainterPath();
253  bool bRoundTop = iRoundTop > 0;
254  bool bRountBottom = iRoundBottom > 0;
255 
256  if (bRoundTop)
257  {
258  emptyPath.moveTo(fullItemRect.topLeft() + QPoint(0, iRoundTop));
259  emptyPath.quadTo(fullItemRect.topLeft(), fullItemRect.topLeft() + QPoint(iRoundTop, 0));
260  }
261  else
262  emptyPath.moveTo(fullItemRect.topLeft());
263 
264  emptyPath.lineTo(fullItemRect.topRight() - QPoint(iRoundTop, 0));
265 
266  if (bRoundTop)
267  emptyPath.quadTo(fullItemRect.topRight(), fullItemRect.topRight() + QPoint(0, iRoundTop));
268 
269  emptyPath.lineTo(fullItemRect.bottomRight() - QPoint(0, iRoundBottom));
270 
271  if (bRountBottom)
272  emptyPath.quadTo(fullItemRect.bottomRight(), fullItemRect.bottomRight() - QPoint(iRoundBottom, 0));
273 
274  emptyPath.lineTo(fullItemRect.bottomLeft() + QPoint(iRoundBottom, 0));
275 
276  if (bRountBottom)
277  emptyPath.quadTo(fullItemRect.bottomLeft(), fullItemRect.bottomLeft() - QPoint(0, iRoundBottom));
278 
279  emptyPath.closeSubpath();
280 }
