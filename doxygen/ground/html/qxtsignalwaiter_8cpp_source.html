115 bool QxtSignalWaiter::wait(int msec, QEventLoop::ProcessEventsFlags flags)
116 {
117  QxtSignalWaiterPrivate& d = qxt_d();
118 
119  // Clear the emission status
120  d.ready = false;
121  d.emitted = false;
122 
123  // Check input parameters
124  if (msec < -1 || msec == 0)
125  return false;
126 
127  // activate the timeout
128  if (msec != -1)
129  d.timerID = startTimer(msec);
130  else
131  d.timerID = 0;
132 
133  // Make sure to wait for events
134  flags |= QEventLoop::WaitForMoreEvents;
135 
136  // Begin waiting
137  d.waiting = true;
138  while (!d.ready && !d.timeout)
139  QCoreApplication::processEvents(flags);
140 
141  // Clean up and return status
142  qxt_d().stopTimer();
143  d.emitted = d.ready;
144  d.waiting = false;
145  return d.ready;
146 }
