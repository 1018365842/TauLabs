190 void Run_BlinkLEDs(const simToPlugin *stp,
191  pluginToSim *pts)
192 {
193  if ((stp->simMenuStatus & MenuEnable) != 0) {
194  pts->newMenuStatus |= MenuLedGreen;
195  int timeout;
196  quint8 armed;
197  quint8 mode;
198  rcvr->getFlighStatus(armed, mode);
199  debugInfo.append(QString("armed: %1, mode: %2\n").arg(armed).arg(mode));
200 
201  if (armed == 0) // disarm
202  timeout = 1000;
203  else if (armed == 1) // arming
204  timeout = 40;
205  else if (armed == 2) // armed
206  timeout = 100;
207  else // unknown
208  timeout = 2000;
209  if (ledTimer.elapsed() > timeout) {
210  ledTimer.restart();
211  pts->newMenuStatus ^= MenuLedBlue;
212  }
213 
214  if (mode == 6) {
215  pts->newMenuStatus |= MenuFMode3;
216  pts->newMenuStatus |= MenuFMode2;
217  pts->newMenuStatus |= MenuFMode1;
218  } else if (mode == 5) {
219  pts->newMenuStatus |= MenuFMode3;
220  pts->newMenuStatus |= MenuFMode2;
221  pts->newMenuStatus &= ~MenuFMode1;
222  } else if (mode == 4) {
223  pts->newMenuStatus |= MenuFMode3;
224  pts->newMenuStatus &= ~MenuFMode2;
225  pts->newMenuStatus |= MenuFMode1;
226  } else if (mode == 3) {
227  pts->newMenuStatus |= MenuFMode3;
228  pts->newMenuStatus &= ~MenuFMode2;
229  pts->newMenuStatus &= ~MenuFMode1;
230  } else if (mode == 2) {
231  pts->newMenuStatus &= ~MenuFMode3;
232  pts->newMenuStatus |= MenuFMode2;
233  pts->newMenuStatus &= ~MenuFMode1;
234  } else if (mode == 1) {
235  pts->newMenuStatus &= ~MenuFMode3;
236  pts->newMenuStatus &= ~MenuFMode2;
237  pts->newMenuStatus |= MenuFMode1;
238  } else /*(mode == 0)*/ {
239  pts->newMenuStatus &= ~MenuFMode3;
240  pts->newMenuStatus &= ~MenuFMode2;
241  pts->newMenuStatus &= ~MenuFMode1;
242  }
243  } else {
244  pts->newMenuStatus = 0;
245  }

353 SIM_DLL_EXPORT void AeroSIMRC_Plugin_Run(const simToPlugin *stp,
354  pluginToSim *pts)
355 {
356  debugInfo = "---\n";
357  // By default do not change the Menu Items of type CheckBox
358  pts->newMenuStatus = stp->simMenuStatus;
359  // Extract Menu Commands from Flags
360  bool isReset = (stp->simMenuStatus & MenuCmdReset) != 0;
361  bool isEnable = (stp->simMenuStatus & MenuEnable) != 0;
362  bool isTxON = (stp->simMenuStatus & MenuTx) != 0;
363  bool isRxON = (stp->simMenuStatus & MenuRx) != 0;
364  bool isScreen = (stp->simMenuStatus & MenuScreen) != 0;
365  bool isNextWp = (stp->simMenuStatus & MenuNextWpt) != 0;
366  // Run commands
367  if (isReset) {
368  Run_Command_Reset(/*stp, pts*/);
369  } else if (isScreen) {
370  Run_Command_WindowSizeAndPos(stp, pts);
371  } else if (isNextWp) {
372  Run_Command_MoveToNextWaypoint(stp, pts);
373  } else {
374  Run_BlinkLEDs(stp, pts);
375  if (isEnable) {
376  if (isTxON)
377  sndr->sendDatagram(stp);
378  if (isRxON)
379  rcvr->setChannels(pts);
380  }
381 
382  // network lag
383  debugInfo.append(QString("out: %1, inp: %2, delta: %3\n")
384  .arg(sndr->pcks() - 1)
385  .arg(rcvr->pcks())
386  .arg(sndr->pcks() - rcvr->pcks() - 1)
387  );
388  }
389 
390  // debug info is shown on the screen
391  InfoText(stp, pts);
392  pts->dbgInfoText = debugInfo.toAscii();
393  isFirstRun = false;
394 }
