677 bool QextSerialPort::open(OpenMode mode)
678 {
679  QMutexLocker lock(mutex);
680  if (mode == QIODevice::NotOpen)
681  return isOpen();
682  if (!isOpen()) {
683  qDebug() << "trying to open file" << port.toAscii();
684  //note: linux 2.6.21 seems to ignore O_NDELAY flag
685  if ((fd = ::open(port.toAscii() ,O_RDWR | O_NOCTTY | O_NDELAY)) != -1) {
686  qDebug("file opened succesfully");
687 
688  setOpenMode(mode); // Flag the port as opened
689  tcgetattr(fd, &old_termios); // Save the old termios
690  Posix_CommConfig = old_termios; // Make a working copy
691  cfmakeraw(&Posix_CommConfig); // Enable raw access
692 
693  /*set up other port settings*/
694  Posix_CommConfig.c_cflag|=CREAD|CLOCAL;
695  Posix_CommConfig.c_lflag&=(~(ICANON|ECHO|ECHOE|ECHOK|ECHONL|ISIG));
696  Posix_CommConfig.c_iflag&=(~(INPCK|IGNPAR|PARMRK|ISTRIP|ICRNL|IXANY));
697  Posix_CommConfig.c_oflag&=(~OPOST);
698  Posix_CommConfig.c_cc[VMIN]= 0;
699 #ifdef _POSIX_VDISABLE // Is a disable character available on this system?
700  // Some systems allow for per-device disable-characters, so get the
701  // proper value for the configured device
702  const long vdisable = fpathconf(fd, _PC_VDISABLE);
703  Posix_CommConfig.c_cc[VINTR] = vdisable;
704  Posix_CommConfig.c_cc[VQUIT] = vdisable;
705  Posix_CommConfig.c_cc[VSTART] = vdisable;
706  Posix_CommConfig.c_cc[VSTOP] = vdisable;
707  Posix_CommConfig.c_cc[VSUSP] = vdisable;
708 #endif //_POSIX_VDISABLE
709  setBaudRate(Settings.BaudRate);
710  setDataBits(Settings.DataBits);
711  setParity(Settings.Parity);
712  setStopBits(Settings.StopBits);
713  setFlowControl(Settings.FlowControl);
714  setTimeout(Settings.Timeout_Millisec);
715  tcsetattr(fd, TCSAFLUSH, &Posix_CommConfig);
716 
717  if (queryMode() == QextSerialPort::EventDriven) {
718  readNotifier = new QSocketNotifier(fd, QSocketNotifier::Read, this);
719  connect(readNotifier, SIGNAL(activated(int)), this, SIGNAL(readyRead()));
720  }
721  } else {
722  qDebug() << "could not open file:" << strerror(errno);
723  lastErr = E_FILE_NOT_FOUND;
724  }
725  }
726  return isOpen();
727 }

733 void QextSerialPort::close()
734 {
735  QMutexLocker lock(mutex);
736  if( isOpen() )
737  {
738  // Force a flush and then restore the original termios
739  flush();
740  // Using both TCSAFLUSH and TCSANOW here discards any pending input
741  tcsetattr(fd, TCSAFLUSH | TCSANOW, &old_termios); // Restore termios
742  // Be a good QIODevice and call QIODevice::close() before POSIX close()
743  // so the aboutToClose() signal is emitted at the proper time
744  QIODevice::close(); // Flag the device as closed
745  // QIODevice::close() doesn't actually close the port, so do that here
746  ::close(fd);
747  if(readNotifier) {
748  delete readNotifier;
749  readNotifier = 0;
750  }
751  }
