98 QList<int> selectedRows(const QAbstractItemView *view)
99 {
100  const QModelIndexList indexList = view->selectionModel()->selectedRows(0);
101  if (indexList.empty())
102  return QList<int>();
103  QList<int> rc;
104  const QModelIndexList::const_iterator cend = indexList.constEnd();

106  rc.push_back(it->row());
107  return rc;

167 void SubmitEditorWidget::registerActions(QAction *editorUndoAction, QAction *editorRedoAction,
168  QAction *submitAction, QAction *diffAction)
169 {
170  if (editorUndoAction) {
171  editorUndoAction->setEnabled(m_d->m_ui.description->document()->isUndoAvailable());
172  connect(m_d->m_ui.description, SIGNAL(undoAvailable(bool)), editorUndoAction, SLOT(setEnabled(bool)));
173  connect(editorUndoAction, SIGNAL(triggered()), m_d->m_ui.description, SLOT(undo()));
174  }
175  if (editorRedoAction) {
176  editorRedoAction->setEnabled(m_d->m_ui.description->document()->isRedoAvailable());
177  connect(m_d->m_ui.description, SIGNAL(redoAvailable(bool)), editorRedoAction, SLOT(setEnabled(bool)));
178  connect(editorRedoAction, SIGNAL(triggered()), m_d->m_ui.description, SLOT(redo()));
179  }
180 
181  if (submitAction) {
182  if (debug) {
183  int count = 0;
184  if (const QAbstractItemModel *model = m_d->m_ui.fileView->model())
185  count = model->rowCount();
186  qDebug() << Q_FUNC_INFO << submitAction << count << "items" << m_d->m_filesChecked;
187  }
188  submitAction->setEnabled(m_d->m_filesChecked);
189  connect(this, SIGNAL(fileCheckStateChanged(bool)), submitAction, SLOT(setEnabled(bool)));
190  m_d->m_ui.buttonLayout->addWidget(new QActionPushButton(submitAction));
191  }
192  if (diffAction) {
193  if (debug)
194  qDebug() << diffAction << m_d->m_filesSelected;
195  diffAction->setEnabled(m_d->m_filesSelected);
196  connect(this, SIGNAL(fileSelectionChanged(bool)), diffAction, SLOT(setEnabled(bool)));
197  connect(diffAction, SIGNAL(triggered()), this, SLOT(triggerDiffSelected()));
198  m_d->m_ui.buttonLayout->addWidget(new QActionPushButton(diffAction));
199  }
200 }

231 // Extract the wrapped text from a text edit, which performs
232 // the wrapping only optically.
233 static QString wrappedText(const QTextEdit *e)
234 {
235  const QChar newLine = QLatin1Char('\n');
236  QString rc;
237  QTextCursor cursor(e->document());
238  cursor.movePosition(QTextCursor::Start);
239  while (!cursor.atEnd()) {
240  cursor.select(QTextCursor::LineUnderCursor);
241  rc += cursor.selectedText();
242  rc += newLine;
243  cursor.movePosition(QTextCursor::EndOfLine); // Mac needs it
244  cursor.movePosition(QTextCursor::Right);
245  }
246  return rc;
247 }

488 void SubmitEditorWidget::editorCustomContextMenuRequested(const QPoint &pos)
489 {

491  // Extend
