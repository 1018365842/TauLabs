223 void HistogramScopeConfig::loadConfiguration(ScopeGadgetWidget *scopeGadgetWidget)
224 {
225  preparePlot(scopeGadgetWidget);
226  scopeGadgetWidget->setScope(this);
227  scopeGadgetWidget->startTimer(m_refreshInterval);
228 
229  // Configure each data source
230  foreach (Plot2dCurveConfiguration* histogramDataSourceConfig, m_HistogramSourceConfigs)
231  {
232  QRgb color = histogramDataSourceConfig->color;
233 
234  // Get and store the units
235  units = getUavObjectFieldUnits(histogramDataSourceConfig->uavObjectName, histogramDataSourceConfig->uavFieldName);
236 
237  HistogramData* histogramData;
238  histogramData = new HistogramData(histogramDataSourceConfig->uavObjectName, histogramDataSourceConfig->uavFieldName, binWidth, maxNumberOfBins);
239 
240  histogramData->setScalePower(histogramDataSourceConfig->yScalePower);
241  histogramData->setMeanSamples(histogramDataSourceConfig->yMeanSamples);
242  histogramData->setMathFunction(histogramDataSourceConfig->mathFunction);
243 
244  //Generate the curve name
245  QString curveName = (histogramData->getUavoName()) + "." + (histogramData->getUavoFieldName());
246  if(histogramData->getHaveSubFieldFlag())
247  curveName = curveName.append("." + histogramData->getUavoSubFieldName());
248 
249  //Get the uav object
250  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
251  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
252  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(objManager->getObject((histogramData->getUavoName())));
253  if(!obj) {
254  qDebug() << "Object " << histogramData->getUavoName() << " is missing";
255  return;
256  }
257 
258  //Get the units
259  QString units = getUavObjectFieldUnits(histogramData->getUavoName(), histogramData->getUavoFieldName());
260 
261  //Generate name with scaling factor appeneded
262  QString histogramNameScaled;
263  if(histogramDataSourceConfig->yScalePower == 0)
264  histogramNameScaled = curveName + "(" + units + ")";
265  else
266  histogramNameScaled = curveName + "(x10^" + QString::number(histogramDataSourceConfig->yScalePower) + " " + units + ")";
267 
268  while(scopeGadgetWidget->getDataSources().keys().contains(histogramNameScaled))
269  histogramNameScaled=histogramNameScaled+"*";
270 
271  // Create the histogram
272  QwtPlotHistogram* plotHistogram = new QwtPlotHistogram(histogramNameScaled);
273  plotHistogram->setStyle( QwtPlotHistogram::Columns );
274  plotHistogram->setBrush(QBrush(QColor(color)));
275  plotHistogram->setData( histogramData->getIntervalSeriesData());
276 
277  plotHistogram->attach(scopeGadgetWidget);
278  histogramData->setHistogram(plotHistogram);
279 
280  // Keep the curve details for later
281  scopeGadgetWidget->insertDataSources(histogramNameScaled, histogramData);
282 
283  // Connect the UAVO
284  scopeGadgetWidget->connectUAVO(obj);
285  }
286  mutex.lock();
287  scopeGadgetWidget->replot();
288  mutex.unlock();
289 }
