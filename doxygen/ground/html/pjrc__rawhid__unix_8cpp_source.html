56 // open - open 1 or more devices
57 //
58 // Inputs:
59 // max = maximum number of devices to open
60 // vid = Vendor ID, or -1 if any
61 // pid = Product ID, or -1 if any
62 // usage_page = unused, argument kept for api consistency
63 // usage = unused, argument kept for api consistency
64 // Output:
65 // actual number of devices opened
66 //
67 int pjrc_rawhid::open(int max, int vid, int pid, int /* usage_page */, int /* usage */)
68 {
69  //check if there are still open devices and close them
70  if (!m_DeviceHandles.empty()) {
71  for (std::size_t iDevice = 0; iDevice < m_DeviceHandles.size(); ++iDevice) {
72  if (m_DeviceHandles[iDevice] != NULL)
73  close(iDevice);
74  }
75  m_DeviceHandles.clear();
76  m_DeviceInterfaces.clear();
77  }
78 
79  int retval;
80 
81  //enumerate all devices
82  libusb_device** list = NULL;
83  ssize_t num_devices;
84  num_devices = libusb_get_device_list(m_pLibraryContext, &list);
85 
86  for (ssize_t iDevice = 0;
87  iDevice < num_devices && m_DeviceHandles.size() < (std::size_t)max;
88  ++iDevice)
89  {
90  libusb_device_descriptor device_desc;
91  retval = libusb_get_device_descriptor(list[iDevice], &device_desc);
92  if (retval < 0) {
93  fprintf(stderr, "pjrc_rawhid_unix: Failed to get device descriptor (%d)\n", retval);
94  continue;
95  }
96 
97  //check if we are interested in this device
98  if (device_desc.idVendor != vid || device_desc.idProduct != pid)
99  continue;
100 
101  //search for proper interface
102  libusb_config_descriptor* config_desc;
103  retval = libusb_get_config_descriptor(list[iDevice], 0, &config_desc);
104  if (retval != 0) {
105  fprintf(stderr, "pjrc_rawhid_unix: libusb_get_config_descriptor error (%d)\n", retval);
106  continue;
107  }
108 
109  int device_interface = -1;
110  const libusb_interface* interface;
111  const libusb_interface_descriptor* interface_desc;
112  for (ssize_t iInterface = 0;
113  iInterface < config_desc->bNumInterfaces && device_interface == -1;
114  ++iInterface)
115  {
116  interface = &config_desc->interface[iInterface];
117 
118  for (ssize_t iAltSetting = 0;
119  iAltSetting < interface->num_altsetting && device_interface == -1;
120  ++iAltSetting)
121  {
122  interface_desc = &interface->altsetting[iAltSetting];
123 
124  if (interface_desc->bInterfaceClass == LIBUSB_CLASS_HID &&
125  interface_desc->bInterfaceSubClass == 0 &&
126  interface_desc->bInterfaceProtocol == 0)
127  {
128  device_interface = iInterface;
129  }
130  }
131  }
132 
133  libusb_free_config_descriptor(config_desc);
134 
135  //check if we found an interface, if not, skip device
136  if (device_interface == -1)
137  continue;
138 
139  libusb_device_handle* device_handle;
140  retval = libusb_open(list[iDevice], &device_handle);
141  //skip if device cant be opened
142  if (retval != 0) {
143  fprintf(stderr, "pjrc_rawhid_unix: Failed to open device (%d)\n", retval);
144  continue;
145  }
146 
147  //detach kernel
148  retval = libusb_detach_kernel_driver(device_handle, device_interface);
149  if (retval != 0 && retval != LIBUSB_ERROR_NOT_FOUND) {
150  fprintf(stderr, "pjrc_rawhid_unix: Unable to detach kernel driver (%d)\n", retval);
151  libusb_close(device_handle);
152  continue;
153  }
154 
155  //claim interface
156  retval = libusb_claim_interface(device_handle, device_interface);
157  if (retval != 0) {
158  fprintf(stderr, "pjrc_rawhid_unix: libusb_claim_interface error (%d)\n", retval);
159  libusb_close(device_handle);
160  continue;
161  }
162 
163  m_DeviceHandles.push_back(device_handle);
164  m_DeviceInterfaces.push_back(device_interface);
165  }
166 
167  libusb_free_device_list(list, 1);
168 
169  return m_DeviceHandles.size();
170 }

172 // recveive - receive a packet
173 // Inputs:
174 // num = device to receive from (zero based)
175 // buf = buffer to receive packet
176 // len = buffer's size
177 // timeout = time to wait, in milliseconds
178 // Output:
179 // number of bytes received, or -1 on error
180 //
181 int pjrc_rawhid::receive(int num, void *buf, int len, int timeout)
182 {
183  if ((std::size_t)num >= m_DeviceHandles.size()) {
184  fprintf(stderr, "pjrc_rawhid_unix: Invalid device number used (%d)\n", num);
185  return -1;
186  }
187  if (m_DeviceHandles[num] == NULL) {
188  fprintf(stderr, "pjrc_rawhid_unix: Tried to use a device which is not open (%d)\n", num);
189  return -1;
190  }
191 
192  int bytes_transferred;
193  int retval;
194 
195  // Read data from the device.
196  retval = libusb_interrupt_transfer(
197  m_DeviceHandles[num],
198  INTERRUPT_IN_ENDPOINT,
199  (unsigned char*)buf,
200  len > MAX_INTERRUPT_IN_TRANSFER_SIZE ? MAX_INTERRUPT_IN_TRANSFER_SIZE : len,
201  &bytes_transferred,
202  timeout);
203 
204  if (retval >= 0) {
205  if (bytes_transferred > 0) {
206  return bytes_transferred;
207  }
208  else {
209  fprintf(stderr, "pjrc_rawhid_unix: No data received in interrupt transfer (%d)\n", retval);
210  return -1;
211  }
212  }
213  else {
214  if (retval == LIBUSB_ERROR_TIMEOUT)
215  return 0;
216  fprintf(stderr, "pjrc_rawhid_unix: Error receiving data via interrupt transfer (%d)\n", retval);
217  return -1;
218  }
219 }
