192 void QwtCounter::keyPressEvent ( QKeyEvent *event )
193 {
194  bool accepted = true;
195 
196  switch ( event->key() )
197  {
198  case Qt::Key_Home:
199  {
200  if ( event->modifiers() & Qt::ControlModifier )
201  setValue( minValue() );
202  else
203  accepted = false;
204  break;
205  }
206  case Qt::Key_End:
207  {
208  if ( event->modifiers() & Qt::ControlModifier )
209  setValue( maxValue() );
210  else
211  accepted = false;
212  break;
213  }
214  case Qt::Key_Up:
215  {
216  incValue( d_data->increment[0] );
217  break;
218  }
219  case Qt::Key_Down:
220  {
221  incValue( -d_data->increment[0] );
222  break;
223  }
224  case Qt::Key_PageUp:
225  case Qt::Key_PageDown:
226  {
227  int increment = d_data->increment[0];
228  if ( d_data->numButtons >= 2 )
229  increment = d_data->increment[1];
230  if ( d_data->numButtons >= 3 )
231  {
232  if ( event->modifiers() & Qt::ShiftModifier )
233  increment = d_data->increment[2];
234  }
235  if ( event->key() == Qt::Key_PageDown )
236  increment = -increment;
237  incValue( increment );
238  break;
239  }
240  default:
241  {
242  accepted = false;
243  }
244  }
245 
246  if ( accepted )
247  {
248  event->accept();
249  return;
250  }
251 
252  QWidget::keyPressEvent ( event );
253 }

259 void QwtCounter::wheelEvent( QWheelEvent *event )
260 {
261  event->accept();
262 
263  if ( d_data->numButtons <= 0 )
264  return;
265 
266  int increment = d_data->increment[0];
267  if ( d_data->numButtons >= 2 )
268  {
269  if ( event->modifiers() & Qt::ControlModifier )
270  increment = d_data->increment[1];
271  }
272  if ( d_data->numButtons >= 3 )
273  {
274  if ( event->modifiers() & Qt::ShiftModifier )
275  increment = d_data->increment[2];
276  }
277 
278  for ( int i = 0; i < d_data->numButtons; i++ )
279  {
280  if ( d_data->buttonDown[i]->geometry().contains( event->pos() ) ||
281  d_data->buttonUp[i]->geometry().contains( event->pos() ) )
282  {
283  increment = d_data->increment[i];
284  }
285  }
286 
287  const int wheel_delta = 120;
288 
289  int delta = event->delta();
290  if ( delta >= 2 * wheel_delta )
291  delta /= 2; // Never saw an abs(delta) < 240
292 
293  incValue( delta / wheel_delta * increment );
294 }

471 QSize QwtCounter::sizeHint() const
472 {
473  QString tmp;
474 
475  int w = tmp.setNum( minValue() ).length();
476  int w1 = tmp.setNum( maxValue() ).length();
477  if ( w1 > w )
478  w = w1;
479  w1 = tmp.setNum( minValue() + step() ).length();
480  if ( w1 > w )
481  w = w1;
482  w1 = tmp.setNum( maxValue() - step() ).length();
483  if ( w1 > w )
484  w = w1;
485 
486  tmp.fill( '9', w );
487 
488  QFontMetrics fm( d_data->valueEdit->font() );
489  w = fm.width( tmp ) + 2;
490  if ( d_data->valueEdit->hasFrame() )
491  w += 2 * style()->pixelMetric( QStyle::PM_DefaultFrameWidth );
492 
493  // Now we replace default sizeHint contribution of d_data->valueEdit by
494  // what we really need.
495 
496  w += QWidget::sizeHint().width() - d_data->valueEdit->sizeHint().width();
497 
498  const int h = qMin( QWidget::sizeHint().height(),
499  d_data->valueEdit->minimumSizeHint().height() );
500  return QSize( w, h );
