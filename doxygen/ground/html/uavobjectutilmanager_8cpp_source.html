119 void UAVObjectUtilManager::saveNextObject()
120 {
121  if ( queue.isEmpty() )
122  {
123  return;
124  }
125 
126  Q_ASSERT(saveState == IDLE);
127 
128  // Get next object from the queue
129  UAVObject* obj = queue.head();
130  qDebug() << "Send save object request to board " << obj->getName();
131 
132  ObjectPersistence* objper = dynamic_cast<ObjectPersistence*>( getObjectManager()->getObject(ObjectPersistence::NAME) );
133  connect(objper, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(objectPersistenceTransactionCompleted(UAVObject*,bool)));
134  connect(objper, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(objectPersistenceUpdated(UAVObject *)));
135  saveState = AWAITING_ACK;
136  if (obj != NULL)
137  {
138  ObjectPersistence::DataFields data;
139  data.Operation = ObjectPersistence::OPERATION_SAVE;
140  data.Selection = ObjectPersistence::SELECTION_SINGLEOBJECT;
141  data.ObjectID = obj->getObjID();
142  data.InstanceID = obj->getInstID();
143  objper->setData(data);
144  objper->updated();
145  }
146  // Now: we are going to get two "objectUpdated" messages (one coming from GCS, one coming from Flight, which
147  // will confirm the object was properly received by both sides) and then one "transactionCompleted" indicating
148  // that the Flight side did not only receive the object but it did receive it without error. Last we will get
149  // a last "objectUpdated" message coming from flight side, where we'll get the results of the objectPersistence
150  // operation we asked for (saved, other).
151 }

162 void UAVObjectUtilManager::objectPersistenceTransactionCompleted(UAVObject* obj, bool success)
163 {
164  if(success) {
165  Q_ASSERT(obj->getName().compare("ObjectPersistence") == 0);
166  Q_ASSERT(saveState == AWAITING_ACK);
167  // Two things can happen:
168  // Either the Object Save Request did actually go through, and then we should get in
169  // "AWAITING_COMPLETED" mode, or the Object Save Request did _not_ go through, for example
170  // because the object does not exist and then we will never get a subsequent update.
171  // For this reason, we will arm a 1 second timer to make provision for this and not block
172  // the queue:
173  saveState = AWAITING_COMPLETED;
174  disconnect(obj, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(objectPersistenceTransactionCompleted(UAVObject*,bool)));
175  failureTimer.start(2000); // Create a timeout
176  } else {
177  // Can be caused by timeout errors on sending. Forget it and send next.
178  qDebug() << "objectPersistenceTranscationCompleted (error)";
179  UAVObject *obj = getObjectManager()->getObject(ObjectPersistence::NAME);
180  obj->disconnect(this);
181  queue.dequeue(); // We can now remove the object, it failed anyway.
182  saveState = IDLE;
183  emit saveCompleted(obj->getField("ObjectID")->getValue().toInt(), false);
184  saveNextObject();
185  }
186 }

222 void UAVObjectUtilManager::objectPersistenceUpdated(UAVObject * obj)
223 {
224  Q_ASSERT(obj);
225  Q_ASSERT(obj->getObjID() == ObjectPersistence::OBJID);
226  ObjectPersistence::DataFields objectPersistence = ((ObjectPersistence *)obj)->getData();
227 
228  if(saveState == AWAITING_COMPLETED && objectPersistence.Operation == ObjectPersistence::OPERATION_ERROR) {
229  failureTimer.stop();
230  objectPersistenceOperationFailed();
231  } else if (saveState == AWAITING_COMPLETED &&
232  objectPersistence.Operation == ObjectPersistence::OPERATION_COMPLETED) {
233  failureTimer.stop();
234  // Check right object saved
235  UAVObject* savingObj = queue.head();
236  if(objectPersistence.ObjectID != savingObj->getObjID() ) {
237  objectPersistenceOperationFailed();
238  return;
239  }
240 
241  obj->disconnect(this);
242  queue.dequeue(); // We can now remove the object, it's done.
243  saveState = IDLE;
244 
245  emit saveCompleted(objectPersistence.ObjectID, true);
246  saveNextObject();
247  }
248 }

611 bool UAVObjectUtilManager::descriptionToStructure(QByteArray desc, deviceDescriptorStruct & struc)
612 {
613  if (desc.startsWith("TlFw") || desc.startsWith("OpFw")) {
614  /*
615  * This looks like a binary with a description at the end
616  * 4 bytes: header: "TlFw"
617  * 4 bytes: git commit hash (short version of SHA1)
618  * 4 bytes: Unix timestamp of last git commit
619  * 2 bytes: target platform. Should follow same rule as BOARD_TYPE and BOARD_REVISION in board define files.
620  * 26 bytes: commit tag if it is there, otherwise "Unreleased". Zero-padded
621  * 20 bytes: SHA1 sum of the firmware.
622  * 20 bytes: SHA1 sum of the UAVO definition files.
623  * 20 bytes: free for now.
624  */
625 
626  // Note: the ARM binary is big-endian:
627  quint32 gitCommitHash = desc.at(7) & 0xFF;
628  for (int i = 1; i < 4; i++) {
629  gitCommitHash = gitCommitHash << 8;
630  gitCommitHash += desc.at(7-i) & 0xFF;
631  }
632  struc.gitHash = QString("%1").arg(gitCommitHash, 8, 16, QChar('0'));
633 
634  quint32 gitDate = desc.at(11) & 0xFF;
635  for (int i = 1; i < 4; i++) {
636  gitDate = gitDate << 8;
637  gitDate += desc.at(11-i) & 0xFF;
638  }
639  struc.gitDate = QDateTime::fromTime_t(gitDate).toUTC().toString("yyyyMMdd HH:mm");
640 
641  QString gitTag = QString(desc.mid(14,26));
642  struc.gitTag = gitTag;
643 
644  // TODO: check platform compatibility
645  QByteArray targetPlatform = desc.mid(12,2);
646  struc.boardType = (int)targetPlatform.at(0);
647  struc.boardRevision = (int)targetPlatform.at(1);
648  struc.fwHash.clear();
649  struc.fwHash=desc.mid(40,20);
650  struc.uavoHash.clear();
651  struc.uavoHash=desc.mid(60,20);
652 
653  return true;
654  }
655  return false;
656 }
