179 void QxtCsvModel::setSource(QIODevice *file, bool withHeader, QChar separator)
180 {
181  QxtCsvModelPrivate* d_ptr = &qxt_d();
182  QString l;
183  int size;
184  bool isQuoted, headerSet = false;
185  if (!file->isOpen()) file->open(QIODevice::ReadOnly);
186  if (withHeader)
187  d_ptr->maxColumn = 0;
188  else
189  d_ptr->maxColumn = d_ptr->header.size();
190  d_ptr->csvData.clear();
191  while (!file->atEnd())
192  {
193  l = file->readLine();
194  l.remove('\n');
195  l.remove('\r');
196  size = l.length();
197  isQuoted = false;
198  for (int i = 0;i < size;i++)
199  {
200  if (i > 0)
201  {
202  if (l[i] == '"' && l[i-1] != '\\') isQuoted = !isQuoted;
203  else if (!isQuoted && l[i] == separator) l[i] = QChar(1);
204  }
205  else
206  {
207  if (l[i] == '"') isQuoted = !isQuoted;
208  else if (!isQuoted && l[i] == separator) l[i] = QChar(1);
209  }
210  }
211  if (l.count(QChar(1)) + 1 > d_ptr->maxColumn) d_ptr->maxColumn = l.count(QChar(1)) + 1;
212  if (withHeader && !headerSet)
213  {
214  d_ptr->header = l.split(QChar(1));
215  headerSet = true;
216  }
217  else
218  d_ptr->csvData.append(l);
219  }
220  file->close();
221 }

381 void QxtCsvModel::toCSV(QIODevice* dest, bool withHeader, QChar separator)
382 {
383  QxtCsvModelPrivate& d_ptr = qxt_d();
384  int row, col, rows, cols;
385  rows = rowCount();
386  cols = columnCount();
387  QString data;
388  if (!dest->isOpen()) dest->open(QIODevice::WriteOnly | QIODevice::Truncate);
389  if (withHeader)
390  {
391  data = "";
392  for (col = 0; col < cols; ++col)
393  {
394  data += '"' + d_ptr.header.at(col) + '"';
395  if (col < cols - 1) data += separator;
396  }
397  data += QChar(10);
398  dest->write(data.toLatin1());
399  }
400  for (row = 0; row < rows; ++row)
401  {
402  data = "";
403  for (col = 0; col < cols; ++col)
404  {
405  data += '"' + d_ptr.csvData[row].section(QChar(1), col, col) + '"';
406  if (col < cols - 1) data += separator;
407  }
408  data += QChar(10);
409  dest->write(data.toLatin1());
410  }
411  dest->close();
412 }
