151 void QxtAbstractHttpConnector::incomingData(QIODevice* device)
152 {
153  if (!device)
154  {
155  device = qobject_cast<QIODevice*>(sender());
156  if (!device) return;
157  }
158  QReadLocker locker(&qxt_d().bufferLock);
159  QByteArray& buffer = qxt_d().buffers[device];
160  buffer.append(device->readAll());
161  if (!canParseRequest(buffer)) return;
162  QHttpRequestHeader header = parseRequest(buffer);
163  QxtWebContent* content = 0;
164  QByteArray start;
165  if (header.contentLength() > 0)
166  {
167  start = buffer.left(header.value("content-length").toInt());
168  buffer = buffer.mid(header.value("content-length").toInt());
169  content = new QxtWebContent(header.contentLength(), start, device);
170  }
171  else if (header.hasKey("connection") && header.value("connection").toLower() == "close")
172  {
173  start = buffer;
174  buffer.clear();
175  content = new QxtWebContent(header.contentLength(), start, device);
176  } // else no content
177  quint32 requestID = qxt_d().getNextRequestID(device);
178  sessionManager()->incomingRequest(requestID, header, content);
179 }
