32 void QxtScreenPrivate::init_sys()
33 {
34 #ifdef HAVE_XRANDR
35  Display* display = XOpenDisplay(NULL);
36  Window root = RootWindow(display, screen);
37  XRRScreenConfiguration* config = XRRGetScreenInfo(display, root);
38 
39  // available resolutions & rates
40  if (availResos.isEmpty() || availRates.isEmpty())
41  {
42  availResos.clear();
43  availRates.clear();
44 
45  int sizeCount = 0;
46  XRRScreenSize* sizes = XRRSizes(display, 0, &sizeCount);
47  for (int i = 0; i < sizeCount; ++i)
48  {
49  QSize reso(sizes[i].width, sizes[i].height);
50  if (!availResos.contains(reso))
51  availResos += reso;
52 
53  int rateCount = 0;
54  short* rates = XRRConfigRates(config, i, &rateCount);
55  for (int j = 0; j < rateCount; ++j)
56  availRates.insertMulti(reso, rates[j]);
57  }
58  }
59 
60  // current resolution & rate
61  if (!currReso.isValid() || currRate < 0)
62  {
63  Rotation rotation;
64  SizeID sizeId = XRRConfigCurrentConfiguration(config, &rotation);
65  currReso = availResos.at(sizeId);
66  currRate = XRRConfigCurrentRate(config);
67  }
68 
69  XRRFreeScreenConfigInfo(config);
70  XCloseDisplay(display);
71 #endif // HAVE_XRANDR
72 }

74 bool QxtScreenPrivate::set(const QSize& reso, int rate, int depth)
75 {
76  bool result = false;
77  Q_UNUSED(reso);
78  Q_UNUSED(rate);
79  Q_UNUSED(depth);
80 #ifdef HAVE_XRANDR
81  Display* display = XOpenDisplay(NULL);
82  Window root = RootWindow(display, screen);
83  XRRScreenConfiguration* config = XRRGetScreenInfo(display, root);
84 
85  int sizeIndex = 0;
86  if (reso.isValid())
87  sizeIndex = availResos.indexOf(reso);
88  else
89  sizeIndex = availResos.indexOf(currReso);
90  Q_ASSERT(sizeIndex != -1);
91 
92  if (rate == -1)
93  result = XRRSetScreenConfig(display, config, root, sizeIndex, RR_Rotate_0, CurrentTime);
94  else
95  result = XRRSetScreenConfigAndRate(display, config, root, sizeIndex, RR_Rotate_0, rate, CurrentTime);
96 
97  XRRFreeScreenConfigInfo(config);
98  XCloseDisplay(display);
99 #endif // HAVE_XRANDR
100  return result;
101 }
