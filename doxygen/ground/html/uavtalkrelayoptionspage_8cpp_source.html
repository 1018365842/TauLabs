94 void UavTalkRelayOptionsPage::apply()
95 {
96  m_config->m_Port=m_page->ListeningPort->value();
97  m_config->m_IpAddress=m_page->ListeningInterface->text();
98  m_config->rules.clear();
99  for(int i=0;i < m_page->twRules->rowCount();++i)
100  {
101  QString host=m_page->twRules->item(i,0)->text();
102  quint32 uavo=m_page->cbAddRuleUAVO->itemData(m_page->cbAddRuleUAVO->findText(m_page->twRules->item(i,1)->text())).toInt();
103  UavTalkRelayComon::accessType aType=(UavTalkRelayComon::accessType)m_page->cbAddRuleAccessType->itemData(m_page->cbAddRuleAccessType->findText(m_page->twRules->item(i,2)->text())).toInt();
104  if(m_config->rules.keys().contains(host))
105  {
106  QHash <quint32,UavTalkRelayComon::accessType> temp=m_config->rules.value(host);
107  temp.insert(uavo,aType);
108  }
109  else
110  {
111  QHash <quint32,UavTalkRelayComon::accessType> temp;
112  temp.insert(uavo,aType);
113  m_config->rules.insert(host,temp);
114  }
115  }
116  emit settingsUpdated();
117 }
