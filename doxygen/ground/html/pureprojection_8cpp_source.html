198  void PureProjection::FromGeodeticToCartesian(double Lat,double Lng,double Height, double &X, double &Y, double &Z)
199  {
200  Lat = (PI / 180) * Lat;
201  Lng = (PI / 180) * Lng;
202 
203  double B = Axis() * (1.0 - Flattening());
204  double ee = 1.0 - (B / Axis()) * (B / Axis());
205  double N = (Axis() / sqrt(1.0 - ee * sin(Lat) * sin(Lat)));
206 
207  X = (N + Height) * cos(Lat) * cos(Lng);
208  Y = (N + Height) * cos(Lat) * sin(Lng);
209  Z = (N * (B / Axis()) * (B / Axis()) + Height) * sin(Lat);
210  }
211  void PureProjection::FromCartesianTGeodetic(const double &X,const double &Y,const double &Z, double &Lat, double &Lng)
212  {
213  double E = Flattening() * (2.0 - Flattening());
214  Lng = atan2(Y, X);
215 
216  double P = sqrt(X * X + Y * Y);
217  double Theta = atan2(Z, (P * (1.0 - Flattening())));
218  double st = sin(Theta);
219  double ct = cos(Theta);
220  Lat = atan2(Z + E / (1.0 - Flattening()) * Axis() * st * st * st, P - E * Axis() * ct * ct * ct);
221 
222  Lat /= (PI / 180);
223  Lng /= (PI / 180);
224  }

237  double PureProjection::DistanceBetweenLatLng(PointLatLng const& p1,PointLatLng const& p2)
238  {
239  double R = 6371; // km
240  double lat1=p1.Lat();
241  double lat2=p2.Lat();
242  double lon1=p1.Lng();
243  double lon2=p2.Lng();
244  double dLat = (lat2-lat1)* (PI / 180);
245  double dLon = (lon2-lon1)* (PI / 180);
246  double a = sin(dLat/2) * sin(dLat/2) + cos(lat1* (PI / 180)) * cos(lat2* (PI / 180)) * sin(dLon/2) * sin(dLon/2);
247  double c = 2 * atan2(sqrt(a), sqrt(1-a));
