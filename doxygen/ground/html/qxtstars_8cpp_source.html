63 int QxtStarsPrivate::pixelPosToRangeValue(int pos) const
64 {
65  const QxtStars* p = &qxt_p();
66  QStyleOptionSlider opt = getStyleOption();
67  QRect gr = p->style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderGroove, p);
68  QRect sr = p->style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderHandle, p);
69  int sliderMin, sliderMax, sliderLength;
70 
71  gr.setSize(qxt_p().sizeHint());
72  if (p->orientation() == Qt::Horizontal)
73  {
74  sliderLength = sr.width();
75  sliderMin = gr.x();
76  sliderMax = gr.right() - sliderLength + 1;
77  }
78  else
79  {
80  sliderLength = sr.height();
81  sliderMin = gr.y();
82  sliderMax = gr.bottom() - sliderLength + 1;
83  }
84  return QStyle::sliderValueFromPosition(p->minimum(), p->maximum(), pos - sliderMin,
85  sliderMax - sliderMin, opt.upsideDown);
86 }

94 QStyleOptionSlider QxtStarsPrivate::getStyleOption() const
95 {
96  const QxtStars* p = &qxt_p();
97  QStyleOptionSlider opt;
98  opt.initFrom(p);
99  opt.subControls = QStyle::SC_None;
100  opt.activeSubControls = QStyle::SC_None;
101  opt.orientation = p->orientation();
102  opt.maximum = p->maximum();
103  opt.minimum = p->minimum();
104  opt.upsideDown = (p->orientation() == Qt::Horizontal) ?
105  (p->invertedAppearance() != (opt.direction == Qt::RightToLeft)) : (!p->invertedAppearance());
106  opt.direction = Qt::LeftToRight; // we use the upsideDown option instead
107  opt.sliderPosition = p->sliderPosition();
108  opt.sliderValue = p->value();
109  opt.singleStep = p->singleStep();
110  opt.pageStep = p->pageStep();
111  if (p->orientation() == Qt::Horizontal)
112  opt.state |= QStyle::State_Horizontal;
113  return opt;

233 void QxtStars::paintEvent(QPaintEvent* event)
234 {
235  QAbstractSlider::paintEvent(event);
236 
237  QPainter painter(this);
238  painter.save();
239  painter.setPen(palette().color(QPalette::Text));
240  painter.setRenderHint(QPainter::Antialiasing);
241 
242  const bool invert = invertedAppearance();
243  const QSize size = qxt_d().getStarSize();
244  const QRectF star = qxt_d().star.boundingRect();
245  painter.scale(size.width() / star.width(), size.height() / star.height());
246  const int count = maximum() - minimum();
247  if (orientation() == Qt::Horizontal)
248  {
249  painter.translate(-star.x(), -star.y());
250  if (invert != isRightToLeft())
251  painter.translate((count - 1) * star.width(), 0);
252  }
253  else
254  {
255  painter.translate(-star.x(), -star.y());
256  if (!invert)
257  painter.translate(0, (count - 1) * star.height());
258  }
259  for (int i = 0; i < count; ++i)
260  {
261  if (value() > minimum() + i)
262  painter.setBrush(palette().highlight());
263  else
264  painter.setBrush(palette().base());
265  painter.drawPath(qxt_d().star);
266 
267  if (orientation() == Qt::Horizontal)
268  painter.translate(invert != isRightToLeft() ? -star.width() : star.width(), 0);
269  else
270  painter.translate(0, invert ? star.height() : -star.height());
271  }
272 
273  painter.restore();
274  if (hasFocus())
275  {
276  QStyleOptionFocusRect opt;
277  opt.initFrom(this);
278  opt.rect.setSize(sizeHint());
279  style()->drawPrimitive(QStyle::PE_FrameFocusRect, &opt, &painter, this);
280  }
281 }

299 void QxtStars::mousePressEvent(QMouseEvent* event)
300 {
301  if (qxt_d().readOnly)
302  {
303  event->ignore();
304  return;
305  }
306  QAbstractSlider::mousePressEvent(event);
307 
308  if (maximum() == minimum() || (event->buttons() ^ event->button()))
309  {
310  event->ignore();
311  return;
312  }
313 
314  event->accept();
315  QStyleOptionSlider opt = qxt_d().getStyleOption();
316  const QRect sliderRect = style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderHandle, this);
317  const QPoint center = sliderRect.center() - sliderRect.topLeft();
318  const int pos = qxt_d().pixelPosToRangeValue(qxt_d().pick(event->pos() - center));
319  setSliderPosition(pos);
320  triggerAction(SliderMove);
321  setRepeatAction(SliderNoAction);
322  qxt_d().snapBackPosition = pos;
323  update();
324 }
