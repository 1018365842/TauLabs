83 void FancyToolButton::paintEvent(QPaintEvent *event)
84 {
85  Q_UNUSED(event)
86  QPainter p(this);
87  QSize sh(sizeHint());
88  double scale = (double)height() / sh.height();
89  if (scale < 1) {
90  p.save();
91  p.scale(1, scale);
92  }
93  p.drawPicture(0, 0, m_buttonElements.value(svgIdButtonBase));
94  p.drawPicture(0, 0, m_buttonElements.value(isDown() ? svgIdButtonPressedBase : svgIdButtonNormalBase));
95 #ifndef Q_WS_MAC // Mac UIs usually don't hover
96  if (underMouse() && isEnabled())
97  p.drawPicture(0, 0, m_buttonElements.value(svgIdButtonHoverOverlay));
98 #endif
99 
100  if (scale < 1)
101  p.restore();
102 
103  if (!icon().isNull()) {
104  icon().paint(&p, rect());
105  } else {
106  const int margin = 4;
107  p.drawText(rect().adjusted(margin, margin, -margin, -margin), Qt::AlignCenter | Qt::TextWordWrap, text());
108 
109  }
110 
111  if (scale < 1) {
112  p.scale(1, scale);
113  }
114 
115  if (isEnabled()) {
116  p.drawPicture(0, 0, m_buttonElements.value(isDown() ?
117  svgIdButtonPressedOverlay : svgIdButtonNormalOverlay));
118  } else {
119  p.drawPicture(0, 0, m_buttonElements.value(svgIdButtonDisabledOverlay));
120  }
121 }

150 void FancyActionBar::insertAction(int index, QAction *action, QMenu *menu)
151 {
152  FancyToolButton *toolButton = new FancyToolButton(this);
153  toolButton->setDefaultAction(action);
154  if (menu) {
155  toolButton->setMenu(menu);
156  toolButton->setPopupMode(QToolButton::DelayedPopup);
157 
158  // execute action also if a context menu item is select
159  connect(toolButton, SIGNAL(triggered(QAction*)),
160  this, SLOT(toolButtonContextMenuActionTriggered(QAction*)),
161  Qt::QueuedConnection);
162  }
163  m_actionsLayout->insertWidget(index, toolButton);
164 }

166 /*
167  This slot is invoked when a context menu action of a tool button is triggered.
168  In this case we also want to trigger the default action of the button.
169 
170  This allows the user e.g. to select and run a specific run configuration with one click.
171  */
172 void FancyActionBar::toolButtonContextMenuActionTriggered(QAction* action)
173 {
174  if (QToolButton *button = qobject_cast<QToolButton*>(sender())) {
