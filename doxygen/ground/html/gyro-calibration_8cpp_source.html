7 /*
8  * Compute basic calibration parameters for a three axis gyroscope.
9  * The measurement equation is
10  * gyro_k = accelSensitivity * \tilde{accel}_k + bias + omega
11  *
12  * where, omega is the true angular rate (assumed to be zero)
13  * bias is the sensor bias
14  * accelSensitivity is the 3x3 matrix of sensitivity scale factors
15  * \tilde{accel}_k is the calibrated measurement of the accelerometer
16  * at time k
17  *
18  * After calibration, the optimized gyro measurement is given by
19  * \tilde{gyro}_k = gyro_k - bias - accelSensitivity * \tilde{accel}_k
20  */
21 void
22 gyroscope_calibration(Vector3f& bias,
23  Matrix3f& accelSensitivity,
24  Vector3f gyroSamples[],
25  Vector3f accelSamples[],
26  size_t n_samples)
27 {
28  // Assume the following measurement model:
29  // y = H*x
30  // where x is the vector of unknowns, and y is the measurement vector.
31  // the vector of unknowns is
32  // [a_x a_y a_z b_x]
33  // The measurement vector is
34  // [gyro_x]
35  // and the measurement matrix H is
36  // [accelSample_x accelSample_y accelSample_z 1]
37 
38  // Note that the individual solutions for x are given by
39  // (H^T \times H)^-1 \times H^T y
40  // Everything is identical except for y and x. So, transform it
41  // into block form X = (H^T \times H)^-1 \times H^T Y
42  // where each column of X contains the partial solution for each
43  // column of y.
44 
45  // Construct the matrix of accelerometer samples. Intermediate results
46  // are computed in "twice the precision that the source provides and the
47  // result deserves" by Kahan's thumbrule to prevent numerical problems.
48  Matrix<double, Dynamic, 4> H(n_samples, 4);
49  // And the matrix of gyro samples.
50  Matrix<double, Dynamic, 3> Y(n_samples, 3);
51  for (unsigned i = 0; i < n_samples; ++i) {
52  H.row(i) << accelSamples[i].transpose().cast<double>(), 1.0;
53  Y.row(i) << gyroSamples[i].transpose().cast<double>();
54  }
55 
56 #if 1
57  Matrix<double, 4, 3> result;
58  // Use the cholesky-based Penrose pseudoinverse method.
59  (H.transpose() * H).ldlt().solve(H.transpose()*Y, &result);
60 
61  // Transpose the result and return it to the caller. Cast back to float
62  // since there really isn't that much accuracy in the result.
63  bias = result.row(3).transpose().cast<float>();
64  accelSensitivity = result.block<3, 3>(0, 0).transpose().cast<float>();
65 #else
66  // TODO: Compare this result with a total-least-squares model.
67  size_t n = 4;
68  Matrix<double, Dynamic, 7> C;
69  C << H, Y;
70  SVD<Matrix<double, Dynamic, 7> > usv(C);
71  Matrix<double, 4, 3> V_ab = usv.matrixV().block<4, 3>(0, n);
72  Matrix<double, Dynamic, 3> V_bb = usv.matrixV().corner(BottomRight, n_samples-4, 3);
73  // X = -V_ab/V_bb;
74  // X^T = (A * B^-1)^T
75  // X^T = (B^-1^T * A^T)
76  // X^T = (B^T^-1 * A^T)
77  // V_bb is orthgonal but not orthonormal. QR decomposition
78  // should be very fast in this case.
79  Matrix<double, 3, 4> result;
80  V_bb.transpose().qr().solve(-V_ab.transpose(), &result);
81 
82  // Results only deserve single precision.
83  bias = result.col(3).cast<float>();
84  accelSensitivity = result.block<3, 3>(0, 0).cast<float>();
