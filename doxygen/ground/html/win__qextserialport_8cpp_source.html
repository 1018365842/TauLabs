252 {
253  QMutexLocker lock( mutex );
254  DWORD retVal = 0;
255  if (queryMode() == QextSerialPort::EventDriven) {
256  OVERLAPPED* newOverlapWrite = new OVERLAPPED;
257  ZeroMemory(newOverlapWrite, sizeof(OVERLAPPED));
258  newOverlapWrite->hEvent = CreateEvent(NULL, true, false, NULL);
259  if (WriteFile(Win_Handle, (void*)data, (DWORD)maxSize, & retVal, newOverlapWrite)) {
260  CloseHandle(newOverlapWrite->hEvent);

271  lastErr = E_WRITE_FAILED;
272  retVal = (DWORD)-1;

274  qDebug() << "serialport: couldn't cancel IO";
275  if(!CloseHandle(newOverlapWrite->hEvent))
276  qDebug() << "serialport: couldn't close OVERLAPPED handle";
277  delete newOverlapWrite;
278  }
279  } else if (!WriteFile(Win_Handle, (void*)data, (DWORD)maxSize, & retVal, NULL)) {
280  lastErr = E_WRITE_FAILED;
281  retVal = (DWORD)-1;
282  }

821 void QextSerialPort::onWinEvent(HANDLE h)
822 {

824  if(h == overlap.hEvent) {
825  if (eventMask & EV_RXCHAR) {
826  if (sender() != this && bytesAvailable() > 0)
827  emit readyRead();
828  }
829  if (eventMask & EV_TXEMPTY) {
830  /*
831  A write completed. Run through the list of OVERLAPPED writes, and if
832  they completed successfully, take them off the list and delete them.
833  Otherwise, leave them on there so they can finish.
834  */
835  qint64 totalBytesWritten = 0;
836  QList<OVERLAPPED*> overlapsToDelete;
837  foreach(OVERLAPPED* o, pendingWrites) {
838  DWORD numBytes = 0;
839  if (GetOverlappedResult(Win_Handle, o, & numBytes, false)) {
840  overlapsToDelete.append(o);
841  totalBytesWritten += numBytes;
842  } else if( GetLastError() != ERROR_IO_INCOMPLETE ) {
843  overlapsToDelete.append(o);

845  }
