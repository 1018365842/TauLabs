72 // for fixed-point arithmetic, we need minimum 32-bit long
73 // long long (64-bit) might be useful for multiplication and division
74 typedef long PFreal;
75 #define PFREAL_SHIFT 10
76 #define PFREAL_ONE (1 << PFREAL_SHIFT)
77 
78 #define IANGLE_MAX 1024
79 #define IANGLE_MASK 1023
80 
81 inline PFreal fmul(PFreal a, PFreal b)
82 {
83  return ((long long)(a))*((long long)(b)) >> PFREAL_SHIFT;
84 }
85 
86 inline PFreal fdiv(PFreal num, PFreal den)
87 {
88  long long p = (long long)(num) << (PFREAL_SHIFT * 2);
89  long long q = p / (long long)den;
90  long long r = q >> PFREAL_SHIFT;
91 
92  return r;
93 }

95 inline PFreal fsin(int iangle)
96 {
97  // warning: regenerate the table if IANGLE_MAX and PFREAL_SHIFT are changed!
98  static const PFreal tab[] =
99  {
100  3, 103, 202, 300, 394, 485, 571, 652,
101  726, 793, 853, 904, 947, 980, 1004, 1019,
102  1023, 1018, 1003, 978, 944, 901, 849, 789,
103  721, 647, 566, 479, 388, 294, 196, 97,
104  -4, -104, -203, -301, -395, -486, -572, -653,
105  -727, -794, -854, -905, -948, -981, -1005, -1020,
106  -1024, -1019, -1004, -979, -945, -902, -850, -790,
107  -722, -648, -567, -480, -389, -295, -197, -98,
108  3
109  };
110 
111  while (iangle < 0)
112  iangle += IANGLE_MAX;
113  iangle &= IANGLE_MASK;
114 
115  int i = (iangle >> 4);
116  PFreal p = tab[i];
117  PFreal q = tab[(i+1)];
118  PFreal g = (q - p);
119  return p + g * (iangle - i*16) / 16;
120 }

122 inline PFreal fcos(int iangle)
123 {
124  return fsin(iangle + (IANGLE_MAX >> 2));
125 }

130 information to be able to render them.

132 QxtFlowViewAnimator is responsible to move the slides during the
133 transition between slides, to achieve the effect similar to Cover Flow,
134 by changing the state.

136 QxtFlowViewSoftwareRenderer (or QxtFlowViewOpenGLRenderer) is
137 the actual 3-d renderer. It should render all slides given the state
