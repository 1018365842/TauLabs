333 void
334 lib3ds_quat_slerp(Lib3dsQuat c, Lib3dsQuat a, Lib3dsQuat b, Lib3dsFloat t)
335 {
336  Lib3dsDouble l;
337  Lib3dsDouble om,sinom;
338  Lib3dsDouble sp,sq;
339  Lib3dsQuat q;
340 
341  l=a[0]*b[0] + a[1]*b[1] + a[2]*b[2] + a[3]*b[3];
342  if ((1.0+l)>LIB3DS_EPSILON) {
343  if (fabs(l)>1.0f) l/=fabs(l);
344  om=acos(l);
345  sinom=sin(om);
346  if (fabs(sinom)>LIB3DS_EPSILON) {
347  sp=sin((1.0f-t)*om)/sinom;
348  sq=sin(t*om)/sinom;
349  }
350  else {
351  sp=1.0f-t;
352  sq=t;
353  }
354  c[0]=(Lib3dsFloat)(sp*a[0] + sq*b[0]);
355  c[1]=(Lib3dsFloat)(sp*a[1] + sq*b[1]);
356  c[2]=(Lib3dsFloat)(sp*a[2] + sq*b[2]);
357  c[3]=(Lib3dsFloat)(sp*a[3] + sq*b[3]);
358  }
359  else {
360  q[0]=-a[1];
361  q[1]=a[0];
362  q[2]=-a[3];
363  q[3]=a[2];
364  sp=sin((1.0-t)*LIB3DS_HALFPI);
365  sq=sin(t*LIB3DS_HALFPI);
366  c[0]=(Lib3dsFloat)(sp*a[0] + sq*q[0]);
367  c[1]=(Lib3dsFloat)(sp*a[1] + sq*q[1]);
368  c[2]=(Lib3dsFloat)(sp*a[2] + sq*q[2]);
369  c[3]=(Lib3dsFloat)(sp*a[3] + sq*q[3]);
370  }
371 }
