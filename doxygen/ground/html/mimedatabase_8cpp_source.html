173 bool HeuristicTextMagicMatcher::isTextFile(const QByteArray &data)
174 {
175  const int size = data.size();
176  for (int i = 0; i < size; i++) {
177  const char c = data.at(i);
178  if (c >= 0x01 && c < 0x09) // Sure-fire binary
179  return false;

204 bool MagicRule::matches(const QByteArray &data) const
205 {
206  // Quick check
207  const int dataSize = data.size();
208  if ((m_startPos + m_pattern.size()) >= dataSize)
209  return false;
210  // Most common: some string at position 0:
211  if (m_startPos == 0 && m_startPos == m_endPos)
212  return data.startsWith(m_pattern);
213  // Range
214  const int index = data.indexOf(m_pattern, m_startPos);
215  return index != -1 && index < m_endPos;

489 unsigned MimeType::matchesFile(Internal::FileMatchContext &c) const
490 {
491  // check globs
492  foreach (QRegExp pattern, m_d->globPatterns) {
493  if (pattern.exactMatch(c.fileName()))
494  return GlobMatchPriority;
495  }
496 
497  // Nope, try magic matchers on context data
498  if (m_d->magicMatchers.isEmpty())
499  return 0;
500 
501  const QByteArray data = c.data();
502  if (!data.isEmpty()) {
503  foreach (MimeTypeData::IMagicMatcherSharedPointer matcher, m_d->magicMatchers) {
504  if (matcher->matches(data))
505  return matcher->priority();
506  }
507  }
508  return 0;
509 }

580 void BaseMimeTypeParser::addGlobPattern(const QString &pattern, MimeTypeData *d) const
581 {
582  if (pattern.isEmpty())
583  return;
584  // Collect patterns as a QRegExp list and filter out the plain
585  // suffix ones for our suffix list. Use first one as preferred
586  const QRegExp wildCard(pattern, Qt::CaseSensitive, QRegExp::Wildcard);
587  if (!wildCard.isValid()) {
588  qWarning("%s: Invalid wildcard '%s'.",
589  Q_FUNC_INFO, pattern.toUtf8().constData());
590  return;
591  }
592 
593  d->globPatterns.push_back(wildCard);
594  if (m_suffixPattern.exactMatch(pattern)) {
595  const QString suffix = pattern.right(pattern.size() - 2);
596  d->suffixes.push_back(suffix);
597  if (d->preferredSuffix.isEmpty())
598  d->preferredSuffix = suffix;
599  }
600 }

602 BaseMimeTypeParser::ParseStage BaseMimeTypeParser::nextStage(ParseStage currentStage, const QStringRef &startElement)
603 {
604  switch (currentStage) {
605  case ParseBeginning:
606  if (startElement == QLatin1String(mimeInfoTagC))
607  return ParseMimeInfo;
608  if (startElement == QLatin1String(mimeTypeTagC))
609  return ParseMimeType;
610  return ParseError;
611  case ParseMimeInfo:
612  return startElement == QLatin1String(mimeTypeTagC) ? ParseMimeType : ParseError;
613  case ParseMimeType:
614  case ParseComment:
615  case ParseGlobPattern:
616  case ParseSubClass:
617  case ParseAlias:
618  case ParseOtherMimeTypeSubTag:
619  case ParseMagicMatchRule:
620  if (startElement == QLatin1String(mimeTypeTagC)) // Sequence of <mime-type>
621  return ParseMimeType;
622  if (startElement == QLatin1String(commentTagC ))
623  return ParseComment;
624  if (startElement == QLatin1String(globTagC))
625  return ParseGlobPattern;
626  if (startElement == QLatin1String(subClassTagC))
627  return ParseSubClass;
628  if (startElement == QLatin1String(aliasTagC))
629  return ParseAlias;
630  if (startElement == QLatin1String(magicTagC))
631  return ParseMagic;
632  return ParseOtherMimeTypeSubTag;
633  case ParseMagic:
634  if (startElement == QLatin1String(matchTagC))
635  return ParseMagicMatchRule;
636  break;
637  case ParseError:
638  break;
639  }
640  return ParseError;
641 }

655 // Evaluate a magic match rule like
656 // <match value="must be converted with BinHex" type="string" offset="11"/>
657 // <match value="0x9501" type="big16" offset="0:64"/>
658 static bool addMagicMatchRule(const QXmlStreamAttributes &atts,
659  MagicRuleMatcher *ruleMatcher,
660  QString *errorMessage)
661 {
662  const QString type = atts.value(QLatin1String(matchTypeAttributeC)).toString();
663  if (type != QLatin1String(matchStringTypeValueC)) {
664  qWarning("%s: match type %s is not supported.", Q_FUNC_INFO, type.toUtf8().constData());
665  return true;
666  }
667  const QString value = atts.value(QLatin1String(matchValueAttributeC)).toString();
668  if (value.isEmpty()) {
669  *errorMessage = QString::fromLatin1("Empty match value detected.");
670  return false;
671  }
672  // Parse for offset as "1" or "1:10"
673  int startPos, endPos;
674  const QString offsetS = atts.value(QLatin1String(matchOffsetAttributeC)).toString();
675  const int colonIndex = offsetS.indexOf(QLatin1Char(':'));
676  const QString startPosS = colonIndex == -1 ? offsetS : offsetS.mid(0, colonIndex);
677  const QString endPosS = colonIndex == -1 ? offsetS : offsetS.mid(colonIndex + 1);
678  if (!parseNumber(startPosS, &startPos, errorMessage) || !parseNumber(endPosS, &endPos, errorMessage))
679  return false;
680  if (debugMimeDB)
681  qDebug() << Q_FUNC_INFO << value << startPos << endPos;
682  ruleMatcher->add(QSharedPointer<MagicRule>(MagicRule::createStringRule(value, startPos, endPos)));
683  return true;
684 }

686 bool BaseMimeTypeParser::parse(QIODevice *dev, const QString &fileName, QString *errorMessage)
687 {
688  MimeTypeData data;
689  MagicRuleMatcher *ruleMatcher = 0;
690 
691  QXmlStreamReader reader(dev);
692  ParseStage ps = ParseBeginning;
693  while (!reader.atEnd()) {
694  switch (reader.readNext()) {
695  case QXmlStreamReader::StartElement:
696  ps = nextStage(ps, reader.name());
697  switch (ps) {
698  case ParseMimeType: { // start parsing a type
699  const QString type = reader.attributes().value(QLatin1String(mimeTypeAttributeC)).toString();
700  if (type.isEmpty()) {
701  reader.raiseError(QString::fromLatin1("Missing 'type'-attribute"));
702  } else {
703  data.type = type;
704  }
705  }
706  break;
707  case ParseGlobPattern:
708  addGlobPattern(reader.attributes().value(QLatin1String(patternAttributeC)).toString(), &data);
709  break;
710  case ParseSubClass: {
711  const QString inheritsFrom = reader.attributes().value(QLatin1String(mimeTypeAttributeC)).toString();
712  if (!inheritsFrom.isEmpty())
713  data.subClassesOf.push_back(inheritsFrom);
714  }
715  break;
716  case ParseComment: {
717  // comments have locale attributes. We want the default, English one
718  QString locale = reader.attributes().value(QLatin1String(localeAttributeC)).toString();
719  const QString comment = QCoreApplication::translate("MimeType", reader.readElementText().toAscii());
720  if (locale.isEmpty()) {
721  data.comment = comment;
722  } else {
723  data.localeComments.insert(locale, comment);
724  }
725  }
726  break;
727  case ParseAlias: {
728  const QString alias = reader.attributes().value(QLatin1String(mimeTypeAttributeC)).toString();
729  if (!alias.isEmpty())
730  data.aliases.push_back(alias);
731  }
732  break;
733  case ParseMagic: {
734  int priority = 0;
735  const QString priorityS = reader.attributes().value(QLatin1String(priorityAttributeC)).toString();
736  if (!priorityS.isEmpty()) {
737  if (!parseNumber(priorityS, &priority, errorMessage))
738  return false;
739 
740  }
741  ruleMatcher = new MagicRuleMatcher;
742  ruleMatcher->setPriority(priority);
743  }
744  break;
745  case ParseMagicMatchRule:
746  if (!addMagicMatchRule(reader.attributes(), ruleMatcher, errorMessage))
747  return false;
748  break;
749  case ParseError:
750  reader.raiseError(QString::fromLatin1("Unexpected element <%1>").arg(reader.name().toString()));
751  break;
752  default:
753  break;
754  } // switch nextStage
755  break;
756  // continue switch QXmlStreamReader::Token...
757  case QXmlStreamReader::EndElement: // Finished element
758  if (reader.name() == QLatin1String(mimeTypeTagC)) {
759  if (!process(MimeType(data), errorMessage))
760  return false;
761  data.clear();
762  } else {
763  // Finished a match sequence
764  if (reader.name() == QLatin1String(QLatin1String(magicTagC))) {
765  data.magicMatchers.push_back(QSharedPointer<IMagicMatcher>(ruleMatcher));
766  ruleMatcher = 0;
767  }
768  }
769  break;
770 
771  default:
772  break;
773  } // switch reader.readNext()
774  }
775 
776  if (reader.hasError()) {
777  *errorMessage = QString::fromLatin1("An error has been encountered at line %1 of %2: %3:").arg(reader.lineNumber()).arg(fileName, reader.errorString());
778  return false;
779  }
780  return true;
781 }

802 /* MimeDatabasePrivate: Requirements for storage:
803  * - Must be robust in case of incomplete hierachies, dangling entries
804  * - Plugins will not load and register their mime types in order
805  * of inheritance.
806  * - Multiple inheritance (several subClassesOf) can occur
807  * - Provide quick lookup by name
808  * - Provide quick lookup by file type.
809  * This basically rules out some pointer-based tree, so the structure choosen
810  * is:
811  * - An alias map <QString->QString> for mapping aliases to types
812  * - A Map <QString-MimeMapEntry> for the types (MimeMapEntry being a pair of
813  * MimeType and (hierarchy) level.
814  * - A map <QString->QString> representing parent->child relations (enabling
815  * recursing over children)
816  * Using strings avoids dangling pointers.
817  * The hierarchy level is used for mapping by file types. When findByFile()
818  * is first called after addMimeType() it recurses over the hierarchy and sets
819  * the hierarchy level of the entries accordingly (0 toplevel, 1 first
820  * order...). It then does several passes over the type map, checking the
821  * globs for maxLevel, maxLevel-1....until it finds a match (idea being to
822  * to check the most specific types first). Starting a recursion from the
823  * leaves is not suitable since it will hit parent nodes several times. */

903 bool MimeDatabasePrivate::addMimeType(MimeType mt)
904 {
905  if (!mt)
906  return false;
907 
908  const QString type = mt.type();
909  // Hack: Add a magic text matcher to "text/plain" and the fallback matcher to
910  // binary types "application/octet-stream"
911  if (type == QLatin1String(textTypeC)) {
912  mt.addMagicMatcher(QSharedPointer<IMagicMatcher>(new Internal::HeuristicTextMagicMatcher));
913  } else {
914  if (type == QLatin1String(binaryTypeC))
915  mt.addMagicMatcher(QSharedPointer<IMagicMatcher>(new Internal::BinaryMatcher));
916  }
917  // insert the type.
918  m_typeMimeTypeMap.insert(type, MimeMapEntry(mt));
919  // Register the children, resolved via alias map. Note that it is still
920  // possible that aliases end up in the map if the parent classes are not inserted
921  // at this point (thus their aliases not known).
922  const QStringList subClassesOf = mt.subClassesOf();
923  if (!subClassesOf.empty()) {
924  const QStringList::const_iterator socend = subClassesOf.constEnd();
925  for (QStringList::const_iterator soit = subClassesOf.constBegin(); soit != socend; ++soit)
926  m_parentChildrenMap.insert(resolveAlias(*soit), type);
927  }
928  // register aliasses
929  const QStringList aliases = mt.aliases();
930  if (!aliases.empty()) {
931  const QStringList::const_iterator cend = aliases.constEnd();
932  for (QStringList::const_iterator it = aliases.constBegin(); it != cend; ++it)
933  m_aliasMap.insert(*it, type);
934  }
935  m_maxLevel = -1; // Mark as dirty
936  return true;
937 }

945 void MimeDatabasePrivate::raiseLevelRecursion(MimeMapEntry &e, int level)
946 {
947  if (e.level == Dangling || e.level < level)
948  e.level = level;
949  if (m_maxLevel < level)
950  m_maxLevel = level;
951  // At all events recurse over children since nodes might have been
952  // added.
953  const QStringList childTypes = m_parentChildrenMap.values(e.type.type());
954  if (childTypes.empty())
955  return;
956  // look them up in the type->mime type map
957  const int nextLevel = level + 1;
958  const TypeMimeTypeMap::iterator tm_end = m_typeMimeTypeMap.end();
959  const QStringList::const_iterator cend = childTypes.constEnd();
960  for (QStringList::const_iterator it = childTypes.constBegin(); it != cend; ++it) {
961  const TypeMimeTypeMap::iterator tm_it = m_typeMimeTypeMap.find(resolveAlias(*it));
962  if (tm_it == tm_end) {
963  qWarning("%s: Inconsistent mime hierarchy detected, child %s of %s cannot be found.",
964  Q_FUNC_INFO, it->toUtf8().constData(), e.type.type().toUtf8().constData());
965  } else {
966  raiseLevelRecursion(*tm_it, nextLevel);
967  }
968  }
969 }

971 void MimeDatabasePrivate::determineLevels()
972 {
973  // Loop over toplevels and recurse down their hierarchies.
974  // Determine top levels by subtracting the children from the parent
975  // set. Note that a toplevel at this point might have 'subclassesOf'
976  // set to some class that is not in the DB, so, checking for an empty
977  // 'subclassesOf' set is not sufficient to find the toplevels.
978  // First, take the parent->child entries whose parent exists and build
979  // sets of parents/children
980  QSet<QString> parentSet, childrenSet;
981  const ParentChildrenMap::const_iterator pcend = m_parentChildrenMap.constEnd();
982  for (ParentChildrenMap::const_iterator it = m_parentChildrenMap.constBegin(); it != pcend; ++it)
983  if (m_typeMimeTypeMap.contains(it.key())) {
984  parentSet.insert(it.key());
985  childrenSet.insert(it.value());
986  }
987  const QSet<QString> topLevels = parentSet.subtract(childrenSet);
988  if (debugMimeDB)
989  qDebug() << Q_FUNC_INFO << "top levels" << topLevels;
990  const TypeMimeTypeMap::iterator tm_end = m_typeMimeTypeMap.end();
991  const QSet<QString>::const_iterator tl_cend = topLevels.constEnd();
992  for (QSet<QString>::const_iterator tl_it = topLevels.constBegin(); tl_it != tl_cend; ++tl_it) {
993  const TypeMimeTypeMap::iterator tm_it = m_typeMimeTypeMap.find(resolveAlias(*tl_it));
994  if (tm_it == tm_end) {
995  qWarning("%s: Inconsistent mime hierarchy detected, top level element %s cannot be found.",
996  Q_FUNC_INFO, tl_it->toUtf8().constData());
997  } else {
998  raiseLevelRecursion(tm_it.value(), 0);
999  }
1000  }
1001 }

1038 MimeType MimeDatabasePrivate::findByFile(const QFileInfo &f, unsigned *priorityPtr) const
1039 {
1040  typedef QList<MimeMapEntry> MimeMapEntryList;
1041 
1042  // Is the hierarchy set up in case we find several matches?
1043  if (m_maxLevel < 0) {
1044  MimeDatabasePrivate *db = const_cast<MimeDatabasePrivate *>(this);
1045  db->determineLevels();
1046  }
1047  // Starting from max level (most specific): Try to find a match of
1048  // best (max) priority. Return if a glob match triggers.
1049  *priorityPtr = 0;
1050  unsigned maxPriority = 0;
1051  MimeType rc;
1052  Internal::FileMatchContext context(f);
1053  const TypeMimeTypeMap::const_iterator cend = m_typeMimeTypeMap.constEnd();
1054  for (int level = m_maxLevel; level >= 0; level--)
1055  for (TypeMimeTypeMap::const_iterator it = m_typeMimeTypeMap.constBegin(); it != cend; ++it)
1056  if (it.value().level == level) {
1057  const unsigned priority = it.value().type.matchesFile(context);
1058  if (debugMimeDB > 1)
1059  qDebug() << "pass" << level << it.value().type.type() << " matches " << priority;
1060  if (priority)
1061  if (priority > maxPriority) {
1062  rc = it.value().type;
1063  maxPriority = priority;
1064  // Glob (exact) match?! We are done
1065  if (maxPriority == MimeType::GlobMatchPriority) {
1066  *priorityPtr = priority;
1067  return rc;
1068  }
1069  }
1070  }
1071  return rc;
1072 }
