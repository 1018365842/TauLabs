221 double QwtKnob::getValue( const QPoint &pos )
222 {
223  const double dx = rect().center().x() - pos.x();
224  const double dy = rect().center().y() - pos.y();
225 
226  const double arc = qAtan2( -dx, dy ) * 180.0 / M_PI;
227 
228  double newValue = 0.5 * ( minValue() + maxValue() )
229  + ( arc + d_data->nTurns * 360.0 ) * ( maxValue() - minValue() )
230  / d_data->totalAngle;
231 
232  const double oneTurn = qFabs( maxValue() - minValue() ) * 360.0 / d_data->totalAngle;
233  const double eqValue = value() + mouseOffset();
234 
235  if ( qFabs( newValue - eqValue ) > 0.5 * oneTurn )
236  {
237  if ( newValue < eqValue )
238  newValue += oneTurn;
239  else
240  newValue -= oneTurn;
241  }
242 
243  return newValue;
244 }

254 void QwtKnob::getScrollMode( const QPoint &pos,
255  QwtAbstractSlider::ScrollMode &scrollMode, int &direction ) const
256 {
257  const int r = d_data->knobRect.width() / 2;
258 
259  const int dx = d_data->knobRect.x() + r - pos.x();
260  const int dy = d_data->knobRect.y() + r - pos.y();
261 
262  if ( ( dx * dx ) + ( dy * dy ) <= ( r * r ) ) // point is inside the knob
263  {
264  scrollMode = QwtAbstractSlider::ScrMouse;
265  direction = 0;
266  }
267  else // point lies outside
268  {
269  scrollMode = QwtAbstractSlider::ScrTimer;
270  double arc = qAtan2( double( -dx ), double( dy ) ) * 180.0 / M_PI;
271  if ( arc < d_data->angle )
272  direction = -1;
273  else if ( arc > d_data->angle )
274  direction = 1;
275  else
276  direction = 0;
277  }
278 }

378 void QwtKnob::drawKnob( QPainter *painter,
379  const QRectF &knobRect ) const
380 {
381  double dim = qMin( knobRect.width(), knobRect.height() );
382  dim -= d_data->borderWidth * 0.5;
383 
384  QRectF aRect( 0, 0, dim, dim );
385  aRect.moveCenter( knobRect.center() );
386 
387  QPen pen( Qt::NoPen );
388  if ( d_data->borderWidth > 0 )
389  {
390  QColor c1 = palette().color( QPalette::Light );
391  QColor c2 = palette().color( QPalette::Dark );
392 
393  QLinearGradient gradient( aRect.topLeft(), aRect.bottomRight() );
394  gradient.setColorAt( 0.0, c1 );
395  gradient.setColorAt( 0.3, c1 );
396  gradient.setColorAt( 0.7, c2 );
397  gradient.setColorAt( 1.0, c2 );
398 
399  pen = QPen( gradient, d_data->borderWidth );
400  }
401 
402  QBrush brush;
403  switch( d_data->knobStyle )
404  {
405  case QwtKnob::Raised:
406  {
407  double off = 0.3 * knobRect.width();
408  QRadialGradient gradient( knobRect.center(),
409  knobRect.width(), knobRect.topLeft() + QPoint( off, off ) );
410 
411  gradient.setColorAt( 0.0, palette().color( QPalette::Midlight ) );
412  gradient.setColorAt( 1.0, palette().color( QPalette::Button ) );
413 
414  brush = QBrush( gradient );
415 
416  break;
417  }
418  case QwtKnob::Sunken:
419  {
420  QLinearGradient gradient(
421  knobRect.topLeft(), knobRect.bottomRight() );
422  gradient.setColorAt( 0.0, palette().color( QPalette::Mid ) );
423  gradient.setColorAt( 0.5, palette().color( QPalette::Button ) );
424  gradient.setColorAt( 1.0, palette().color( QPalette::Midlight ) );
425  brush = QBrush( gradient );
426 
427  break;
428  }
429  default:
430  brush = palette().brush( QPalette::Button );
431  }
432 
433  painter->setPen( pen );

445 void QwtKnob::drawMarker( QPainter *painter,
446  const QRectF &rect, double angle ) const
447 {
448  if ( d_data->markerStyle == NoMarker || !isValid() )
449  return;
450 
451  const double radians = angle * M_PI / 180.0;
452  const double sinA = -qFastSin( radians );
453  const double cosA = qFastCos( radians );
454 
455  const double xm = rect.center().x();
456  const double ym = rect.center().y();
457  const double margin = 4.0;
458 
459  double radius = 0.5 * ( rect.width() - d_data->borderWidth ) - margin;
460  if ( radius < 1.0 )
461  radius = 1.0;
462 
463  switch ( d_data->markerStyle )
464  {
465  case Notch:
466  case Nub:
467  {
468  const double dotWidth =
469  qMin( double( d_data->markerSize ), radius);
470 
471  const double dotCenterDist = radius - 0.5 * dotWidth;
472  if ( dotCenterDist > 0.0 )
473  {
474  const QPointF center( xm - sinA * dotCenterDist,
475  ym - cosA * dotCenterDist );
476 
477  QRectF ellipse( 0.0, 0.0, dotWidth, dotWidth );
478  ellipse.moveCenter( center );
479 
480  QColor c1 = palette().color( QPalette::Light );
481  QColor c2 = palette().color( QPalette::Mid );
482 
483  if ( d_data->markerStyle == Notch )
484  qSwap( c1, c2 );
485 
486  QLinearGradient gradient(
487  ellipse.topLeft(), ellipse.bottomRight() );
488  gradient.setColorAt( 0.0, c1 );
489  gradient.setColorAt( 1.0, c2 );
490 
491  painter->setPen( Qt::NoPen );
492  painter->setBrush( gradient );
493 
494  painter->drawEllipse( ellipse );
495  }
496  break;
497  }
498  case Dot:
499  {
500  const double dotWidth =
501  qMin( double( d_data->markerSize ), radius);
502 
503  const double dotCenterDist = radius - 0.5 * dotWidth;
504  if ( dotCenterDist > 0.0 )
505  {
506  const QPointF center( xm - sinA * dotCenterDist,
507  ym - cosA * dotCenterDist );
508 
509  QRectF ellipse( 0.0, 0.0, dotWidth, dotWidth );
510  ellipse.moveCenter( center );
511 
512  painter->setPen( Qt::NoPen );
513  painter->setBrush( palette().color( QPalette::ButtonText ) );
514  painter->drawEllipse( ellipse );
515  }
516 
517  break;
518  }
519  case Tick:
520  {
521  const double rb = qMax( radius - d_data->markerSize, 1.0 );
522  const double re = radius;
523 
524  const QLine line( xm - sinA * rb, ym - cosA * rb,
525  xm - sinA * re, ym - cosA * re );
526 
527  QPen pen( palette().color( QPalette::ButtonText ), 0 );
528  pen.setCapStyle( Qt::FlatCap );
529  painter->setPen( pen );
530  painter->drawLine ( line );
531 
532  break;
533  }
534 #if 0
535  case Triangle:
536  {
537  const double rb = qMax( radius - d_data->markerSize, 1.0 );
538  const double re = radius;
539 
540  painter->translate( rect.center() );
541  painter->rotate( angle - 90.0 );
542 
543  QPolygonF polygon;
544  polygon += QPointF( re, 0.0 );
545  polygon += QPointF( rb, 0.5 * ( re - rb ) );
546  polygon += QPointF( rb, -0.5 * ( re - rb ) );
547 
548  painter->setPen( Qt::NoPen );
549  painter->setBrush( palette().color( QPalette::Text ) );
550  painter->drawPolygon( polygon );
551  break;
552  }
553 #endif
554  default:
555  break;
556  }
557 }

616 void QwtKnob::recalcAngle()
617 {
618  //
619  // calculate the angle corresponding to the value
620  //
621  if ( maxValue() == minValue() )
622  {
623  d_data->angle = 0;
624  d_data->nTurns = 0;
625  }
626  else
627  {
628  d_data->angle = ( value() - 0.5 * ( minValue() + maxValue() ) )
629  / ( maxValue() - minValue() ) * d_data->totalAngle;
630  d_data->nTurns = qFloor( ( d_data->angle + 180.0 ) / 360.0 );
631  d_data->angle = d_data->angle - d_data->nTurns * 360.0;
632  }
633 }
