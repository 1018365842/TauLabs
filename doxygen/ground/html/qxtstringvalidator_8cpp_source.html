42 QModelIndex QxtStringValidatorPrivate::lookupPartialMatch(const QString &value) const
43 {
44  //a empty string is always invalid
45  if (value.isEmpty())
46  return QModelIndex();
47 
48  Qt::MatchFlags matchFlags = Qt::MatchStartsWith | userFlags;
49  if (cs == Qt::CaseSensitive)
50  matchFlags |= Qt::MatchCaseSensitive;
51 
52  return lookup(value, matchFlags);
53 }

55 QModelIndex QxtStringValidatorPrivate::lookupExactMatch(const QString &value) const
56 {
57  Qt::MatchFlags matchFlags = Qt::MatchFixedString | userFlags;
58  if (cs == Qt::CaseSensitive)
59  matchFlags |= Qt::MatchCaseSensitive;
60 
61  return lookup(value, matchFlags);

64 QModelIndex QxtStringValidatorPrivate::lookup(const QString &value, const Qt::MatchFlags &matchFlags) const
65 {
66  QModelIndex startIndex = lookupStartModelIndex.isValid() ? lookupStartModelIndex : model->index(0, 0);
67 
68  QModelIndexList list = model->match(startIndex, lookupRole, value, 1, matchFlags);
69 
70  if (list.size() > 0)
71  return list[0];
72  return QModelIndex();
73 }

76 /*
77  \class QxtStringValidator
78  \inmodule QxtGui
79  \brief The QxtStringValidator class provides validation on a QStringList or a QAbstractItemModel
80 
81  It provides a String based validation in a stringlist or a custom model.
82  QxtStringValidator uses QAbstractItemModel::match() to validate the input.
83  For a partial match it returns QValidator::Intermediate and for a full match QValidator::Acceptable.
84 
85  Example usage:
86  \code
87 
88  QLineEdit * testLineEdit = new QLineEdit();
89 
90  QStringList testList;
91  testList << "StringTestString" << "sTrInGCaSe"<< "StringTest"<< "String"<< "Foobar"<< "BarFoo"<< "QxtLib";
92 
93  QxtStringValidator *validator = new QxtStringValidator(ui.lineEdit);
94  validator->setStringList(testList);
95 
96  //change lookup case sensitivity
97  validator->setCaseSensitivity(Qt::CaseInsensitive);
98 
99  testLineEdit->setValidator(validator);
100 
101  \endcode
102  */

155 QValidator::State QxtStringValidator::validate(QString & input, int & pos) const
156 {
157  Q_UNUSED(pos);
158 
159  // no model or a empty model has only Acceptable values (like no validator was set)
160  if (!qxt_d().model)
161  return QValidator::Acceptable;
162 
163  if (qxt_d().model->rowCount() == 0)
164  return QValidator::Acceptable;
165 
166  if (input.isEmpty())
167  return QValidator::Intermediate;
168 
169  if (qxt_d().lookupExactMatch(input).isValid())
170  {
171  qDebug() << input << " is QValidator::Acceptable";
172  return QValidator::Acceptable;
173  }
174 
175  if (qxt_d().lookupPartialMatch(input).isValid())
176  {
177  qDebug() << input << " is QValidator::Intermediate";
178  return QValidator::Intermediate;
179  }
180 
181  qDebug() << input << " is QValidator::Invalid";
182  return QValidator::Invalid;
183 }
