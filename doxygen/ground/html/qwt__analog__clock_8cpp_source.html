175 void QwtAnalogClock::drawNeedle( QPainter *painter, const QPointF &center,
176  double radius, double dir, QPalette::ColorGroup colorGroup ) const
177 {
178  Q_UNUSED( dir );
179 
180  if ( isValid() )
181  {
182  const double hours = value() / ( 60.0 * 60.0 );
183  const double minutes =
184  ( value() - qFloor(hours) * 60.0 * 60.0 ) / 60.0;
185  const double seconds = value() - qFloor(hours) * 60.0 * 60.0
186  - qFloor(minutes) * 60.0;
187 
188  double angle[NHands];

190  angle[MinuteHand] = 360.0 * minutes / 60.0;
191  angle[SecondHand] = 360.0 * seconds / 60.0;
192

194  {

196  if ( direction() == Clockwise )
197  d = 360.0 - d;

202  }
203  }
