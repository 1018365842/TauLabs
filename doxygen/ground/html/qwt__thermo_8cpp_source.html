41 QVector<double> qwtTickList( const QwtScaleDiv &scaleDiv, double value )
42 {
43  QVector<double> values;
44 
45  double lowerLimit = scaleDiv.interval().minValue();
46  double upperLimit = scaleDiv.interval().maxValue();
47 
48  if ( upperLimit < lowerLimit )
49  qSwap( lowerLimit, upperLimit );
50 
51  if ( value < lowerLimit )
52  return values;
53 
54  if ( value < upperLimit )
55  upperLimit = value;
56 
57  values += lowerLimit;
58 
59  for ( int tickType = QwtScaleDiv::MinorTick;
60  tickType < QwtScaleDiv::NTickTypes; tickType++ )
61  {
62  const QList<double> ticks = scaleDiv.ticks( tickType );
63 
64  for ( int i = 0; i < ticks.count(); i++ )
65  {
66  const double v = ticks[i];
67  if ( v > lowerLimit && v < upperLimit )
68  values += v;
69  }
70  }
71 
72  values += upperLimit;
73 
74  return values;
75 }

337 {

339  const int bw = d_data->borderWidth + d_data->spacing;
340  const bool inverted = ( maxValue() < minValue() );
341 
342  int from, to;
343 
344  if ( d_data->orientation == Qt::Horizontal )
345  {
346  from = tRect.left();
347  to = tRect.right();
348 
349  if ( d_data->rangeFlags & QwtInterval::ExcludeMinimum )
350  {
351  if ( inverted )
352  to++;
353  else
354  from--;
355  }
356  if ( d_data->rangeFlags & QwtInterval::ExcludeMaximum )
357  {
358  if ( inverted )
359  from--;
360  else
361  to++;
362  }
363 
364  switch ( d_data->scalePos )
365  {
366  case TopScale:
367  {
368  scaleDraw()->setAlignment( QwtScaleDraw::TopScale );
369  scaleDraw()->move( from, tRect.top() - bw );
370  scaleDraw()->setLength( to - from );
371  break;
372  }
373 
374  case BottomScale:
375  case NoScale:
376  default:
377  {
378  scaleDraw()->setAlignment( QwtScaleDraw::BottomScale );
379  scaleDraw()->move( from, tRect.bottom() + bw );
380  scaleDraw()->setLength( to - from );
381  break;
382  }
383  }
384 
385  d_data->map.setPaintInterval( from, to );
386  }
387  else // Qt::Vertical
388  {
389  from = tRect.top();
390  to = tRect.bottom();
391 
392  if ( d_data->rangeFlags & QwtInterval::ExcludeMinimum )
393  {
394  if ( inverted )
395  from--;
396  else
397  to++;
398  }
399  if ( d_data->rangeFlags & QwtInterval::ExcludeMaximum )
400  {
401  if ( inverted )
402  to++;
403  else
404  from--;
405  }

407  switch ( d_data->scalePos )
408  {
409  case RightScale:

412  scaleDraw()->move( tRect.right() + bw, from );

619 void QwtThermo::drawLiquid(
620  QPainter *painter, const QRect &pipeRect ) const
621 {
622  painter->save();
623  painter->setClipRect( pipeRect, Qt::IntersectClip );
624 
625  const bool inverted = ( maxValue() < minValue() );
626  if ( d_data->colorMap != NULL )
627  {
628  QwtInterval interval( d_data->minValue, d_data->maxValue );
629  interval = interval.normalized();
630 
631  // Because the positions of the ticks are rounded
632  // we calculate the colors for the rounded tick values
633 
634  QVector<double> values = qwtTickList(
635  scaleDraw()->scaleDiv(), d_data->value );
636 
637  if ( d_data->map.isInverting() )
638  qSort( values.begin(), values.end(), qGreater<double>() );
639  else
640  qSort( values.begin(), values.end(), qLess<double>() );
641 
642  int from;
643  if ( !values.isEmpty() )
644  {
645  from = qRound( d_data->map.transform( values[0] ) );
646  qwtDrawLine( painter, from,
647  d_data->colorMap->color( interval, values[0] ),
648  pipeRect, d_data->orientation );
649  }
650 
651  for ( int i = 1; i < values.size(); i++ )
652  {
653  const int to = qRound( d_data->map.transform( values[i] ) );
654 
655  for ( int pos = from + 1; pos < to; pos++ )
656  {
657  const double v = d_data->map.invTransform( pos );
658 
659  qwtDrawLine( painter, pos,
660  d_data->colorMap->color( interval, v ),
661  pipeRect, d_data->orientation );
662  }
663  qwtDrawLine( painter, to,
664  d_data->colorMap->color( interval, values[i] ),
665  pipeRect, d_data->orientation );
666 
667  from = to;
668  }
669  }
670  else
671  {
672  const int tval = qRound( d_data->map.transform( d_data->value ) );
673 
674  QRect fillRect = pipeRect;
675  if ( d_data->orientation == Qt::Horizontal )
676  {
677  if ( inverted )
678  fillRect.setLeft( tval );
679  else
680  fillRect.setRight( tval );
681  }
682  else // Qt::Vertical
683  {
684  if ( inverted )
685  fillRect.setBottom( tval );
686  else
687  fillRect.setTop( tval );
688  }
689 
690  if ( d_data->alarmEnabled &&
691  d_data->value >= d_data->alarmLevel )
692  {
693  QRect alarmRect = fillRect;
694 
695  const int taval = qRound( d_data->map.transform( d_data->alarmLevel ) );
696  if ( d_data->orientation == Qt::Horizontal )
697  {
698  if ( inverted )
699  alarmRect.setRight( taval );
700  else
701  alarmRect.setLeft( taval );
702  }
703  else
704  {
705  if ( inverted )
706  alarmRect.setTop( taval );
707  else
708  alarmRect.setBottom( taval );
709  }
710 
711  fillRect = QRegion( fillRect ).subtracted( alarmRect ).boundingRect();
712 
713  painter->fillRect( alarmRect, palette().brush( QPalette::Highlight ) );
714  }
715 
716  painter->fillRect( fillRect, palette().brush( QPalette::ButtonText ) );
717  }
718 
719  painter->restore();
720 }

1004 QSize QwtThermo::minimumSizeHint() const
1005 {

1009  {

1011  const int sdLength = scaleDraw()->minLength( font() );
1012 
1013  w = sdLength;
1014  h = d_data->pipeWidth + sdExtent +
1015  d_data->borderWidth + d_data->spacing;
1016 
1017  }

1019  {
1020  w = 200;
1021  h = d_data->pipeWidth;
1022  }
1023 
1024  if ( d_data->orientation == Qt::Vertical )
1025  qSwap( w, h );
1026 
1027  w += 2 * d_data->borderWidth;
1028  h += 2 * d_data->borderWidth;
1029 
1030  int left, right, top, bottom;
1031  getContentsMargins( &left, &top, &right, &bottom );
1032  w += left + right;
