32 template<typename XprType, typename MatrixType>
33 void ei_compute_inverse_in_size2_case(const XprType& matrix, MatrixType* result)
34 {
35  typedef typename MatrixType::Scalar Scalar;
36  const Scalar invdet = Scalar(1) / matrix.determinant();
37  result->coeffRef(0,0) = matrix.coeff(1,1) * invdet;
38  result->coeffRef(1,0) = -matrix.coeff(1,0) * invdet;
39  result->coeffRef(0,1) = -matrix.coeff(0,1) * invdet;
40  result->coeffRef(1,1) = matrix.coeff(0,0) * invdet;
41 }

43 template<typename XprType, typename MatrixType>
44 bool ei_compute_inverse_in_size2_case_with_check(const XprType& matrix, MatrixType* result)
45 {
46  typedef typename MatrixType::Scalar Scalar;
47  const Scalar det = matrix.determinant();
48  if(ei_isMuchSmallerThan(det, matrix.cwise().abs().maxCoeff())) return false;
49  const Scalar invdet = Scalar(1) / det;
50  result->coeffRef(0,0) = matrix.coeff(1,1) * invdet;
51  result->coeffRef(1,0) = -matrix.coeff(1,0) * invdet;
52  result->coeffRef(0,1) = -matrix.coeff(0,1) * invdet;
53  result->coeffRef(1,1) = matrix.coeff(0,0) * invdet;
54  return true;
55 }

57 template<typename Derived, typename OtherDerived>
58 void ei_compute_inverse_in_size3_case(const Derived& matrix, OtherDerived* result)
59 {
60  typedef typename Derived::Scalar Scalar;
61  const Scalar det_minor00 = matrix.minor(0,0).determinant();
62  const Scalar det_minor10 = matrix.minor(1,0).determinant();
63  const Scalar det_minor20 = matrix.minor(2,0).determinant();
64  const Scalar invdet = Scalar(1) / ( det_minor00 * matrix.coeff(0,0)
65  - det_minor10 * matrix.coeff(1,0)
66  + det_minor20 * matrix.coeff(2,0) );
67  result->coeffRef(0, 0) = det_minor00 * invdet;
68  result->coeffRef(0, 1) = -det_minor10 * invdet;
69  result->coeffRef(0, 2) = det_minor20 * invdet;
70  result->coeffRef(1, 0) = -matrix.minor(0,1).determinant() * invdet;
71  result->coeffRef(1, 1) = matrix.minor(1,1).determinant() * invdet;
72  result->coeffRef(1, 2) = -matrix.minor(2,1).determinant() * invdet;
73  result->coeffRef(2, 0) = matrix.minor(0,2).determinant() * invdet;
74  result->coeffRef(2, 1) = -matrix.minor(1,2).determinant() * invdet;
75  result->coeffRef(2, 2) = matrix.minor(2,2).determinant() * invdet;
76 }

78 template<typename Derived, typename OtherDerived, typename Scalar = typename Derived::Scalar>
79 struct ei_compute_inverse_in_size4_case
80 {
81  static void run(const Derived& matrix, OtherDerived& result)
82  {
83  result.coeffRef(0,0) = matrix.minor(0,0).determinant();
84  result.coeffRef(1,0) = -matrix.minor(0,1).determinant();
85  result.coeffRef(2,0) = matrix.minor(0,2).determinant();
86  result.coeffRef(3,0) = -matrix.minor(0,3).determinant();
87  result.coeffRef(0,2) = matrix.minor(2,0).determinant();
88  result.coeffRef(1,2) = -matrix.minor(2,1).determinant();
89  result.coeffRef(2,2) = matrix.minor(2,2).determinant();
90  result.coeffRef(3,2) = -matrix.minor(2,3).determinant();
91  result.coeffRef(0,1) = -matrix.minor(1,0).determinant();
92  result.coeffRef(1,1) = matrix.minor(1,1).determinant();
93  result.coeffRef(2,1) = -matrix.minor(1,2).determinant();
94  result.coeffRef(3,1) = matrix.minor(1,3).determinant();
95  result.coeffRef(0,3) = -matrix.minor(3,0).determinant();
96  result.coeffRef(1,3) = matrix.minor(3,1).determinant();
97  result.coeffRef(2,3) = -matrix.minor(3,2).determinant();
98  result.coeffRef(3,3) = matrix.minor(3,3).determinant();
99  result /= (matrix.col(0).cwise()*result.row(0).transpose()).sum();
100  }
101 };

103 #ifdef EIGEN_VECTORIZE_SSE
104 // The SSE code for the 4x4 float matrix inverse in this file comes from the file
105 // ftp://download.intel.com/design/PentiumIII/sml/24504301.pdf
106 // its copyright information is:
107 // Copyright (C) 1999 Intel Corporation
108 // See page ii of that document for legal stuff. Not being lawyers, we just assume
109 // here that if Intel makes this document publically available, with source code
110 // and detailed explanations, it's because they want their CPUs to be fed with
111 // good code, and therefore they presumably don't mind us using it in Eigen.
112 template<typename Derived, typename OtherDerived>
113 struct ei_compute_inverse_in_size4_case<Derived, OtherDerived, float>
114 {
115  static void run(const Derived& matrix, OtherDerived& result)
116  {
117  // Variables (Streaming SIMD Extensions registers) which will contain cofactors and, later, the
118  // lines of the inverted matrix.
119  __m128 minor0, minor1, minor2, minor3;
120 
121  // Variables which will contain the lines of the reference matrix and, later (after the transposition),
122  // the columns of the original matrix.
123  __m128 row0, row1, row2, row3;
124 
125  // Temporary variables and the variable that will contain the matrix determinant.
126  __m128 det, tmp1;
127 
128  // Matrix transposition
129  const float *src = matrix.data();
130  tmp1 = _mm_loadh_pi(_mm_castpd_ps(_mm_load_sd((double*)src)), (__m64*)(src+ 4));
131  row1 = _mm_loadh_pi(_mm_castpd_ps(_mm_load_sd((double*)(src+8))), (__m64*)(src+12));
132  row0 = _mm_shuffle_ps(tmp1, row1, 0x88);
133  row1 = _mm_shuffle_ps(row1, tmp1, 0xDD);
134  tmp1 = _mm_loadh_pi(_mm_castpd_ps(_mm_load_sd((double*)(src+ 2))), (__m64*)(src+ 6));
135  row3 = _mm_loadh_pi(_mm_castpd_ps(_mm_load_sd((double*)(src+10))), (__m64*)(src+14));
136  row2 = _mm_shuffle_ps(tmp1, row3, 0x88);
137  row3 = _mm_shuffle_ps(row3, tmp1, 0xDD);
138 
139 
140  // Cofactors calculation. Because in the process of cofactor computation some pairs in three-
141  // element products are repeated, it is not reasonable to load these pairs anew every time. The
142  // values in the registers with these pairs are formed using shuffle instruction. Cofactors are
143  // calculated row by row (4 elements are placed in 1 SP FP SIMD floating point register).
144 
145  tmp1 = _mm_mul_ps(row2, row3);
146  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0xB1);
147  minor0 = _mm_mul_ps(row1, tmp1);
148  minor1 = _mm_mul_ps(row0, tmp1);
149  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0x4E);
150  minor0 = _mm_sub_ps(_mm_mul_ps(row1, tmp1), minor0);
151  minor1 = _mm_sub_ps(_mm_mul_ps(row0, tmp1), minor1);
152  minor1 = _mm_shuffle_ps(minor1, minor1, 0x4E);
153  // -----------------------------------------------
154  tmp1 = _mm_mul_ps(row1, row2);
155  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0xB1);
156  minor0 = _mm_add_ps(_mm_mul_ps(row3, tmp1), minor0);
157  minor3 = _mm_mul_ps(row0, tmp1);
158  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0x4E);
159  minor0 = _mm_sub_ps(minor0, _mm_mul_ps(row3, tmp1));
160  minor3 = _mm_sub_ps(_mm_mul_ps(row0, tmp1), minor3);
161  minor3 = _mm_shuffle_ps(minor3, minor3, 0x4E);
162  // -----------------------------------------------
163  tmp1 = _mm_mul_ps(_mm_shuffle_ps(row1, row1, 0x4E), row3);
164  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0xB1);
165  row2 = _mm_shuffle_ps(row2, row2, 0x4E);
166  minor0 = _mm_add_ps(_mm_mul_ps(row2, tmp1), minor0);
167  minor2 = _mm_mul_ps(row0, tmp1);
168  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0x4E);
169  minor0 = _mm_sub_ps(minor0, _mm_mul_ps(row2, tmp1));
170  minor2 = _mm_sub_ps(_mm_mul_ps(row0, tmp1), minor2);
171  minor2 = _mm_shuffle_ps(minor2, minor2, 0x4E);
172  // -----------------------------------------------
173  tmp1 = _mm_mul_ps(row0, row1);
174  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0xB1);
175  minor2 = _mm_add_ps(_mm_mul_ps(row3, tmp1), minor2);
176  minor3 = _mm_sub_ps(_mm_mul_ps(row2, tmp1), minor3);
177  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0x4E);
178  minor2 = _mm_sub_ps(_mm_mul_ps(row3, tmp1), minor2);
179  minor3 = _mm_sub_ps(minor3, _mm_mul_ps(row2, tmp1));
180  // -----------------------------------------------
181  tmp1 = _mm_mul_ps(row0, row3);
182  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0xB1);
183  minor1 = _mm_sub_ps(minor1, _mm_mul_ps(row2, tmp1));
184  minor2 = _mm_add_ps(_mm_mul_ps(row1, tmp1), minor2);
185  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0x4E);
186  minor1 = _mm_add_ps(_mm_mul_ps(row2, tmp1), minor1);
187  minor2 = _mm_sub_ps(minor2, _mm_mul_ps(row1, tmp1));
188  // -----------------------------------------------
189  tmp1 = _mm_mul_ps(row0, row2);
190  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0xB1);
191  minor1 = _mm_add_ps(_mm_mul_ps(row3, tmp1), minor1);
192  minor3 = _mm_sub_ps(minor3, _mm_mul_ps(row1, tmp1));
193  tmp1 = _mm_shuffle_ps(tmp1, tmp1, 0x4E);
194  minor1 = _mm_sub_ps(minor1, _mm_mul_ps(row3, tmp1));
195  minor3 = _mm_add_ps(_mm_mul_ps(row1, tmp1), minor3);
196 
197  // Evaluation of determinant and its reciprocal value. In the original Intel document,
198  // 1/det was evaluated using a fast rcpps command with subsequent approximation using
199  // the Newton-Raphson algorithm. Here, we go for a IEEE-compliant division instead,
200  // so as to not compromise precision at all.
201  det = _mm_mul_ps(row0, minor0);
202  det = _mm_add_ps(_mm_shuffle_ps(det, det, 0x4E), det);
203  det = _mm_add_ss(_mm_shuffle_ps(det, det, 0xB1), det);
204 // tmp1= _mm_rcp_ss(det);
205 // det= _mm_sub_ss(_mm_add_ss(tmp1, tmp1), _mm_mul_ss(det, _mm_mul_ss(tmp1, tmp1)));
206  det = _mm_div_ss(_mm_set_ss(1.0f), det); // <--- yay, one original line not copied from Intel
207  det = _mm_shuffle_ps(det, det, 0x00);
208  // warning, Intel's variable naming is very confusing: now 'det' is 1/det !
209 
210  // Multiplication of cofactors by 1/det. Storing the inverse matrix to the address in pointer src.
211  minor0 = _mm_mul_ps(det, minor0);
212  float *dst = result.data();
213  _mm_storel_pi((__m64*)(dst), minor0);
214  _mm_storeh_pi((__m64*)(dst+2), minor0);
215  minor1 = _mm_mul_ps(det, minor1);
216  _mm_storel_pi((__m64*)(dst+4), minor1);
217  _mm_storeh_pi((__m64*)(dst+6), minor1);
218  minor2 = _mm_mul_ps(det, minor2);
219  _mm_storel_pi((__m64*)(dst+ 8), minor2);
220  _mm_storeh_pi((__m64*)(dst+10), minor2);
221  minor3 = _mm_mul_ps(det, minor3);
222  _mm_storel_pi((__m64*)(dst+12), minor3);
223  _mm_storeh_pi((__m64*)(dst+14), minor3);
224  }
225 };
226 #endif
