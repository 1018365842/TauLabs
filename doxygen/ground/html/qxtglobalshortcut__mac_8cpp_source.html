174 bool QxtGlobalShortcutPrivate::registerShortcut(quint32 nativeKey, quint32 nativeMods)
175 {
176  if (!qxt_mac_handler_installed)
177  {
178  EventTypeSpec t;
179  t.eventClass = kEventClassKeyboard;
180  t.eventKind = kEventHotKeyPressed;
181  InstallApplicationEventHandler(&qxt_mac_handle_hot_key, 1, &t, NULL, NULL);
182  }
183 
184  EventHotKeyID keyID;
185  keyID.signature = 'cute';
186  keyID.id = ++hotKeySerial;
187 
188  EventHotKeyRef ref = 0;
189  bool rv = !RegisterEventHotKey(nativeKey, nativeMods, keyID, GetApplicationEventTarget(), 0, &ref);
190  if (rv)
191  {
192  keyIDs.insert(Identifier(nativeMods, nativeKey), keyID.id);
193  keyRefs.insert(keyID.id, ref);
194  }
195  qDebug() << ref;
196  return rv;
197 }
