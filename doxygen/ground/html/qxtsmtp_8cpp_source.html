178 void QxtSmtpPrivate::socketRead()
179 {
180  buffer += socket->readAll();
181  while (true)
182  {
183  int pos = buffer.indexOf("\r\n");
184  if (pos < 0) return;
185  QByteArray line = buffer.left(pos);
186  buffer = buffer.mid(pos + 2);
187  QByteArray code = line.left(3);
188  switch (state)
189  {
190  case StartState:
191  if (code[0] != '2')
192  {
193  socket->disconnectFromHost();
194  }
195  else
196  {
197  ehlo();
198  }
199  break;
200  case HeloSent:
201  case EhloSent:
202  case EhloGreetReceived:
203  parseEhlo(code, (line[3] != ' '), line.mid(4));
204  break;
205 #ifndef QT_NO_OPENSSL
206  case StartTLSSent:
207  if (code == "220")
208  {
209  socket->startClientEncryption();
210  ehlo();
211  }
212  else
213  {
214  authenticate();
215  }
216  break;
217 #endif
218  case AuthRequestSent:
219  case AuthUsernameSent:
220  if (authType == AuthPlain) authPlain();
221  else if (authType == AuthLogin) authLogin();
222  else authCramMD5(line.mid(4));
223  break;
224  case AuthSent:
225  if (code[0] == '2')
226  {
227  state = Authenticated;
228  emit qxt_p().authenticated();
229  }
230  else
231  {
232  state = Disconnected;
233  emit qxt_p().authenticationFailed();
234  emit qxt_p().authenticationFailed( line );
235  emit socket->disconnectFromHost();
236  }
237  break;
238  case MailToSent:
239  case RcptAckPending:
240  if (code[0] != '2') {
241  emit qxt_p().mailFailed( pending.first().first, code.toInt() );
242  emit qxt_p().mailFailed(pending.first().first, code.toInt(), line);
243  // pending.removeFirst();
244  // DO NOT remove it, the body sent state needs this message to assigned the next mail failed message that will
245  // the sendNext
246  // a reset will be sent to clear things out
247  sendNext();
248  state = BodySent;
249  }
250  else
251  sendNextRcpt(code, line);
252  break;
253  case SendingBody:
254  sendBody(code, line);
255  break;
256  case BodySent:
257  if ( pending.count() )
258  {
259  // if you removeFirst in RcpActpending/MailToSent on an error, and the queue is now empty,
260  // you will get into this state and then crash because no check is done. CHeck added but shouldnt
261  // be necessary since I commented out the removeFirst
262  if (code[0] != '2')
263  {
264  emit qxt_p().mailFailed(pending.first().first, code.toInt() );
265  emit qxt_p().mailFailed(pending.first().first, code.toInt(), line);
266  }
267  else
268  emit qxt_p().mailSent(pending.first().first);
269  pending.removeFirst();
270  }
271  sendNext();
272  break;
273  case Resetting:
274  if (code[0] != '2') {
275  emit qxt_p().connectionFailed();
276  emit qxt_p().connectionFailed( line );
277  }
278  else {
279  state = Waiting;
280  sendNext();
281  }
282  break;
283  }
284  }

302 void QxtSmtpPrivate::parseEhlo(const QByteArray& code, bool cont, const QString& line)
303 {
304  if (code != "250")
305  {
306  // error!
307  if (state != HeloSent)
308  {
309  // maybe let's try HELO
310  socket->write("helo\r\n");
311  state = HeloSent;
312  }
313  else
314  {
315  // nope
316  socket->write("QUIT\r\n");
317  socket->flush();
318  socket->disconnectFromHost();
319  }
320  return;
321  }
322  else if (state != EhloGreetReceived)
323  {
324  if (!cont)
325  {
326  // greeting only, no extensions
327  state = EhloDone;
328  }
329  else
330  {
331  // greeting followed by extensions
332  state = EhloGreetReceived;
333  return;
334  }
335  }
336  else
337  {
338  extensions[line.section(' ', 0, 0).toUpper()] = line.section(' ', 1);
339  if (!cont)
340  state = EhloDone;
341  }
342  if (state != EhloDone) return;
343  if (extensions.contains("STARTTLS") && !disableStartTLS)
344  {
345  startTLS();
346  }
347  else
348  {
349  authenticate();
350  }
351 }

363 void QxtSmtpPrivate::authenticate()
364 {
365  if (!extensions.contains("AUTH") || username.isEmpty() || password.isEmpty())
366  {
367  state = Authenticated;
368  emit qxt_p().authenticated();
369  }
370  else
371  {
372  QStringList auth = extensions["AUTH"].toUpper().split(' ', QString::SkipEmptyParts);
373  if (auth.contains("CRAM-MD5"))
374  {
375  authCramMD5();
376  }
377  else if (auth.contains("PLAIN"))
378  {
379  authPlain();
380  }
381  else if (auth.contains("LOGIN"))
382  {
383  authLogin();
384  }
385  else
386  {
387  state = Authenticated;
388  emit qxt_p().authenticated();
389  }
390  }

393 void QxtSmtpPrivate::authCramMD5(const QByteArray& challenge)
394 {
395  if (state != AuthRequestSent)
396  {
397  socket->write("auth cram-md5\r\n");
398  authType = AuthCramMD5;
399  state = AuthRequestSent;
400  }
401  else
402  {
403  QxtHmac hmac(QCryptographicHash::Md5);
404  hmac.setKey(password);
405  hmac.addData(QByteArray::fromBase64(challenge));
406  QByteArray response = username + ' ' + hmac.result().toHex();
407  socket->write(response.toBase64() + "\r\n");
408  state = AuthSent;
409  }

452 static QByteArray qxt_extract_address(const QString& address)
453 {
454  int parenDepth = 0;
455  int addrStart = -1;
456  bool inQuote = false;
457  int ct = address.length();
458 
459  for (int i = 0; i < ct; i++)
460  {
461  QChar ch = address[i];
462  if (inQuote)
463  {
464  if (ch == '"')
465  inQuote = false;
466  }
467  else if (addrStart != -1)
468  {
469  if (ch == '>')
470  return address.mid(addrStart, (i - addrStart)).toAscii();
471  }
472  else if (ch == '(')
473  {
474  parenDepth++;
475  }
476  else if (ch == ')')
477  {
478  parenDepth--;
479  if (parenDepth < 0) parenDepth = 0;
480  }
481  else if (ch == '"')
482  {
483  if (parenDepth == 0)
484  inQuote = true;
485  }
486  else if (ch == '<')
487  {
488  if (!inQuote && parenDepth == 0)
489  addrStart = i + 1;
490  }
491  }
492  return address.toAscii();
493 }

495 void QxtSmtpPrivate::sendNext()
496 {
497  if (state == Disconnected)
498  {
499  // leave the mail in the queue if not ready to send
500  return;
501  }
502 
503  if (pending.isEmpty())
504  {
505  // if there are no additional mails to send, finish up
506  state = Waiting;
507  emit qxt_p().finished();
508  return;
509  }
510 
511  if(state != Waiting) {
512  state = Resetting;
513  socket->write("rset\r\n");
514  return;
515  }
516  const QxtMailMessage& msg = pending.first().second;
517  rcptNumber = rcptAck = mailAck = 0;
518  recipients = msg.recipients(QxtMailMessage::To) +
519  msg.recipients(QxtMailMessage::Cc) +
520  msg.recipients(QxtMailMessage::Bcc);
521  if (recipients.count() == 0)
522  {
523  // can't send an e-mail with no recipients
524  emit qxt_p().mailFailed(pending.first().first, QxtSmtp::NoRecipients );
525  emit qxt_p().mailFailed(pending.first().first, QxtSmtp::NoRecipients, QByteArray( "e-mail has no recipients" ) );
526  pending.removeFirst();
527  sendNext();
528  return;
529  }
530  // We explicitly use lowercase keywords because for some reason gmail
531  // interprets any string starting with an uppercase R as a request
532  // to renegotiate the SSL connection.
533  socket->write("mail from:<" + qxt_extract_address(msg.sender()) + ">\r\n");
534  if (extensions.contains("PIPELINING")) // almost all do nowadays
535  {
536  foreach(const QString& rcpt, recipients)
537  {
538  socket->write("rcpt to:<" + qxt_extract_address(rcpt) + ">\r\n");
539  }
540  state = RcptAckPending;
541  }
542  else
543  {
544  state = MailToSent;
545  }
546 }

548 void QxtSmtpPrivate::sendNextRcpt(const QByteArray& code, const QByteArray&line)
549 {
550  int messageID = pending.first().first;
551  const QxtMailMessage& msg = pending.first().second;
552 
553  if (code[0] != '2')
554  {
555  // on failure, emit a warning signal
556  if (!mailAck)
557  {
558  emit qxt_p().senderRejected(messageID, msg.sender());
559  emit qxt_p().senderRejected(messageID, msg.sender(), line );
560  }
561  else
562  {
563  emit qxt_p().recipientRejected(messageID, msg.sender());
564  emit qxt_p().recipientRejected(messageID, msg.sender(), line);
565  }
566  }
567  else if (!mailAck)
568  {
569  mailAck = true;
570  }
571  else
572  {
573  rcptAck++;
574  }
575 
576  if (rcptNumber == recipients.count())
577  {
578  // all recipients have been sent
579  if (rcptAck == 0)
580  {
581  // no recipients were considered valid
582  emit qxt_p().mailFailed(messageID, code.toInt() );
583  emit qxt_p().mailFailed(messageID, code.toInt(), line);
584  pending.removeFirst();
585  sendNext();
586  }
587  else
588  {
589  // at least one recipient was acknowledged, send mail body
590  socket->write("data\r\n");
591  state = SendingBody;
592  }
593  }
594  else if (state != RcptAckPending)
595  {
596  // send the next recipient unless we're only waiting on acks
597  socket->write("rcpt to:<" + qxt_extract_address(recipients[rcptNumber]) + ">\r\n");
598  rcptNumber++;
599  }
600  else
601  {
602  // If we're only waiting on acks, just count them
603  rcptNumber++;
604  }
605 }

607 void QxtSmtpPrivate::sendBody(const QByteArray& code, const QByteArray & line)
608 {
609  int messageID = pending.first().first;
610  const QxtMailMessage& msg = pending.first().second;
611 
612  if (code[0] != '3')
613  {
614  emit qxt_p().mailFailed(messageID, code.toInt() );
615  emit qxt_p().mailFailed(messageID, code.toInt(), line);
616  pending.removeFirst();
617  sendNext();
618  return;
619  }
620 
621  socket->write(msg.rfc2822());
622  socket->write(".\r\n");
623  state = BodySent;
