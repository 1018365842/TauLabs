159 inline Scalar AlignedBox<Scalar,AmbiantDim>::squaredExteriorDistance(const VectorType& p) const
160 {
161  Scalar dist2 = 0.;
162  Scalar aux;
163  for (int k=0; k<dim(); ++k)
164  {
165  if ((aux = (p[k]-m_min[k]))<0.)
166  dist2 += aux*aux;
167  else if ( (aux = (m_max[k]-p[k]))<0. )

169  }
