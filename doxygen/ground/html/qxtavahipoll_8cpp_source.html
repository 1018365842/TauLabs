128 void AvahiWatch::setEventType(AvahiWatchEvent event)
129 {
130  if (_notifier != 0)
131  {
132  if (_notifier->thread() == QThread::currentThread())
133  delete _notifier;
134  else
135  _notifier->deleteLater();
136  _notifier = 0;
137  }
138  _event = event;
139  switch (_event)
140  {
141  case AVAHI_WATCH_IN:
142  {
143  _notifier = new QSocketNotifier(_fd, QSocketNotifier::Read, this);
144  connect(_notifier, SIGNAL(activated(int)), this, SLOT(activated(int)));
145  break;
146  }
147  case AVAHI_WATCH_OUT:
148  {
149  _notifier = new QSocketNotifier(_fd, QSocketNotifier::Write, this);
150  connect(_notifier, SIGNAL(activated(int)), this, SLOT(activated(int)));
151  break;
152  }
153  default: //The constructor should only get in or out...
154  qWarning("Bad event type passed to AvahiWatch constructor");
155  break;
156  }
157 }

181 void AvahiTimeout::updateTimeout(const struct timeval *tv)
182 {
183  if(tv == 0)
184  {
185  _timer.stop();
186  return;
187  }
188  int msecs = (tv->tv_sec * 1000) + (tv->tv_usec / 1000);
189  if (msecs > 1)
190  _timer.start(msecs);
191  else
192  _timer.stop();
193 }
