199 void XplaneSimulator::processUpdate(const QByteArray& dataBuf)
200 {
201  float altitude_msl = 0;
202  float altitude_agl = 0;
203  float latitude = 0;
204  float longitude = 0;
205  float airspeed_keas = 0;
206  float airspeed_ktas = 0;
207  float groundspeed_ktgs = 0;
208  float alpha_D=0;
209  float beta_D=0;
210  float pitch = 0;
211  float roll = 0;
212  float heading = 0;
213  float pressure = 0;
214  float temperature = 0;
215  float velX = 0;
216  float velY = 0;
217  float velZ = 0;
218  float dstX = 0;
219  float dstY = 0;
220  float dstZ = 0;
221  float accX = 0;
222  float accY = 0;
223  float accZ = 0;
224  float rollRate_rad=0;
225  float pitchRate_rad=0;
226  float yawRate_rad=0;
227 
228  // QString str;
229  QByteArray& buf = const_cast<QByteArray&>(dataBuf);
230  QString data(buf);
231 
232  if(data.left(4) == "DATA") // check type of packet
233  {
234  buf.remove(0,5);
235  if(dataBuf.size() % 36)
236  {
237  qxtLog->info("incorrect length of UDP packet: ",buf);
238  return; // incorrect length of struct
239  }
240  // check correctness of data length, length must be multiple of (id_size+8*float_size)=4+8*4=36
241  int channelCounter = dataBuf.size() / 36;
242  do
243  {
244  switch(buf[0]) // switch by id
245  {
246  case XplaneSimulator::LatitudeLongitudeAltitude:
247  latitude = *((float*)(buf.data()+4*1));
248  longitude = *((float*)(buf.data()+4*2));
249  altitude_msl = *((float*)(buf.data()+4*3)) * FEET2MILES;
250  altitude_agl = *((float*)(buf.data()+4*4)) * FEET2MILES;
251  break;
252 
253  case XplaneSimulator::Speed:
254  airspeed_keas = *((float*)(buf.data()+4*2));
255  airspeed_ktas = *((float*)(buf.data()+4*3));
256  groundspeed_ktgs = *((float*)(buf.data()+4*4));
257  break;
258 
259  case XplaneSimulator::AoA:
260  alpha_D = *((float*)(buf.data()+4*1));
261  beta_D = *((float*)(buf.data()+4*2));
262  break;
263 
264  case XplaneSimulator::PitchRollHeading:
265  pitch = *((float*)(buf.data()+4*1));
266  roll = *((float*)(buf.data()+4*2));
267  heading = *((float*)(buf.data()+4*3));
268  break;
269 
270  /*
271  case XplaneSimulator::SystemPressures:
272  pressure = *((float*)(buf.data()+4*1));
273  break;
274  */
275 
276  case XplaneSimulator::AtmosphereWeather:
277  pressure = *((float*)(buf.data()+4*1)) * INCHES_MERCURY2KPA;
278  temperature = *((float*)(buf.data()+4*2));
279  break;
280 
281  case XplaneSimulator::LocVelDistTraveled:
282  dstX = *((float*)(buf.data()+4*1));
283  dstY = - *((float*)(buf.data()+4*3));
284  dstZ = *((float*)(buf.data()+4*2));
285  velX = *((float*)(buf.data()+4*4));
286  velY = - *((float*)(buf.data()+4*6));
287  velZ = *((float*)(buf.data()+4*5));
288  break;
289 
290  case XplaneSimulator::AngularVelocities: //In [rad/s]
291  pitchRate_rad = *((float*)(buf.data()+4*1));
292  rollRate_rad = *((float*)(buf.data()+4*2));
293  yawRate_rad = *((float*)(buf.data()+4*3));
294  break;
295 
296  case XplaneSimulator::Gload:
297  accX = *((float*)(buf.data()+4*6)) * GRAVITY;
298  accY = *((float*)(buf.data()+4*7)) * GRAVITY;
299  accZ = *((float*)(buf.data()+4*5)) * GRAVITY;
300  break;
301 
302  default:
303  break;
304  }
305  channelCounter--;
306  buf.remove(0,36);
307  } while (channelCounter);
308 
309 
311  // Output formatting
313  Output2Hardware out;
314  memset(&out, 0, sizeof(Output2Hardware));
315 
316  // Update GPS Position objects
317  out.latitude = latitude * 1e7;
318  out.longitude = longitude * 1e7;
319  out.altitude = altitude_msl;
320  out.agl = altitude_agl;
321  out.groundspeed = groundspeed_ktgs*1.15*1.6089/3.6; //Convert from [kts] to [m/s]
322 
323  out.calibratedAirspeed = airspeed_keas*1.15*1.6089/3.6; //Convert from [kts] to [m/s]
324  out.trueAirspeed = airspeed_ktas*1.15*1.6089/3.6; //Convert from [kts] to [m/s]
325  out.angleOfAttack=alpha_D;
326  out.angleOfSlip=beta_D;
327 
328  // Update BaroAltitude object
329  out.temperature = temperature;
330  out.pressure = pressure;
331 
332  // Update attActual object
333  out.roll = roll; //roll;
334  out.pitch = pitch; // pitch
335  out.heading = heading; // yaw
336 
337 
338  out.dstN=dstY;
339  out.dstE=dstX;
340  out.dstD=-dstZ;
341 
342  // Update VelocityActual.{North,East,Down}
343  out.velNorth = velY;
344  out.velEast = velX;
345  out.velDown = -velZ;
346 
347  //Update gyroscope sensor data
348  out.rollRate = rollRate_rad;
349  out.pitchRate = pitchRate_rad;
350  out.yawRate = yawRate_rad;
351 
352  //Update accelerometer sensor data
353  out.accX = accX;
354  out.accY = accY;
355  out.accZ = -accZ;
356 
357  updateUAVOs(out);
358  }
359  // issue manual update
360  //attActual->updated();
361  //altActual->updated();
362  //posActual->updated();
