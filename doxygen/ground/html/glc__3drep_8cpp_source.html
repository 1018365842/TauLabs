459 QDataStream &operator>>(QDataStream & stream, GLC_3DRep & rep)
460 {
461  Q_ASSERT(rep.isEmpty());
462 
463  quint32 chunckId;
464  stream >> chunckId;
465  Q_ASSERT(chunckId == GLC_3DRep::m_ChunkId);
466 
467  // The rep name
468  QString name;
469  stream >> name;
470  rep.setName(name);
471 
472  // Retrieve the list of rep materials
473  QList<GLC_Material> materialsList;
474  stream >> materialsList;
475  MaterialHash materialHash;
476  // Update mesh materials hash table
477  QHash<GLC_uint, GLC_uint> materialIdMap;
478  const int materialsCount= materialsList.size();
479  for (int i= 0; i < materialsCount; ++i)
480  {
481  GLC_Material* pMaterial= new GLC_Material(materialsList.at(i));
482  pMaterial->setId(glc::GLC_GenID());
483  materialIdMap.insert(materialsList.at(i).id(), pMaterial->id());
484  materialHash.insert(pMaterial->id(), pMaterial);
485  }
486 
487  int meshNumber;
488  stream >> meshNumber;
489  for (int i= 0; i < meshNumber; ++i)
490  {
491  GLC_Mesh* pMesh= new GLC_Mesh();
492  pMesh->loadFromDataStream(stream, materialHash, materialIdMap);
493 
494  rep.addGeom(pMesh);
495  }
496 
497  return stream;
