95  UAVObjectUtilManager* utilMngr = pm->getObject<UAVObjectUtilManager>();
96  int id = utilMngr->getBoardModel();

98  switch (id) {
99  case 0x0301:
100  return SetupWizard::CONTROLLER_PIPX;

102  return SetupWizard::CONTROLLER_CC;
103  case 0x0402:
104  return SetupWizard::CONTROLLER_CC3D;
105  case 0x0901:

108  return SetupWizard::CONTROLLER_UNKNOWN;
109  }

140 void ControllerPage::devicesChanged(QLinkedList<Core::DevListItem> devices)
141 {
142  // Get the selected item before the update if any
143  QString currSelectedDeviceName = ui->deviceCombo->currentIndex() != -1 ?
144  ui->deviceCombo->itemData(ui->deviceCombo->currentIndex(), Qt::ToolTipRole).toString() : "";
145 
146  // Clear the box
147  ui->deviceCombo->clear();
148 
149  int indexOfSelectedItem = -1;
150  int i = 0;
151 
152  // Loop and fill the combo with items from connectionmanager
153  foreach (Core::DevListItem deviceItem, devices)
154  {
155  ui->deviceCombo->addItem(deviceItem.getConName());
156  QString deviceName = (const QString)deviceItem.getConName();
157  ui->deviceCombo->setItemData(ui->deviceCombo->count() - 1, deviceName, Qt::ToolTipRole);
158  if(!deviceName.startsWith("USB:", Qt::CaseInsensitive)) {
159  ui->deviceCombo->setItemData(ui->deviceCombo->count() - 1, QVariant(0), Qt::UserRole - 1);
160  }
161  if(currSelectedDeviceName != "" && currSelectedDeviceName == deviceName) {
162  indexOfSelectedItem = i;
163  }
164  i++;
165  }
166 
167  // Re select the item that was selected before if any
168  if(indexOfSelectedItem != -1) {
169  ui->deviceCombo->setCurrentIndex(indexOfSelectedItem);
170  }
171  //connectionStatusChanged();
172 }

174 void ControllerPage::connectionStatusChanged()
175 {
176  if(m_connectionManager->isConnected()) {
177  ui->deviceCombo->setEnabled(false);
178  ui->connectButton->setText(tr("Disconnect"));
179  ui->boardTypeCombo->setEnabled(false);
180  QString connectedDeviceName = m_connectionManager->getCurrentDevice().getConName();
181  for(int i = 0; i < ui->deviceCombo->count(); ++i) {
182  if(connectedDeviceName == ui->deviceCombo->itemData(i, Qt::ToolTipRole).toString()) {
183  ui->deviceCombo->setCurrentIndex(i);
184  break;
185  }
186  }
187 
188  SetupWizard::CONTROLLER_TYPE type = getControllerType();
189  setControllerType(type);
190  qDebug() << "Connection status changed: Connected, controller type: " << getControllerType();
191  }
192  else {
193  ui->deviceCombo->setEnabled(true);
194  ui->connectButton->setText(tr("Connect"));
195  ui->boardTypeCombo->setEnabled(false);
196  ui->boardTypeCombo->model()->setData(ui->boardTypeCombo->model()->index(0, 0), QVariant(0), Qt::UserRole - 1);
197  setControllerType(SetupWizard::CONTROLLER_UNKNOWN);
198  qDebug() << "Connection status changed: Disconnected";
199  }
200  emit completeChanged();
201 }
