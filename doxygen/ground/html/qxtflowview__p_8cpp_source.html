69 // readjust the settings, call this when slide dimension is changed
70 void QxtFlowViewState::reposition()
71 {
72  angle = 70 * IANGLE_MAX / 360; // approx. 70 degrees tilted
73 
74  offsetX = slideWidth / 2 * (PFREAL_ONE - fcos(angle));
75  offsetY = slideWidth / 2 * fsin(angle);
76  offsetX += slideWidth * PFREAL_ONE;
77  offsetY += slideWidth * PFREAL_ONE / 4;
78  spacing = 40;
79 }

81 // adjust slides so that they are in "steady state" position
82 void QxtFlowViewState::reset()
83 {
84  centerSlide.angle = 0;
85  centerSlide.cx = 0;
86  centerSlide.cy = 0;
87  centerSlide.slideIndex = centerIndex;
88  centerSlide.blend = 256;
89 
90  leftSlides.resize(6);
91  for (int i = 0; i < (int)leftSlides.count(); i++)
92  {
93  SlideInfo& si = leftSlides[i];
94  si.angle = angle;
95  si.cx = -(offsetX + spacing * i * PFREAL_ONE);
96  si.cy = offsetY;
97  si.slideIndex = centerIndex - 1 - i;
98  si.blend = 256;
99  if (i == (int)leftSlides.count() - 2)
100  si.blend = 128;
101  if (i == (int)leftSlides.count() - 1)
102  si.blend = 0;
103  }
104 
105  rightSlides.resize(6);
106  for (int i = 0; i < (int)rightSlides.count(); i++)
107  {
108  SlideInfo& si = rightSlides[i];
109  si.angle = -angle;
110  si.cx = offsetX + spacing * i * PFREAL_ONE;
111  si.cy = offsetY;
112  si.slideIndex = centerIndex + 1 + i;
113  si.blend = 256;
114  if (i == (int)rightSlides.count() - 2)
115  si.blend = 128;
116  if (i == (int)rightSlides.count() - 1)
117  si.blend = 0;

146 void QxtFlowViewAnimator::update()
147 {
148  if (!animateTimer.isActive())
149  return;
150  if (step == 0)
151  return;
152  if (!state)
153  return;
154 
155  int speed = 16384 / 4;
156 
157 #if 1
158  // deaccelerate when approaching the target
159  const int max = 2 * 65536;
160 
161  int fi = frame;
162  fi -= (target << 16);
163  if (fi < 0)
164  fi = -fi;
165  fi = qMin(fi, max);
166 
167  int ia = IANGLE_MAX * (fi - max / 2) / (max * 2);
168  speed = 512 + 16384 * (PFREAL_ONE + fsin(ia)) / PFREAL_ONE;
169 #endif
170 
171  frame += speed * step;
172 
173  int index = frame >> 16;
174  int pos = frame & 0xffff;
175  int neg = 65536 - pos;
176  int tick = (step < 0) ? neg : pos;
177  PFreal ftick = (tick * PFREAL_ONE) >> 16;
178 
179  if (step < 0)
180  index++;
181 
182  if (state->centerIndex != index)
183  {
184  state->centerIndex = index;
185  frame = index << 16;
186  state->centerSlide.slideIndex = state->centerIndex;
187  for (int i = 0; i < (int)state->leftSlides.count(); i++)
188  state->leftSlides[i].slideIndex = state->centerIndex - 1 - i;
189  for (int i = 0; i < (int)state->rightSlides.count(); i++)
190  state->rightSlides[i].slideIndex = state->centerIndex + 1 + i;
191  }
192 
193  state->centerSlide.angle = (step * tick * state->angle) >> 16;
194  state->centerSlide.cx = -step * fmul(state->offsetX, ftick);
195  state->centerSlide.cy = fmul(state->offsetY, ftick);
196 
197  if (state->centerIndex == target)
198  {
199  stop(target);
200  state->reset();
201  return;
202  }
203 
204  for (int i = 0; i < (int)state->leftSlides.count(); i++)
205  {
206  SlideInfo& si = state->leftSlides[i];
207  si.angle = state->angle;
208  si.cx = -(state->offsetX + state->spacing * i * PFREAL_ONE + step * state->spacing * ftick);
209  si.cy = state->offsetY;
210  }
211 
212  for (int i = 0; i < (int)state->rightSlides.count(); i++)
213  {
214  SlideInfo& si = state->rightSlides[i];
215  si.angle = -state->angle;
216  si.cx = state->offsetX + state->spacing * i * PFREAL_ONE - step * state->spacing * ftick;
217  si.cy = state->offsetY;
218  }
219 
220  if (step > 0)
221  {
222  PFreal ftick = (neg * PFREAL_ONE) >> 16;
223  state->rightSlides[0].angle = -(neg * state->angle) >> 16;
224  state->rightSlides[0].cx = fmul(state->offsetX, ftick);
225  state->rightSlides[0].cy = fmul(state->offsetY, ftick);
226  }
227  else
228  {
229  PFreal ftick = (pos * PFREAL_ONE) >> 16;
230  state->leftSlides[0].angle = (pos * state->angle) >> 16;
231  state->leftSlides[0].cx = -fmul(state->offsetX, ftick);
232  state->leftSlides[0].cy = fmul(state->offsetY, ftick);
233  }
234 
235  // must change direction ?
236  if (target < index) if (step > 0)
237  step = -1;
238  if (target > index) if (step < 0)
239  step = 1;
240 
241  // the first and last slide must fade in/fade out
242  int nleft = state->leftSlides.count();
243  int nright = state->rightSlides.count();
244  int fade = pos / 256;
245 
246  for (int index = 0; index < nleft; index++)
247  {
248  int blend = 256;
249  if (index == nleft - 1)
250  blend = (step > 0) ? 0 : 128 - fade / 2;
251  if (index == nleft - 2)
252  blend = (step > 0) ? 128 - fade / 2 : 256 - fade / 2;
253  if (index == nleft - 3)
254  blend = (step > 0) ? 256 - fade / 2 : 256;
255  state->leftSlides[index].blend = blend;
256  }
257  for (int index = 0; index < nright; index++)
258  {
259  int blend = (index < nright - 2) ? 256 : 128;
260  if (index == nright - 1)
261  blend = (step > 0) ? fade / 2 : 0;
262  if (index == nright - 2)
263  blend = (step > 0) ? 128 + fade / 2 : fade / 2;
264  if (index == nright - 3)
265  blend = (step > 0) ? 256 : 128 + fade / 2;
266  state->rightSlides[index].blend = blend;
267  }
268 
269 }

313 {
314  if (!widget)
315  return;
316 
317  surfaceCache.clear();
318  blankSurface = 0;
319 
320  size = widget->size();
321  int ww = size.width();
322  int wh = size.height();
323  int w = (ww + 1) / 2;
324  int h = (wh + 1) / 2;
325 
326 #ifdef PICTUREFLOW_QT4
327  buffer = QImage(ww, wh, QImage::Format_RGB32);
328 #endif

331 #endif
332  buffer.fill(bgcolor);
333 
334  rays.resize(w*2);
335  for (int i = 0; i < w; i++)
336  {
337  PFreal gg = ((PFREAL_ONE >> 1) + i * PFREAL_ONE) / (2 * h);
338  rays[w-i-1] = -gg;

341 
342  dirty = true;

345 // TODO: optimize this with lookup tables
346 static QRgb blendColor(QRgb c1, QRgb c2, int blend)
347 {
348  int r = qRed(c1) * blend / 256 + qRed(c2) * (256 - blend) / 256;
349  int g = qGreen(c1) * blend / 256 + qGreen(c2) * (256 - blend) / 256;
350  int b = qBlue(c1) * blend / 256 + qBlue(c2) * (256 - blend) / 256;
351  return qRgb(r, g, b);
352 }

355 static QImage* prepareSurface(const QImage* slideImage, int w, int h, QRgb bgcolor,
356  QxtFlowView::ReflectionEffect reflectionEffect)
357 {
358 #ifdef PICTUREFLOW_QT4
359  Qt::TransformationMode mode = Qt::SmoothTransformation;
360  QImage img = slideImage->scaled(w, h, Qt::IgnoreAspectRatio, mode);
361 #endif
362 #if defined(PICTUREFLOW_QT3) || defined(PICTUREFLOW_QT2)
363  QImage img = slideImage->smoothScale(w, h);
364 #endif
365 
366  // slightly larger, to accommodate for the reflection
367  int hs = h * 2;
368  int hofs = h / 3;
369 
370  // offscreen buffer: black is sweet
371 #ifdef PICTUREFLOW_QT4
372  QImage* result = new QImage(hs, w, QImage::Format_RGB32);
373 #endif
374 #if defined(PICTUREFLOW_QT3) || defined(PICTUREFLOW_QT2)
375  QImage* result = new QImage;
376  result->create(hs, w, 32);
377 #endif
378  result->fill(bgcolor);
379 
380  // transpose the image, this is to speed-up the rendering
381  // because we process one column at a time
382  // (and much better and faster to work row-wise, i.e in one scanline)
383  for (int x = 0; x < w; x++)
384  for (int y = 0; y < h; y++)
385  result->setPixel(hofs + y, x, img.pixel(x, y));
386 
387  if (reflectionEffect != QxtFlowView::NoReflection)
388  {
389  // create the reflection
390  int ht = hs - h - hofs;
391  int hte = ht;
392  for (int x = 0; x < w; x++)
393  for (int y = 0; y < ht; y++)
394  {
395  QRgb color = img.pixel(x, img.height() - y - 1);
396  result->setPixel(h + hofs + y, x, blendColor(color, bgcolor, 128*(hte - y) / hte));
397  }
398 
399  if (reflectionEffect == QxtFlowView::BlurredReflection)
400  {
401  // blur the reflection everything first
402  // Based on exponential blur algorithm by Jani Huhtanen
403  QRect rect(hs / 2, 0, hs / 2, w);
404  rect &= result->rect();
405 
406  int r1 = rect.top();
407  int r2 = rect.bottom();
408  int c1 = rect.left();
409  int c2 = rect.right();
410 
411  int bpl = result->bytesPerLine();
412  int rgba[4];
413  unsigned char* p;
414 
415  // how many times blur is applied?
416  // for low-end system, limit this to only 1 loop
417  for (int loop = 0; loop < 2; loop++)
418  {
419  for (int col = c1; col <= c2; col++)
420  {
421  p = result->scanLine(r1) + col * 4;
422  for (int i = 0; i < 3; i++)
423  rgba[i] = p[i] << 4;
424 
425  p += bpl;
426  for (int j = r1; j < r2; j++, p += bpl)
427  for (int i = 0; i < 3; i++)
428  p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
429  }
430 
431  for (int row = r1; row <= r2; row++)
432  {
433  p = result->scanLine(row) + c1 * 4;
434  for (int i = 0; i < 3; i++)
435  rgba[i] = p[i] << 4;
436 
437  p += 4;
438  for (int j = c1; j < c2; j++, p += 4)
439  for (int i = 0; i < 3; i++)
440  p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
441  }
442 
443  for (int col = c1; col <= c2; col++)
444  {
445  p = result->scanLine(r2) + col * 4;
446  for (int i = 0; i < 3; i++)
447  rgba[i] = p[i] << 4;
448 
449  p -= bpl;
450  for (int j = r1; j < r2; j++, p -= bpl)
451  for (int i = 0; i < 3; i++)
452  p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
453  }
454 
455  for (int row = r1; row <= r2; row++)
456  {
457  p = result->scanLine(row) + c2 * 4;
458  for (int i = 0; i < 3; i++)
459  rgba[i] = p[i] << 4;
460 
461  p -= 4;
462  for (int j = c1; j < c2; j++, p -= 4)
463  for (int i = 0; i < 3; i++)
464  p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
465  }
466  }
467 
468  // overdraw to leave only the reflection blurred (but not the actual image)
469  for (int x = 0; x < w; x++)
470  for (int y = 0; y < h; y++)
471  result->setPixel(hofs + y, x, img.pixel(x, y));
472  }
473  }
474 
475  return result;
476 }

478 QImage* QxtFlowViewSoftwareRenderer::surface(int slideIndex)
479 {
480  if (!state)
481  return 0;
482  if (slideIndex < 0)
483  return 0;
484  if (slideIndex >= (int)state->slideImages.count())
485  return 0;
486 
487 #ifdef PICTUREFLOW_QT4
488  int key = slideIndex;
489 #endif
490 #if defined(PICTUREFLOW_QT3) || defined(PICTUREFLOW_QT2)
491  QString key = QString::number(slideIndex);
492 #endif
493 
494  QImage* img = state->slideImages.at(slideIndex);
495  bool empty = img ? img->isNull() : true;
496  if (empty)
497  {
498  surfaceCache.remove(key);
499  imageHash.remove(slideIndex);
500  if (!blankSurface)
501  {
502  int sw = state->slideWidth;
503  int sh = state->slideHeight;
504 
505 #ifdef PICTUREFLOW_QT4
506  QImage img = QImage(sw, sh, QImage::Format_RGB32);
507 
508  QPainter painter(&img);
509  QPoint p1(sw*4 / 10, 0);
510  QPoint p2(sw*6 / 10, sh);
511  QLinearGradient linearGrad(p1, p2);
512  linearGrad.setColorAt(0, Qt::black);
513  linearGrad.setColorAt(1, Qt::white);
514  painter.setBrush(linearGrad);
515  painter.fillRect(0, 0, sw, sh, QBrush(linearGrad));
516 
517  painter.setPen(QPen(QColor(64, 64, 64), 4));
518  painter.setBrush(QBrush());
519  painter.drawRect(2, 2, sw - 3, sh - 3);
520  painter.end();
521 #endif
522 #if defined(PICTUREFLOW_QT3) || defined(PICTUREFLOW_QT2)
523  QPixmap pixmap(sw, sh, 32);
524  QPainter painter(&pixmap);
525  painter.fillRect(pixmap.rect(), QColor(192, 192, 192));
526  painter.fillRect(5, 5, sw - 10, sh - 10, QColor(64, 64, 64));
527  painter.end();
528  QImage img = pixmap.convertToImage();
529 #endif
530 
531  blankSurface = prepareSurface(&img, sw, sh, bgcolor, state->reflectionEffect);
532  }
533  return blankSurface;
534  }
535 
536 #ifdef PICTUREFLOW_QT4
537  bool exist = imageHash.contains(slideIndex);
538  if (exist)
539  if (img == imageHash.find(slideIndex).value())
540 #endif
541 #ifdef PICTUREFLOW_QT3
542  bool exist = imageHash.find(slideIndex) != imageHash.end();
543  if (exist)
544  if (img == imageHash.find(slideIndex).data())
545 #endif
546 #ifdef PICTUREFLOW_QT2
547  if (img == imageHash[slideIndex])
548 #endif
549  if (surfaceCache.contains(key))
550  return surfaceCache[key];
551 
552  QImage* sr = prepareSurface(img, state->slideWidth, state->slideHeight, bgcolor, state->reflectionEffect);
553  surfaceCache.insert(key, sr);
554  imageHash.insert(slideIndex, img);
555 
556  return sr;
557 }

559 // Renders a slide to offscreen buffer. Returns a rect of the rendered area.
560 // col1 and col2 limit the column for rendering.
561 QRect QxtFlowViewSoftwareRenderer::renderSlide(const SlideInfo &slide, int col1, int col2)
562 {
563  int blend = slide.blend;
564  if (!blend)
565  return QRect();
566 
567  QImage* src = surface(slide.slideIndex);
568  if (!src)
569  return QRect();
570 
571  QRect rect(0, 0, 0, 0);
572 
573  int sw = src->height();
574  int sh = src->width();
575  int h = buffer.height();
576  int w = buffer.width();
577 
578  if (col1 > col2)
579  {
580  int c = col2;
581  col2 = col1;
582  col1 = c;
583  }
584 
585  col1 = (col1 >= 0) ? col1 : 0;
586  col2 = (col2 >= 0) ? col2 : w - 1;
587  col1 = qMin(col1, w - 1);
588  col2 = qMin(col2, w - 1);
589 
590  int zoom = 100;
591  int distance = h * 100 / zoom;
592  PFreal sdx = fcos(slide.angle);
593  PFreal sdy = fsin(slide.angle);
594  PFreal xs = slide.cx - state->slideWidth * sdx / 2;
595  PFreal ys = slide.cy - state->slideWidth * sdy / 2;
596  PFreal dist = distance * PFREAL_ONE;
597 
598  int xi = qMax((PFreal)0, ((w * PFREAL_ONE / 2) + fdiv(xs * h, dist + ys)) >> PFREAL_SHIFT);
599  if (xi >= w)
600  return rect;
601 
602  bool flag = false;
603  rect.setLeft(xi);
604  for (int x = qMax(xi, col1); x <= col2; x++)
605  {
606  PFreal hity = 0;
607  PFreal fk = rays[x];
608  if (sdy)
609  {
610  fk = fk - fdiv(sdx, sdy);
611  hity = -fdiv((rays[x] * distance - slide.cx + slide.cy * sdx / sdy), fk);
612  }
613 
614  dist = distance * PFREAL_ONE + hity;
615  if (dist < 0)
616  continue;
617 
618  PFreal hitx = fmul(dist, rays[x]);
619  PFreal hitdist = fdiv(hitx - slide.cx, sdx);
620 
621  int column = sw / 2 + (hitdist >> PFREAL_SHIFT);
622  if (column >= sw)
623  break;
624  if (column < 0)
625  continue;
626 
627  rect.setRight(x);
628  if (!flag)
629  rect.setLeft(x);
630  flag = true;
631 
632  int y1 = h / 2;
633  int y2 = y1 + 1;
634  QRgb* pixel1 = (QRgb*)(buffer.scanLine(y1)) + x;
635  QRgb* pixel2 = (QRgb*)(buffer.scanLine(y2)) + x;
636  QRgb pixelstep = pixel2 - pixel1;
637 
638  int center = (sh / 2);
639  int dy = dist / h;
640  int p1 = center * PFREAL_ONE - dy / 2;
641  int p2 = center * PFREAL_ONE + dy / 2;
642 
643  const QRgb *ptr = (const QRgb*)(src->scanLine(column));
644  if (blend == 256)
645  while ((y1 >= 0) && (y2 < h) && (p1 >= 0))
646  {
647  *pixel1 = ptr[p1 >> PFREAL_SHIFT];
648  *pixel2 = ptr[p2 >> PFREAL_SHIFT];
649  p1 -= dy;
650  p2 += dy;
651  y1--;
652  y2++;
653  pixel1 -= pixelstep;
654  pixel2 += pixelstep;
655  }
656  else
657  while ((y1 >= 0) && (y2 < h) && (p1 >= 0))
658  {
659  QRgb c1 = ptr[p1 >> PFREAL_SHIFT];
660  QRgb c2 = ptr[p2 >> PFREAL_SHIFT];
661  *pixel1 = blendColor(c1, bgcolor, blend);
662  *pixel2 = blendColor(c2, bgcolor, blend);
663  p1 -= dy;
664  p2 += dy;
665  y1--;
666  y2++;
667  pixel1 -= pixelstep;
668  pixel2 += pixelstep;
669  }
670  }
671 
672  rect.setTop(0);
673  rect.setBottom(h - 1);
674  return rect;
