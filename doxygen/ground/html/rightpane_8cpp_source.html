69 void RightPanePlaceHolder::applyStoredSize(int width)
70 {
71  if (width) {
72  QSplitter *splitter = qobject_cast<QSplitter *>(parentWidget());
73  if (splitter) {
74  // A splitter we need to resize the splitter sizes
75  QList<int> sizes = splitter->sizes();
76  int index = splitter->indexOf(this);
77  int diff = width - sizes.at(index);
78  int adjust = sizes.count() > 1 ? (diff / (sizes.count() - 1)) : 0;
79  for (int i = 0; i < sizes.count(); ++i) {
80  if (i != index)
81  sizes[i] -= adjust;
82  }
83  sizes[index]= width;
84  splitter->setSizes(sizes);
85  } else {
86  QSize s = size();
87  s.setWidth(width);
88  resize(s);
89  }
90  }
91 }

99 void RightPanePlaceHolder::currentModeChanged(Core::IMode *mode)
100 {
101  if (m_current == this) {
102  m_current = 0;
103  RightPaneWidget::instance()->setParent(0);
104  RightPaneWidget::instance()->hide();
105  }
106  if (m_mode == mode) {
107  m_current = this;
108 
109  int width = RightPaneWidget::instance()->storedWidth();
110 
111  layout()->addWidget(RightPaneWidget::instance());
112  RightPaneWidget::instance()->show();
113 
114  applyStoredSize(width);
115  setVisible(RightPaneWidget::instance()->isShown());
116  }
