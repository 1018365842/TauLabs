77 void QxtXmlFileLoggerEngine::initLoggerEngine()
78 {
79  QxtAbstractFileLoggerEngine::initLoggerEngine();
80 
81  // Mkay, we have an open file. We need to check that it's all valid.
82  // at the end of this block of code, we either can't log, or the carat is ready for writing.
83  /*
84  <?xml version="1.0" encoding="UTF-8"?>
85  <log>
86  <entry type="Error" time="sometime">
87  <message>What's going on?</message>
88  <message?Hi there</message>
89  </entry>
90  </log>
91  */
92  QIODevice* file = device();
93  Q_ASSERT(file);
94  if (file->size() == 0)
95  {
96  file->write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
97  file->write("<log>\n");
98  file->write("</log>");
99  }
100  else
101  {
102  QByteArray data = file->read(64);
103  if (!data.startsWith(QByteArray("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<log>")))
104  {
105  QFile* ptr_fileTarget = static_cast<QFile*>(file);
106  qxtLog->warning(QString(" is not a valid XML log file.").prepend(ptr_fileTarget->fileName()));
107  killLoggerEngine();
108  return;
109  }
110  }
111 }

116 void QxtXmlFileLoggerEngine::writeToFile(const QString &level, const QVariantList &messages)
117 {
118  QIODevice* ptr_fileTarget = device();
119  Q_ASSERT(ptr_fileTarget);
120  ptr_fileTarget->seek(ptr_fileTarget->size() - 6);
121  ptr_fileTarget->write(qxt_d().tab.toUtf8());
122  ptr_fileTarget->write("<entry type=\"");
123  ptr_fileTarget->write(level.toUtf8());
124  ptr_fileTarget->write("\" time=\"");
125  ptr_fileTarget->write(QTime::currentTime().toString("hh:mm:ss.zzzz").toUtf8());
126  ptr_fileTarget->write("\">");
127  ptr_fileTarget->write("\n");
128  Q_FOREACH(const QVariant& m, messages)
129  {
130  ptr_fileTarget->write(qxt_d().tab.toUtf8());
131  ptr_fileTarget->write(qxt_d().tab.toUtf8());
132  ptr_fileTarget->write("<message>");
133  ptr_fileTarget->write(toXmlSafeString(m.toString()).toUtf8());
134  ptr_fileTarget->write("</message>\n");
135  }
136 
137  ptr_fileTarget->write(qxt_d().tab.toUtf8());
138  ptr_fileTarget->write("</entry>");
139  ptr_fileTarget->write("\n");
140  ptr_fileTarget->write("</log>");
141 }

146 QString QxtXmlFileLoggerEngine::toXmlSafeString(const QString &raw)
147 {
148  /* Reserved characters:
149  < &lt;
150  & &amp;
151  > &lt;
152  ' &apos;
153  " &quot;
154 
155  Convert ampersands first, then the rest.
156  */
157  return QByteArray(raw.toUtf8()).replace('&', "&amp;").replace('<', "&lt;").replace('>', "&gt;").replace('\'', "&apos;").replace('"', "&quot;");
158 }
