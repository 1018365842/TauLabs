84 void GeneralSettings::fillLanguageBox() const
85 {
86  const QString currentLocale = language();
87 
88  m_page->languageBox->addItem(tr("<System Language>"), QString());
89  // need to add this explicitly, since there is no qm file for English
90  m_page->languageBox->addItem(QLatin1String("English"), QLatin1String("C"));
91  if (currentLocale == QLatin1String("C"))
92  m_page->languageBox->setCurrentIndex(m_page->languageBox->count() - 1);
93 
94  const QString creatorTrPath =
95  Core::ICore::instance()->resourcePath() + QLatin1String("/translations");
96  const QStringList languageFiles = QDir(creatorTrPath).entryList(QStringList(QLatin1String("openpilotgcs*.qm")));
97 
98  Q_FOREACH(const QString &languageFile, languageFiles)
99  {
100  int start = languageFile.indexOf(QLatin1Char('_'))+1;
101  int end = languageFile.lastIndexOf(QLatin1Char('.'));
102  const QString locale = languageFile.mid(start, end-start);
103  // no need to show a language that creator will not load anyway
104  if (hasQmFilesForLocale(locale, creatorTrPath)) {
105  m_page->languageBox->addItem(QLocale::languageToString(QLocale(locale).language()), locale);
106  if (locale == currentLocale)
107  m_page->languageBox->setCurrentIndex(m_page->languageBox->count() - 1);
108  }
109  }
110 }
