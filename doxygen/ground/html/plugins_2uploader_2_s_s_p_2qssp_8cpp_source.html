168 int16_t qssp::ssp_SendProcess( )
169 {
170  int16_t value = SSP_TX_WAITING;
171 
172  if (thisport->SendState == SSP_AWAITING_ACK ) {
173  if (sf_CheckTimeout() == TRUE) {
174  if (thisport->retryCount < thisport->maxRetryCount) {
175  // Try again
176  sf_SendPacket();
177  sf_SetSendTimeout();
178  value = SSP_TX_WAITING;
179  } else {
180  // Give up, # of trys has exceded the limit
181  value = SSP_TX_TIMEOUT;
182  CLEARBIT( thisport->flags, ACK_RECEIVED);
183  thisport->SendState = SSP_IDLE;
184  if (debug)
185  qDebug()<<"Send TimeOut!";
186  }
187  } else {
188  value = SSP_TX_WAITING;
189  }
190  } else if( thisport->SendState == SSP_ACKED ) {
191  SETBIT( thisport->flags, ACK_RECEIVED);
192  value = SSP_TX_ACKED;
193  thisport->SendState = SSP_IDLE;
194  } else {
195  thisport->SendState = SSP_IDLE;
196  value = SSP_TX_IDLE;
197  }
198  return value;
199 }

210 {
211  int16_t b;

213 
214  do {

287 int16_t qssp::ssp_SendData(const uint8_t *data, const uint16_t length )
288 {
289 
290  int16_t value = SSP_TX_WAITING;
291 
292  if( (length + 2) > thisport->txBufSize ) {
293  // TRYING to send too much data.
294  value = SSP_TX_BUFOVERRUN;
295  } else if( thisport->SendState == SSP_IDLE ) {
296 #ifdef ACTIVE_SYNCH
297  if( thisport->sendSynch == TRUE ) {
298  sf_SendSynchPacket();
299  }
300 #endif
301 
302 #ifdef SYNCH_SEND
303  if( length == 0 ) {
304  // TODO this method could allow a task/user to start a synchronisation step if a zero is mistakenly passed to this function.
305  // could add a check for a NULL data pointer, or use some sort of static flag that can only be accessed by a static function
306  // that must be called before calling this function.
307  // we are attempting to send a synch packet
308  thisport->txSeqNo = 0; // make this zero to cause the other end to re-synch with us
309  SETBIT(thisport->flags, SENT_SYNCH);
310  } else {
311  // we are sending a data packet
312  CLEARBIT( thisport->txSeqNo, ACK_BIT ); // make sure we are not sending a ACK packet
313  thisport->txSeqNo++; // update the sequence number.
314  if( thisport->txSeqNo > 0x7F) { // check for sequence number rollover
315  thisport->txSeqNo = 1; // if we do have rollover then reset to 1 not zero,
316  // zero is reserviced for synchronization requests
317  }
318  }
319 
320 #else
321  CLEARBIT( thisport->txSeqNo, ACK_BIT ); // make sure we are not sending a ACK packet
322  thisport->txSeqNo++; // update the sequence number.
323  if( thisport->txSeqNo > 0x7F) { // check for sequence number rollover
324  thisport->txSeqNo = 1; // if we do have rollover then reset to 1 not zero,
325  // zero is reserved for synchronization requests
326  }
327 #endif
328  CLEARBIT( thisport->flags, ACK_RECEIVED);
329  thisport->SendState = SSP_AWAITING_ACK;
330  value = SSP_TX_WAITING;
331  thisport->retryCount = 0; // zero out the retry counter for this transmission
332  sf_MakePacket( thisport->txBuf, data, length, thisport->txSeqNo );
333  sf_SendPacket( ); // punch out the packet to the serial port
334  sf_SetSendTimeout( ); // do the timeout values
335  if (debug)
336  qDebug()<<"Sent DATA PACKET:"<<thisport->txSeqNo;
337  } else {
338  // error we are already sending a packet. Need to wait for the current packet to be acked or timeout.
339  value = SSP_TX_BUSY;
340  if (debug)
341  qDebug()<<"Error sending TX was busy";
342  }
343  return value;
344 }

437 void qssp::sf_MakePacket( uint8_t *txBuf, const uint8_t * pdata, uint16_t length, uint8_t seqNo )
438 {
439  uint16_t crc = 0xffff;
440  uint16_t bufPos = 0;
441  uint8_t b;
442 
443  // add 1 for the seq. number
444  txBuf[LENGTH] = length + 1;
445  txBuf[SEQNUM] = seqNo;
446  crc = sf_crc16( crc, seqNo );
447 
448  length = length + 2; // add two for the length and seqno bytes which are added before the loop.
449  for( bufPos = 2; bufPos < length; bufPos++ ) {
450  b = *pdata++;
451  txBuf[bufPos] = b;
452  crc = sf_crc16( crc, b ); // update CRC value
453  }
454  txBuf[bufPos++] = LOWERBYTE(crc);
455  txBuf[bufPos] = UPPERBYTE(crc);
456

591 {
592  int16_t retval = SSP_RX_RECEIVING;
593 
594  switch( thisport->InputState ) {

597  thisport->DecodeState = decode_len1_e;

612  }

640 int16_t qssp::sf_DecodeState( uint8_t c )
641 {
642  int16_t retval;
643  switch( thisport->DecodeState ) {
644  case decode_idle_e:
645  // 'c' is ignored in this state as the only way to leave the idle state is
646  // recognition of the SYNC byte in the sf_ReceiveState function.
647  retval = SSP_RX_IDLE;
648  break;
649  case decode_len1_e:
650  thisport->rxBuf[LENGTH]= c;
651  thisport->rxBufLen = c;
652  if( thisport->rxBufLen <= thisport->rxBufSize ) {
653  thisport->DecodeState = decode_seqNo_e;
654  retval = SSP_RX_RECEIVING;
655  } else {
656  thisport->DecodeState = decode_idle_e;
657  retval = SSP_RX_IDLE;
658  }
659  break;
660  case decode_seqNo_e:
661  thisport->rxBuf[SEQNUM] = c;
662  thisport->crc = 0xffff;
663  thisport->rxBufLen--; // subtract 1 for the seq. no.
664  thisport->rxBufPos = 2;
665 
666  thisport->crc = sf_crc16( thisport->crc, c );
667  if( thisport->rxBufLen > 0 ) {
668  thisport->DecodeState = decode_data_e;
669  } else {
670  thisport->DecodeState = decode_crc1_e;
671  }
672  retval = SSP_RX_RECEIVING;
673  break;
674  case decode_data_e:
675  thisport->rxBuf[ (thisport->rxBufPos)++] = c;
676  thisport->crc = sf_crc16( thisport->crc, c );
677  if( thisport->rxBufPos == (thisport->rxBufLen+2) ) {
678  thisport->DecodeState = decode_crc1_e;
679  }
680  retval = SSP_RX_RECEIVING;
681  break;
682  case decode_crc1_e:
683  thisport->crc = sf_crc16( thisport->crc, c );
684  thisport->DecodeState = decode_crc2_e;
685  retval = SSP_RX_RECEIVING;
686  break;
687  case decode_crc2_e:
688  thisport->DecodeState = decode_idle_e;
689  // verify the CRC value for the packet
690  if( sf_crc16( thisport->crc, c) == 0) {
691  // TODO shouldn't the return value of sf_ReceivePacket() be checked?
692  sf_ReceivePacket();
693  retval = SSP_RX_COMPLETE;
694  } else {
695  thisport->RxError++;
696  retval = SSP_RX_IDLE;
697  }
698  break;
699  default:
700  thisport->DecodeState = decode_idle_e; // unknown state so reset to idle state and wait for the next start of a packet.
701  retval = SSP_RX_IDLE;
702  break;
703  }
704  return retval;
705 }

735  if( ISBITSET(thisport->rxBuf[SEQNUM], ACK_BIT ) ) {
736  // Received an ACK packet, need to check if it matches the previous sent packet
737  if( ( thisport->rxBuf[SEQNUM] & 0x7F) == (thisport->txSeqNo & 0x7f)) {
738  // It matches the last packet sent by us
739  SETBIT( thisport->txSeqNo, ACK_BIT );
740  thisport->SendState = SSP_ACKED;
741  value = FALSE;
742  if (debug)
743  qDebug()<<"Received ACK:"<<(thisport->txSeqNo & 0x7F);
744  }
745  // else ignore the ACK packet
746  } else {
747  // Received a 'data' packet, figure out what type of packet we received...
748  if( thisport->rxBuf[SEQNUM] == 0 ) {
749  if (debug)
750  qDebug()<<"Received SYNC Request";
751  // Synchronize sequence number with host
752 #ifdef ACTIVE_SYNCH
753  thisport->sendSynch = TRUE;
754 #endif
755  sf_SendAckPacket(thisport->rxBuf[SEQNUM] );
756  thisport->rxSeqNo = 0;
757  value = FALSE;
758  } else if( thisport->rxBuf[SEQNUM] == thisport->rxSeqNo ) {
759  // Already seen this packet, just ack it, don't act on the packet.
760  sf_SendAckPacket(thisport->rxBuf[SEQNUM] );
761  value = FALSE;
762  } else {
763  //New Packet
764  thisport->rxSeqNo = thisport->rxBuf[SEQNUM];
765  // Let the application do something with the data/packet.
766 
767  // skip the first two bytes (length and seq. no.) in the buffer.
768  if (debug)
769  qDebug()<<"Received DATA PACKET seq="<<thisport->rxSeqNo<<"Data="<<(uint8_t)thisport->rxBuf[2]<<(uint8_t)thisport->rxBuf[3]<<(uint8_t)thisport->rxBuf[4];
770  pfCallBack( &(thisport->rxBuf[2]), thisport->rxBufLen);
771 
772  // after we send the ACK, it is possible for the host to send a new packet.
773  // Thus the application needs to copy the data and reset the receive buffer
774  // inside of thisport->pfCallBack()
775  sf_SendAckPacket(thisport->rxBuf[SEQNUM] );
776  value = TRUE;
777  }
778  }
779  return value;
780 }
