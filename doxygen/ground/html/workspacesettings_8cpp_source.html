149 void WorkspaceSettings::apply()
150 {
151  selectWorkspace(m_currentIndex, true);
152 
153  saveSettings(Core::ICore::instance()->settings());
154 
155  if (m_numberOfWorkspaces != m_previousNumberOfWorkspaces) {
156  Core::ICore::instance()->readMainSettings(Core::ICore::instance()->settings(), true);
157  m_previousNumberOfWorkspaces = m_numberOfWorkspaces;
158  }
159 
160  ModeManager* modeManager = Core::ICore::instance()->modeManager();
161  for (int i = 0; i < MAX_WORKSPACES; ++i) {
162  IMode *baseMode = modeManager->mode(modeName(i));
163  Core::UAVGadgetManager *mode = qobject_cast<Core::UAVGadgetManager*>(baseMode);
164  if (mode) {
165  modeManager->updateModeNameIcon(mode, QIcon(iconName(i)), name(i));
166  }
167  }
168  m_tabBarPlacementIndex = m_page->comboBoxTabBarPlacement->currentIndex();
169  m_allowTabBarMovement = m_page->checkBoxAllowTabMovement->isChecked();
170  QTabWidget::TabPosition pos = m_tabBarPlacementIndex == 0 ? QTabWidget::North : QTabWidget::South;
171  emit tabBarSettingsApplied(pos, m_allowTabBarMovement);
172 }

223 void WorkspaceSettings::newModeOrder(QVector<IMode*> modes)
224 {
225  QList<int> priorities;
226  QStringList modeNames;
227  for (int i = 0; i < modes.count(); ++i) {
228  Core::UAVGadgetManager *mode = qobject_cast<Core::UAVGadgetManager*>(modes.at(i));
229  if (mode) {
230  priorities.append(mode->priority());
231  modeNames.append(mode->uniqueModeName());
232  }
233  }
234  // Bubble sort
235  bool swapped = false;
236  do {
237  swapped = false;
238  for (int i = 0; i < m_names.count()-1; ++i) {
239  int j = i+1;
240  int p = modeNames.indexOf(m_modeNames.at(i));
241  int q = modeNames.indexOf(m_modeNames.at(j));
242  bool nonShowingMode = (p == -1 && q >=0);
243  bool pqBothFound = (p >= 0 && q >= 0);
244  if (nonShowingMode || (pqBothFound && (priorities.at(q) > priorities.at(p)))) {
245  m_names.swap(i, j);
246  m_iconNames.swap(i, j);
247  m_modeNames.swap(i, j);
248  swapped = true;
249  }
250  }
251  } while (swapped);
252 }
