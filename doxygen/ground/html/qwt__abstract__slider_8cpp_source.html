148 void QwtAbstractSlider::mousePressEvent( QMouseEvent *e )
149 {
150  if ( isReadOnly() )
151  {
152  e->ignore();
153  return;
154  }
155  if ( !isValid() )
156  return;
157 
158  const QPoint &p = e->pos();
159 
160  d_data->timerTick = 0;
161 
162  getScrollMode( p, d_data->scrollMode, d_data->direction );
163  stopMoving();
164 
165  switch ( d_data->scrollMode )
166  {
167  case ScrPage:
168  case ScrTimer:
169  d_data->mouseOffset = 0;
170  d_data->tmrID = startTimer( qMax( 250, 2 * d_data->updTime ) );
171  break;
172 
173  case ScrMouse:
174  d_data->time.start();
175  d_data->speed = 0;
176  d_data->mouseOffset = getValue( p ) - value();
177  Q_EMIT sliderPressed();
178  break;
179 
180  default:
181  d_data->mouseOffset = 0;
182  d_data->direction = 0;
183  break;
184  }
185 }

200 void QwtAbstractSlider::mouseReleaseEvent( QMouseEvent *e )
201 {
202  if ( isReadOnly() )
203  {
204  e->ignore();
205  return;
206  }
207  if ( !isValid() )
208  return;
209 
210  const double inc = step();
211 
212  switch ( d_data->scrollMode )
213  {
214  case ScrMouse:
215  {
216  setPosition( e->pos() );
217  d_data->direction = 0;
218  d_data->mouseOffset = 0;
219  if ( d_data->mass > 0.0 )
220  {
221  const int ms = d_data->time.elapsed();
222  if ( ( qFabs( d_data->speed ) > 0.0 ) && ( ms < 50 ) )
223  d_data->tmrID = startTimer( d_data->updTime );
224  }
225  else
226  {
227  d_data->scrollMode = ScrNone;
228  buttonReleased();
229  }
230  Q_EMIT sliderReleased();
231 
232  break;
233  }
234 
235  case ScrDirect:
236  {
237  setPosition( e->pos() );
238  d_data->direction = 0;
239  d_data->mouseOffset = 0;
240  d_data->scrollMode = ScrNone;
241  buttonReleased();
242  break;
243  }
244 
245  case ScrPage:
246  {
247  stopMoving();
248  if ( !d_data->timerTick )
249  QwtDoubleRange::incPages( d_data->direction );
250  d_data->timerTick = 0;
251  buttonReleased();
252  d_data->scrollMode = ScrNone;
253  break;
254  }
255 
256  case ScrTimer:
257  {
258  stopMoving();
259  if ( !d_data->timerTick )
260  QwtDoubleRange::fitValue( value() + double( d_data->direction ) * inc );
261  d_data->timerTick = 0;
262  buttonReleased();
263  d_data->scrollMode = ScrNone;
264  break;
265  }
266 
267  default:
268  {
269  d_data->scrollMode = ScrNone;
270  buttonReleased();
271  }
272  }
273 }

309 void QwtAbstractSlider::mouseMoveEvent( QMouseEvent *e )
310 {
311  if ( isReadOnly() )
312  {
313  e->ignore();
314  return;
315  }
316 
317  if ( !isValid() )
318  return;
319 
320  if ( d_data->scrollMode == ScrMouse )
321  {
322  setPosition( e->pos() );
323  if ( d_data->mass > 0.0 )
324  {
325  double ms = double( d_data->time.elapsed() );
326  if ( ms < 1.0 )
327  ms = 1.0;
328  d_data->speed = ( exactValue() - exactPrevValue() ) / ms;
329  d_data->time.start();
330  }
331  if ( value() != prevValue() )
332  Q_EMIT sliderMoved( value() );
333  }

425 void QwtAbstractSlider::timerEvent( QTimerEvent * )
426 {
427  const double inc = step();
428 
429  switch ( d_data->scrollMode )
430  {
431  case ScrMouse:
432  {
433  if ( d_data->mass > 0.0 )
434  {
435  d_data->speed *= qExp( - double( d_data->updTime ) * 0.001 / d_data->mass );
436  const double newval =
437  exactValue() + d_data->speed * double( d_data->updTime );
438  QwtDoubleRange::fitValue( newval );
439  // stop if d_data->speed < one step per second
440  if ( qFabs( d_data->speed ) < 0.001 * qFabs( step() ) )
441  {
442  d_data->speed = 0;
443  stopMoving();
444  buttonReleased();
445  }
446 
447  }
448  else
449  stopMoving();
450  break;
451  }
452 
453  case ScrPage:
454  {
455  QwtDoubleRange::incPages( d_data->direction );
456  if ( !d_data->timerTick )
457  {
458  killTimer( d_data->tmrID );
459  d_data->tmrID = startTimer( d_data->updTime );
460  }
461  break;
462  }
463  case ScrTimer:
464  {
465  QwtDoubleRange::fitValue( value() + double( d_data->direction ) * inc );
466  if ( !d_data->timerTick )
467  {
468  killTimer( d_data->tmrID );
469  d_data->tmrID = startTimer( d_data->updTime );
470  }
471  break;
472  }
473  default:
474  {
475  stopMoving();
476  break;
477  }
478  }
479 
480  d_data->timerTick = 1;
481 }
