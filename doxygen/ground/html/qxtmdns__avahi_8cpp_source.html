60 void QxtMDNSPrivate::avahiClientCallback(AvahiClient *s, AvahiClientState state, void *userdata)
61 {
62  QxtMDNSPrivate* self = static_cast<QxtMDNSPrivate*>(userdata);
63  self->client = s;
64  if (state == AVAHI_CLIENT_S_RUNNING)
65  {
66  self->recordbrowser = avahi_record_browser_new(
67  s,
68  AVAHI_IF_UNSPEC,
69  AVAHI_PROTO_INET,
70  self->name.toAscii().constData(),
71  AVAHI_DNS_CLASS_IN,
72  AVAHI_DNS_TYPE_A,
73  (AvahiLookupFlags)0,
74  QxtMDNSPrivate::avahiRecordBrowserCallback,
75  self);
76  if (self->recordbrowser == NULL)
77  {
78  self->info.setError(QHostInfo::UnknownError);
79  self->info.setErrorString(avahi_strerror(avahi_client_errno(self->client)));
80  QMetaObject::invokeMethod(self->receiver, self->member, Q_ARG(QHostInfo, self->info));
81  self->sent = true;
82  self->qxt_p().cancelLookup();
83  }
84  }
85 }

87 void QxtMDNSPrivate::avahiRecordBrowserCallback(AvahiRecordBrowser *b, AvahiIfIndex interface, AvahiProtocol protocol, AvahiBrowserEvent event, const char *name, uint16_t clazz, uint16_t type, const void *rdata, size_t size, AvahiLookupResultFlags flags, void *userdata)
88 {
90  Q_UNUSED(interface);
91  Q_UNUSED(protocol);
92  Q_UNUSED(name);
93  Q_UNUSED(clazz);
94  Q_UNUSED(type);
95  Q_UNUSED(size);
96  Q_UNUSED(flags);
97  QxtMDNSPrivate* self = static_cast<QxtMDNSPrivate*>(userdata);
98  self->recordbrowser = b;
99 // qDebug() << "Return thing" << md->name << name << size;
100  switch (event)
101  {
102  case AVAHI_BROWSER_NEW:
103  {
104  //Found an entry!
105  uint32_t ip = qFromBigEndian(*static_cast<const uint32_t*>(rdata));
106  if (self->sent)
107  {
108  QHostInfo info(self->info.lookupId());
109  info.setAddresses(QList<QHostAddress>() << QHostAddress(ip));
110  QMetaObject::invokeMethod(self->receiver, self->member, Q_ARG(QHostInfo, info));
111  }
112  else
113  {
114  self->addresses << QHostAddress(ip);
115  }
116  break;
117  }
118  case AVAHI_BROWSER_REMOVE:
119  {
120  uint32_t ip = qFromBigEndian(*static_cast<const uint32_t*>(rdata));
121  self->addresses.removeAll(QHostAddress(ip));
122  break;
123  }
124  case AVAHI_BROWSER_CACHE_EXHAUSTED:
125  break;
126  case AVAHI_BROWSER_ALL_FOR_NOW:
127  if (self->addresses.count() == 0)
128  {
129  self->info.setError(QHostInfo::HostNotFound);
130  self->info.setErrorString("The host was not found.");
131  }
132  else
133  {
134  self->info.setAddresses(self->addresses);
135  self->addresses.clear();
136  }
137  QMetaObject::invokeMethod(self->receiver, self->member, Q_ARG(QHostInfo, self->info));
138  self->sent = true;
139  break;
140  case AVAHI_BROWSER_FAILURE:
141 
142  if (self->sent)
143  {
144  QHostInfo info(self->info.lookupId());
145  info.setError(QHostInfo::UnknownError);
146  info.setErrorString(avahi_strerror(avahi_client_errno(self->client)));
147  info.setAddresses(self->addresses);
148  QMetaObject::invokeMethod(self->receiver, self->member, Q_ARG(QHostInfo, info));
149  }
150  else
151  {
152  self->info.setError(QHostInfo::UnknownError);
153  self->info.setErrorString(avahi_strerror(avahi_client_errno(self->client)));
154  self->info.setAddresses(self->addresses);
155  self->addresses.clear();
156  QMetaObject::invokeMethod(self->receiver, self->member, Q_ARG(QHostInfo, self->info));
157  self->sent = true;
158  }
159  break;
160  }
161 }
