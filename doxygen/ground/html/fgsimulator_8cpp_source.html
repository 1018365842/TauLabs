151 void FGSimulator::transmitUpdate()
152 {
153  ActuatorDesired::DataFields actData;
154  FlightStatus::DataFields flightStatusData = flightStatus->getData();
155  ManualControlCommand::DataFields manCtrlData = manCtrlCommand->getData();
156  float ailerons = -1;
157  float elevator = -1;
158  float rudder = -1;
159  float throttle = -1;
160 
161  if(flightStatusData.FlightMode == FlightStatus::FLIGHTMODE_MANUAL)
162  {
163  // Read joystick input
164  if(flightStatusData.Armed == FlightStatus::ARMED_ARMED)
165  {
166  // Note: Pitch sign is reversed in FG ?
167  ailerons = manCtrlData.Roll;
168  elevator = -manCtrlData.Pitch;
169  rudder = manCtrlData.Yaw;
170  throttle = manCtrlData.Throttle;
171  }
172  }
173  else
174  {
175  // Read ActuatorDesired from autopilot
176  actData = actDesired->getData();
177 
178  ailerons = actData.Roll;
179  elevator = -actData.Pitch;
180  rudder = actData.Yaw;
181  throttle = actData.Throttle;
182  }
183 
184  int allowableDifference = 10;
185 
186  //qDebug() << "UDP sent:" << udpCounterGCSsend << " - UDP Received:" << udpCounterFGrecv;
187 
188  if(udpCounterFGrecv == udpCounterGCSsend)
189  udpCounterGCSsend = 0;
190 
191  if((udpCounterGCSsend < allowableDifference) || (udpCounterFGrecv==0) ) //FG udp queue is not delayed
192  {
193  udpCounterGCSsend++;
194 
195  // Send update to FlightGear
196  QString cmd;
197  cmd = QString("%1,%2,%3,%4,%5\n")
198  .arg(ailerons) //ailerons
199  .arg(elevator) //elevator
200  .arg(rudder) //rudder
201  .arg(throttle) //throttle
202  .arg(udpCounterGCSsend); //UDP packet counter delay
203 
204  QByteArray data = cmd.toAscii();
205 
206  if(outSocket->writeDatagram(data, QHostAddress(settings.remoteAddress), settings.outPort) == -1)
207  {
208  emit processOutput("Error sending UDP packet to FG: " + outSocket->errorString() + "\n");
209  }
210  }
211  else
212  {
213  // don't send new packet. Flightgear cannot process UDP fast enough.
214  // V1.9.1 reads udp packets at set frequency and will get delayed if packets are sent too fast
215  // V2.0 does not currently work with --generic-protocol
216  }
217 
218  if(settings.manualControlEnabled)
219  {
220  actData.Roll = ailerons;
221  actData.Pitch = -elevator;
222  actData.Yaw = rudder;
223  actData.Throttle = throttle;
224  //actData.NumLongUpdates = (float)udpCounterFGrecv;
225  //actData.UpdateTime = (float)udpCounterGCSsend;
226  actDesired->setData(actData);
227  }
228 }
