212 void Simulator::receiveUpdate()
213 {
214  // Update connection timer and status
215  simTimer->setInterval(simTimeout);
216  simTimer->stop();
217  simTimer->start();
218  if ( !simConnectionStatus )
219  {
220  simConnectionStatus = true;
221  emit simulatorConnected();
222  }
223 
224  // Process data
225  while(inSocket->hasPendingDatagrams()) {
226  // Receive datagram
227  QByteArray datagram;
228  datagram.resize(inSocket->pendingDatagramSize());
229  QHostAddress sender;
230  quint16 senderPort;
231  inSocket->readDatagram(datagram.data(), datagram.size(),
232  &sender, &senderPort);
233 
234  // Process incomming data
235  processUpdate(datagram);
236  }

239 void Simulator::setupUAVObjects()
240 {
241  UAVObjectUtilManager* utilMngr = getObjectUtilManager();
242 
243  // Store original metadata, but only if it hasn't already been set
244  // by another class instance.
245  if (originalMetaData.empty()) {
246  originalMetaData = utilMngr->readAllNonSettingsMetadata();
247  }
248 
249  uint16_t slowUpdate = 5000; // in [ms]
250 
251  // Iterate over list of UAVObjects, setting all dynamic data metadata objects to slow update rate.
252  UAVObjectManager *objManager = getObjectManager();
253  QVector< QVector<UAVDataObject*> > objList = objManager->getDataObjects();
254  foreach (QVector<UAVDataObject*> list, objList) {
255  foreach (UAVDataObject* obj, list) {
256  if(!obj->isSettings()) {
257  UAVObject::Metadata mdata = obj->getMetadata();
258  UAVObject::SetFlightTelemetryUpdateMode(mdata, UAVObject::UPDATEMODE_PERIODIC);
259 
260  // Default configuration for all UAVO data rates is "slowUpdate"
261  mdata.flightTelemetryUpdatePeriod = slowUpdate;
262 
263  metaDataList.insert(obj->getName(), mdata);
264  }
265  }
266  }
267 
268  //-----------------------------------------//
269  // Configure metadata for individual UAVOs //
270  //-----------------------------------------//
271 
272  // Most simulators use the flight controller's ActuatorDesired UAVO
273  // for control output...
274  if (settings.simulatorId == "FG" ||
275  settings.simulatorId == "IL2" ||
276  settings.simulatorId == "X-Plane")
277  {
278  setupInputObject(actDesired, settings.minOutputPeriod);
279  }
280  else if(settings.simulatorId == "ASimRC")
281  {
282  //...however, AeroSimRC uses the servo PWM output (i.e. ActuatorCommand)
283  setupInputObject(actCommand, settings.minOutputPeriod);
284  }
285 
286  if (settings.gcsReceiverEnabled) {
287  setupOutputObject(gcsReceiver, settings.minOutputPeriod);
288  } else if (settings.manualControlEnabled) {
289  setupOutputObject(manCtrlCommand, settings.minOutputPeriod);
290  }
291 
292  if (settings.gpsPositionEnabled){
293  setupOutputObject(gpsPos, settings.gpsPosRate);
294  setupOutputObject(gpsVel, settings.gpsPosRate);
295  }
296 
297  if (settings.groundTruthEnabled){
298  setupOutputObject(posActual, settings.groundTruthRate);
299  setupOutputObject(velActual, settings.groundTruthRate);
300  }
301 
302  if (settings.attRawEnabled) {
303  setupOutputObject(accels, settings.attRawRate);
304  setupOutputObject(gyros, settings.attRawRate);
305  }
306 
307  if (settings.attActualEnabled && settings.attActHW) {
308  setupOutputObject(accels, settings.attRawRate);
309  setupOutputObject(gyros, settings.attRawRate);
310  }
311 
312  if (settings.attActualEnabled && !settings.attActHW)
313  setupOutputObject(attActual, 20); //Hardcoded? Bleh.
314  else
315  setupInputObject(attActual, 100); //Hardcoded? Bleh.
316 
317  if(settings.airspeedActualEnabled)
318  setupOutputObject(airspeedActual, settings.airspeedActualRate);
319 
320  if(settings.baroAltitudeEnabled)
321  setupOutputObject(baroAlt, settings.baroAltRate);
322 
323  // Set new metadata
324  utilMngr->setAllNonSettingsMetadata(metaDataList);
325 }

409 void Simulator::updateUAVOs(Output2Hardware out){
410 
411  QTime currentTime = QTime::currentTime();
412 
413  Noise noise;
414  HitlNoiseGeneration noiseSource;
415 
416  if(settings.addNoise){
417  noise = noiseSource.generateNoise();
418  }
419  else{
420  memset(&noise, 0, sizeof(Noise));
421  }
422 
423  /*******************************/
424  HomeLocation::DataFields homeData = posHome->getData();
425  if(!once)
426  {
427  // Upon startup, we reset the HomeLocation object to
428  // the plane's location:
429  memset(&homeData, 0, sizeof(HomeLocation::DataFields));
430  // Update homelocation
431  homeData.Latitude = out.latitude; //Already in *10^7 integer format
432  homeData.Longitude = out.longitude; //Already in *10^7 integer format

444 
445  // Compute initial distance

448  initD = out.dstD;
449

451  }
452 
453  /*******************************/
454  //Copy everything to the ground truth object. GroundTruth is Noise-free.
455  GroundTruth::DataFields groundTruthData;
456  groundTruthData = groundTruth->getData();
457 
458  groundTruthData.CalibratedAirspeed=out.calibratedAirspeed;
459  groundTruthData.TrueAirspeed=out.trueAirspeed;
460  groundTruthData.AngleOfAttack=out.angleOfAttack;
461  groundTruthData.AngleOfSlip=out.angleOfSlip;
462 
463  groundTruthData.PositionNED[0]=out.dstN-initN;
464  groundTruthData.PositionNED[1]=out.dstE-initD;
465  groundTruthData.PositionNED[2]=out.dstD-initD;
466 
467  groundTruthData.VelocityNED[0]=out.velNorth;
468  groundTruthData.VelocityNED[1]=out.velEast;
469  groundTruthData.VelocityNED[2]=out.velDown;
470 
471  groundTruthData.AccelerationXYZ[0]=out.accX;
472  groundTruthData.AccelerationXYZ[1]=out.accY;
473  groundTruthData.AccelerationXYZ[2]=out.accZ;
474 
475  groundTruthData.RPY[0]=out.roll;
476  groundTruthData.RPY[0]=out.pitch;
477  groundTruthData.RPY[0]=out.heading;
478 
479  groundTruthData.AngularRates[0]=out.rollRate;
480  groundTruthData.AngularRates[1]=out.pitchRate;
481  groundTruthData.AngularRates[2]=out.yawRate;
482 
483  //Set UAVO
484  groundTruth->setData(groundTruthData);
485 
486 /*******************************/
487  // Update attActual object
488  AttitudeActual::DataFields attActualData;
489  attActualData = attActual->getData();
490 
491  if (settings.attActHW) {
492  // do nothing
493  /*****************************************/
494  } else if (settings.attActSim) {
495  // take all data from simulator
496  attActualData.Roll = out.roll + noise.attActualData.Roll; //roll;
497  attActualData.Pitch = out.pitch + noise.attActualData.Pitch; // pitch
498  attActualData.Yaw = out.heading + noise.attActualData.Yaw; // Yaw
499  float rpy[3];
500  float quat[4];
501  rpy[0] = attActualData.Roll;
502  rpy[1] = attActualData.Pitch;
503  rpy[2] = attActualData.Yaw;
504  Utils::CoordinateConversions().RPY2Quaternion(rpy,quat);
505  attActualData.q1 = quat[0];
506  attActualData.q2 = quat[1];
507  attActualData.q3 = quat[2];
508  attActualData.q4 = quat[3];
509 
510  //Set UAVO
511  attActual->setData(attActualData);
512  /*****************************************/
513  } else if (settings.attActCalc) {
514  // calculate RPY with code from Attitude module
515  static float q[4] = {1, 0, 0, 0};
516  static float gyro_correct_int2 = 0;
517 
518  float dT = out.delT;
519 
520  AttitudeSettings::DataFields attitudeSettingsData = attitudeSettings->getData();
521  float accelKp = attitudeSettingsData.AccelKp * 0.1666666666666667;
522  float accelKi = attitudeSettingsData.AccelKp * 0.1666666666666667;
523  float yawBiasRate = attitudeSettingsData.YawBiasRate;
524 
525  // calibrate sensors on arming
526  if (flightStatus->getData().Armed == FlightStatus::ARMED_ARMING) {
527  accelKp = 2.0;
528  accelKi = 0.9;
529  }
530 
531  float gyro[3] = {out.rollRate, out.pitchRate, out.yawRate};
532  float attRawAcc[3] = {out.accX, out.accY, out.accZ};
533 
534  // code from Attitude module begin ///////////////////////////////
535  float *accels = attRawAcc;
536  float grot[3];
537  float accel_err[3];
538 
539  // Rotate gravity to body frame and cross with accels
540  grot[0] = -(2 * (q[1] * q[3] - q[0] * q[2]));
541  grot[1] = -(2 * (q[2] * q[3] + q[0] * q[1]));
542  grot[2] = -(q[0] * q[0] - q[1]*q[1] - q[2]*q[2] + q[3]*q[3]);
543 
544  // CrossProduct
545  {
546  accel_err[0] = accels[1]*grot[2] - grot[1]*accels[2];
547  accel_err[1] = grot[0]*accels[2] - accels[0]*grot[2];
548  accel_err[2] = accels[0]*grot[1] - grot[0]*accels[1];
549  }
550 
551  // Account for accel magnitude
552  float accel_mag = sqrt(accels[0] * accels[0] + accels[1] * accels[1] + accels[2] * accels[2]);
553  accel_err[0] /= accel_mag;
554  accel_err[1] /= accel_mag;
555  accel_err[2] /= accel_mag;
556 
557  // Accumulate integral of error. Scale here so that units are (deg/s) but Ki has units of s
558  gyro_correct_int2 += -gyro[2] * yawBiasRate;
559 
560  // Correct rates based on error, integral component dealt with in updateSensors
561  gyro[0] += accel_err[0] * accelKp / dT;
562  gyro[1] += accel_err[1] * accelKp / dT;
563  gyro[2] += accel_err[2] * accelKp / dT + gyro_correct_int2;
564 
565  // Work out time derivative from INSAlgo writeup
566  // Also accounts for the fact that gyros are in deg/s
567  float qdot[4];
568  qdot[0] = (-q[1] * gyro[0] - q[2] * gyro[1] - q[3] * gyro[2]) * dT * M_PI / 180 / 2;
569  qdot[1] = (+q[0] * gyro[0] - q[3] * gyro[1] + q[2] * gyro[2]) * dT * M_PI / 180 / 2;
570  qdot[2] = (+q[3] * gyro[0] + q[0] * gyro[1] - q[1] * gyro[2]) * dT * M_PI / 180 / 2;
571  qdot[3] = (-q[2] * gyro[0] + q[1] * gyro[1] + q[0] * gyro[2]) * dT * M_PI / 180 / 2;
572 
573  // Take a time step
574  q[0] += qdot[0];
575  q[1] += qdot[1];
576  q[2] += qdot[2];
577  q[3] += qdot[3];
578 
579  if(q[0] < 0) {
580  q[0] = -q[0];
581  q[1] = -q[1];
582  q[2] = -q[2];
583  q[3] = -q[3];
584  }
585 
586  // Renomalize
587  float qmag = sqrt((q[0] * q[0]) + (q[1] * q[1]) + (q[2] * q[2]) + (q[3] * q[3]));
588  q[0] /= qmag;
589  q[1] /= qmag;
590  q[2] /= qmag;
591  q[3] /= qmag;
592 
593  // If quaternion has become inappropriately short or is nan reinit.
594  // THIS SHOULD NEVER ACTUALLY HAPPEN
595  if((fabs(qmag) < 1e-3) || (qmag != qmag)) {
596  q[0] = 1;
597  q[1] = 0;
598  q[2] = 0;
599  q[3] = 0;
600  }
601 
602  float rpy2[3];
603  // Quaternion2RPY
604  {
605  float q0s, q1s, q2s, q3s;
606  q0s = q[0] * q[0];
607  q1s = q[1] * q[1];
608  q2s = q[2] * q[2];
609  q3s = q[3] * q[3];
610 
611  float R13, R11, R12, R23, R33;
612  R13 = 2 * (q[1] * q[3] - q[0] * q[2]);
613  R11 = q0s + q1s - q2s - q3s;
614  R12 = 2 * (q[1] * q[2] + q[0] * q[3]);
615  R23 = 2 * (q[2] * q[3] + q[0] * q[1]);
616  R33 = q0s - q1s - q2s + q3s;
617 
618  rpy2[1] = RAD2DEG * asinf(-R13); // pitch always between -pi/2 to pi/2
619  rpy2[2] = RAD2DEG * atan2f(R12, R11);
620  rpy2[0] = RAD2DEG * atan2f(R23, R33);
621  }
622 
623  attActualData.Roll = rpy2[0];
624  attActualData.Pitch = rpy2[1];
625  attActualData.Yaw = rpy2[2];
626  attActualData.q1 = q[0];
627  attActualData.q2 = q[1];
628  attActualData.q3 = q[2];
629  attActualData.q4 = q[3];
630 
631  //Set UAVO
632  attActual->setData(attActualData);
633  /*****************************************/
634  }
635
