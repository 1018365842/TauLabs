169 void Telemetry::connectToObjectInstances(UAVObject* obj, quint32 eventMask)
170 {
171  QVector<UAVObject*> objs = objMngr->getObjectInstances(obj->getObjID());
172  int objsSize = objs.size();
173  for (int n = 0; n < objsSize; ++n)
174  {
175  // Disconnect all
176  objs[n]->disconnect(this);
177  // Connect only the selected events
178  if ( (eventMask&EV_UNPACKED) != 0)
179  {
180  connect(objs[n], SIGNAL(objectUnpacked(UAVObject*)), this, SLOT(objectUnpacked(UAVObject*)));
181  }
182  if ( (eventMask&EV_UPDATED) != 0)
183  {
184  connect(objs[n], SIGNAL(objectUpdatedAuto(UAVObject*)), this, SLOT(objectUpdatedAuto(UAVObject*)));
185  }
186  if ( (eventMask&EV_UPDATED_MANUAL) != 0)
187  {
188  connect(objs[n], SIGNAL(objectUpdatedManual(UAVObject*)), this, SLOT(objectUpdatedManual(UAVObject*)));
189  }
190  if ( (eventMask&EV_UPDATED_PERIODIC) != 0)
191  {
192  connect(objs[n], SIGNAL(objectUpdatedPeriodic(UAVObject*)), this, SLOT(objectUpdatedPeriodic(UAVObject*)));
193  }
194  if ( (eventMask&EV_UPDATE_REQ) != 0)
195  {
196  connect(objs[n], SIGNAL(updateRequested(UAVObject*)), this, SLOT(updateRequested(UAVObject*)));
197  }
198  }

210 void Telemetry::updateObject(UAVObject* obj, quint32 eventType)
211 {
212  // Get metadata
213  UAVObject::Metadata metadata = obj->getMetadata();
214  UAVObject::UpdateMode updateMode = UAVObject::GetGcsTelemetryUpdateMode(metadata);
215 
216  // Setup object depending on update mode
217  qint32 eventMask;
218  if ( updateMode == UAVObject::UPDATEMODE_PERIODIC )
219  {
220  // Set update period
221  setUpdatePeriod(obj, metadata.gcsTelemetryUpdatePeriod);
222  // Connect signals for all instances
223  eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ | EV_UPDATED_PERIODIC;
224  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
225  connectToObjectInstances(obj, eventMask);
226  }
227  else if ( updateMode == UAVObject::UPDATEMODE_ONCHANGE )
228  {
229  // Set update period
230  setUpdatePeriod(obj, 0);
231  // Connect signals for all instances
232  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
233  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
234  connectToObjectInstances(obj, eventMask);
235  }
236  else if ( updateMode == UAVObject::UPDATEMODE_THROTTLED )
237  {
238  // If we received a periodic update, we can change back to update on change
239  if ((eventType == EV_UPDATED_PERIODIC) || (eventType == EV_NONE)) {
240  // Set update period
241  if (eventType == EV_NONE)
242  setUpdatePeriod(obj, metadata.gcsTelemetryUpdatePeriod);
243  // Connect signals for all instances
244  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ | EV_UPDATED_PERIODIC;
245  }
246  else
247  {
248  // Otherwise, we just received an object update, so switch to periodic for the timeout period to prevent more updates
249  // Connect signals for all instances
250  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
251  }
252  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
253  connectToObjectInstances(obj, eventMask);
254  }
255  else if ( updateMode == UAVObject::UPDATEMODE_MANUAL )
256  {
257  // Set update period
258  setUpdatePeriod(obj, 0);
259  // Connect signals for all instances
260  eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ;
261  eventMask |= EV_UNPACKED; // we also need to act on remote updates (unpack events)
262  connectToObjectInstances(obj, eventMask);
263  }

372 {
373 
374  // Initiate transaction
375  if (transInfo->objRequest)
376  { // We are requesting an object from the remote end
377  utalk->sendObjectRequest(transInfo->obj, transInfo->allInstances);
378  }
379  else

439 void Telemetry::processObjectQueue()
440 {
441  if (objQueue.length() > 1)
442  qDebug() << "[telemetry.cpp] **************** Object Queue above 1 in backlog ****************";
443  // Get object information from queue (first the priority and then the regular queue)
444  ObjectQueueInfo objInfo;
445  if ( !objPriorityQueue.isEmpty() )
446  {
447  objInfo = objPriorityQueue.dequeue();
448  }
449  else if ( !objQueue.isEmpty() )
450  {
451  objInfo = objQueue.dequeue();
452  }
453  else
454  {
455  return;
456  }
457 
458  // Check if a connection has been established, only process GCSTelemetryStats updates
459  // (used to establish the connection)
460  GCSTelemetryStats::DataFields gcsStats = gcsStatsObj->getData();
461  if ( gcsStats.Status != GCSTelemetryStats::STATUS_CONNECTED )
462  {
463  objQueue.clear();
464  if ( objInfo.obj->getObjID() != GCSTelemetryStats::OBJID && objInfo.obj->getObjID() != OPLinkSettings::OBJID && objInfo.obj->getObjID() != ObjectPersistence::OBJID )
465  {
466  objInfo.obj->emitTransactionCompleted(false);
467  return;
468  }
469  }
470 
471  // Setup transaction (skip if unpack event)
472  UAVObject::Metadata metadata = objInfo.obj->getMetadata();
473  UAVObject::UpdateMode updateMode = UAVObject::GetGcsTelemetryUpdateMode(metadata);
474  if ( ( objInfo.event != EV_UNPACKED ) && ( ( objInfo.event != EV_UPDATED_PERIODIC ) || ( updateMode != UAVObject::UPDATEMODE_THROTTLED ) ) )
475  {
476  // We are either going to send an object, or are requesting one:
477  if (transMap.contains(TransactionKey(objInfo.obj, objInfo.event == EV_UPDATE_REQ))) {
478  qDebug() << "[telemetry.cpp] Warning: Got request for " << objInfo.obj->getName() << " for which a request is already in progress. Not doing it";
479  // We will not re-request it, then, we should wait for a timeout or success...
480  } else
481  {
482  UAVObject::Metadata metadata = objInfo.obj->getMetadata();
483  ObjectTransactionInfo *transInfo = new ObjectTransactionInfo(this);
484  transInfo->obj = objInfo.obj;
485  transInfo->allInstances = objInfo.allInstances;
486  transInfo->retriesRemaining = MAX_RETRIES;
487  transInfo->acked = UAVObject::GetGcsTelemetryAcked(metadata);
488  if ( objInfo.event == EV_UPDATED || objInfo.event == EV_UPDATED_MANUAL || objInfo.event == EV_UPDATED_PERIODIC )
489  {
490  transInfo->objRequest = false;
491  }
492  else if ( objInfo.event == EV_UPDATE_REQ )
493  {
494  transInfo->objRequest = true;
495  }
496  transInfo->telem = this;
497  // Insert the transaction into the transaction map.
498  TransactionKey key(objInfo.obj, transInfo->objRequest);
499  transMap.insert(key, transInfo);
500  processObjectTransaction(transInfo);
501  }
502  }
503 
504  // If this is a metaobject then make necessary telemetry updates
505  // to the connections of this object to Telemetry (this) :
506  UAVMetaObject* metaobj = dynamic_cast<UAVMetaObject*>(objInfo.obj);
507  if ( metaobj != NULL )
508  {
509  updateObject( metaobj->getParentObject(), EV_NONE );
510  }
511  else if ( updateMode != UAVObject::UPDATEMODE_THROTTLED )
512  {
513  updateObject( objInfo.obj, objInfo.event );
514  }
515 
516  // We received an "unpacked" event, check whether
517  // this is for an object we were expecting
518  if ( objInfo.event == EV_UNPACKED ) {
519  // TODO: Check here this is for a OBJ_REQ
520  if (transMap.contains(TransactionKey(objInfo.obj, true))) {
521  qDebug() << "[telemetry.cpp] EV_UNPACKED " << objInfo.obj->getName() << " inst " << objInfo.obj->getInstID();
522  transactionRequestCompleted(objInfo.obj);
523  } else
524  {
525  processObjectQueue();
526  }
527  }
528 }

534 void Telemetry::processPeriodicUpdates()
535 {
536  QMutexLocker locker(mutex);
537 
538  // Stop timer
539  updateTimer->stop();
540 
541  // Iterate through each object and update its timer, if zero then transmit object.
542  // Also calculate smallest delay to next update (will be used for setting timeToNextUpdateMs)
543  qint32 minDelay = MAX_UPDATE_PERIOD_MS;
544  qint32 elapsedMs = 0;
545  QTime time;
546  qint32 offset;
547 
548  const QVector<ObjectTimeInfo>::iterator objInfoEnd = objList.end();
549  for (QVector<ObjectTimeInfo>::iterator objinfo = objList.begin(); objinfo != objInfoEnd; ++objinfo)
550  {
551  // If object is configured for periodic updates
552  // FixMe: This is an inefficient process as it depends on polling. It would be better to have
553  // this functionality be driven through timers.
554  if (objinfo->updatePeriodMs > 0)
555  {
556  objinfo->timeToNextUpdateMs -= timeToNextUpdateMs;
557  // Check if time for the next update
558  if (objinfo->timeToNextUpdateMs <= 0)
559  {
560  // Reset timer
561  offset = (-objinfo->timeToNextUpdateMs) % objinfo->updatePeriodMs;
562  objinfo->timeToNextUpdateMs = objinfo->updatePeriodMs - offset;
563  // Send object
564  time.start();
565  processObjectUpdates(objinfo->obj, EV_UPDATED_PERIODIC, true, false);
566  elapsedMs = time.elapsed();
567  // Update timeToNextUpdateMs with the elapsed delay of sending the object;
568  timeToNextUpdateMs += elapsedMs;
569  }
570  // Update minimum delay
571  if (objinfo->timeToNextUpdateMs < minDelay)
572  {
573  minDelay = objinfo->timeToNextUpdateMs;
574  }
575  }
576  }
577 
578  // Check if delay for the next update is too short
579  if (minDelay < MIN_UPDATE_PERIOD_MS)
580  {
581  minDelay = MIN_UPDATE_PERIOD_MS;
582  }
583 
584  // Done
585  timeToNextUpdateMs = minDelay;
586 
587  // Restart timer
588  updateTimer->start(timeToNextUpdateMs);
589 }
