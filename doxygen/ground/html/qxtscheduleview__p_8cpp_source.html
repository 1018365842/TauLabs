64 QVector< QRect > QxtScheduleViewPrivate::calculateRangeGeometries(const int iStartOffset, const int iEndOffset) const
65 {
66  QVector<QRect> rects;
67 
68  if (iStartOffset < 0 || iEndOffset < 0)
69  return rects;
70 
71  if (iEndOffset < iStartOffset)
72  return rects;
73 
74  int iCurrentStartOffset = iStartOffset;
75  int iCurrentEndOffset;
76 
77  do
78  {
79  if (offsetToVisualColumn(iCurrentStartOffset) == offsetToVisualColumn(iEndOffset))
80  iCurrentEndOffset = iEndOffset;
81  else
82  iCurrentEndOffset = visualIndexToOffset(m_vHeader->count() - 1, offsetToVisualColumn(iCurrentStartOffset));
83 
84  qint32 iLeft = m_hHeader->sectionPosition(offsetToVisualColumn(iCurrentStartOffset));
85  qint32 iTop = m_vHeader->sectionPosition(offsetToVisualRow(iCurrentStartOffset));
86  qint32 iBottom = m_vHeader->sectionPosition(offsetToVisualRow(iCurrentEndOffset)) + m_vHeader->sectionSize(offsetToVisualRow(iCurrentEndOffset));
87  qint32 iRight = m_hHeader->sectionPosition(offsetToVisualColumn(iCurrentEndOffset)) + m_hHeader->sectionSize(offsetToVisualColumn(iCurrentEndOffset));
88  rects.append(QRect(iLeft + 1, iTop + 1, iRight - iLeft - 1, iBottom - iTop - 1));
89 
90  iCurrentStartOffset = visualIndexToOffset(0, offsetToVisualColumn(iCurrentEndOffset) + 1);
91 
92  }
93  while (iCurrentEndOffset < iEndOffset);
94 
95  return rects;
96 
97 }

173 QList< QLinkedList<QxtScheduleInternalItem *> > QxtScheduleViewPrivate::findConcurrentItems(const int from, const int to) const
174 {
175  QList< QLinkedList<QxtScheduleInternalItem *> > allConcurrentItems;
176 
177  QList<QxtScheduleInternalItem *> allItemsSorted = m_Items;
178 
179  if(m_Items.size() == 0)
180  return allConcurrentItems;
181 
182  qSort(allItemsSorted.begin(), allItemsSorted.end(), qxtScheduleItemLessThan);
183 
184  int startItem = 0;
185  int endItem = allItemsSorted.size() - 1;
186 
187  //find the startitem that interferes with our range
188  for (int i = 0; i < allItemsSorted.size(); i++)
189  {
190  if (i > 0)
191  {
192  if (!(allItemsSorted.at(i - 1)->visualEndTableOffset() >= allItemsSorted.at(i)->visualStartTableOffset()
193  && allItemsSorted.at(i - 1)->visualStartTableOffset() <= allItemsSorted.at(i)->visualEndTableOffset()))
194  startItem = i;
195  }
196 
197  if (allItemsSorted.at(i)->visualEndTableOffset() >= from && allItemsSorted.at(i)->visualStartTableOffset() <= to)
198  break;
199  }
200 
201  //find the last item that interferes with our range
202  for (int i = allItemsSorted.size() - 1; i >= 0 ; i--)
203  {
204  if (i < allItemsSorted.size() - 1)
205  {
206  if (!(allItemsSorted.at(i + 1)->visualEndTableOffset() >= allItemsSorted.at(i)->visualStartTableOffset()
207  && allItemsSorted.at(i + 1)->visualStartTableOffset() <= allItemsSorted.at(i)->visualEndTableOffset()))
208  endItem = i;
209  }
210 
211  if (allItemsSorted.at(i)->visualEndTableOffset() >= from && allItemsSorted.at(i)->visualStartTableOffset() <= to)
212  break;
213  }
214 
215  int startOffset = allItemsSorted.at(startItem)->visualStartTableOffset();
216  int endOffset = allItemsSorted.at(endItem)->visualEndTableOffset();
217 
218  /*now we have to populate a list with all items that interfere with our range */
219  QLinkedList<QxtScheduleInternalItem *> concurrentItems;
220  for (int iAllItemLoop = startItem; iAllItemLoop <= endItem; iAllItemLoop++)
221  {
222  int tempStartOffset = allItemsSorted.at(iAllItemLoop)->visualStartTableOffset();
223  int tempEndOffset = allItemsSorted.at(iAllItemLoop)->visualEndTableOffset();
224 
225  if (tempEndOffset >= startOffset && tempStartOffset <= endOffset)
226  {
227 
228  if (concurrentItems.size() >= 1)
229  {
230  bool bAppend = false;
231  /*check all items in the list if the current items interfers although the items are ordered by startIndex
232  *we can loose some of them if the endTime of the last Item is before the endTime of the pre last item
233  */
234 
235  for (QLinkedList<QxtScheduleInternalItem *>::iterator it = concurrentItems.begin(); it != concurrentItems.end(); ++it)
236  {
237  int lastStartOffset = (*it)->visualStartTableOffset();
238  int lastEndOffset = (*it)->visualEndTableOffset();
239 
240  if (tempEndOffset >= lastStartOffset && tempStartOffset <= lastEndOffset)
241  {
242  bAppend = true;
243  break;
244  }
245  }
246 
247  if (bAppend)
248  {
249  concurrentItems.append(allItemsSorted.at(iAllItemLoop));
250  }
251  else
252  {
253  allConcurrentItems.append(concurrentItems);
254  concurrentItems.clear();
255  concurrentItems.append(allItemsSorted.at(iAllItemLoop));
256  }
257  }
258  else
259  concurrentItems.append(allItemsSorted.at(iAllItemLoop));
260 
261  if (tempStartOffset < startOffset)
262  startOffset = tempStartOffset;
263 
264  if (tempEndOffset > endOffset)
265  endOffset = tempEndOffset;
266  }
267  }
268  if (concurrentItems.size() > 0)
269  allConcurrentItems.append(concurrentItems);
270 
271  return allConcurrentItems;
272 }

274 void QxtScheduleViewPrivate::handleItemConcurrency(const int from, const int to)
275 {
276  /*collect all items that interfere only in that range*/
277  if (from < 0 || to < 0 || m_Items.size() == 0){
278  //do a update or we may have artifacts
279  qxt_p().viewport()->update();
280  return;
281  }
282 
283  qDebug() << "handleItemConcurrency";
284 
285  if (handlesConcurrency)
286  return;
287 
288  handlesConcurrency = true;
289 
290  QList< QLinkedList<QxtScheduleInternalItem *> > allConcurrentItems = findConcurrentItems(from, to);
291 
292  /*thx to ahigerd for suggesting that algorithm*/
293  //[16:24] <ahigerd> Start with the first event. Put it in a list.
294  //[16:25] <ahigerd> Iterate until you find an event that doesn't overlap with the first event. Put it in the list. Repeat until you've reached the last event.
295  //[16:25] <ahigerd> This fills the left column optimally.
296  //[16:25] <ahigerd> Repeat the algorithm for the second column, etc., until there aren't any events left that don't have a column.
297  //[16:27] <ahigerd> This algorithm is O(n*m), where n is the number of events and m is the maximum number of overlapping events.
298 
299  QList< QList< QxtScheduleInternalItem *> >virtualTable;
300 
301  for (int iListLoop = 0; iListLoop < allConcurrentItems.size(); iListLoop++)
302  {
303  QLinkedList<QxtScheduleInternalItem *> & currentItems = allConcurrentItems[iListLoop];
304  QList< QxtScheduleInternalItem * > currentColumn;
305 
306  qDebug() << "handle overlapping for " << currentItems.size() << " Items";
307 
308  virtualTable.clear();
309 
310  //we iterate over the currect collection and remove every item that can be placed in the current column
311  //when the collection is empty we are done
312  while (currentItems.size())
313  {
314  QMutableLinkedListIterator< QxtScheduleInternalItem * > iter(currentItems);
315 
316  while (iter.hasNext())
317  {
318  iter.next();
319  //initialize the current column
320  if (currentColumn.isEmpty() || currentColumn[currentColumn.size()-1]->visualEndTableOffset() < iter.value()->visualStartTableOffset())
321  {
322  currentColumn.append(iter.value());
323  iter.remove();
324  continue;
325  }
326  }
327 
328  if (!currentColumn.isEmpty())
329  {
330  virtualTable.append(currentColumn);
331  currentColumn.clear();
332  }
333  }
334 
335  qDebug() << "Found columns" << virtualTable.size();
336 
337  //this code part resizes the item geometries
338  for (int col = 0; col < virtualTable.size(); col++)
339  {
340  for (int item = 0; item < virtualTable.at(col).size() ; item++)
341  {
342  int startVisualCol = offsetToVisualColumn(virtualTable[col][item]->visualStartTableOffset());
343  QVector<QRect> geo = virtualTable[col][item]->geometry();
344 
345  for (int rect = 0; rect < geo.size(); rect++)
346  {
347  int sectionStart = m_hHeader->sectionPosition(startVisualCol);
348  int fullWidth = m_hHeader->sectionSize(startVisualCol);
349  int oneItemWidth = fullWidth / virtualTable.size();
350  int itemWidth = oneItemWidth;
351  int itemXStart = (col * oneItemWidth) + sectionStart;
352  int overlap = oneItemWidth / 10;
353  int adjustX1 = 0;
354  int adjustX2 = 0;
355 
356  //this is very expensive.I try to check if my item can span over more than one col
357  int possibleCols = 1;
358  bool foundCollision;
359 
360  for (int tmpCol = col + 1; tmpCol < virtualTable.size(); tmpCol++)
361  {
362  foundCollision = false;
363  for (int tmpItem = 0; tmpItem < virtualTable.at(tmpCol).size() ; tmpItem++)
364  {
365  if ((virtualTable[tmpCol][tmpItem]->visualEndTableOffset() >= virtualTable[col][item]->visualStartTableOffset()
366  && virtualTable[tmpCol][tmpItem]->visualStartTableOffset() <= virtualTable[col][item]->visualEndTableOffset()))
367  {
368  foundCollision = true;
369  break;
370  }
371  }
372 
373  if (!foundCollision)
374  possibleCols++;
375  else
376  break;
377  }
378  //now lets adjust the size to get a nice overlapping of items
379  if (virtualTable.size() > 1)
380  {
381  if (col == 0)
382  adjustX2 = overlap;
383  else if (col == virtualTable.size() - 1)
384  {
385  adjustX1 = -overlap;
386  adjustX2 = overlap;
387  }
388  else
389  {
390  if (col + possibleCols == virtualTable.size())
391  adjustX2 = overlap;
392  else
393  adjustX2 = overlap * 2;
394 
395  adjustX1 = -overlap;
396  }
397  }
398 
399  // possibleCols = 1;
400  itemWidth = oneItemWidth * possibleCols;
401 
402  qDebug() << "orginial rect" << geo[rect];
403  geo[rect].setLeft(itemXStart + adjustX1);
404  geo[rect].setWidth(itemWidth + adjustX2);
405  qDebug() << "new rect" << geo[rect];
406 
407 
408  startVisualCol++;
409  }
410  virtualTable[col][item]->setGeometry(geo);
411  }
412  }
413  }
414  handlesConcurrency = false;
415  qxt_p().viewport()->update();
416 }

628 int QxtScheduleInternalItem::startTableOffset() const
629 {
630  if (parentView() && parentView()->model())
631  {
632  int startTime = data(Qxt::ItemStartTimeRole).toInt();
633  int zoomDepth = parentView()->currentZoomDepth(Qxt::Second);
634 
635  qint32 offset = startTime - parentView()->qxt_d().m_startUnixTime;
636 
637  //the start of the current item does not fit in the view
638  //so we have to align it to the nearest boundaries
639  if (offset % zoomDepth)
640  {
641  int lower = offset / zoomDepth * zoomDepth;
642  int upper = lower + zoomDepth;
643 
644  offset = (offset - lower >= upper - offset ? upper : lower);
645 
646  return parentView()->qxt_d().unixTimeToOffset(offset + parentView()->qxt_d().m_startUnixTime);
647  }
648 
649  return parentView()->qxt_d().unixTimeToOffset(startTime);
650  }
651  return -1;

684  //so we have to align it to the nearest boundaries
685  if (iNumSecs % zoomDepth)
686  {
687  int lower = iNumSecs / zoomDepth * zoomDepth;
688  int upper = lower + zoomDepth;
689 
690  return (iNumSecs - lower >= upper - iNumSecs ? (upper / zoomDepth) : (lower / zoomDepth));
691 
692  }
693  return (iNumSecs / zoomDepth);
694  }
695  return -1;
696 }
