59 void AeroSimRCSimulator::transmitUpdate()
60 {
61  // read actuator output
62  ActuatorCommand::DataFields actCmdData;
63  actCmdData = actCommand->getData();
64  float channels[10];
65  for (int i = 0; i < 10; ++i) {
66  qint16 ch = actCmdData.Channel[i];
67  float out = -1.0;
68  if (ch >= 1000 && ch <= 2000) {
69  ch -= 1000;
70  out = ((float)ch / 500.0) - 1.0;
71  }
72  channels[i] = out;
73  }
74 
75  // read flight status
76  FlightStatus::DataFields flightData;
77  flightData = flightStatus->getData();
78  quint8 armed;
79  quint8 mode;
80  armed = flightData.Armed;
81  mode = flightData.FlightMode;
82 
83  QByteArray data;
84  // 50 - current size of values, 4(quint32) + 10*4(float) + 2(quint8) + 4(quint32)
85  data.resize(50);
86  QDataStream stream(&data, QIODevice::WriteOnly);
87  stream.setFloatingPointPrecision(QDataStream::SinglePrecision);
88  stream << quint32(0x52434D44); // magic header, "RCMD"
89  for (int i = 0; i < 10; ++i)
90  stream << channels[i]; // channels
91  stream << armed << mode; // flight status
92  stream << udpCounterASrecv; // packet counter
93 
94  if (outSocket->writeDatagram(data, QHostAddress(settings.remoteAddress), settings.outPort) == -1) {
95  qDebug() << "write failed: " << outSocket->errorString();
96  }
97 
98 #ifdef DBG_TIMERS
99  static int cntTX = 0;
100  if (cntTX >= 100) {
101  qDebug() << "TX=" << 1000.0 * 100 / timeTX.restart();
102  cntTX = 0;
103  } else {
104  ++cntTX;
105  }
106 #endif
107 }

237 void AeroSimRCSimulator::asMatrix2Quat(const QMatrix4x4 &m, QQuaternion &q)
238 {
239  qreal w, x, y, z;
240 
241  // w always >= 0
242  w = qSqrt(qMax(0.0, 1.0 + m(0, 0) + m(1, 1) + m(2, 2))) / 2.0;
243  x = qSqrt(qMax(0.0, 1.0 + m(0, 0) - m(1, 1) - m(2, 2))) / 2.0;
244  y = qSqrt(qMax(0.0, 1.0 - m(0, 0) + m(1, 1) - m(2, 2))) / 2.0;
245  z = qSqrt(qMax(0.0, 1.0 - m(0, 0) - m(1, 1) + m(2, 2))) / 2.0;
246 
247  x = copysign(x, (m(1, 2) - m(2, 1)));
248  y = copysign(y, (m(2, 0) - m(0, 2)));
249  z = copysign(z, (m(0, 1) - m(1, 0)));
250 
251  q.setScalar(w);
252  q.setX(x);
253  q.setY(y);
254  q.setZ(z);
255 }

257 void AeroSimRCSimulator::asMatrix2RPY(const QMatrix4x4 &m, QVector3D &rpy)
258 {
259  qreal roll, pitch, yaw;
260 
261  if (qFabs(m(0, 2)) > 0.998) {
262  // ~86.3°, gimbal lock
263  roll = 0.0;
264  pitch = copysign(M_PI_2, -m(0, 2));
265  yaw = qAtan2(-m(1, 0), m(1, 1));
266  } else {
267  roll = qAtan2(m(1, 2), m(2, 2));
268  pitch = qAsin(-m(0, 2));
269  yaw = qAtan2(m(0, 1), m(0, 0));
270  }
271 
272  rpy.setX(roll * RAD2DEG);
273  rpy.setY(pitch * RAD2DEG);
274  rpy.setZ(yaw * RAD2DEG);
275 }
