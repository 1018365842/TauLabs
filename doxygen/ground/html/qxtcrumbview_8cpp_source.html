250 void QxtCrumbView::enterTree(const QModelIndex& index) {
251  if(!model()) return;
252  if(!model()->hasChildren(index)) {
253  emit activated(index);
254  return;
255  }
256  if(index == qxt_d().view->rootIndex()) {
257  // do nothing but reload the view
258  } else if(index.parent() != qxt_d().view->rootIndex()) {
259  foreach(QxtCrumbViewButton* b, qxt_d().buttons) b->deleteLater();
260  qxt_d().crumbs.clear();
261  qxt_d().buttons.clear();
262  qxt_d().addCrumb(QModelIndex());
263  QList<QModelIndex> chain;
264  QModelIndex pos = index;
265  while(pos.isValid()) {
266  chain.append(pos);
267  pos = pos.parent();
268  }
269  while(!chain.isEmpty()) {
270  qxt_d().addCrumb(chain.last());
271  chain.removeLast();
272  }
273  } else {
274  qxt_d().addCrumb(index);
275  }
276  qxt_d().view->setRootIndex(index);
277 }

279 void QxtCrumbView::back() {
280  if(qxt_d().crumbs.count() <= 1) return;
281  qxt_d().buttons.last()->deleteLater();
282  qxt_d().buttons.removeLast();
283  qxt_d().crumbs.removeLast();
284  qxt_d().view->setRootIndex(qxt_d().crumbs.last());
285 }
