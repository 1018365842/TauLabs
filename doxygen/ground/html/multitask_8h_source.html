59 template <typename Class, typename R>
60 class MultiTask : public MultiTaskBase
61 {
62 public:
63  MultiTask(void (Class::*fn)(QFutureInterface<R> &), const QList<Class *> &objects)
64  : fn(fn),
65  objects(objects)
66  {
67  maxProgress = 100*objects.size();
68  }
69 
70  QFuture<R> future()
71  {
72  futureInterface.reportStarted();
73  return futureInterface.future();
74  }
75 
76  void run()
77  {
78  QThreadPool::globalInstance()->releaseThread();
79  futureInterface.setProgressRange(0, maxProgress);
80  foreach (Class *object, objects) {
81  QFutureWatcher<R> *watcher = new QFutureWatcher<R>();
82  watchers.insert(object, watcher);
83  finished.insert(watcher, false);
84  connect(watcher, SIGNAL(finished()), this, SLOT(setFinished()));
85  connect(watcher, SIGNAL(progressRangeChanged(int,int)), this, SLOT(setProgressRange(int,int)));
86  connect(watcher, SIGNAL(progressValueChanged(int)), this, SLOT(setProgressValue(int)));
87  connect(watcher, SIGNAL(progressTextChanged(QString)), this, SLOT(setProgressText(QString)));
88  watcher->setFuture(QtConcurrent::run(fn, object));
89  }
90  selfWatcher = new QFutureWatcher<R>();
91  connect(selfWatcher, SIGNAL(canceled()), this, SLOT(cancelSelf()));
92  selfWatcher->setFuture(futureInterface.future());
93  loop = new QEventLoop;
94  loop->exec();
95  futureInterface.reportFinished();
96  QThreadPool::globalInstance()->reserveThread();
97  qDeleteAll(watchers.values());
98  delete selfWatcher;
99  delete loop;
100  }
101 protected:
102  void cancelSelf()
103  {
104  foreach (QFutureWatcher<R> *watcher, watchers)
105  watcher->future().cancel();
106  }
107 
108  void setFinished()
109  {
110  updateProgress();
111  QFutureWatcher<R> *watcher = static_cast<QFutureWatcher<R> *>(sender());
112  if (finished.contains(watcher))
113  finished[watcher] = true;
114  bool allFinished = true;
115  const QList<bool> finishedValues = finished.values();
116  foreach (bool isFinished, finishedValues) {
117  if (!isFinished) {
118  allFinished = false;
119  break;
120  }
121  }
122  if (allFinished)
123  loop->quit();
124  }

126  void setProgressRange(int min, int max)
127  {
128  Q_UNUSED(min)
129  Q_UNUSED(max)
130  updateProgress();
131  }

133  void setProgressValue(int value)
134  {
135  Q_UNUSED(value)

137  }

139  void setProgressText(QString value)
140  {
141  Q_UNUSED(value)
142  updateProgressText();
143  }
144 private:
145  void updateProgress()
146  {
147  int progressSum = 0;
148  const QList<QFutureWatcher<R> *> watchersValues = watchers.values();
149  foreach (QFutureWatcher<R> *watcher, watchersValues) {
150  if (watcher->progressMinimum() == watcher->progressMaximum()) {
151  if (watcher->future().isFinished() && !watcher->future().isCanceled())
152  progressSum += 100;
153  } else {
154  progressSum += 100*(watcher->progressValue()-watcher->progressMinimum())/(watcher->progressMaximum()-watcher->progressMinimum());
155  }
156  }
157  futureInterface.setProgressValue(progressSum);
158  }
159 
160  void updateProgressText()
161  {
162  QString text;
163  const QList<QFutureWatcher<R> *> watchersValues = watchers.values();
164  foreach (QFutureWatcher<R> *watcher, watchersValues) {
165  if (!watcher->progressText().isEmpty())
166  text += watcher->progressText() + "\n";
167  }
168  text = text.trimmed();
169  futureInterface.setProgressValueAndText(futureInterface.progressValue(), text);
170  }

183 template <typename Class, typename T>
184 QFuture<T> run(void (Class::*fn)(QFutureInterface<T> &), const QList<Class *> &objects, int priority = 0)
185 {
186  MultiTask<Class, T> *task = new MultiTask<Class, T>(fn, objects);
187  QFuture<T> future = task->future();
188  QThreadPool::globalInstance()->start(task, priority);
