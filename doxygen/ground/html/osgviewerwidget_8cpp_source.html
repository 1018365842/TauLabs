223 void OsgViewerWidget::updateAirframe(UAVObject *obj)
224 {
225  if (obj->getObjID() == SystemSettings::OBJID && rotateModelNED != NULL) {
226  SystemSettings *systemSettingsObj = dynamic_cast<SystemSettings *>(obj);
227  if (systemSettingsObj == NULL)
228  return;
229  SystemSettings::DataFields systemSettings = systemSettingsObj->getData();
230 
231  osg::Node *uav;
232 
233  QString modelFile;
234  QString textureFile;
235  switch(systemSettings.AirframeType) {
236  case SystemSettings::AIRFRAMETYPE_FIXEDWING:
237  case SystemSettings::AIRFRAMETYPE_FIXEDWINGELEVON:
238  case SystemSettings::AIRFRAMETYPE_FIXEDWINGVTAIL:
239  modelFile = QDir::temp().filePath("model.3ds");
240  QFile::copy( AIRPLANE_MODEL, modelFile );
241  qDebug() << "Copied to " << modelFile;
242 
243  textureFile = QDir::temp().filePath("texture.jpg");
244  QFile::copy( AIRPLANE_TEXTURE, textureFile );
245  qDebug() << "Copied to " << textureFile;
246  break;
247  default:
248  modelFile = QDir::temp().filePath("model.3ds");
249  QFile::copy( MULTIROTOR_MODEL, modelFile );
250  qDebug() << "Copied to " << modelFile;
251 
252  textureFile = QDir::temp().filePath("TEXTURE.PNG");
253  QFile::copy( MULTIROTOR_TEXTURE, textureFile );
254  qDebug() << "Copied to " << textureFile;
255  }
256 
257  uav = osgDB::readNodeFile(modelFile.toUtf8().constData());
258 
259  QFile::remove(modelFile);
260  QFile::remove(modelFile);
261 
262  if (uav) {
263  rotateModelNED->removeChild(0,1);
264  rotateModelNED->addChild(uav);
265  } else {
266  qDebug() << "Unable to load the model file";
267  }
268 
269  }

272 osg::Node* OsgViewerWidget::createAirplane()
273 {
274  osg::Group* model = new osg::Group;
275 
276  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
277  Q_ASSERT(pm);
278  UAVObjectManager * objMngr = pm->getObject<UAVObjectManager>();
279  Q_ASSERT(objMngr);
280 
281  SystemSettings *systemSettingsObj = SystemSettings::GetInstance(objMngr);
282 
283  uavAttitudeAndScale = new osg::MatrixTransform();
284  uavAttitudeAndScale->setMatrix(osg::Matrixd::scale(0.2e0,0.2e0,0.2e0));
285 
286  // Apply a rotation so model is NED before any other rotations
287  rotateModelNED = new osg::MatrixTransform();
288  rotateModelNED->setMatrix(osg::Matrixd::scale(0.05e0,0.05e0,0.05e0) * osg::Matrixd::rotate(M_PI, osg::Vec3d(0,0,1)));
289 
290  uavAttitudeAndScale->addChild( rotateModelNED );
291 
292  model->addChild(uavAttitudeAndScale);
293 
294  // Attach the specific airframe
295  updateAirframe(systemSettingsObj);
296 
297  return model;
298 }

300 void OsgViewerWidget::paintEvent( QPaintEvent* event )
301 {
302  Q_UNUSED(event);
303  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
304  UAVObjectManager * objMngr = pm->getObject<UAVObjectManager>();
305 
306  PositionActual *positionActualObj = PositionActual::GetInstance(objMngr);
307  PositionActual::DataFields positionActual = positionActualObj->getData();
308  double NED[3] = {positionActual.North, positionActual.East, positionActual.Down};
309 
310  bool positionActualUpdate = true;
311  if(positionActualUpdate) {
312  HomeLocation *homeLocationObj = HomeLocation::GetInstance(objMngr);
313  HomeLocation::DataFields homeLocation = homeLocationObj->getData();
314  double homeLLA[3] = {homeLocation.Latitude / 10.0e6, homeLocation.Longitude / 10.0e6, homeLocation.Altitude};
315 
316  double LLA[3];
317  CoordinateConversions().NED2LLA_HomeLLA(homeLLA, NED, LLA);
318  uavPos->getLocator()->setPosition( osg::Vec3d(LLA[1], LLA[0], LLA[2]) ); // Note this takes longtitude first
319  } else {
320  GPSPosition *gpsPosObj = GPSPosition::GetInstance(objMngr);
321  GPSPosition::DataFields gpsPos = gpsPosObj->getData();
322  uavPos->getLocator()->setPosition( osg::Vec3d(gpsPos.Longitude / 10.0e6, gpsPos.Latitude / 10.0e6, gpsPos.Altitude));
323  }
324 
325  // Set the attitude (reverse the attitude)
326  AttitudeActual *attitudeActualObj = AttitudeActual::GetInstance(objMngr);
327  AttitudeActual::DataFields attitudeActual = attitudeActualObj->getData();
328  osg::Quat quat(attitudeActual.q2, attitudeActual.q3, attitudeActual.q4, attitudeActual.q1);
329 
330  // Have to rotate the axes from OP NED frame to OSG frame (X east, Y north, Z down)
331  double angle;
332  osg::Vec3d axis;
333  quat.getRotate(angle,axis);
334  quat.makeRotate(angle, osg::Vec3d(axis[1],axis[0],-axis[2]));
335  osg::Matrixd rot = osg::Matrixd::rotate(quat);
336 
337  uavAttitudeAndScale->setMatrix(rot);
338 
339  frame();
340 }
