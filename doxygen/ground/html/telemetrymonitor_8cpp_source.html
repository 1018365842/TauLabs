75 void TelemetryMonitor::startRetrievingObjects()
76 {
77  // Clear object queue
78  queue.clear();
79  // Get all objects, add metaobjects, settings and data objects with OnChange update mode to the queue
80  QVector< QVector<UAVObject*> > objs = objMngr->getObjects();
81  const int objsSize = objs.size();
82  for (int n = 0; n < objsSize; ++n)
83  {
84  UAVObject* obj = objs[n][0];
85  UAVMetaObject* mobj = dynamic_cast<UAVMetaObject*>(obj);
86  UAVDataObject* dobj = dynamic_cast<UAVDataObject*>(obj);
87  UAVObject::Metadata mdata = obj->getMetadata();
88  if ( mobj != NULL )
89  {
90  queue.enqueue(obj);
91  }
92  else if ( dobj != NULL )
93  {
94  if ( dobj->isSettings() )
95  {
96  queue.enqueue(obj);
97  }
98  else
99  {
100  if ( UAVObject::GetFlightTelemetryUpdateMode(mdata) == UAVObject::UPDATEMODE_ONCHANGE )
101  {
102  queue.enqueue(obj);
103  }
104  }
105  }
106  }
107  // Start retrieving
108  qxtLog->debug(tr("Starting to retrieve meta and settings objects from the autopilot (%1 objects)")
109  .arg( queue.length()) );
110  retrieveNextObject();
111 }

125 void TelemetryMonitor::retrieveNextObject()
126 {
127  // If queue is empty return
128  if ( queue.isEmpty() )
129  {
130  qxtLog->debug("Object retrieval completed");
131  emit connected();
132  return;
133  }
134  // Get next object from the queue
135  UAVObject* obj = queue.dequeue();
136  //qxtLog->trace( tr("Retrieving object: %1").arg(obj->getName()) );
137  // Connect to object
138  connect(obj, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(transactionCompleted(UAVObject*,bool)));
139  // Request update
140  obj->requestUpdate();
141  objPending = obj;
142 }

149  Q_UNUSED(success);

187 void TelemetryMonitor::processStatsUpdates()
188 {
189  QMutexLocker locker(mutex);
190 
191  // Get telemetry stats
192  GCSTelemetryStats::DataFields gcsStats = gcsStatsObj->getData();
193  FlightTelemetryStats::DataFields flightStats = flightStatsObj->getData();
194  Telemetry::TelemetryStats telStats = tel->getStats();
195  tel->resetStats();
196 
197  // Update stats object
198  gcsStats.RxDataRate = (float)telStats.rxBytes / ((float)statsTimer->interval()/1000.0);
199  gcsStats.TxDataRate = (float)telStats.txBytes / ((float)statsTimer->interval()/1000.0);
200  gcsStats.RxFailures += telStats.rxErrors;
201  gcsStats.TxFailures += telStats.txErrors;
202  gcsStats.TxRetries += telStats.txRetries;
203 
204  // Check for a connection timeout
205  bool connectionTimeout;
206  if ( telStats.rxObjects > 0 )
207  {
208  connectionTimer->start();
209  }
210  if ( connectionTimer->elapsed() > CONNECTION_TIMEOUT_MS )
211  {
212  connectionTimeout = true;
213  }
214  else
215  {
216  connectionTimeout = false;
217  }
218 
219  // Update connection state
220  int oldStatus = gcsStats.Status;
221  if ( gcsStats.Status == GCSTelemetryStats::STATUS_DISCONNECTED )
222  {
223  // Request connection
224  gcsStats.Status = GCSTelemetryStats::STATUS_HANDSHAKEREQ;
225  }
226  else if ( gcsStats.Status == GCSTelemetryStats::STATUS_HANDSHAKEREQ )
227  {
228  // Check for connection acknowledge
229  if ( flightStats.Status == FlightTelemetryStats::STATUS_HANDSHAKEACK )
230  {
231  gcsStats.Status = GCSTelemetryStats::STATUS_CONNECTED;
232  }
233  }
234  else if ( gcsStats.Status == GCSTelemetryStats::STATUS_CONNECTED )
235  {
236  // Check if the connection is still active and the the autopilot is still connected
237  if (flightStats.Status == FlightTelemetryStats::STATUS_DISCONNECTED || connectionTimeout)
238  {
239  gcsStats.Status = GCSTelemetryStats::STATUS_DISCONNECTED;
240  }
241  }
242 
243  emit telemetryUpdated((double)gcsStats.TxDataRate, (double)gcsStats.RxDataRate);
244 
245  // Set data
246  gcsStatsObj->setData(gcsStats);
247 
248  // Force telemetry update if not yet connected
249  if ( gcsStats.Status != GCSTelemetryStats::STATUS_CONNECTED ||
250  flightStats.Status != FlightTelemetryStats::STATUS_CONNECTED )
251  {
252  gcsStatsObj->updated();
253  }
254 
255  // Act on new connections or disconnections
256  if (gcsStats.Status == GCSTelemetryStats::STATUS_CONNECTED && gcsStats.Status != oldStatus)
257  {
258  statsTimer->setInterval(STATS_UPDATE_PERIOD_MS);
259  qxtLog->info("Connection with the autopilot established");
260  startRetrievingObjects();
261  }
262  if (gcsStats.Status == GCSTelemetryStats::STATUS_DISCONNECTED && gcsStats.Status != oldStatus)
263  {
264  statsTimer->setInterval(STATS_CONNECT_PERIOD_MS);
265  qxtLog->info("Connection with the autopilot lost");
266  qxtLog->info("Trying to connect to the autopilot");
267  emit disconnected();
268  }
