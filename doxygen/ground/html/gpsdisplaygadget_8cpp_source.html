64  GpsDisplayGadgetConfiguration *gpsDisplayConfig = qobject_cast< GpsDisplayGadgetConfiguration*>(config);
65 
66  if (gpsDisplayConfig->connectionMode() == "Serial") {
67  PortSettings portsettings;
68  portsettings.BaudRate=gpsDisplayConfig->speed();
69  portsettings.DataBits=gpsDisplayConfig->dataBits();
70  portsettings.FlowControl=gpsDisplayConfig->flow();
71  portsettings.Parity=gpsDisplayConfig->parity();
72  portsettings.StopBits=gpsDisplayConfig->stopBits();
73  portsettings.Timeout_Millisec=gpsDisplayConfig->timeOut();
74 
75  // In case we find no port, buttons disabled
76  m_widget->connectButton->setEnabled(false);
77  m_widget->disconnectButton->setEnabled(false);
78 
79  QList<QextPortInfo> ports = QextSerialEnumerator::getPorts();
80  foreach( QextPortInfo nport, ports ) {
81  if(nport.friendName == gpsDisplayConfig->port())
82  {
83  qDebug() << "Using Serial parser";
84  parser = new NMEAParser();
85 
86 #ifdef Q_OS_WIN
87  port=new QextSerialPort(nport.portName,portsettings,QextSerialPort::EventDriven);
88 #else
89  port=new QextSerialPort(nport.physName,portsettings,QextSerialPort::EventDriven);
90 #endif
91  m_widget->connectButton->setEnabled(true);
92  m_widget->disconnectButton->setEnabled(false);
93  m_widget->connectButton->setHidden(false);
94  m_widget->disconnectButton->setHidden(false);
95 
96  connect(port, SIGNAL(readyRead()), this, SLOT(onDataAvailable()));
97  }
98  }
99  m_widget->dataStreamGroupBox->setHidden(false);
100  } else if (gpsDisplayConfig->connectionMode() == "Telemetry") {
101  qDebug() << "Using Telemetry parser";
102  parser = new TelemetryParser();
103  m_widget->disconnectButton->setHidden(true);
104  m_widget->connectButton->setHidden(true);
105  m_widget->dataStreamGroupBox->setHidden(true);
106  } else if (gpsDisplayConfig->connectionMode() == "Network") {
107  // Not implemented for now...

110  m_widget->dataStreamGroupBox->setHidden(false);

153 void GpsDisplayGadget::onDataAvailable() {
154  int avail = port->bytesAvailable();
155  if( avail > 0 ) {
156  QByteArray serialData;
157  serialData.resize(avail);
158  int bytesRead = port->read(serialData.data(), serialData.size());
159  if( bytesRead > 0 ) {
160  processNewSerialData(serialData);
161  }
162  }
163 }

165 void GpsDisplayGadget::processNewSerialData(QByteArray serialData) {
