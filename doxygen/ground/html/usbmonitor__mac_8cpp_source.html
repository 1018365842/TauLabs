46 USBMonitor::USBMonitor(QObject *parent): QThread(parent) {
47  hid_manager=NULL;
48  IOReturn ret;
49 
50  m_instance = this;
51 
52  listMutex = new QMutex();
53  knowndevices.clear();
54 
55  hid_manager = IOHIDManagerCreate(kCFAllocatorDefault, kIOHIDOptionsTypeNone);
56  if (hid_manager == NULL || CFGetTypeID(hid_manager) != IOHIDManagerGetTypeID()) {
57  if (hid_manager) CFRelease(hid_manager);
58  Q_ASSERT(0);
59  }
60 
61  // No matching filter
62  IOHIDManagerSetDeviceMatching(hid_manager, NULL);
63 
64  // set up a callbacks for device attach & detach
65  IOHIDManagerScheduleWithRunLoop(hid_manager, CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
66  IOHIDManagerRegisterDeviceMatchingCallback(hid_manager, attach_callback, NULL);
67  IOHIDManagerRegisterDeviceRemovalCallback(hid_manager, detach_callback, NULL);
68  ret = IOHIDManagerOpen(hid_manager, kIOHIDOptionsTypeNone);
69  if (ret != kIOReturnSuccess) {
70  IOHIDManagerUnscheduleFromRunLoop(hid_manager,
71  CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
72  CFRelease(hid_manager);
73  return;
74  }
75 
76  start();
77 }

132 void USBMonitor::attach_callback(void *context, IOReturn r, void *hid_mgr, IOHIDDeviceRef dev)
133 {
134  Q_UNUSED(context);
135  Q_UNUSED(r);
136  Q_UNUSED(hid_mgr);
137 
138  bool got_properties = true;
139 
140  USBPortInfo deviceInfo;
141 
142  deviceInfo.dev_handle = dev;
143 
144  qDebug() << "USBMonitor: Device attached event";
145 
146  // Populate the device info structure
147  got_properties &= HID_GetIntProperty(dev, CFSTR( kIOHIDVendorIDKey ), &deviceInfo.vendorID);
148  got_properties &= HID_GetIntProperty(dev, CFSTR( kIOHIDProductIDKey ), &deviceInfo.productID);
149  got_properties &= HID_GetIntProperty(dev, CFSTR( kIOHIDVersionNumberKey ), &deviceInfo.bcdDevice);
150  got_properties &= HID_GetStrProperty(dev, CFSTR( kIOHIDSerialNumberKey ), deviceInfo.serialNumber);
151  got_properties &= HID_GetStrProperty(dev, CFSTR( kIOHIDProductKey ), deviceInfo.product);
152  got_properties &= HID_GetStrProperty(dev, CFSTR( kIOHIDManufacturerKey ), deviceInfo.manufacturer);
153  // TOOD: Eventually want to take array of usages if devices start needing that
154  got_properties &= HID_GetIntProperty(dev, CFSTR( kIOHIDPrimaryUsageKey ), &deviceInfo.Usage);
155  got_properties &= HID_GetIntProperty(dev, CFSTR( kIOHIDPrimaryUsagePageKey ), &deviceInfo.UsagePage);
156 
157  // Currently only enumerating objects that have the complete list of properties
158  if(got_properties) {
159  qDebug() << "USBMonitor: Adding device";
160  instance()->addDevice(deviceInfo);
161  }
162 }

184 QList<USBPortInfo> USBMonitor::availableDevices(int vid, int pid, int bcdDeviceMSB, int bcdDeviceLSB)
185 {
186  QList<USBPortInfo> allPorts = availableDevices();
187  QList<USBPortInfo> thePortsWeWant;
