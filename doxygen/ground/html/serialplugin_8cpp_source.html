55 void SerialEnumerationThread::run()
56 {
57  QList <Core::IDevice*> devices = m_serial->availableDevices();
58 
59  while(m_running)
60  {
61  if(!m_serial->deviceOpened())
62  {
63  QList <Core::IDevice*> newDev = m_serial->availableDevices();
64  // Note: if(devices != newDev) does not work here (QList of pointers)...
65  bool different = false;
66  if (newDev.length()!= devices.length()) {
67  different = true;
68  } else {
69  for (int i= 0; i< newDev.length(); i++) {
70  Core::IDevice* oldd = devices.at(i);
71  Core::IDevice* newd = newDev.at(i);
72  different |= !(oldd->equals(newd));
73  }
74  }
75  if (different) {
76  devices = newDev;
77  emit enumerationChanged();
78  }
79  }
80  msleep(2000); //update available devices every two seconds (doesn't need more)
81  }
82 }

130 QList <IDevice *> SerialConnection::availableDevices()
131 {
132  QList <Core::IDevice*> list;
133 
134  if (enablePolling) {
135  QList<QextPortInfo> ports = QextSerialEnumerator::getPorts();
136 
137  //sort the list by port number (nice idea from PT_Dreamer :))
138  qSort(ports.begin(), ports.end(),sortPorts);
139  foreach( QextPortInfo port, ports ) {
140  SerialDevice* d = new SerialDevice();
141  d->setDisplayName(port.friendName);
142  d->setName(port.physName);
143  list.append(d);
144  }
145  }
146 
147  return list;
148 }

150 QIODevice *SerialConnection::openDevice(IDevice *deviceName)
151 {
152  if (serialHandle){
153  closeDevice(deviceName->getName());
154  }
155  QList<QextPortInfo> ports = QextSerialEnumerator::getPorts();
156  foreach( QextPortInfo port, ports ) {
157  if(port.physName == deviceName->getName())
158  {
159  //we need to handle port settings here...
160  PortSettings set;
161  set.BaudRate = stringToBaud(m_config->speed());
162  qDebug()<<"Serial telemetry running at "<<m_config->speed();
163  set.DataBits = DATA_8;
164  set.Parity = PAR_NONE;
165  set.StopBits = STOP_1;
166  set.FlowControl = FLOW_OFF;
167  set.Timeout_Millisec = 500;
168 #ifdef Q_OS_WIN
169  serialHandle = new QextSerialPort(port.portName, set);
170 #else
171  serialHandle = new QextSerialPort(port.physName, set);
172 #endif
173  m_deviceOpened = true;
174  return serialHandle;
175  }
176  }
177  return NULL;
178 }
