85 void SystemHealthGadgetWidget::updateAlarms(UAVObject* systemAlarm)
86 {
87  // This code does not know anything about alarms beforehand, and
88  // I found no efficient way to locate items inside the scene by
89  // name, so it's just as simple to reset the scene:
90  // And add the one with the right name.
91  QGraphicsScene *m_scene = scene();
92  foreach ( QGraphicsItem* item ,background->childItems()){
93  m_scene->removeItem(item);
94  delete item; // removeItem does _not_ delete the item.
95  }
96 
97  UAVObjectField *field = systemAlarm->getField("Alarm");
98  Q_ASSERT(field);
99  if (field == NULL)
100  return;
101 
102  for (uint i = 0; i < field->getNumElements(); ++i) {
103  QString element = field->getElementNames()[i];
104  QString value = field->getValue(i).toString();
105  if (m_renderer->elementExists(element)) {
106  QMatrix blockMatrix = m_renderer->matrixForElement(element);
107  qreal startX = blockMatrix.mapRect(m_renderer->boundsOnElement(element)).x();
108  qreal startY = blockMatrix.mapRect(m_renderer->boundsOnElement(element)).y();
109  QString element2 = element + "-" + value;
110  if (m_renderer->elementExists(element2)) {
111  QGraphicsSvgItem *ind = new QGraphicsSvgItem();
112  ind->setSharedRenderer(m_renderer);
113  ind->setElementId(element2);
114  ind->setParentItem(background);
115  QTransform matrix;
116  matrix.translate(startX,startY);
117  ind->setTransform(matrix,false);
118  } else {
119  if (value.compare("Uninitialised")!=0)qDebug() << "Warning: element " << element2 << " not found in SVG.";
120  }
121  } else {
122  qDebug() << "Warning: Element " << element << " not found in SVG.";
123  }
124  }
125 }

133 void SystemHealthGadgetWidget::setSystemFile(QString dfn)
134 {
135  setBackgroundBrush(QBrush(Utils::StyleHelper::baseColor()));
136  if (QFile::exists(dfn)) {
137  m_renderer->load(dfn);
138  if(m_renderer->isValid()) {
139  fgenabled = false;
140  background->setSharedRenderer(m_renderer);
141  background->setElementId("background");
142 
143  if (m_renderer->elementExists("foreground")) {
144  foreground->setSharedRenderer(m_renderer);
145  foreground->setElementId("foreground");
146  foreground->setZValue(99);
147  fgenabled = true;
148  }
149  if (m_renderer->elementExists("nolink")) {
150  nolink->setSharedRenderer(m_renderer);
151  nolink->setElementId("nolink");
152  nolink->setZValue(100);
153  }
154 
155  QGraphicsScene *l_scene = scene();
156  l_scene->setSceneRect(background->boundingRect());
157  fitInView(background, Qt::KeepAspectRatio );
158 
159  // Check whether the autopilot is connected already, by the way:
160  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
161  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
162  TelemetryManager* telMngr = pm->getObject<TelemetryManager>();
163  if (telMngr->isConnected()) {
164  onAutopilotConnect();
165  SystemAlarms* obj = dynamic_cast<SystemAlarms*>(objManager->getObject(QString("SystemAlarms")));
166  updateAlarms(obj);
167  }
168  }
169  }
170  else
171  { qDebug() <<"SystemHealthGadget: no file"; }
172 }

203 void SystemHealthGadgetWidget::mousePressEvent ( QMouseEvent * event )
204 {
205  QGraphicsScene *graphicsScene = scene();
206  if(graphicsScene){
207  QPoint point = event->pos();
208  bool haveAlarmItem = false;
209  foreach(QGraphicsItem* sceneItem, items(point)){
210  QGraphicsSvgItem *clickedItem = dynamic_cast<QGraphicsSvgItem*>(sceneItem);
211 
212  if(clickedItem){
213  if((clickedItem != foreground) && (clickedItem != background)){
214  // Clicked an actual alarm. We need to set haveAlarmItem to true
215  // as two of the items in this loop will always be foreground and
216  // background. Without this flag, at some point in the loop we
217  // would always call showAllAlarmDescriptions...
218  haveAlarmItem = true;
219  QString itemId = clickedItem->elementId();
220  if(itemId.contains("OK")){
221  // No alarm set for this item
222  showAlarmDescriptionForItemId("AlarmOK", event->globalPos());
223  }else{
224  // Warning, error or critical alarm
225  showAlarmDescriptionForItemId(itemId, event->globalPos());
226  }
227  }
228  }
229  }
230  if(!haveAlarmItem){
231  // Clicked foreground or background
232  showAllAlarmDescriptions(event->globalPos());
233  }
234  }
235 }

245 void SystemHealthGadgetWidget::showAllAlarmDescriptions(const QPoint& location){
246  QGraphicsScene *graphicsScene = scene();
247  if(graphicsScene){
248  QString alarmsText;
249 
250  // Loop through all items in the scene looking for svg items that represent alarms
251  foreach(QGraphicsItem* curItem, graphicsScene->items()){
252  QGraphicsSvgItem* curSvgItem = dynamic_cast<QGraphicsSvgItem*>(curItem);
253  if(curSvgItem && (curSvgItem != foreground) && (curSvgItem != background)){
254  QString elementId = curSvgItem->elementId();
255  if(!elementId.contains("OK")){
256  // Found an alarm, get its corresponding alarm html file contents
257  // and append to the cumulative string for all alarms.
258  QFile alarmDescription(":/systemhealth/html/" + elementId + ".html");
259  if(alarmDescription.open(QIODevice::ReadOnly | QIODevice::Text)){
260  QTextStream textStream(&alarmDescription);
261  alarmsText.append(textStream.readAll());
262  }
263  }
264  }
265  }
266 
267  // Show alarms text if we have any
268  if(alarmsText.length() > 0){
269  QWhatsThis::showText(location, alarmsText);
270  }
