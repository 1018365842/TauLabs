35 QStringList AbstractProcess::fixWinEnvironment(const QStringList &env)
36 {
37  QStringList envStrings = env;
38  // add PATH if necessary (for DLL loading)
39  if (envStrings.filter(QRegExp(QLatin1String("^PATH="),Qt::CaseInsensitive)).isEmpty()) {
40  QByteArray path = qgetenv("PATH");
41  if (!path.isEmpty())
42  envStrings.prepend(QString(QLatin1String("PATH=%1")).arg(QString::fromLocal8Bit(path)));
43  }
44  // add systemroot if needed
45  if (envStrings.filter(QRegExp(QLatin1String("^SystemRoot="),Qt::CaseInsensitive)).isEmpty()) {
46  QByteArray systemRoot = qgetenv("SystemRoot");
47  if (!systemRoot.isEmpty())
48  envStrings.prepend(QString(QLatin1String("SystemRoot=%1")).arg(QString::fromLocal8Bit(systemRoot)));
49  }
50  return envStrings;
51 }

53 QString AbstractProcess::createWinCommandline(const QString &program, const QStringList &args)
54 {
55  const QChar doubleQuote = QLatin1Char('"');
56  const QChar blank = QLatin1Char(' ');
57  const QChar backSlash = QLatin1Char('\\');
58 
59  QString programName = program;
60  if (!programName.startsWith(doubleQuote) && !programName.endsWith(doubleQuote) && programName.contains(blank)) {
61  programName.insert(0, doubleQuote);
62  programName.append(doubleQuote);
63  }
64  // add the prgram as the first arrg ... it works better
65  programName.replace(QLatin1Char('/'), backSlash);
66  QString cmdLine = programName;
67  if (args.empty())
68  return cmdLine;
69 
70  cmdLine += blank;
71  for (int i = 0; i < args.size(); ++i) {
72  QString tmp = args.at(i);
73  // in the case of \" already being in the string the \ must also be escaped
74  tmp.replace(QLatin1String("\\\""), QLatin1String("\\\\\""));
75  // escape a single " because the arguments will be parsed
76  tmp.replace(QString(doubleQuote), QLatin1String("\\\""));
77  if (tmp.isEmpty() || tmp.contains(blank) || tmp.contains('\t')) {
78  // The argument must not end with a \ since this would be interpreted
79  // as escaping the quote -- rather put the \ behind the quote: e.g.
80  // rather use "foo"\ than "foo\"
81  QString endQuote(doubleQuote);
82  int i = tmp.length();
83  while (i > 0 && tmp.at(i - 1) == backSlash) {
84  --i;
85  endQuote += backSlash;
86  }
87  cmdLine += QLatin1String(" \"");
88  cmdLine += tmp.left(i);
89  cmdLine += endQuote;
90  } else {
91  cmdLine += blank;
92  cmdLine += tmp;
93  }
94  }
95  return cmdLine;
96 }

98 QByteArray AbstractProcess::createWinEnvironment(const QStringList &env)
99 {
100  QByteArray envlist;
