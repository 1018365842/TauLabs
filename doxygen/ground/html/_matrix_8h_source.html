235  inline void resize(int rows, int cols)
236  {
237  ei_assert((MaxRowsAtCompileTime == Dynamic || MaxRowsAtCompileTime >= rows)
238  && (RowsAtCompileTime == Dynamic || RowsAtCompileTime == rows)
239  && (MaxColsAtCompileTime == Dynamic || MaxColsAtCompileTime >= cols)
