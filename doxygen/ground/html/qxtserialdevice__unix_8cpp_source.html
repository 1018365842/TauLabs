61 bool QxtSerialDevice::open(OpenMode mode) {
62  if(qxt_d().notifier) {
63  delete qxt_d().notifier;
64  qxt_d().notifier = 0;
65  }
66  int m;
67  if(mode & QIODevice::ReadOnly && mode & QIODevice::WriteOnly) { // r/w
68  m = O_RDWR;
69  } else if(mode & QIODevice::ReadOnly) {
70  m = O_WRONLY;
71  } else {
72  m = O_RDONLY;
73  }
74  qxt_d().fd = ::open(qxt_d().device.toLocal8Bit().constData(), m | O_NOCTTY);
75  if(qxt_d().fd < 0) {
76  // TODO: set error message
77  return false;
78  }
79  fcntl(qxt_d().fd, F_SETFL, O_NONBLOCK);
80  tcgetattr(qxt_d().fd, &qxt_d().reset);
81  cfmakeraw(&qxt_d().settings);
82 
83  qxt_d().notifier = new QSocketNotifier(qxt_d().fd, QSocketNotifier::Read, this);
84  if(mode & QIODevice::Unbuffered) {
85  QObject::connect(qxt_d().notifier, SIGNAL(activated(int)), this, SIGNAL(readyRead()));
86  } else {
87  QObject::connect(qxt_d().notifier, SIGNAL(activated(int)), &qxt_d(), SLOT(fillBuffer()));
88  }
89  setOpenMode(mode);
90  return qxt_d().updateSettings();
91 }

93 int QxtSerialDevicePrivate::constFillBuffer() const {
94  notifier->setEnabled(false);
95  int sz = deviceBuffer();
96  QVarLengthArray<char, 1024> buf(sz);
97  int rv = read(fd, buf.data(), sz);
98  if(rv < 0) {
99  notifier->setEnabled(true);
100  if(errno == EAGAIN) return 0; // harmless
101  return errno;
102  }
103  buffer += QByteArray(buf.constData(), rv);
104  notifier->setEnabled(true);
105  return 0;
106 }

151 qint64 QxtSerialDevice::readData(char* data, qint64 maxSize) {
152  int b = bytesAvailable();
153  if(maxSize > b) maxSize = b;
154  if(!(openMode() & QIODevice::Unbuffered)) {
155  if(qxt_d().fillBuffer()) return 0;
156  }
157  int bufsize = qxt_d().buffer.length();
158  int rv = 0;
159  if(maxSize <= bufsize) {
160  memcpy(data, qxt_d().buffer.constData(), maxSize);
161  qxt_d().buffer = qxt_d().buffer.mid(maxSize);
162  return maxSize;
163  } else {
164  memcpy(data, qxt_d().buffer.constData(), bufsize);
165  qxt_d().buffer.clear();
166  rv = bufsize;
167  data += bufsize;
168  maxSize -= bufsize;
169  }
170  int readVal = ::read(qxt_d().fd, data, maxSize);
171  if(readVal < 0) {
172  setErrorString(strerror(errno));
173  } else {
174  rv += readVal;
175  }
176  return rv;
177 }

187 bool QxtSerialDevicePrivate::setPortSettings(QxtSerialDevice::PortSettings setup) {
188  int bits = setup & QxtSerialDevice::BitMask;
189  switch(bits) {
190  case QxtSerialDevice::Bit8: format = CS8; break;
191  case QxtSerialDevice::Bit7: format = CS7; break;
192  case QxtSerialDevice::Bit6: format = CS6; break;
193  case QxtSerialDevice::Bit5: format = CS5; break;
194  };
195  if(setup & QxtSerialDevice::Stop2)
196  format |= CSTOPB;
197  int parity = setup & QxtSerialDevice::ParityMask;
198  if(parity != QxtSerialDevice::ParityNone) {
199  format |= PARENB;
200  if(parity == QxtSerialDevice::ParityOdd) {
201  format |= PARODD;
202  } else if(parity == QxtSerialDevice::ParityMark || parity == QxtSerialDevice::ParitySpace) {
203 #ifdef CMSPAR
204  format |= CMSPAR;
205  if(parity == QxtSerialDevice::ParityMark)
206  format |= PARODD;
207 #else
208  qxt_p().setErrorString("Space/Mark parity not supported");
209  return false;
210 #endif
211  }
212  }
213  int flowbits = setup & QxtSerialDevice::FlowMask;
214  if(flowbits == QxtSerialDevice::FlowRtsCts) {
215 #ifdef CRTSCTS
216  flow = CRTSCTS;
217 #else
218  qxt_p().setErrorString("Hardware flow control not supported");
219  return false;
220 #endif
221  } else if(flowbits == QxtSerialDevice::FlowXonXoff) {
222  flow = IXON | IXOFF;
223  }
224  return updateSettings();
225 }

227 bool QxtSerialDevicePrivate::updateSettings() {
228  if(qxt_p().isOpen()) {
229  settings.c_cflag = baud | flow | format | CLOCAL | CREAD;
230  tcflush(fd, TCIFLUSH);
231  if(tcsetattr(fd, TCSANOW, &settings)) {
232  qxt_p().setErrorString(strerror(errno));
233  return false;
234  }
235  }
236  return true;
