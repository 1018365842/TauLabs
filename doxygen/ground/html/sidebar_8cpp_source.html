187 void SideBar::activateItem(SideBarItem *item)
188 {
189  QMap<QString, SideBarItem*>::const_iterator it = m_itemMap.constBegin();
190  QString title;
191  while (it != m_itemMap.constEnd()) {
192  if (it.value() == item) {
193  title = it.key();
194  break;
195  }
196  ++it;
197  }
198 
199  if (title.isEmpty())
200  return;
201 
202  for (int i = 0; i < m_widgets.count(); ++i) {
203  if (m_widgets.at(i)->currentItemTitle() == title) {
204  item->widget()->setFocus();
205  return;
206  }
207  }
208 
209  SideBarWidget *widget = m_widgets.first();
210  widget->setCurrentItem(title);
211  updateWidgets();
212  item->widget()->setFocus();
213 }

307 void SideBarWidget::updateAvailableItems()
308 {
309  bool blocked = m_comboBox->blockSignals(true);
310  QString current = m_comboBox->currentText();
311  m_comboBox->clear();
312  QStringList itms = m_sideBar->availableItems();
313  if (!current.isEmpty() && !itms.contains(current))
314  itms.append(current);
315  qSort(itms);
316  m_comboBox->addItems(itms);
317  int idx = m_comboBox->findText(current);
318  if (idx < 0)
319  idx = 0;
320  m_comboBox->setCurrentIndex(idx);
321  m_splitButton->setEnabled(itms.count() > 1);
322  m_comboBox->blockSignals(blocked);
