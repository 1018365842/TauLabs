133 bool SpectrogramData::append(UAVObject* multiObj)
134 {
135  QDateTime NOW = QDateTime::currentDateTime(); //TODO: Upgrade this to show UAVO time and not system time
136 
137  // Check to make sure it's the correct UAVO
138  if (uavObjectName == multiObj->getName()) {
139 
140  // Only run on UAVOs that have multiple instances
141  if (multiObj->isSingleInstance())
142  return false;
143 
144  //Instantiate object manager
145  UAVObjectManager *objManager;
146 
147  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
148  Q_ASSERT(pm != NULL);
149  objManager = pm->getObject<UAVObjectManager>();
150  Q_ASSERT(objManager != NULL);
151 
152 
153  // Get list of object instances
154  QVector<UAVObject*> list = objManager->getObjectInstances(multiObj->getName());
155 
156  // Remove a row's worth of data.
157  unsigned int spectrogramWidth = list.size();
158 
159  // Check that there is a full window worth of data. While GCS is starting up, the size of
160  // multiple instance UAVOs is 1, so it's possible for spurious data to come in before
161  // the flight controller board has had time to initialize the UAVO size.
162  if (spectrogramWidth != windowWidth){
163  qDebug() << "Incomplete data set in" << multiObj->getName() << "." << uavFieldName << "spectrogram: " << spectrogramWidth << " samples provided, but expected " << windowWidth;
164  return false;
165  }
166 
167  //Initialize vector where we will read out an entire row of multiple instance UAVO
168  QVector<double> values;
169 
170  timeDataHistory->append(NOW.toTime_t() + NOW.time().msec() / 1000.0);
171  UAVObjectField* multiField = multiObj->getField(uavFieldName);
172  Q_ASSERT(multiField);
173  if (multiField ) {
174 
175  // Get the field of interest
176  foreach (UAVObject *obj, list) {
177  UAVObjectField* field = obj->getField(uavFieldName);
178 
179  double currentValue = valueAsDouble(obj, field, haveSubField, uavSubFieldName) * pow(10, scalePower);
180 
181  double vecVal = currentValue;
182  //Normally some math would go here, modifying vecVal before appending it to values
183  // .
184  // .
185  // .
186 
187 
188  // Second to last step, see if autoscale is turned on and if the value exceeds the maximum for the scope.
189  if ( zMaximum == 0 && vecVal > rasterData->interval(Qt::ZAxis).maxValue()){
190  // Change scope maximum and color depth
191  rasterData->setInterval(Qt::ZAxis, QwtInterval(0, vecVal) );
192  autoscaleValueUpdated = vecVal;
193  }
194  // Last step, assign value to vector
195  values += vecVal;
196  }
197 
198  while (timeDataHistory->back() - timeDataHistory->front() > timeHorizon){
199  timeDataHistory->pop_front();
200  zDataHistory->remove(0, fminl(spectrogramWidth, zDataHistory->size()));
201  }
202 
203  // Doublecheck that there are the right number of samples. This can occur if the "field" assert fails
204  if(values.size() == (int) windowWidth){
205  *zDataHistory << values;
206  }
207 
208  return true;
209  }
210  }
211 
212  return false;
213 }
