53  // A Qt invokable, such as a signal or slot, can be identified by the metaobject containing its description plus
54  // its signature or name. It is worth noting that QxtRPCService uses the same structure for both signals and slots,
55  // but in slightly different ways: For identifying incoming signals, this structure contains the signature. For
56  // identifying outgoing slots, this structure only contains the name, and the parameters coming from the signal are
57  // used to determine which overload to use.
58  // TODO: Is this actually safe?
59  typedef QPair<const QMetaObject*, QByteArray> MetaMethodDef;

61  // A slot connection can be identified by the object receiving it and the name of the function. Additionally, a
62  // connection can be Direct, Queued, or BlockingQueued.
63  struct SlotDef
64  {
65  QObject* recv;
66  QByteArray slot;
67  Qt::ConnectionType type;
68  inline bool operator==(const SlotDef& other) const {
69  // Two slots are equivalent only if they refer to the same slot on the same object with the same
70  // connection type.
71  return (recv == other.recv) && (slot == other.slot) && (type == other.type);
72  }
73  };
