149 void PathPlannerGadgetWidget::on_tbFilletPath_clicked()
150 {
151  // Create a copy of the model before filleting
152  if (!prevModel)
153  prevModel = new FlightDataModel(this);
154  Q_ASSERT(prevModel);
155  if (prevModel)
156  prevModel->replaceData(model);
157 
158  IPathAlgorithm * algo = new PathFillet(this);
159  // Only process is successfully configured and the verification of the model succeeds
160  QString err;
161  if(algo->configure(this) && algo->verifyPath(model, err)) {
162  // If unsuccessful delete the cached model
163  if (!algo->processPath(model)) {
164  delete prevModel;
165  prevModel = NULL;
166  }
167  }
168 }
