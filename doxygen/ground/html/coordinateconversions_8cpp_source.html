68 void CoordinateConversions::LLA2ECEF(double LLA[3], double ECEF[3]){
69  const double a = R_equator; // Equatorial Radius
70  const double e = eccentricity; // Eccentricity
71  double sinLat, sinLon, cosLat, cosLon;
72  double N;
73 
74  sinLat=sin(DEG2RAD*LLA[0]);
75  sinLon=sin(DEG2RAD*LLA[1]);
76  cosLat=cos(DEG2RAD*LLA[0]);
77  cosLon=cos(DEG2RAD*LLA[1]);
78 
79  N = a / sqrt(1.0 - e*e*sinLat*sinLat); //prime vertical radius of curvature
80 
81  ECEF[0] = (N+LLA[2])*cosLat*cosLon;
82  ECEF[1] = (N+LLA[2])*cosLat*sinLon;
83  ECEF[2] = ((1-e*e)*N + LLA[2]) * sinLat;
84 }

91 int CoordinateConversions::ECEF2LLA(double ECEF[3], double LLA[3])
92 {
93  const double a = R_equator; // Equatorial Radius
94  const double e = eccentricity; // Eccentricity
95  double x=ECEF[0], y=ECEF[1], z=ECEF[2];
96  double Lat, N, NplusH, delta, esLat;
97  uint16_t iter;
98 
99  LLA[1] = RAD2DEG*atan2(y,x);
100  N = a;
101  NplusH = N;
102  delta = 1;
103  Lat = 1;
104  iter=0;
105 
106  while (((delta > 1.0e-14)||(delta < -1.0e-14)) && (iter < 100))
107  {
108  delta = Lat - atan(z / (sqrt(x*x + y*y)*(1-(N*e*e/NplusH))));
109  Lat = Lat-delta;
110  esLat = e*sin(Lat);
111  N = a / sqrt(1 - esLat*esLat);
112  NplusH = sqrt(x*x + y*y)/cos(Lat);
113  iter += 1;
114  }
115 
116  LLA[0] = RAD2DEG*Lat;
117  LLA[2] = NplusH - N;
118 
119  if (iter==500) return (0);
120  else return (1);
121 }

210 void CoordinateConversions::LLA2NED_HomeLLA(double LLA[3], double homeLLA[3], double NED[3])
211 {
212  double lat = homeLLA[0] * DEG2RAD;
213  double alt = homeLLA[2];
214 
215  float T[3];
216  T[0] = alt+6.378137E6;
217  T[1] = cos(lat)*(alt+6.378137E6);

230 void CoordinateConversions::Quaternion2RPY(const float q[4], float rpy[3])
231 {
232  float R13, R11, R12, R23, R33;
233  float q0s = q[0] * q[0];
234  float q1s = q[1] * q[1];
235  float q2s = q[2] * q[2];
236  float q3s = q[3] * q[3];
237 
238  R13 = 2 * (q[1] * q[3] - q[0] * q[2]);
239  R11 = q0s + q1s - q2s - q3s;
240  R12 = 2 * (q[1] * q[2] + q[0] * q[3]);
241  R23 = 2 * (q[2] * q[3] + q[0] * q[1]);
242  R33 = q0s - q1s - q2s + q3s;
243 
244  rpy[1] = RAD2DEG * asinf(-R13); // pitch always between -pi/2 to pi/2
245  rpy[2] = RAD2DEG * atan2f(R12, R11);
246  rpy[0] = RAD2DEG * atan2f(R23, R33);
247 
248  //TODO: consider the cases where |R13| ~= 1, |pitch| ~= pi/2
249 }

251 // ****** find quaternion from roll, pitch, yaw ********
252 void CoordinateConversions::RPY2Quaternion(const float rpy[3], float q[4])
253 {
254  float phi, theta, psi;
255  float cphi, sphi, ctheta, stheta, cpsi, spsi;
256 
257  phi = DEG2RAD * rpy[0] / 2;
258  theta = DEG2RAD * rpy[1] / 2;
259  psi = DEG2RAD * rpy[2] / 2;
260  cphi = cosf(phi);
261  sphi = sinf(phi);
262  ctheta = cosf(theta);
263  stheta = sinf(theta);
264  cpsi = cosf(psi);
265  spsi = sinf(psi);
266 
267  q[0] = cphi * ctheta * cpsi + sphi * stheta * spsi;
268  q[1] = sphi * ctheta * cpsi - cphi * stheta * spsi;
269  q[2] = cphi * stheta * cpsi + sphi * ctheta * spsi;
270  q[3] = cphi * ctheta * spsi - sphi * stheta * cpsi;
271 
272  if (q[0] < 0) { // q0 always positive for uniqueness
273  q[0] = -q[0];
274  q[1] = -q[1];
275  q[2] = -q[2];
276  q[3] = -q[3];
277  }

281 void CoordinateConversions::Quaternion2R(const float q[4], float Rbe[3][3])
282 {
283 
284  float q0s = q[0] * q[0], q1s = q[1] * q[1], q2s = q[2] * q[2], q3s = q[3] * q[3];
285 
286  Rbe[0][0] = q0s + q1s - q2s - q3s;
287  Rbe[0][1] = 2 * (q[1] * q[2] + q[0] * q[3]);
288  Rbe[0][2] = 2 * (q[1] * q[3] - q[0] * q[2]);
289  Rbe[1][0] = 2 * (q[1] * q[2] - q[0] * q[3]);
290  Rbe[1][1] = q0s - q1s + q2s - q3s;
291  Rbe[1][2] = 2 * (q[2] * q[3] + q[0] * q[1]);
292  Rbe[2][0] = 2 * (q[1] * q[3] + q[0] * q[2]);
293  Rbe[2][1] = 2 * (q[2] * q[3] - q[0] * q[1]);
294  Rbe[2][2] = q0s - q1s - q2s + q3s;
295 }

297 //** Find quaternion vector from a rotation matrix, Rbe, a matrix which rotates a vector from earth frame to body frame **
298 void CoordinateConversions::R2Quaternion(float const Rbe[3][3], float q[4])
299 {
300  qreal w, x, y, z;
301 
302  // w always >= 0
303  w = sqrt(std::max(0.0, 1.0 + Rbe[0][0] + Rbe[1][1] + Rbe[2][2])) / 2.0;
304  x = sqrt(std::max(0.0, 1.0 + Rbe[0][0] - Rbe[1][1] - Rbe[2][2])) / 2.0;
305  y = sqrt(std::max(0.0, 1.0 - Rbe[0][0] + Rbe[1][1] - Rbe[2][2])) / 2.0;
306  z = sqrt(std::max(0.0, 1.0 - Rbe[0][0] - Rbe[1][1] + Rbe[2][2])) / 2.0;
307 
308  x = copysign(x, (Rbe[1][2] - Rbe[2][1]));
309  y = copysign(y, (Rbe[2][0] - Rbe[0][2]));
310  z = copysign(z, (Rbe[0][1] - Rbe[1][0]));
311 
312  q[0]=w;
313  q[1]=x;
314  q[2]=y;
315  q[3]=z;
316 }
