67 void QxtAbstractFileLoggerEngine::initLoggerEngine()
68 {
69  // Are we already logging to a file? If so, close it and disable logging.
70  killLoggerEngine();
71 
72  // If the file exists, check if we can write to it. If we can, we append!
73  // If the file doesn't exits, try to create it.
74  // If we can't write to a file, disable this plugin.
75  if (qxt_d().logFile.isEmpty()) return; // if there's no filename, disable the engine until one is given
76 
77  setDevice(new QFile(qxt_d().logFile));
78  if (!device()->open(qxt_d().mode)
79  || !device()->isWritable())
80  {
81  killLoggerEngine();
82  return;
83  }
84 
85  enableLogging();
86 }
