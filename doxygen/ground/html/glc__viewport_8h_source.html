405 GLC_Matrix4x4 GLC_Viewport::compositionMatrix() const
406 {
407  // Get the viewport projection matrix
408  GLC_Matrix4x4 projectionMatrix= m_ProjectionMatrix;
409  // Get the camera modelView matrix
410  GLC_Matrix4x4 modelViewMatrix= m_pViewCam->modelViewMatrix();
411  // Composition matrix
412  GLC_Matrix4x4 compMatrix= projectionMatrix * modelViewMatrix;
413 
414  return compMatrix;
415 }
