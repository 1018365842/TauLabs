186 template<typename MatrixType,
187  bool IsSquare = (MatrixType::RowsAtCompileTime == MatrixType::ColsAtCompileTime) && MatrixType::RowsAtCompileTime!=Dynamic>
188 struct ei_inplace_transpose_selector;

190 template<typename MatrixType>
191 struct ei_inplace_transpose_selector<MatrixType,true> { // square matrix
192  static void run(MatrixType& m) {
193  m.template part<StrictlyUpperTriangular>().swap(m.transpose());
194  }
195 };
196 
197 template<typename MatrixType>
198 struct ei_inplace_transpose_selector<MatrixType,false> { // non square matrix
199  static void run(MatrixType& m) {
200  if (m.rows()==m.cols())
201  m.template part<StrictlyUpperTriangular>().swap(m.transpose());
202  else
203  m = m.transpose().eval();
204  }
205 };
