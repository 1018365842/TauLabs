113 void DefaultHwSettingsWidget::updateFields()
114 {
115  Q_ASSERT(settingSelected);

117 
118  QLayout *layout = ui->portSettingsFrame->layout();
119  for (int i = 0; i < fieldWidgets.size(); i++)
120  layout->removeWidget(fieldWidgets[i]);
121  fieldWidgets.clear();
122 
123  QList <UAVObjectField*> fields = hwSettingsObject->getFields();
124  for (int i = 0; i < fields.size(); i++) {
125  if (fields[i]->getType() != UAVObjectField::ENUM)
126  continue;
127  HwFieldSelector *sel = new HwFieldSelector(this);
128  layout->addWidget(sel);
