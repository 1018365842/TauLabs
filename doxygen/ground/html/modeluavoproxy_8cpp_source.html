53 void ModelUavoProxy::modelToObjects()
54 {
55  Waypoint *wp = Waypoint::GetInstance(objManager,0);
56  Q_ASSERT(wp);
57  if (wp == NULL)
58  return;
59 
60  // Make sure the object is acked
61  UAVObject::Metadata initialMeta = wp->getMetadata();
62  UAVObject::Metadata meta = initialMeta;
63  UAVObject::SetFlightTelemetryAcked(meta, true);
64  wp->setMetadata(meta);
65 
66  double homeLLA[3];
67  double NED[3];
68  double LLA[3];
69  getHomeLocation(homeLLA);
70 
71  for(int x=0;x<myModel->rowCount();++x)
72  {
73  Waypoint *wp = NULL;
74 
75  // Get the number of existing waypoints
76  int instances=objManager->getNumInstances(Waypoint::OBJID);
77 
78  // Create new instances of waypoints if this is more than exist
79  if(x>instances-1)
80  {
81  wp=new Waypoint;
82  wp->initialize(x,wp->getMetaObject());
83  objManager->registerObject(wp);
84  }
85  else
86  {
87  wp=Waypoint::GetInstance(objManager,x);
88  }
89 
90  Q_ASSERT(wp);
91  Waypoint::DataFields waypoint = wp->getData();
92 
93  // Convert from LLA to NED for sending to the model
94  LLA[0] = myModel->data(myModel->index(x,FlightDataModel::LATPOSITION)).toDouble();
95  LLA[1] = myModel->data(myModel->index(x,FlightDataModel::LNGPOSITION)).toDouble();
96  LLA[2] = myModel->data(myModel->index(x,FlightDataModel::ALTITUDE)).toDouble();
97  Utils::CoordinateConversions().LLA2NED_HomeLLA(LLA, homeLLA, NED);
98 
99  // Fetch the data from the internal model
100  waypoint.Velocity=myModel->data(myModel->index(x,FlightDataModel::VELOCITY)).toFloat();
101  waypoint.Position[Waypoint::POSITION_NORTH] = NED[0];
102  waypoint.Position[Waypoint::POSITION_EAST] = NED[1];
103  waypoint.Position[Waypoint::POSITION_DOWN] = NED[2];
104  waypoint.Mode = myModel->data(myModel->index(x,FlightDataModel::MODE), Qt::UserRole).toInt();
105  waypoint.ModeParameters = myModel->data(myModel->index(x,FlightDataModel::MODE_PARAMS)).toFloat();
106 
107  if (robustUpdate(waypoint, x))
108  qDebug() << "Successfully updated";
109  else {
110  qDebug() << "Upload failed";
111  break;
112  }
113  }
114  wp->setMetadata(initialMeta);
115 }

123 bool ModelUavoProxy::robustUpdate(Waypoint::DataFields data, int instance)
124 {
125  Waypoint *wp = Waypoint::GetInstance(objManager, instance);
126  connect(wp, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(waypointTransactionCompleted(UAVObject *, bool)));
127  for (int i = 0; i < 10; i++) {
128  QEventLoop m_eventloop;
129  QTimer::singleShot(500, &m_eventloop, SLOT(quit()));
130  connect(this, SIGNAL(waypointTransactionSucceeded()), &m_eventloop, SLOT(quit()));
131  connect(this, SIGNAL(waypointTransactionFailed()), &m_eventloop, SLOT(quit()));
132  waypointTransactionResult.insert(instance, false);
133  wp->setData(data);
134  wp->updated();
135  m_eventloop.exec();
136  if (waypointTransactionResult.value(instance)) {
137  disconnect(wp, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(waypointTransactionCompleted(UAVObject *, bool)));
138  return true;
139  }
140 
141  // Wait a second for next attempt
142  QTimer::singleShot(500, &m_eventloop, SLOT(quit()));
143  m_eventloop.exec();
144  }
145 
146  disconnect(wp, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(waypointTransactionCompleted(UAVObject *, bool)));
147 
148  // None of the attempt got an ack
149  return false;
150 }

172 void ModelUavoProxy::objectsToModel()
173 {
174  double homeLLA[3];
175  getHomeLocation(homeLLA);
176  double LLA[3];
177 
178  myModel->removeRows(0,myModel->rowCount());
179  for(int x=0;x<objManager->getNumInstances(waypointObj->getObjID());++x)
180  {
181  Waypoint * wp;
182  Waypoint::DataFields wpfields;
183 
184  wp = Waypoint::GetInstance(objManager,x);
185  Q_ASSERT(wp);
186 
187  if(!wp)
188  continue;
189 
190  // Get the waypoint data from the object manager and prepare a row in the internal model
191  wpfields = wp->getData();
192  myModel->insertRow(x);
193 
194  // Compute the coordinates in LLA
195  double NED[3] = {wpfields.Position[0], wpfields.Position[1], wpfields.Position[2]};
196  Utils::CoordinateConversions().NED2LLA_HomeLLA(homeLLA, NED, LLA);
197 
198  // Store the data
199  myModel->setData(myModel->index(x,FlightDataModel::LATPOSITION), LLA[0]);
200  myModel->setData(myModel->index(x,FlightDataModel::LNGPOSITION), LLA[1]);
201  myModel->setData(myModel->index(x,FlightDataModel::VELOCITY), wpfields.Velocity);
202  myModel->setData(myModel->index(x,FlightDataModel::MODE), wpfields.Mode);
203  myModel->setData(myModel->index(x,FlightDataModel::MODE_PARAMS), wpfields.ModeParameters);
204  }
