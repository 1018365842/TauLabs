56  * An item in the sidebar. Has a widget that is displayed in the sidebar and
57  * optionally a list of tool buttons that are added to the toolbar above it.
58  * The window title of the widget is displayed in the combo box.
59  *
60  * The SideBarItem takes ownership over the widget.
61  */

64 public:

67  {}
68 
69  virtual ~SideBarItem();
70 
71  QWidget *widget() { return m_widget; }
72 
73  /* Should always return a new set of tool buttons.
74  *
75  * Workaround since there doesn't seem to be a nice way to remove widgets
76  * that have been added to a QToolBar without either not deleting the
77  * associated QAction or causing the QToolButton to be deleted.
78  */
79  virtual QList<QToolButton *> createToolBarWidgets()
80  {
81  return QList<QToolButton *>();
82  }
83 
84 private:
