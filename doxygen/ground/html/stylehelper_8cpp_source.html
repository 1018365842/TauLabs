56 QColor StyleHelper::mergedColors(const QColor &colorA, const QColor &colorB, int factor)
57 {
58  const int maxFactor = 100;
59  QColor tmp = colorA;
60  tmp.setRed((tmp.red() * factor) / maxFactor + (colorB.red() * (maxFactor - factor)) / maxFactor);
61  tmp.setGreen((tmp.green() * factor) / maxFactor + (colorB.green() * (maxFactor - factor)) / maxFactor);
62  tmp.setBlue((tmp.blue() * factor) / maxFactor + (colorB.blue() * (maxFactor - factor)) / maxFactor);
63  return tmp;
64 }

172 static void horizontalGradientHelper(QPainter *p, const QRect &spanRect, const
173 QRect &rect)
174 {
175  QColor base = StyleHelper::baseColor();
176  QLinearGradient grad(rect.topLeft(), rect.bottomLeft());
177  grad.setColorAt(0, StyleHelper::highlightColor().lighter(120));
178  if (rect.height() == StyleHelper::navigationWidgetHeight()) {
179  grad.setColorAt(0.4, StyleHelper::highlightColor());
180  grad.setColorAt(0.401, base);
181  }
182  grad.setColorAt(1, StyleHelper::shadowColor());
183  p->fillRect(rect, grad);
184 
185  QLinearGradient shadowGradient(spanRect.topLeft(), spanRect.topRight());
186  shadowGradient.setColorAt(0, QColor(0, 0, 0, 30));
187  QColor highlight = StyleHelper::highlightColor().lighter(130);
188  highlight.setAlpha(100);
189  shadowGradient.setColorAt(0.7, highlight);
190  shadowGradient.setColorAt(1, QColor(0, 0, 0, 40));
191  p->fillRect(rect, shadowGradient);
