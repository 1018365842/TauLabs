113 void OutputChannelForm::linkToggled(bool state)
114 {
115  Q_UNUSED(state)
116 
117  if (!m_inChannelTest)
118  return; // we are not in Test Output mode
119 
120  // find the minimum slider value for the linked ones
121  if (!parent()) return;
122  int min = 10000;
123  int linked_count = 0;
124  QList<OutputChannelForm*> outputChannelForms = parent()->findChildren<OutputChannelForm*>();
125  // set the linked channels of the parent widget to the same value
126  foreach(OutputChannelForm *outputChannelForm, outputChannelForms)
127  {
128  if (!outputChannelForm->ui.actuatorLink->checkState())
129  continue;
130  if (this == outputChannelForm)
131  continue;
132  int value = outputChannelForm->ui.actuatorNeutral->value();
133  if(min > value) min = value;
134  linked_count++;
135  }
136 
137  if (linked_count <= 0)
138  return; // no linked channels
139 
140  // set the linked channels to the same value
141  foreach(OutputChannelForm *outputChannelForm, outputChannelForms)
142  {
143  if (!outputChannelForm->ui.actuatorLink->checkState())
144  continue;
145  outputChannelForm->ui.actuatorNeutral->setValue(min);
146  }
147 }

240 void OutputChannelForm::reverseChannel(bool state)
241 {
242  // Sanity check: if state became true, make sure the Maxvalue was higher than Minvalue
243  // the situations below can happen!
244  if (state && (ui.actuatorMax->value() < ui.actuatorMin->value()))
245  return;
246  if (!state && (ui.actuatorMax->value() > ui.actuatorMin->value()))
247  return;
248 
249  // Now, swap the min & max values (only on the spinboxes, the slider
250  // does not change!
251  int temp = ui.actuatorMax->value();
252  ui.actuatorMax->setValue(ui.actuatorMin->value());
253  ui.actuatorMin->setValue(temp);
254 
255  // Also update the channel value
256  // This is a trick to force the slider to update its value and
257  // emit the right signal itself, because our sendChannelTest(int) method
258  // relies on the object sender's identity.
259  if (ui.actuatorNeutral->value() < ui.actuatorNeutral->maximum())
260  {
261  ui.actuatorNeutral->setValue(ui.actuatorNeutral->value()+1);
262  ui.actuatorNeutral->setValue(ui.actuatorNeutral->value()-1);
263  }
264  else
265  {
266  ui.actuatorNeutral->setValue(ui.actuatorNeutral->value()-1);
267  ui.actuatorNeutral->setValue(ui.actuatorNeutral->value()+1);
268  }
269 }

275 void OutputChannelForm::sendChannelTest(int value)
276 {
277  int in_value = value;
278 
279  QSlider *ob = (QSlider *)QObject::sender();
280  if (!ob)
281  return;
282 
283  if (ui.actuatorRev->isChecked())
284  value = ui.actuatorMin->value() - value + ui.actuatorMax->value(); // the channel is reversed
285 
286  // update the label
287  ui.actuatorValue->setText(QString::number(value));
288 
289  if (ui.actuatorLink->checkState() && parent())
290  { // the channel is linked to other channels
291  QList<OutputChannelForm*> outputChannelForms = parent()->findChildren<OutputChannelForm*>();
292  // set the linked channels of the parent widget to the same value
293  foreach(OutputChannelForm *outputChannelForm, outputChannelForms)
294  {
295  if (this == outputChannelForm) continue;
296  if (!outputChannelForm->ui.actuatorLink->checkState()) continue;
297 
298  int val = in_value;
299  if (val < outputChannelForm->ui.actuatorNeutral->minimum())
300  val = outputChannelForm->ui.actuatorNeutral->minimum();
301  if (val > outputChannelForm->ui.actuatorNeutral->maximum())
302  val = outputChannelForm->ui.actuatorNeutral->maximum();
303 
304  if (outputChannelForm->ui.actuatorNeutral->value() == val) continue;
305 
306  outputChannelForm->ui.actuatorNeutral->setValue(val);
307  outputChannelForm->ui.actuatorValue->setText(QString::number(val));
308  }
309  }
310 
311  if (!m_inChannelTest)
312  return; // we are not in Test Output mode
313 
314  emit channelChanged(index(), value);
315 }
