59 QMap<QString, QKeySequence> CommandsFile::importCommands() const
60 {
61  QMap<QString, QKeySequence> result;
62 
63  QFile file(m_filename);
64  if (!file.open(QIODevice::ReadOnly))
65  return result;
66 
67  QDomDocument doc("KeyboardMappingScheme");
68  if (!doc.setContent(&file))
69  return result;
70 
71  QDomElement root = doc.documentElement();
72  if (root.nodeName() != QLatin1String("mapping"))
73  return result;
74 
75  QDomElement ks = root.firstChildElement();
76  for (; !ks.isNull(); ks = ks.nextSiblingElement()) {
77  if (ks.nodeName() == QLatin1String("shortcut")) {
78  QString id = ks.attribute(QLatin1String("id"));
79  QKeySequence shortcutkey;
80  QDomElement keyelem = ks.firstChildElement("key");
81  if (!keyelem.isNull())
82  shortcutkey = QKeySequence(keyelem.attribute("value"));
83  result.insert(id, shortcutkey);
84  }
85  }
86 
87  file.close();
88  return result;
89 }
