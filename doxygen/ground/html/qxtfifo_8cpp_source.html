100 qint64 QxtFifo::readData ( char * data, qint64 maxSize )
101 {
102  int bytes = qxt_d().available, step;
103  if(!bytes) return 0;
104  if(bytes > maxSize) bytes = maxSize;
105  int written = bytes;
106  char* writePos = data;
107  QxtFifoNode* node;
108  while(bytes > 0) {
109  node = qxt_d().head;
110  step = node->content.size();
111  if(step >= bytes) {
112  int rem = step - bytes;
113  memcpy(writePos, node->content.constData(), bytes);
114  step = bytes;
115  node->content = node->content.right(rem);
116  } else {
117  memcpy(writePos, node->content.constData(), step);
118  qxt_d().head.QXT_EXCHANGE(node->next);
119  delete node;
120  node = qxt_d().head;
121  }
122  writePos += step;
123  bytes -= step;
124  }
125  qxt_d().available.QXT_ADD(-written);
126  return written;
127 }

132 qint64 QxtFifo::writeData ( const char * data, qint64 maxSize )
133 {
134  if(maxSize > 0) {
135  if(maxSize > INT_MAX) maxSize = INT_MAX; // qint64 could easily exceed QAtomicInt, so let's play it safe
136  QxtFifoNode* newData = new QxtFifoNode(data, maxSize);
137  qxt_d().tail->next.QXT_EXCHANGE(newData);
138  qxt_d().tail.QXT_EXCHANGE(newData);
139  qxt_d().available.QXT_ADD(maxSize);
140  QMetaObject::invokeMethod(this, "bytesWritten", Qt::QueuedConnection, Q_ARG(qint64, maxSize));
141  QMetaObject::invokeMethod(this, "readyRead", Qt::QueuedConnection);
142  }
143  return maxSize;
144 }

164 void QxtFifo::clear()
165 {
166  qxt_d().available.QXT_EXCHANGE(0);
167  qxt_d().tail.QXT_EXCHANGE(qxt_d().head);
168  QxtFifoNode* node = qxt_d().head->next.QXT_EXCHANGE(NULL);
169  while (node && node->next)
170  {
171  QxtFifoNode* next = node->next.QXT_EXCHANGE(NULL);
172  delete node;
173  node = next;
174  }
175  qxt_d().head->content = QByteArray();
176 }
