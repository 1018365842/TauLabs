461 GLC_Matrix4x4& GLC_Matrix4x4::setMatRot(const GLC_Vector3d &Vect, const double &dAngleRad)
462 {
463  // Normalize the vector
464  GLC_Vector3d VectRot(Vect);
465  VectRot.normalize();
466 
467  // Code optimisation
468  const double SinAngleSur2= sin(dAngleRad / 2.0);
469 
470  // Quaternion computation
471  const double q0= cos(dAngleRad / 2);
472  const double q1= VectRot.m_Vector[0] * SinAngleSur2;
473  const double q2= VectRot.m_Vector[1] * SinAngleSur2;
474  const double q3= VectRot.m_Vector[2] * SinAngleSur2;
475 
476  // Code optimisation
477  const double q0Carre= (q0 * q0);
478  const double q1Carre= (q1 * q1);
479  const double q2Carre= (q2 * q2);
480  const double q3Carre= (q3 * q3);
481 
482  m_Matrix[0]= q0Carre + q1Carre - q2Carre - q3Carre;
483  m_Matrix[1]= 2.0 * (q1 *q2 + q0 * q3);
484  m_Matrix[2]= 2.0 * (q1 * q3 - q0 * q2);
485  m_Matrix[3]= 0.0;
486  m_Matrix[4]= 2.0 * (q1 * q2 - q0 * q3);
487  m_Matrix[5]= q0Carre + q2Carre - q3Carre - q1Carre;
488  m_Matrix[6]= 2.0 * (q2 * q3 + q0 * q1);
489  m_Matrix[7]= 0.0;
490  m_Matrix[8]= 2.0 * (q1 * q3 + q0 * q2);
491  m_Matrix[9]= 2.0 * (q2 * q3 - q0 * q1);
492  m_Matrix[10]= q0Carre + q3Carre - q1Carre - q2Carre;
493  m_Matrix[11]= 0.0;
494 
495  m_Matrix[12]= 0.0; //TX
496  m_Matrix[13]= 0.0; //TY
497  m_Matrix[14]= 0.0; //TZ
498  m_Matrix[15]= 1.0;
499 
500  m_Type= Direct;
501 
502  return *this;
503 }
