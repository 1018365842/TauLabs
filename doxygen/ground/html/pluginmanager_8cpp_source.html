357  const QString &opt, const QString &parm, const QString &description,
358  int optionIndentation, int descriptionIndentation)
359 {
360  int remainingIndent = descriptionIndentation - optionIndentation - opt.size();
361  indent(str, optionIndentation);
362  str << opt;
363  if (!parm.isEmpty()) {
364  str << " <" << parm << '>';
365  remainingIndent -= 3 + parm.size();
366  }
367  indent(str, qMax(0, remainingIndent));
368  str << description << '\n';

611 bool PluginManagerPrivate::loadQueue(PluginSpec *spec, QList<PluginSpec *> &queue,
612  QList<PluginSpec *> &circularityCheckQueue)
613 {
614  if (queue.contains(spec))
615  return true;
616  // check for circular dependencies
617  if (circularityCheckQueue.contains(spec)) {
618  spec->d->hasError = true;
619  spec->d->errorString = PluginManager::tr("Circular dependency detected:\n");
620  int index = circularityCheckQueue.indexOf(spec);
621  for (int i = index; i < circularityCheckQueue.size(); ++i) {
622  spec->d->errorString.append(PluginManager::tr("%1(%2) depends on\n")
623  .arg(circularityCheckQueue.at(i)->name()).arg(circularityCheckQueue.at(i)->version()));
624  }
625  spec->d->errorString.append(PluginManager::tr("%1(%2)").arg(spec->name()).arg(spec->version()));
626  return false;
627  }
628  circularityCheckQueue.append(spec);
629  // check if we have the dependencies
630  if (spec->state() == PluginSpec::Invalid || spec->state() == PluginSpec::Read) {
631  spec->d->hasError = true;
632  spec->d->errorString += "\n";
633  spec->d->errorString += PluginManager::tr("Cannot load plugin because dependencies are not resolved");
634  return false;
635  }
636  // add dependencies
637  foreach (PluginSpec *depSpec, spec->dependencySpecs()) {
638  if (!loadQueue(depSpec, queue, circularityCheckQueue)) {
639  spec->d->hasError = true;
640  spec->d->errorString =
641  PluginManager::tr("Cannot load plugin because dependency failed to load: %1(%2)\nReason: %3")
642  .arg(depSpec->name()).arg(depSpec->version()).arg(depSpec->errorString());
643  return false;
644  }
645  }
646  // add self
647  queue.append(spec);
648  return true;
649 }

655 void PluginManagerPrivate::loadPlugin(PluginSpec *spec, PluginSpec::State destState)
656 {
657  if (spec->hasError())
658  return;
659  if (destState == PluginSpec::Running) {
660  spec->d->initializeExtensions();
661  return;
662  } else if (destState == PluginSpec::Deleted) {
663  spec->d->kill();
664  return;
665  }
666  foreach (PluginSpec *depSpec, spec->dependencySpecs()) {
667  if (depSpec->state() != destState) {
668  spec->d->hasError = true;
669  spec->d->errorString =
670  PluginManager::tr("Cannot load plugin because dependency failed to load: %1(%2)\nReason: %3")
671  .arg(depSpec->name()).arg(depSpec->version()).arg(depSpec->errorString());
672  return;
673  }
674  }
675  if (destState == PluginSpec::Loaded)
676  spec->d->loadLibrary();
677  else if (destState == PluginSpec::Initialized)
678  spec->d->initializePlugin();
679  else if (destState == PluginSpec::Stopped)
680  spec->d->stop();
681 }

697 void PluginManagerPrivate::readPluginPaths()
698 {
699  qDeleteAll(pluginSpecs);
700  pluginSpecs.clear();
701 
702  QStringList specFiles;
703  QStringList searchPaths = pluginPaths;
704  while (!searchPaths.isEmpty()) {
705  const QDir dir(searchPaths.takeFirst());
706  const QFileInfoList files = dir.entryInfoList(QStringList() << QString("*.%1").arg(extension), QDir::Files);
707  foreach (const QFileInfo &file, files)
708  specFiles << file.absoluteFilePath();
709  const QFileInfoList dirs = dir.entryInfoList(QDir::Dirs|QDir::NoDotAndDotDot);
710  foreach (const QFileInfo &subdir, dirs)
711  searchPaths << subdir.absoluteFilePath();
712  }
713  foreach (const QString &specFile, specFiles) {
714  PluginSpec *spec = new PluginSpec;
715  spec->d->read(specFile);
716  pluginSpecs.append(spec);
717  }
718  resolveDependencies();
719  // ensure deterministic plugin load order by sorting
720  qSort(pluginSpecs.begin(), pluginSpecs.end(), lessThanByPluginName);
721  emit q->pluginsChanged();
722 }
