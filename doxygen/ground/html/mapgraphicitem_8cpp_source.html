477  void MapGraphicItem::SetZoom(double const& value)
478  {
479  if(ZoomTotal() != value)
480  {
481  //Saturate zoom, including digital zoom
482  if(value > MaxZoom())
483  {
484  zoomReal = MaxZoom();
485  zoomDigi = value-MaxZoom();
486  }
487  else if(value < MinZoom())
488  {
489  zoomDigi=0;
490  zoomReal = MinZoom();
491  }
492  else
493  {
494  zoomDigi=0;
495  zoomReal = value;
496  }
497 
498  //Handle digital zoom and non-integer zoom values
499  double integer;
500  double remainder = modf (value , &integer);
501  if(zoomDigi !=0 || remainder != 0)
502  {
503  float scaleValue = zoomDigi+remainder + 1;
504  MapRenderTransform = scaleValue;
505  if(integer>MaxZoom()){
506  integer=MaxZoom();
507  }
508  SetZoomStep((qint32)(integer));
509  // core->GoToCurrentPositionOnZoom();
510  this->update();
511  }
512  else
513  {
514  MapRenderTransform = 1;
515 
516  SetZoomStep ((qint32)(value));
517  zoomReal = ZoomStep();
518  this->update();
519  }
520  }
