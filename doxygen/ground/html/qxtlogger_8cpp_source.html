550 /*******************************************************************************
551  Message Handler for qdebug, qerror, qwarning, etc...
552  When QxtLogger is enabled as a message handler for Qt, this function
553  redirects message calls like qdebug, qwarning, qfatal.
554  \sa QxtLogger::installAsMessageHandler
555  \sa QxtLogger::removeAsMessageHandler
556 *******************************************************************************/
557 void QxtLoggerMessageHandler(QtMsgType type, const char *msg)
558 {
559  switch (type)
560  {
561  case QtDebugMsg:
562  QxtLogger::getInstance()->debug(msg, "qdebug");
563  break;
564  case QtWarningMsg:
565  QxtLogger::getInstance()->warning(msg, "qwarning");
566  break;
567  case QtCriticalMsg:
568  QxtLogger::getInstance()->critical(msg, "qcritical");
569  break;
570  case QtFatalMsg:
571  QxtLogger::getInstance()->fatal(msg, "qfatal");
572  abort();
573  }
574 }
