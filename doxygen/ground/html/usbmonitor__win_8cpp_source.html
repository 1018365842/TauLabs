179 bool USBMonitor::matchAndDispatchChangedDevice(const QString & deviceID, const GUID & guid, WPARAM wParam)
180 {
181  qDebug()<<"USB_MONITOR matchAndDispatchChangedDevice deviceID="<<deviceID;
182  bool rv = false;
183  DWORD dwFlag = (DBT_DEVICEARRIVAL == wParam) ? DIGCF_PRESENT : DIGCF_ALLCLASSES;
184  HDEVINFO devInfo;
185  if( (devInfo = SetupDiGetClassDevs(&guid,NULL,NULL, dwFlag | DIGCF_DEVICEINTERFACE)) != INVALID_HANDLE_VALUE )
186  {
187  SP_DEVINFO_DATA spDevInfoData;
188  spDevInfoData.cbSize = sizeof(SP_DEVINFO_DATA);
189  for(DWORD i=0; SetupDiEnumDeviceInfo(devInfo, i, &spDevInfoData); i++)
190  {
191  DWORD nSize=0 ;
192  TCHAR buf[MAX_PATH];
193  if ( SetupDiGetDeviceInstanceId(devInfo, &spDevInfoData, buf, MAX_PATH, &nSize) &&
194  deviceID.contains(TCHARToQString(buf))) // we found a match
195  {
196  USBPortInfo info;
197  info.devicePath=deviceID;
198  if( wParam == DBT_DEVICEARRIVAL )
199  {
200  qDebug()<<"USB_MONITOR INSERTION";
201  if(infoFromHandle(guid,info,devInfo,i)!=1)
202  {
203  qDebug()<<"USB_MONITOR infoFromHandle failed on matchAndDispatchChangedDevice";
204  break;
205  }
206  bool m_break=false;
207  foreach (USBPortInfo m_info, knowndevices) {
208  if(m_info.serialNumber==info.serialNumber && m_info.productID==info.productID && m_info.bcdDevice==info.bcdDevice && m_info.devicePath==info.devicePath)
209  {
210  qDebug()<<"USB_MONITOR device already present don't emit signal";
211  m_break=true;
212  }
213 
214  }
215  if(m_break)
216  break;
217  if(info.bcdDevice==0 || info.product.isEmpty())
218  {
219  qDebug()<<"USB_MONITOR empty information on device not emiting signal";
220  break;
221  }
222  knowndevices.append(info);
223  qDebug()<<"USB_MONITOR emit device discovered on device:"<<info.product<<info.bcdDevice;
224  emit deviceDiscovered(info);
225  break;
226 
227  }
228  else if( wParam == DBT_DEVICEREMOVECOMPLETE )
229  {
230  bool found=false;
231  for(int x=0;x<knowndevices.count();++x)
232  {
233  if(knowndevices[x].devicePath==deviceID)
234  {
235  USBPortInfo temp=knowndevices.at(x);
236  knowndevices.removeAt(x);
237  qDebug()<<"USB_MONITOR emit device removed on device:"<<temp.product<<temp.bcdDevice;
238  emit deviceRemoved(temp);
239  found=true;
240  break;
241  }
242  }
243  if(!found)
244  {
245  qDebug()<<"USB_MONITOR emit device removed on unknown device";
246  emit deviceRemoved(info);
247  }
248  }
249  break;
250 
251  }
252  }
253  SetupDiDestroyDeviceInfoList(devInfo);
254  }
255  return rv;
256 }

302 int USBMonitor::infoFromHandle(const GUID & guid,USBPortInfo & info,HDEVINFO & devInfo,DWORD & index)
303 {
304  //qDebug()<<"index0="<<index;
305  bool ret;
306  HANDLE h;
307  SP_DEVICE_INTERFACE_DATA iface;
308  SP_DEVICE_INTERFACE_DETAIL_DATA *details;
309  DWORD reqd_size;
310  HIDD_ATTRIBUTES attrib;
311  PHIDP_PREPARSED_DATA hid_data;
312  HIDP_CAPS capabilities;
313 
314  iface.cbSize = sizeof(SP_DEVICE_INTERFACE_DATA);
315 
316  ret = SetupDiEnumDeviceInterfaces(devInfo, NULL, &guid, index, &iface);
317  if (!ret) return 0;
318  //qDebug()<<"index1="<<index;
319  SetupDiGetInterfaceDeviceDetail(devInfo, &iface, NULL, 0, &reqd_size, NULL);
320  details = (SP_DEVICE_INTERFACE_DETAIL_DATA *)malloc(reqd_size);
321  if (details == NULL) return 2;
322  //qDebug()<<"index2="<<index;
323  memset(details, 0, reqd_size);
324  details->cbSize = sizeof(SP_DEVICE_INTERFACE_DETAIL_DATA);
325  ret = SetupDiGetDeviceInterfaceDetail(devInfo, &iface, details, reqd_size, NULL, NULL);
326  if (!ret)
327  {
328  free(details);
329  return 2;
330  }
331  //qDebug()<<"index3="<<index;
332  h = CreateFile(details->DevicePath, GENERIC_READ|GENERIC_WRITE, FILE_SHARE_READ|FILE_SHARE_WRITE, NULL, OPEN_EXISTING, FILE_FLAG_OVERLAPPED, NULL);
333 
334  if (h == INVALID_HANDLE_VALUE)
335  {
336  DWORD err = GetLastError();
337 
338  // I get ERROR_ACCESS_DENIED with most/all my input devices (mice/trackballs/tablet).
339  // Let's not log it :)
340  if (err == ERROR_ACCESS_DENIED)
341  {
342  free(details);
343  return 2;
344  }
345 
346  qDebug() << "Problem opening handle, path: " << QString().fromWCharArray(details->DevicePath);
347 
348  free(details);
349  return 2;
350  }
351  //qDebug()<<"index4="<<index;
352  free(details);
353  //qDebug()<<"DETAILS???"<<QString().fromWCharArray(details->DevicePath).toUpper().replace("#", "\\");
354  attrib.Size = sizeof(HIDD_ATTRIBUTES);
355  ret = HidD_GetAttributes(h, &attrib);
356  info.vendorID=attrib.VendorID;
357  info.productID=attrib.ProductID;
358  info.bcdDevice=attrib.VersionNumber;
359 
360 
361  if (!ret || !HidD_GetPreparsedData(h, &hid_data))
362  {
363  CloseHandle(h);
364  return 2;
365  }
366  //qDebug()<<"index5="<<index;
367  if (!HidP_GetCaps(hid_data, &capabilities))
368  {
369  HidD_FreePreparsedData(hid_data);
370  CloseHandle(h);
371  return 2;
372  }
373  //qDebug()<<"index6="<<index;
374  info.UsagePage=capabilities.UsagePage;
375  info.Usage=capabilities.Usage;
376  HidD_FreePreparsedData(hid_data);
377  char temp[126];
378  HidD_GetSerialNumberString(h, temp, sizeof(temp));
379  info.serialNumber= QString().fromUtf16((ushort*)temp,-1);
380  HidD_GetManufacturerString(h, temp, sizeof(temp));
381  info.manufacturer= QString().fromUtf16((ushort*)temp,-1);
382  HidD_GetProductString(h, temp, sizeof(temp));
383  info.product= QString().fromUtf16((ushort*)temp,-1);
384  //qDebug()<<"index="<<index<<"ProductID="<<info.product;
385  CloseHandle(h);
386  h = NULL;
387  return 1;
