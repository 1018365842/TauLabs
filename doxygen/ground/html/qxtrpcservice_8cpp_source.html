66  * QxtRPCServiceIntrospector is a customized QObject subclass that implements dynamic slots instead of using a static
67  * metaobject. Trolltech/Nokia has published a document on the concept in Qt Quarterly, available at
68  * <http://doc.trolltech.com/qq/qq16-dynamicqobject.html>, although work on QxtRPCService (originally QxRPCPeer) had
69  * begun before this document was released to non-subscribers.
70  *
71  * This class is responsible for dealing with incoming Qt signals.
72  */
73 class QxtRPCServiceIntrospector : public QObject
74 {
75 public:
76  // Maintain a pointer back to the service object that owns the introspector.
77  QxtRPCService* rpc;
78 
79  // Keep track of the ID of the next virtual slot to be defined.
80  quint32 nextSlotID;
81 
82  // Maps an incoming signal's metamethod to an entry in signalParameters.
83  QHash<QxtRPCServicePrivate::MetaMethodDef, int> signalIDs;
84 
85  // Maps an incoming signal's metamethod to its index in its metaobject.
86  QHash<QxtRPCServicePrivate::MetaMethodDef, int> methodIDs;
87 
88  // Each signal and slot has a list of parameters associated with it.
89  // QMetaType provides a type ID for every recognized data type.
90  QList<QList<int> > signalParameters;
91 
92  // A signal connection can be identified by the object emitting it and the signature of the signal.
93  typedef QPair<QObject*, QByteArray> SignalDef;
94 
95  // Maps a signal connection to one or more dynamic slot IDs.
96  QMultiHash<SignalDef, int> signalToId;
97 
98  // Maps a dynamic slot ID to an entry in signalParameters.
99  QHash<int, int> idToParams;
100 
101  // Keep track of which RPC messages should be emitted for each dynamically created slot.
102  QMultiHash<int, QString> idToRpc;
103

117 bool QxtRPCServiceIntrospector::addSignal(QObject* obj, const char* signature, const QString& rpcFunction)
118 {
119  const QMetaObject* meta = obj->metaObject();
120  QByteArray norm = QxtMetaObject::methodSignature(signature);
121  QxtRPCServicePrivate::MetaMethodDef signal = qMakePair(meta, norm);
122  int sigID, methodID;
123 
124  if(!signalIDs.count(signal)) {
125  // This signal hasn't been encountered before, so read the metaobject and cache the results.
126  methodID = meta->indexOfMethod(norm.constData());
127  if(methodID < 0) {
128  // indexOfMethod() returns -1 if the signal was not found, so report a warning and return an error.
129  qWarning() << "QxtRPCService::attachSignal: " << obj << "::" << signature << " does not exist";
130  return false;
131  }
132 
133  // Look up the signal's parameter list, ensure each parameter is queueable, and cache the type IDs.
134  QList<QByteArray> types = meta->method(methodID).parameterTypes();
135  QList<int> typeIDs;
136  int ct = types.count();
137  for(int i = 0; i < ct; i++) {
138  int typeID = QMetaType::type(types.value(i).constData());
139  if(typeID <= 0) {
140  qWarning() << "QxtRPCService::attachSignal: cannot queue arguments of type " << types.value(i);
141  return false;
142  }
143  typeIDs.append(typeID);
144  }
145 
146  // Cache the looked-up parameter list, associate the list with the signal definition, and cache the signal's
147  // QMetaObject method index.
148  int nextSignalID = signalParameters.count();
149  signalParameters.append(typeIDs);
150  signalIDs[signal] = nextSignalID;
151  methodIDs[signal] = methodID;
152 
153  // And finally, set the signal ID for the connect call below.
154  sigID = nextSignalID;
155  } else {
156  // Use the cached values from before.
157  sigID = signalIDs.value(signal);
158  methodID = methodIDs.value(signal);
159  }
160 
161  // Use an undocumented function from QMetaObject to connect the incoming signal to the next dynamic slot.
162  bool success = QMetaObject::connect(obj, methodID, this, nextSlotID);
163  if(!success) {
164  // Presumedly connect() will output its own warning.
165  // The remaining setup is unnecessary if the connection failed.
166  return false;
167  }
168 
169  // Associate the signal and source object with the dynamic slot.
170  signalToId.insert(qMakePair(obj, norm), nextSlotID);
171 
172  // Associate the new dynamic slot with an RPC message, or use the signal signature as the default message.
173  idToRpc.insert(nextSlotID, rpcFunction.isEmpty() ? norm : rpcFunction);
174 
175  // Associate the new dynamic slot with the parameter list definition.
176  idToParams[nextSlotID] = sigID;
177 
178  do {
179  // This variable is perfectly allowed to wrap around; it's terribly unlikely to have four billion
180  // concurrent connections but a particularly long-lived object with a lot of disconnects and
181  // reconnects might survive to see a wraparound.
182  nextSlotID++;
183  // But of course we need to make sure the next ID isn't in use.
184  } while(nextSlotID < quint32(QObject::staticMetaObject.methodCount()) || idToParams.contains(nextSlotID));
185 
186  return true;
187 }

189 void QxtRPCServiceIntrospector::disconnectObject(QObject* obj)
190 {
191  const QMetaObject* meta = obj->metaObject();
192 
193  foreach(const SignalDef& sig, signalToId.keys()) {
194  // Iterate through all tracked connections and skip any that don't match the incoming object.
195  if(sig.first != obj) continue;
196 
197  int methodID = methodIDs[qMakePair(meta, sig.second)];
198  foreach(int id, signalToId.values(sig)) {
199  // Iterate through all of the different connections for the object and explicitly disconnect them.
200  QMetaObject::disconnect(obj, methodID, this, id);
201 
202  // Remove the connection from our mappings.
203  idToRpc.remove(id);
204  idToParams.remove(id);
205  }
206 
207  // Remove the object/signal from the mapping.
208  signalToId.remove(sig);
209  }
210 }

212 int QxtRPCServiceIntrospector::qt_metacall(QMetaObject::Call _c, int _id, void **_a)
213 {
214  // Qt's signal dispatch mechanism invokes qt_metacall for each slot connected to the object.
215 
216  // The metacall protocol expects slots to be checked from the bottom up, starting with QObject and proceeding to
217  // the more derived classes. qt_metacall returns a negative number to indicate that the request was processed or a
218  // positive number to indicate the greatest method ID that was checked. moc-generated qt_metacall implementations
219  // subtract the return value from _id so that slots on a given class can be counted starting at 0, allowing the
220  // subclasses to add new signals or slots without breaking compatibility. QxtRPCService doesn't need this because
221  // it just asks its base class how many methods it has before adding slots.
222  if(QObject::qt_metacall(_c, _id, _a) < 0)
223  return _id;
224 
225  // qt_metacall is also used for other behaviors besides just invoking methods; we don't implement any of these, so
226  // we just return here.
227  if(_c != QMetaObject::InvokeMetaMethod)
228  return _id;
229 
230  // Construct an array of QVariants based on the parameters passed through _a.
231  QVariant v[8];
232  const QList<int>& types = signalParameters.at(idToParams.value(_id));
233  int ct = types.count();
234  for(int i = 0; i < ct; i++) {
235  // The qt_metacall implementation is expected to already know the data types in _a, so that's why we tracked it.
236  v[i] = QVariant(types.at(i), _a[i+1]);
237  }
238 
239  foreach(const QString& rpcName, idToRpc.values(_id)) {
240  // Invoke each RPC message connected to the requested dynamic slot ID.
241  rpc->call(rpcName, v[0], v[1], v[2], v[3], v[4], v[5], v[6], v[7]);
242  }
243 
244  // Inform the calling function that we handled the call.
245  return -1;
246 }

255 void QxtRPCServicePrivate::clientConnected(QIODevice* dev, quint64 id)
256 {

258  QxtMetaObject::connect(dev, SIGNAL(readyRead()),
259  QxtMetaObject::bind(this, SLOT(clientData(quint64)), Q_ARG(quint64, id)));

285 void QxtRPCServicePrivate::clientData(quint64 id)
286 {
287  // Get the device from the connection manager.
288  QIODevice* dev = manager->client(id);
289 
290  // Cache a reference to the buffer.
291  QByteArray& buf = buffers[id];
292 
293  // Read all available data on the device.
294  buf.append(dev->readAll());
295 
296  while(serializer->canDeserialize(buf)) {
297  // Extract one deserialized signal from the buffer.
298  QxtAbstractSignalSerializer::DeserializedData data = serializer->deserialize(buf);
299 
300  // Check to see if it's a blank command.
301  if(serializer->isNoOp(data))
302  continue;
303 
304  // Check for protocol errors.
305  if(serializer->isProtocolError(data)) {
306  qWarning() << "QxtRPCService: Invalid data received; disconnecting";
307  qxt_p().disconnectClient(id);
308  return;
309  }
310 
311  // Pad the arguments to 8, because that's what dispatchFromClient() expects.
312  while(data.second.count() < 8)
313  data.second << QVariant();
314 
315  // And finally, invoke the dispatcher.
316  dispatchFromClient(id, data.first, data.second[0], data.second[1], data.second[2], data.second[3],
317  data.second[4], data.second[5], data.second[6], data.second[7]);
318  }
319 }

358 void QxtRPCServicePrivate::dispatchFromServer(const QString& fn, const QVariant& p0, const QVariant& p1,
359  const QVariant& p2, const QVariant& p3, const QVariant& p4, const QVariant& p5, const QVariant& p6,
360  const QVariant& p7) const
361 {
362  // If the received message is not connected to any slots, ignore it.
363  if(!connectedSlots.contains(fn)) return;
364 
365  foreach(const SlotDef& slot, connectedSlots.value(fn)) {
366  // Look up the parameters for each slot based on its metamethod definition.
367  MetaMethodDef method = qMakePair(slot.recv->metaObject(), slot.slot);
368  const QList<QByteArray>& params = slotParameters.value(method);
369  int numParams = params.count();
370 
371  // Invoke the specified slot on the receiver object using the arguments passed to the function. The
372  // QGenericArgument stuff is done here for safety, as it's not inconceivable (but it IS dangerous) for
373  // different slots to have different parameter lists.
374  if(!QMetaObject::invokeMethod(slot.recv, slot.slot.constData(), slot.type, QXT_ARG(0), QXT_ARG(1), QXT_ARG(2),
375  QXT_ARG(3), QXT_ARG(4), QXT_ARG(5), QXT_ARG(6), QXT_ARG(7))) {
376  qWarning() << "QxtRPCService: invokeMethod for " << slot.recv << "::" << slot.slot << " failed";
377  }
378  }
379 }

664 bool QxtRPCService::attachSlot(const QString& rpcFunction, QObject* recv, const char* slot, Qt::ConnectionType type)
665 {
666  const QMetaObject* meta = recv->metaObject();
667  QByteArray name = QxtMetaObject::methodName(slot);
668  QxtRPCServicePrivate::MetaMethodDef info = qMakePair(meta, name);
669 
670  if(!qxt_d().slotParameters.count(info)) {
671  // This method hasn't been encountered before, so read the metaobject and cache the results.
672  QByteArray norm = QxtMetaObject::methodSignature(slot);
673  int methodID = meta->indexOfMethod(norm.constData());
674  if(methodID < 0) {
675  // indexOfMethod() returns -1 if the method was not found, so report a warning and return an error.
676  qWarning() << "QxtRPCService::attachSlot: " << recv << "::" << slot << " does not exist";
677  return false;
678  }
679 
680  // Look up the method's parameter list, ensure each parameter is queueable, and cache the type IDs.
681  QList<QByteArray> types = meta->method(methodID).parameterTypes();
682  int ct = types.count();
683  for(int i = 0; i < ct; i++) {
684  int typeID = QMetaType::type(types.value(i).constData());
685  if(typeID <= 0) {
686  qWarning() << "QxtRPCService::attachSlot: cannot queue arguments of type " << types.value(i);
687  return false;
688  }
689  }
690 
691  // Cache the looked-up parameter list.
692  qxt_d().slotParameters[info] = types;
693  }
694 
695  // If the RPC function name appears to be a signal or slot, normalize the signature.
696  QString rpcFunc = rpcFunction;
697  if(QxtMetaObject::isSignalOrSlot(rpcFunction.toAscii().constData()))
698  rpcFunc = QxtMetaObject::methodSignature(rpcFunction.toAscii().constData());
699 
700  // Construct a slot definition and associate it with the RPC function name.
701  QxtRPCServicePrivate::SlotDef slotDef;
702  slotDef.recv = recv;
703  slotDef.slot = name;
704  slotDef.type = type;
705  qxt_d().connectedSlots[rpcFunc].append(slotDef);
706 
707  return true;
708 }

730 void QxtRPCService::detachSlots(QObject* obj)
731 {
