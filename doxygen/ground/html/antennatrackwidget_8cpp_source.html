131 void AntennaTrackWidget::calcAntennaPosition(void)
132 {
134  double lat1, lat2, lon1, lon2, a, c, d, x, y, brng;
135  double azimuth, elevation;
136  double gcsAlt=TrackData.HomeAltitude; // Home MSL altitude
137  double uavAlt=TrackData.Altitude; // UAV MSL altitude
138  double dAlt=uavAlt-gcsAlt; // Altitude difference
139 
140  // Convert to radians
141  lat1 = TrackData.HomeLatitude*(M_PI/180); // Home lat
142  lon1 = TrackData.HomeLongitude*(M_PI/180); // Home lon
143  lat2 = TrackData.Latitude*(M_PI/180); // UAV lat
144  lon2 = TrackData.Longitude*(M_PI/180); // UAV lon
145 
146  // Bearing
153  y = sin(lon2-lon1) * cos(lat2);
154  x = cos(lat1) * sin(lat2) - sin(lat1) * cos(lat2) * cos(lon2-lon1);
155  brng = atan2((sin(lon2-lon1)*cos(lat2)),(cos(lat1)*sin(lat2)-sin(lat1)*cos(lat2)*cos(lon2-lon1)))*(180/M_PI);
156  if(brng<0)
157  brng+=360;
158 
159  // bearing to stepper
160  azimuth = brng;
161 
162  // Haversine formula for distance
173  a = sin((lat2-lat1)/2) * sin((lat2-lat1)/2) +
174  cos(lat1) * cos(lat2) *
175  sin((lon2-lon1)/2) * sin((lon2-lon1)/2);
176  c = 2 * atan2(sqrt(a), sqrt(1-a));
177  d = WGS84_RADIUS_EARTH_KM * 1000 * c;
178 
179  // Elevation v depends servo direction
180  if(d!=0)
181  elevation = 90-(atan(dAlt/d)*(180/M_PI));
182  else
183  elevation = 0;
185 
186  QString str3;
187  str3.sprintf("%.0f deg", azimuth);
188  azimuth_value->setText(str3);
189 
190  str3.sprintf("%.0f deg", elevation);
191  elevation_value->setText(str3);
192 
193  //servo value 2000-4000
194  int servo = (int)(2000.0/180*elevation+2000);
195  int stepper = (int)(400.0/360*(azimuth-azimuth_old));
196 
197  // send azimuth and elevation to tracker hardware
198  str3.sprintf("move %d 2000 2000 2000 %d\r", stepper,servo);
199  if(port->isOpen())
200  {
201  if(azimuth_old!=azimuth || elevation!=elevation_old)
202  port->write(str3.toAscii());
203  azimuth_old = azimuth;
204  elevation_old = elevation;
205  }
206 
207 }
