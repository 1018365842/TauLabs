88  // Need to divide these by 100 because that is what the .ui file does
89  // to get the UAVO
90  const double gain_ratio_r = m_autotune->rateTuning->value() / 100.0;
91  const double zero_ratio_r = m_autotune->rateTuning->value() / 100.0;

94 
95  // For now just run over roll and pitch
96  for (int i = 0; i < 2; i++) {
97  if (relayTuningData.Period[i] == 0 || relayTuningData.Gain[i] == 0)
98  continue;
99 
100  double wu = 1000.0 * 2 * M_PI / relayTuningData.Period[i]; // ultimate freq = output osc freq (rad/s)
101 
102  double wc = wu * gain_ratio_r; // target openloop crossover frequency (rad/s)
103  double zc = wc * zero_ratio_r; // controller zero location (rad/s)
104  double kpu = 4.0f / M_PI / relayTuningData.Gain[i]; // ultimate gain, i.e. the proportional gain for instablity
105  double kp = kpu * gain_ratio_r; // proportional gain
106  double ki = zc * kp; // integral gain
107 
108  // Now calculate gains for the next loop out knowing it is the integral of
109  // the inner loop -- the plant is position/velocity = scale*1/s
110  double wc2 = wc * gain_ratio_p; // crossover of the attitude loop
111  double kp2 = wc2; // kp of attitude
112  double ki2 = wc2 * zero_ratio_p * kp2; // ki of attitude
113 
114  switch(i) {
115  case 0: // Roll
116 
117  stabSettings.RollRatePID[StabilizationSettings::ROLLRATEPID_KP] = kp;
118  stabSettings.RollRatePID[StabilizationSettings::ROLLRATEPID_KI] = ki;
119  stabSettings.RollPI[StabilizationSettings::ROLLPI_KP] = kp2;
120  stabSettings.RollPI[StabilizationSettings::ROLLPI_KI] = ki2;
121  break;
122  case 1: // Pitch
123  stabSettings.PitchRatePID[StabilizationSettings::PITCHRATEPID_KP] = kp;
124  stabSettings.PitchRatePID[StabilizationSettings::PITCHRATEPID_KI] = ki;
125  stabSettings.PitchPI[StabilizationSettings::PITCHPI_KP] = kp2;
126  stabSettings.PitchPI[StabilizationSettings::PITCHPI_KI] = ki2;
127  break;
128  }
129  }
130 
131  // Display these computed settings
