35 bool QxtCheckComboBoxPrivate::eventFilter(QObject* receiver, QEvent* event)
36 {
37  switch (event->type())
38  {
39  case QEvent::KeyPress:
40  case QEvent::KeyRelease:
41  {
42  QKeyEvent* keyEvent = static_cast<QKeyEvent*>(event);
43  if (receiver == &qxt_p() && (keyEvent->key() == Qt::Key_Up || keyEvent->key() == Qt::Key_Down))
44  {
45  qxt_p().showPopup();
46  return true;
47  }
48  else if (keyEvent->key() == Qt::Key_Enter ||
49  keyEvent->key() == Qt::Key_Return ||
50  keyEvent->key() == Qt::Key_Escape)
51  {
52  // it is important to call QComboBox implementation
53  qxt_p().QComboBox::hidePopup();
54  if (keyEvent->key() != Qt::Key_Escape)
55  return true;
56  }
57  }
58  case QEvent::MouseButtonPress:
59  containerMousePress = (receiver == qxt_p().view()->window());
60  break;
61  case QEvent::MouseButtonRelease:
62  containerMousePress = false;;
63  break;
64  default:
65  break;
66  }
67  return false;
68 }

208 QStringList QxtCheckComboBox::checkedItems() const
209 {
210  QStringList items;
211  if (model())
212  {
213  QModelIndex index = model()->index(0, modelColumn(), rootModelIndex());
214  QModelIndexList indexes = model()->match(index, Qt::CheckStateRole, Qt::Checked, -1, Qt::MatchExactly);
215  foreach(const QModelIndex& index, indexes)
216  items += index.data().toString();
217  }
218  return items;
219 }
