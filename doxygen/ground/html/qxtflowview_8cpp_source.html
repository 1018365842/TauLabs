315 void QxtFlowView::showPrevious()
316 {
317  int step = d->animator->step;
318  int center = d->state->centerIndex;
319 
320  if (step > 0)
321  d->animator->start(center);
322 
323  if (step == 0)
324  if (center > 0)
325  d->animator->start(center - 1);
326 
327  if (step < 0)
328  d->animator->target = qMax(0, center - 2);
329 }

332 void QxtFlowView::showNext()
333 {
334  int step = d->animator->step;
335  int center = d->state->centerIndex;
336 
337  if (step < 0)
338  d->animator->start(center);
339 
340  if (step == 0)
341  if (center < d->state->slideImages.count() - 1)
342  d->animator->start(center + 1);
343 
344  if (step > 0)
345  d->animator->target = qMin(center + 2, d->state->slideImages.count() - 1);
346 }

429 void QxtFlowView::wheelEvent(QWheelEvent * event)
430 {
431 
432  if (event->orientation() == Qt::Horizontal)
433  {
434  event->ignore();
435  }
436  else
437  {
438  int numSteps = -((event->delta() / 8) / 15);
439 
440 
441 
442  if (numSteps > 0)
443  {
444  for (int i = 0;i < numSteps;i++)
445  {
446  showNext();
447  }
448  }
449  else
450  {
451  for (int i = numSteps;i < 0;i++)
452  {
453  showPrevious();
454  }
455  }
456  event->accept();
457  }

733 {

735  if (model)
736  {
737  for (int i = 0;i < model->rowCount(rootindex);i++)
738  {
739  QModelIndex idx = model->index(i, piccolumn, rootindex);
740  insertSlide(i, qvariant_cast<QImage>(model->data(idx, picrole)));
