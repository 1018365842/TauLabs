93 template<typename MatrixType>
94 void LDLT<MatrixType>::compute(const MatrixType& a)
95 {
96  assert(a.rows()==a.cols());
97  const int size = a.rows();
98  m_matrix.resize(size, size);
99  m_isPositiveDefinite = true; // always true. This decomposition is not rank-revealing anyway.
100 
101  if (size<=1)
102  {
103  m_matrix = a;
104  return;
105  }
106 
107  // Let's preallocate a temporay vector to evaluate the matrix-vector product into it.
108  // Unlike the standard LLT decomposition, here we cannot evaluate it to the destination
109  // matrix because it a sub-row which is not compatible suitable for efficient packet evaluation.
110  // (at least if we assume the matrix is col-major)
111  Matrix<Scalar,MatrixType::RowsAtCompileTime,1> _temporary(size);
112 
113  // Note that, in this algorithm the rows of the strict upper part of m_matrix is used to store
114  // column vector, thus the strange .conjugate() and .transpose()...
115 
116  m_matrix.row(0) = a.row(0).conjugate();
117  m_matrix.col(0).end(size-1) = m_matrix.row(0).end(size-1) / m_matrix.coeff(0,0);
118  for (int j = 1; j < size; ++j)
119  {
120  RealScalar tmp = ei_real(a.coeff(j,j) - (m_matrix.row(j).start(j) * m_matrix.col(j).start(j).conjugate()).coeff(0,0));
121  m_matrix.coeffRef(j,j) = tmp;
122 
123  int endSize = size-j-1;
124  if (endSize>0)
125  {
126  _temporary.end(endSize) = ( m_matrix.block(j+1,0, endSize, j)
127  * m_matrix.col(j).start(j).conjugate() ).lazy();
128 
129  m_matrix.row(j).end(endSize) = a.row(j).end(endSize).conjugate()
130  - _temporary.end(endSize).transpose();
131 
132  if(tmp != RealScalar(0))
133  m_matrix.col(j).end(endSize) = m_matrix.row(j).end(endSize) / tmp;
134  }
135  }
136 }

168 template<typename MatrixType>
169 template<typename Derived>
170 bool LDLT<MatrixType>::solveInPlace(MatrixBase<Derived> &bAndX) const
171 {
172  const int size = m_matrix.rows();
173  ei_assert(size==bAndX.rows());
174  if (!m_isPositiveDefinite)
175  return false;
176  matrixL().solveTriangularInPlace(bAndX);
177  bAndX = (m_matrix.cwise().inverse().template part<Diagonal>() * bAndX).lazy();
178  m_matrix.adjoint().template part<UnitUpperTriangular>().solveTriangularInPlace(bAndX);
179  return true;
180 }
