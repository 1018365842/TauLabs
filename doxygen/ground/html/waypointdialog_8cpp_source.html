110 void WaypointDialog::setupModeWidgets()

113  switch(mode)

115  case Waypoint::MODE_FLYCIRCLERIGHT:

190 void WaypointDialog::enableEditWidgets()
191 {
192  int row = itemSelection->currentIndex().row();
193  bool value = model->data(model->index(row,FlightDataModel::LOCKED)).toBool();
194  QWidget * w;
195  foreach(QWidget * obj,this->findChildren<QWidget *>())
196  {
197  w=qobject_cast<QComboBox*>(obj);
198  if(w)
199  w->setEnabled(!value);
200  w=qobject_cast<QLineEdit*>(obj);
201  if(w)
202  w->setEnabled(!value);
203  w=qobject_cast<QDoubleSpinBox*>(obj);
204  if(w)
205  w->setEnabled(!value);
206  w=qobject_cast<QSpinBox*>(obj);
207  if(w)
208  w->setEnabled(!value);
209  }
