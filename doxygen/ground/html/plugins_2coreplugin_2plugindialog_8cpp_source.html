101 void PluginDialog::openDetails(ExtensionSystem::PluginSpec *spec)
102 {
103  if (!spec)
104  return;
105  QDialog dialog(this);
106  dialog.setWindowTitle(tr("Plugin Details of %1").arg(spec->name()));
107  QVBoxLayout *layout = new QVBoxLayout;
108  dialog.setLayout(layout);
109  ExtensionSystem::PluginDetailsView *details = new ExtensionSystem::PluginDetailsView(&dialog);
110  layout->addWidget(details);
111  details->update(spec);
112  QDialogButtonBox *buttons = new QDialogButtonBox(QDialogButtonBox::Close, Qt::Horizontal, &dialog);
113  layout->addWidget(buttons);
114  connect(buttons, SIGNAL(accepted()), &dialog, SLOT(accept()));
115  connect(buttons, SIGNAL(rejected()), &dialog, SLOT(reject()));
116  dialog.resize(400, 500);
117  dialog.exec();
118 }
