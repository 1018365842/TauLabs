52 bool QxtGlobalShortcutPrivate::eventFilter(void* message)
53 {
54  XEvent* event = static_cast<XEvent*>(message);
55  if (event->type == KeyPress)
56  {
57  XKeyEvent* key = (XKeyEvent*) event;
58  activateShortcut(key->keycode,
59  // Mod1Mask == Alt, Mod4Mask == Meta
60  key->state & (ShiftMask | ControlMask | Mod1Mask | Mod4Mask));
61  }
62  return false;
63 }

88 bool QxtGlobalShortcutPrivate::registerShortcut(quint32 nativeKey, quint32 nativeMods)
89 {
90  Display* display = QX11Info::display();
91  Window window = QX11Info::appRootWindow();
92  Bool owner = True;
93  int pointer = GrabModeAsync;
94  int keyboard = GrabModeAsync;
95  error = false;
96  original_x_errhandler = XSetErrorHandler(qxt_x_errhandler);
97  XGrabKey(display, nativeKey, nativeMods, window, owner, pointer, keyboard);
98  XGrabKey(display, nativeKey, nativeMods | Mod2Mask, window, owner, pointer, keyboard); // allow numlock
99  XSync(display, False);
100  XSetErrorHandler(original_x_errhandler);
101  return !error;
102 }
