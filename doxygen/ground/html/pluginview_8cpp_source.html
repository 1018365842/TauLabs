114 void PluginView::updateList()
115 {
116  static QIcon okIcon(":/extensionsystem/images/ok.png");
117  static QIcon errorIcon(":/extensionsystem/images/error.png");
118  QList<QTreeWidgetItem *> items;
119  QTreeWidgetItem *currentItem = 0;
120  PluginSpec *currPlugin = currentPlugin();
121  foreach (PluginSpec *spec, p->manager->plugins()) {
122  QTreeWidgetItem *item = new QTreeWidgetItem(QStringList()
123  << ""
124  << spec->name()
125  << QString("%1 (%2)").arg(spec->version()).arg(spec->compatVersion())
126  << spec->vendor()
127  << QDir::toNativeSeparators(spec->filePath()));
128  item->setToolTip(4, QDir::toNativeSeparators(spec->filePath()));
129  item->setIcon(0, spec->hasError() ? errorIcon : okIcon);
130  item->setData(0, Qt::UserRole, qVariantFromValue(spec));
131  items.append(item);
132  if (currPlugin == spec)
133  currentItem = item;
134  }
135  m_ui->pluginWidget->clear();
136  if (!items.isEmpty())
137  m_ui->pluginWidget->addTopLevelItems(items);
138  if (currentItem)
139  m_ui->pluginWidget->setCurrentItem(currentItem);
140  else if (!items.isEmpty())
141  m_ui->pluginWidget->setCurrentItem(items.first());
142 }
