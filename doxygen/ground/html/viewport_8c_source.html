37 Lib3dsBool
38 lib3ds_viewport_read(Lib3dsViewport *viewport, Lib3dsIo *io)
39 {
40  Lib3dsChunk c;
41  Lib3dsWord chunk;
42 
43  if (!lib3ds_chunk_read_start(&c, 0, io)) {
44  return(LIB3DS_FALSE);
45  }
46 
47  switch (c.chunk) {
48  case LIB3DS_VIEWPORT_LAYOUT:
49  {
50  int cur=0;
51  viewport->layout.style=lib3ds_io_read_word(io);
52  viewport->layout.active=lib3ds_io_read_intw(io);
53  lib3ds_io_read_intw(io);
54  viewport->layout.swap=lib3ds_io_read_intw(io);
55  lib3ds_io_read_intw(io);
56  viewport->layout.swap_prior=lib3ds_io_read_intw(io);
57  viewport->layout.swap_view=lib3ds_io_read_intw(io);
58  lib3ds_chunk_read_tell(&c, io);
59  while ((chunk=lib3ds_chunk_read_next(&c, io))!=0) {
60  switch (chunk) {
61  case LIB3DS_VIEWPORT_SIZE:
62  {
63  viewport->layout.position[0]=lib3ds_io_read_word(io);
64  viewport->layout.position[1]=lib3ds_io_read_word(io);
65  viewport->layout.size[0]=lib3ds_io_read_word(io);
66  viewport->layout.size[1]=lib3ds_io_read_word(io);
67  }
68  break;
69  case LIB3DS_VIEWPORT_DATA_3:
70  {
71  lib3ds_viewport_set_views(viewport,cur+1);
72  lib3ds_io_read_intw(io);
73  viewport->layout.viewL[cur].axis_lock=lib3ds_io_read_word(io);
74  viewport->layout.viewL[cur].position[0]=lib3ds_io_read_intw(io);
75  viewport->layout.viewL[cur].position[1]=lib3ds_io_read_intw(io);
76  viewport->layout.viewL[cur].size[0]=lib3ds_io_read_intw(io);
77  viewport->layout.viewL[cur].size[1]=lib3ds_io_read_intw(io);
78  viewport->layout.viewL[cur].type=lib3ds_io_read_word(io);
79  viewport->layout.viewL[cur].zoom=lib3ds_io_read_float(io);
80  lib3ds_io_read_vector(io, viewport->layout.viewL[cur].center);
81  viewport->layout.viewL[cur].horiz_angle=lib3ds_io_read_float(io);
82  viewport->layout.viewL[cur].vert_angle=lib3ds_io_read_float(io);
83  lib3ds_io_read(io, viewport->layout.viewL[cur].camera, 11);
84  ++cur;
85  }
86  break;
87  case LIB3DS_VIEWPORT_DATA:
88  /* 3DS R2 & R3 chunk
89  unsupported */
90  break;
91  default:
92  lib3ds_chunk_unknown(chunk);
93  }
94  }
95  }
96  break;
97  case LIB3DS_DEFAULT_VIEW:
98  {
99  memset(&viewport->default_view,0,sizeof(Lib3dsDefaultView));
100  while ((chunk=lib3ds_chunk_read_next(&c, io))!=0) {
101  switch (chunk) {
102  case LIB3DS_VIEW_TOP:
103  {
104  viewport->default_view.type=LIB3DS_VIEW_TYPE_TOP;
105  lib3ds_io_read_vector(io, viewport->default_view.position);
106  viewport->default_view.width=lib3ds_io_read_float(io);
107  }
108  break;
109  case LIB3DS_VIEW_BOTTOM:
110  {
111  viewport->default_view.type=LIB3DS_VIEW_TYPE_BOTTOM;
112  lib3ds_io_read_vector(io, viewport->default_view.position);
113  viewport->default_view.width=lib3ds_io_read_float(io);
114  }
115  break;
116  case LIB3DS_VIEW_LEFT:
117  {
118  viewport->default_view.type=LIB3DS_VIEW_TYPE_LEFT;
119  lib3ds_io_read_vector(io, viewport->default_view.position);
120  viewport->default_view.width=lib3ds_io_read_float(io);
121  }
122  break;
123  case LIB3DS_VIEW_RIGHT:
124  {
125  viewport->default_view.type=LIB3DS_VIEW_TYPE_RIGHT;
126  lib3ds_io_read_vector(io, viewport->default_view.position);
127  viewport->default_view.width=lib3ds_io_read_float(io);
128  }
129  break;
130  case LIB3DS_VIEW_FRONT:
131  {
132  viewport->default_view.type=LIB3DS_VIEW_TYPE_FRONT;
133  lib3ds_io_read_vector(io, viewport->default_view.position);
134  viewport->default_view.width=lib3ds_io_read_float(io);
135  }
136  break;
137  case LIB3DS_VIEW_BACK:
138  {
139  viewport->default_view.type=LIB3DS_VIEW_TYPE_BACK;
140  lib3ds_io_read_vector(io, viewport->default_view.position);
141  viewport->default_view.width=lib3ds_io_read_float(io);
142  }
143  break;
144  case LIB3DS_VIEW_USER:
145  {
146  viewport->default_view.type=LIB3DS_VIEW_TYPE_USER;
147  lib3ds_io_read_vector(io, viewport->default_view.position);
148  viewport->default_view.width=lib3ds_io_read_float(io);
149  viewport->default_view.horiz_angle=lib3ds_io_read_float(io);
150  viewport->default_view.vert_angle=lib3ds_io_read_float(io);
151  viewport->default_view.roll_angle=lib3ds_io_read_float(io);
152  }
153  break;
154  case LIB3DS_VIEW_CAMERA:
155  {
156  viewport->default_view.type=LIB3DS_VIEW_TYPE_CAMERA;
157  lib3ds_io_read(io, viewport->default_view.camera, 11);
158  }
159  break;
160  default:
161  lib3ds_chunk_unknown(chunk);
162  }
163  }
164  }
165  break;
166  }
167 
168  lib3ds_chunk_read_end(&c, io);
169  return(LIB3DS_TRUE);
170 }

177 lib3ds_viewport_set_views(Lib3dsViewport *viewport, Lib3dsDword views)
178 {
179  ASSERT(viewport);
180  if (viewport->layout.views) {
181  if (views) {
182  viewport->layout.views=views;
183  viewport->layout.viewL=(Lib3dsView*)realloc(viewport->layout.viewL, sizeof(Lib3dsView)*views);
184  }
185  else {
186  free(viewport->layout.viewL);
187  viewport->layout.views=0;
188  viewport->layout.viewL=0;
189  }
190  }
191  else {
192  if (views) {
193  viewport->layout.views=views;
194  viewport->layout.viewL=(Lib3dsView*)calloc(sizeof(Lib3dsView),views);
195  }
196  }
