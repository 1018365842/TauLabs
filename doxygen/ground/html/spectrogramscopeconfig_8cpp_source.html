209 void SpectrogramScopeConfig::loadConfiguration(ScopeGadgetWidget *scopeGadgetWidget)
210 {
211  preparePlot(scopeGadgetWidget);
212  scopeGadgetWidget->setScope(this);
213  scopeGadgetWidget->startTimer(m_refreshInterval);
214 
215  //There should be only one spectrogram per plot //TODO: Upgrade this to handle multiple spectrograms on a single axis
216  if ( m_spectrogramSourceConfigs.length() != 1)
217  return;
218 
219  Plot3dCurveConfiguration* spectrogramSourceConfigs = m_spectrogramSourceConfigs.front();
220  QString uavObjectName = spectrogramSourceConfigs->uavObjectName;
221  QString uavFieldName = spectrogramSourceConfigs->uavFieldName;
222 
223  // Get and store the units
224  units = getUavObjectFieldUnits(uavObjectName, uavFieldName);
225 
226  SpectrogramData* spectrogramData = new SpectrogramData(uavObjectName, uavFieldName, samplingFrequency, windowWidth, timeHorizon);
227  spectrogramData->setXMinimum(0);
228  spectrogramData->setXMaximum(samplingFrequency/2);
229  spectrogramData->setYMinimum(0);
230  spectrogramData->setYMaximum(timeHorizon);
231  spectrogramData->setZMaximum(zMaximum);
232  spectrogramData->setScalePower(spectrogramSourceConfigs->yScalePower);
233  spectrogramData->setMeanSamples(spectrogramSourceConfigs->yMeanSamples);
234  spectrogramData->setMathFunction(spectrogramSourceConfigs->mathFunction);
235 
236  //Generate the waterfall name
237  QString waterfallName = (spectrogramData->getUavoName()) + "." + (spectrogramData->getUavoFieldName());
238  if(spectrogramData->getHaveSubFieldFlag())
239  waterfallName = waterfallName.append("." + spectrogramData->getUavoSubFieldName());
240 
241  //Get the uav object
242  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
243  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
244  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(objManager->getObject((spectrogramData->getUavoName())));
245  if(!obj) {
246  qDebug() << "Object " << spectrogramData->getUavoName() << " is missing";
247  return;
248  }
249 
250  //Get the units
251  QString units = getUavObjectFieldUnits(spectrogramData->getUavoName(), spectrogramData->getUavoFieldName());
252 
253  //Generate name with scaling factor appeneded
254  QString waterfallNameScaled;
255  if(spectrogramSourceConfigs->yScalePower == 0)
256  waterfallNameScaled = waterfallName + "(" + units + ")";
257  else
258  waterfallNameScaled = waterfallName + "(x10^" + QString::number(spectrogramSourceConfigs->yScalePower) + " " + units + ")";
259 
260  //Create the waterfall plot
261  QwtPlotSpectrogram* plotSpectrogram = new QwtPlotSpectrogram(waterfallNameScaled);
262  plotSpectrogram->setRenderThreadCount( 0 ); // use system specific thread count
263  plotSpectrogram->setRenderHint(QwtPlotItem::RenderAntialiased);
264  plotSpectrogram->setColorMap(new ColorMap(colorMapType) );
265 
266  // Initial raster data
267 
268  QDateTime NOW = QDateTime::currentDateTime(); //TODO: Upgrade this to show UAVO time and not system time
269  for ( uint i = 0; i < timeHorizon; i++ ){
270  spectrogramData->timeDataHistory->append(NOW.toTime_t() + NOW.time().msec() / 1000.0 + i);
271  }
272 
273  if (((double) windowWidth) * timeHorizon < (double) 10000000.0 * sizeof(spectrogramData->zDataHistory->front())){ //Don't exceed 10MB for memory
274  for ( uint i = 0; i < windowWidth*timeHorizon; i++ ){
275  spectrogramData->zDataHistory->append(0);
276  }
277  }
278  else{
279  qDebug() << "For some reason, we're trying to allocate a gigantic spectrogram. This probably represents a problem in the configuration file. TimeHorizion: "<< timeHorizon << ", windowWidth: "<< windowWidth;
280  Q_ASSERT(0);
281  return;
282  }
283 
284  //Set up colorbar on right axis
285  spectrogramData->rightAxis = scopeGadgetWidget->axisWidget( QwtPlot::yRight );
286  spectrogramData->rightAxis->setTitle( "Intensity" );
287  spectrogramData->rightAxis->setColorBarEnabled( true );
288  spectrogramData->rightAxis->setColorMap( QwtInterval(0, zMaximum), new ColorMap(colorMapType));
289  scopeGadgetWidget->setAxisScale( QwtPlot::yRight, 0, zMaximum);
290  scopeGadgetWidget->enableAxis( QwtPlot::yRight );
291 
292  plotSpectrogram->setData(spectrogramData->getRasterData());
293 
294  plotSpectrogram->attach(scopeGadgetWidget);
295  spectrogramData->setSpectrogram(plotSpectrogram);
296 
297  //Keep the curve details for later
298  scopeGadgetWidget->insertDataSources(waterfallNameScaled, spectrogramData);
299 
300  // Connect the UAVO
301  scopeGadgetWidget->connectUAVO(obj);
302 
303  mutex.lock();
304  scopeGadgetWidget->replot();
305  mutex.unlock();
306 }
