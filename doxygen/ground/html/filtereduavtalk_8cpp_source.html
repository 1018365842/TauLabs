63 bool FilteredUavTalk::receiveObject(quint8 type, quint32 objId, quint16 instId, quint8 *data, qint32 length)
64 {
65  Q_UNUSED(length);
66  UAVObject* obj = NULL;
67  bool error = false;
68  bool allInstances = (instId == ALL_INSTANCES);
69  UavTalkRelayComon::accessType access=m_rules.value(objId,m_defaultRule);
70  if(access==UavTalkRelayComon::ReadOnly || access==UavTalkRelayComon::None)
71  return false;
72  if (obj == NULL)
73  error = true;
74  // Process message type
75  switch (type) {
76  case TYPE_OBJ: // We have received an object.
77  // All instances, not allowed for OBJ messages
78  if (!allInstances)
79  {
80  // Get object and update its data
81  UAVObject* tobj = objMngr->getObject(objId);
82  bool wasCon=disconnect(tobj, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(sendObjectSlot(UAVObject*)));
83  obj = updateObject(objId, instId, data);
84  if(wasCon)
85  connect(tobj, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(sendObjectSlot(UAVObject*)));
86  UAVMetaObject * mobj=dynamic_cast<UAVMetaObject*>(tobj);
87  if(mobj)
88  tobj->updated();
89  if (obj == NULL)
90  error = true;
91  }
92  else
93  {
94  error = true;
95  }
96  break;
97  case TYPE_OBJ_ACK: // We have received an object and are asked for an ACK
98  // All instances, not allowed for OBJ_ACK messages
99  if (!allInstances)
100  {
101  // Get object and update its data
102  UAVObject* tobj = objMngr->getObject(objId);
103  bool wasCon=disconnect(tobj, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(sendObjectSlot(UAVObject*)));
104  obj = updateObject(objId, instId, data);
105  if(wasCon)
106  connect(tobj, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(sendObjectSlot(UAVObject*)));
107  UAVMetaObject * mobj=dynamic_cast<UAVMetaObject*>(tobj);
108  if(mobj)
109  tobj->updated();
110  // Transmit ACK
111  if ( obj != NULL )
112  {
113  transmitObject(obj, TYPE_ACK, false);
114  }
115  else
116  {
117  error = true;
118  }
119  }
120  else
121  {
122  error = true;
123  }
124  break;
125  case TYPE_OBJ_REQ: // We are being asked for an object
126  // Get object, if all instances are requested get instance 0 of the object
127  if(access==UavTalkRelayComon::WriteOnly || access==UavTalkRelayComon::None)
128  break;
129  if (allInstances)
130  {
131  obj = objMngr->getObject(objId);
132  }
133  else
134  {
135  obj = objMngr->getObject(objId, instId);
136  }
137  // If object was found transmit it
138  if (obj != NULL)
139  {
140  transmitObject(obj, TYPE_OBJ, allInstances);
141  }
142  else
143  {
144  // Object was not found, transmit a NACK with the
145  // objId which was not found.
146  transmitNack(objId);
147  error = true;
148  }
149  break;
150  case TYPE_NACK: // We have received a NACK for an object that does not exist on the far end.
151  // (but should exist on our end)
152  // All instances, not allowed for NACK messages
153  if (!allInstances)
154  {
155  // Get object
156  obj = objMngr->getObject(objId, instId);
157  // Check if object exists:
158  if (obj != NULL)
159  {
160  emit nackReceived(obj);
161  }
162  else
163  {
164  error = true;
165  }
166  }
167  break;
168  case TYPE_ACK: // We have received a ACK, supposedly after sending an object with OBJ_ACK
169  // All instances, not allowed for ACK messages
170  if (!allInstances)
171  {
172  // Get object
173  obj = objMngr->getObject(objId, instId);
174  // Check if we actually know this object (tiny chance the ObjID
175  // could be unknown and got through CRC check...)
176  if (obj != NULL)
177  {
178  emit ackReceived(obj);
179  }
180  else
181  {
182  error = true;
183  }
184  }
185  break;
186  default:
187  error = true;
188  }
189  // Done (exit value is "success", hence the "!" below)
190  return !error;
191 }
