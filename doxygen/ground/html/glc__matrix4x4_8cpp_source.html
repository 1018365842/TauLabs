77 {
78  double angle_x;

84 
85  if (!qFuzzyCompare(C, 0.0)) // Gimball lock?
86  {
87  tracex= m_Matrix[10] / C;

89  angle_x= atan2( tracey, tracex);
90 
91  tracex= m_Matrix[0] / C;
92  tracey= - m_Matrix[4] / C;
93  angle_z= atan2( tracey, tracex);
94  }
95  else // Gimball lock?
