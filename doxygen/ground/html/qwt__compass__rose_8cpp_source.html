131 void QwtSimpleCompassRose::drawRose(
132  QPainter *painter,
133  const QPalette &palette,
134  const QPointF &center, double radius, double north, double width,
135  int numThorns, int numThornLevels, double shrinkFactor )
136 {
137  if ( numThorns < 4 )
138  numThorns = 4;
139 
140  if ( numThorns % 4 )
141  numThorns += 4 - numThorns % 4;
142 
143  if ( numThornLevels <= 0 )
144  numThornLevels = numThorns / 4;
145 
146  if ( shrinkFactor >= 1.0 )
147  shrinkFactor = 1.0;
148 
149  if ( shrinkFactor <= 0.5 )
150  shrinkFactor = 0.5;
151 
152  painter->save();
153 
154  painter->setPen( Qt::NoPen );
155 
156  for ( int j = 1; j <= numThornLevels; j++ )
157  {
158  double step = qPow( 2.0, j ) * M_PI / numThorns;
159  if ( step > M_PI_2 )
160  break;
161 
162  double r = radius;
163  for ( int k = 0; k < 3; k++ )
164  {
165  if ( j + k < numThornLevels )
166  r *= shrinkFactor;
167  }
168 
169  double leafWidth = r * width;
170  if ( 2.0 * M_PI / step > 32 )
171  leafWidth = 16;
172 
173  const double origin = north / 180.0 * M_PI;
174  for ( double angle = origin;
175  angle < 2.0 * M_PI + origin; angle += step )
176  {
177  const QPointF p = qwtPolar2Pos( center, r, angle );
178  const QPointF p1 = qwtPolar2Pos( center, leafWidth, angle + M_PI_2 );
179  const QPointF p2 = qwtPolar2Pos( center, leafWidth, angle - M_PI_2 );
180  const QPointF p3 = qwtPolar2Pos( center, r, angle + step / 2.0 );
181  const QPointF p4 = qwtPolar2Pos( center, r, angle - step / 2.0 );
182 
183  QPainterPath darkPath;
184  darkPath.moveTo( center );
185  darkPath.lineTo( p );
186  darkPath.lineTo( qwtIntersection( center, p3, p1, p ) );
187 
188  painter->setBrush( palette.brush( QPalette::Dark ) );
189  painter->drawPath( darkPath );
190 
191  QPainterPath lightPath;
192  lightPath.moveTo( center );
193  lightPath.lineTo( p );
194  lightPath.lineTo( qwtIntersection( center, p4, p2, p ) );
195 
196  painter->setBrush( palette.brush( QPalette::Light ) );
197  painter->drawPath( lightPath );
198  }
199  }
200  painter->restore();
201 }
