306  void Core::SetMapType(const MapType::Types &value)
307  {
308 
309  if(value != GetMapType())
310  {
311  mapType = value;
312 
313  switch(value)
314  {
315 
316 
317  case MapType::ArcGIS_Map:
318  case MapType::ArcGIS_Satellite:
319  case MapType::ArcGIS_ShadedRelief:
320  case MapType::ArcGIS_Terrain:
321  {
322  if(Projection()->Type()!="PlateCarreeProjection")
323  {
324  SetProjection(new PlateCarreeProjection());
325  }
326  maxzoom=13;
327  }
328  break;
329  case MapType::ArcGIS_MapsLT_Map_Hybrid:
330  case MapType::ArcGIS_MapsLT_Map_Labels:
331  case MapType::ArcGIS_MapsLT_Map:
332  case MapType::ArcGIS_MapsLT_OrtoFoto:
333  {
334  if(Projection()->Type()!="LKS94Projection")
335  {
336  SetProjection(new LKS94Projection());
337  }
338  maxzoom=11;
339  }
340  break;
341 
342  case MapType::PergoTurkeyMap:
343  {
344  if(Projection()->Type()!="PlateCarreeProjectionPergo")
345  {
346  SetProjection(new PlateCarreeProjectionPergo());
347  }
348  maxzoom=17;
349  }
350  break;
351 
352  case MapType::YandexMapRu:
353  {
354  if(Projection()->Type()!="MercatorProjectionYandex")
355  {
356  SetProjection(new MercatorProjectionYandex());
357  }
358  maxzoom=13;
359  }
360  break;
361 
362  case MapType::UserImage:
363  {
364  if(Projection()->Type()!="MercatorProjection")
365  {
366  SetProjection(new MercatorProjection());
367  }
368  maxzoom=32;
369  }
370  break;
371  default:
372  {
373  if(Projection()->Type()!="MercatorProjection")
374  {
375  SetProjection(new MercatorProjection());
376  }
377  maxzoom=21;
378  }
379  break;
380  }
381 
382  //Calculate the number of bits required to hold the tile size
383  quint8 numBits=1;
384  while( (1 << (numBits-1)) < projection->TileSize().Width()){
385  numBits++;
386  }
387 
388  //Ensure that no matter what the zoom can never exceed the number of bits required to display it
389  if (numBits + maxzoom > sizeof(((core::Point *) 0)->X())*8 - 1){ //Remove one because of the sign bit.
390  maxzoom = sizeof(((core::Point *) 0)->X())*8 - 1 - numBits;
391  }
392 
393  minOfTiles = Projection()->GetTileMatrixMinXY(Zoom());
394  maxOfTiles = Projection()->GetTileMatrixMaxXY(Zoom());
395  SetCurrentPositionGPixel(Projection()->FromLatLngToPixel(CurrentPosition(), Zoom()));
396 
397  if(started)
398  {
399  CancelAsyncTasks();
400  OnMapSizeChanged(Width, Height);
401  GoToCurrentPosition();
402  ReloadMap();
403  GoToCurrentPosition();
404  emit OnMapTypeChanged(value);
405 
406  }
407  }
