214 QAction *SavedAction::updatedAction(const QString &text0)
215 {
216  QString text = text0;
217  bool enabled = true;
218  if (!m_textPattern.isEmpty()) {
219  if (text.isEmpty()) {
220  text = m_textPattern;
221  text.remove("\"%1\"");
222  text.remove("%1");
223  enabled = false;
224  } else {
225  text = m_textPattern.arg(text0);
226  }
227  }
228  this->setEnabled(enabled);
229  this->setData(text0);
230  this->setText(text);
231  return this;

240 void SavedAction::readSettings(QSettings *settings)
241 {
242  if (m_settingsGroup.isEmpty() || m_settingsKey.isEmpty())
243  return;
244  settings->beginGroup(m_settingsGroup);
245  QVariant var = settings->value(m_settingsKey, m_defaultValue);
246  // work around old ini files containing @Invalid() entries
247  if (isCheckable() && !var.isValid())
248  var = false;
249  setValue(var);
250  //qDebug() << "READING: " << var.isValid() << m_settingsKey << " -> " << m_value
251  // << " (default: " << m_defaultValue << ")" << var;
252  settings->endGroup();
253 }

271 /*
272  A \c SavedAction can be connected to a widget, typically a
273  checkbox, radiobutton, or a lineedit in some configuration dialog.
274 
275  The widget will retrieve its contents from the SavedAction's
276  value, and - depending on the \a ApplyMode - either write
277  changes back immediately, or when \s SavedAction::apply()
278  is called explicitly.
279 
280  \sa apply(), disconnectWidget()
281 */
282 void SavedAction::connectWidget(QWidget *widget, ApplyMode applyMode)
283 {
284  QTC_ASSERT(!m_widget,
285  qDebug() << "ALREADY CONNECTED: " << widget << m_widget << toString(); return);
286  m_widget = widget;
287  m_applyMode = applyMode;
288 
289  if (QAbstractButton *button = qobject_cast<QAbstractButton *>(widget)) {
290  if (button->isCheckable()) {
291  button->setChecked(m_value.toBool());
292  connect(button, SIGNAL(clicked(bool)),
293  this, SLOT(checkableButtonClicked(bool)));
294  } else {
295  connect(button, SIGNAL(clicked()),
296  this, SLOT(uncheckableButtonClicked()));
297  }
298  } else if (QSpinBox *spinBox = qobject_cast<QSpinBox *>(widget)) {
299  spinBox->setValue(m_value.toInt());
300  //qDebug() << "SETTING VALUE" << spinBox->value();
301  connect(spinBox, SIGNAL(valueChanged(int)),
302  this, SLOT(spinBoxValueChanged(int)));
303  connect(spinBox, SIGNAL(valueChanged(QString)),
304  this, SLOT(spinBoxValueChanged(QString)));
305  } else if (QDoubleSpinBox *doubleSpinBox = qobject_cast<QDoubleSpinBox *>(widget)) {
306  doubleSpinBox->setValue(m_value.toDouble());
307  //qDebug() << "SETTING VALUE" << doubleSpinBox->value();
308  connect(doubleSpinBox, SIGNAL(valueChanged(double)),
309  this, SLOT(doubleSpinBoxValueChanged(double)));
310  connect(doubleSpinBox, SIGNAL(valueChanged(QString)),
311  this, SLOT(doubleSpinBoxValueChanged(QString)));
312  } else if (QLineEdit *lineEdit = qobject_cast<QLineEdit *>(widget)) {
313  lineEdit->setText(m_value.toString());
314  //qDebug() << "SETTING TEXT" << lineEdit->text();
315  connect(lineEdit, SIGNAL(editingFinished()),
316  this, SLOT(lineEditEditingFinished()));
317  } else if (PathChooser *pathChooser = qobject_cast<PathChooser *>(widget)) {
318  pathChooser->setPath(m_value.toString());
319  connect(pathChooser, SIGNAL(editingFinished()),
320  this, SLOT(pathChooserEditingFinished()));
321  connect(pathChooser, SIGNAL(browsingFinished()),
322  this, SLOT(pathChooserEditingFinished()));
323  } else {
324  qDebug() << "Cannot connect widget " << widget << toString();
325  }
326 }

338 void SavedAction::apply(QSettings *s)
339 {
340  if (QAbstractButton *button = qobject_cast<QAbstractButton *>(m_widget))
341  setValue(button->isChecked());
342  else if (QLineEdit *lineEdit = qobject_cast<QLineEdit *>(m_widget))
343  setValue(lineEdit->text());
344  else if (QSpinBox *spinBox = qobject_cast<QSpinBox *>(m_widget))
345  setValue(spinBox->value());
346  else if (QDoubleSpinBox *doubleSpinBox = qobject_cast<QDoubleSpinBox *>(m_widget))
347  setValue(doubleSpinBox->value());
348  else if (PathChooser *pathChooser = qobject_cast<PathChooser *>(m_widget))
349  setValue(pathChooser->path());
350  if (s)
351  writeSettings(s);
352 }
