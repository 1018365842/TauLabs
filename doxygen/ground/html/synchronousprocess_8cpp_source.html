99 /* Check for complete lines read from the device and return them, moving the
100  * buffer position. This is based on the assumption that '\n' is the new line
101  * marker in any sane codec. */
102 QByteArray ChannelBuffer::linesRead()
103 {
104  // Any new lines?
105  const int lastLineIndex = data.lastIndexOf('\n');
106  if (lastLineIndex == -1 || lastLineIndex <= bufferPos)
107  return QByteArray();
108  const int nextBufferPos = lastLineIndex + 1;
109  const QByteArray lines = data.mid(bufferPos, nextBufferPos - bufferPos);
110  bufferPos = nextBufferPos;
111  return lines;
112 }

242 SynchronousProcessResponse SynchronousProcess::run(const QString &binary,
243  const QStringList &args)
244 {
245  if (debug)
246  qDebug() << '>' << Q_FUNC_INFO << binary << args;
247 
248  m_d->clearForRun();
249 
250  // On Windows, start failure is triggered immediately if the
251  // executable cannot be found in the path. Do not start the
252  // event loop in that case.
253  m_d->m_process.start(binary, args, QIODevice::ReadOnly);
254  if (!m_d->m_startFailure) {
255  m_d->m_timer.start();
256  QApplication::setOverrideCursor(Qt::WaitCursor);
257  m_d->m_eventLoop.exec(QEventLoop::ExcludeUserInputEvents);
258  if (m_d->m_result.result == SynchronousProcessResponse::Finished || m_d->m_result.result == SynchronousProcessResponse::FinishedError) {
259  processStdOut(false);
260  processStdErr(false);
261  }
262 
263  m_d->m_result.stdOut = convertStdOut(m_d->m_stdOut.data);
264  m_d->m_result.stdErr = convertStdErr(m_d->m_stdErr.data);
265 
266  m_d->m_timer.stop();
267  QApplication::restoreOverrideCursor();
268  }
269 
270  if (debug)
271  qDebug() << '<' << Q_FUNC_INFO << binary << m_d->m_result;
272  return m_d->m_result;
273 }

288 void SynchronousProcess::finished(int exitCode, QProcess::ExitStatus e)
289 {
290  if (debug)
291  qDebug() << Q_FUNC_INFO << exitCode << e;
292  m_d->m_hangTimerCount = 0;
293  switch (e) {
294  case QProcess::NormalExit:
295  m_d->m_result.result = exitCode ? SynchronousProcessResponse::FinishedError : SynchronousProcessResponse::Finished;
296  m_d->m_result.exitCode = exitCode;
297  break;
298  case QProcess::CrashExit:
299  // Was hang detected before and killed?
300  if (m_d->m_result.result != SynchronousProcessResponse::Hang)
301  m_d->m_result.result = SynchronousProcessResponse::TerminatedAbnormally;
302  m_d->m_result.exitCode = -1;
303  break;
304  }
305  m_d->m_eventLoop.quit();
306 }

343 void SynchronousProcess::processStdOut(bool emitSignals)
344 {
345  // Handle binary data
346  const QByteArray ba = m_d->m_process.readAllStandardOutput();
347  if (debug > 1)
348  qDebug() << Q_FUNC_INFO << emitSignals << ba;
349  if (!ba.isEmpty()) {
350  m_d->m_stdOut.data += ba;
351  if (emitSignals) {
352  // Emit binary signals
353  emit stdOut(ba, m_d->m_stdOut.firstData);
354  m_d->m_stdOut.firstData = false;
355  // Buffered. Emit complete lines?
356  if (m_d->m_stdOut.bufferedSignalsEnabled) {
357  const QByteArray lines = m_d->m_stdOut.linesRead();
358  if (!lines.isEmpty()) {
359  emit stdOutBuffered(convertStdOut(lines), m_d->m_stdOut.firstBuffer);
360  m_d->m_stdOut.firstBuffer = false;
361  }
362  }
363  }
364  }

405 // Locate a binary in a directory, applying all kinds of
406 // extensions the operating system supports.
407 static QString checkBinary(const QDir &dir, const QString &binary)
408 {
409  // naive UNIX approach
410  const QFileInfo info(dir.filePath(binary));
411  if (info.isFile() && info.isExecutable())
412  return info.absoluteFilePath();
413 
414  // Does the OS have some weird extension concept or does the
415  // binary have a 3 letter extension?
416  if (pathOS == OS_Unix)
417  return QString();
418  const int dotIndex = binary.lastIndexOf(QLatin1Char('.'));
419  if (dotIndex != -1 && dotIndex == binary.size() - 4)
420  return QString();
421 
422  switch (pathOS) {
423  case OS_Unix:
424  break;
425  case OS_Windows: {
426  static const char *windowsExtensions[] = {".cmd", ".bat", ".exe", ".com" };
427  // Check the Windows extensions using the order
428  const int windowsExtensionCount = sizeof(windowsExtensions)/sizeof(const char*);
429  for (int e = 0; e < windowsExtensionCount; e ++) {
430  const QFileInfo windowsBinary(dir.filePath(binary + QLatin1String(windowsExtensions[e])));
431  if (windowsBinary.isFile() && windowsBinary.isExecutable())
432  return windowsBinary.absoluteFilePath();
433  }
434  }
435  break;
436  case OS_Mac: {
437  // Check for Mac app folders
438  const QFileInfo appFolder(dir.filePath(binary + QLatin1String(".app")));
439  if (appFolder.isDir()) {
440  QString macBinaryPath = appFolder.absoluteFilePath();
441  macBinaryPath += QLatin1String("/Contents/MacOS/");
442  macBinaryPath += binary;
443  const QFileInfo macBinary(macBinaryPath);
444  if (macBinary.isFile() && macBinary.isExecutable())
445  return macBinary.absoluteFilePath();
446  }
447  }
448  break;
449  }
450  return QString();
451 }

453 QString SynchronousProcess::locateBinary(const QString &path, const QString &binary)
454 {
455  // Absolute file?
456  const QFileInfo absInfo(binary);
457  if (absInfo.isAbsolute())
458  return checkBinary(absInfo.dir(), absInfo.fileName());
459 
460  // Windows finds binaries in the current directory
461  if (pathOS == OS_Windows) {
462  const QString currentDirBinary = checkBinary(QDir::current(), binary);
463  if (!currentDirBinary.isEmpty())
464  return currentDirBinary;
465  }
466 
467  const QStringList paths = path.split(pathSeparator());
468  if (paths.empty())
469  return QString();
470  const QStringList::const_iterator cend = paths.constEnd();
471  for (QStringList::const_iterator it = paths.constBegin(); it != cend; ++it) {
472  const QDir dir(*it);
473  const QString rc = checkBinary(dir, binary);
474  if (!rc.isEmpty())
475  return rc;
476  }
477  return QString();
478 }
