70 void QwtSlider::initSlider( Qt::Orientation orientation,
71  ScalePos scalePos, BackgroundStyles bgStyle )
72 {
73  if ( orientation == Qt::Vertical )
74  setSizePolicy( QSizePolicy::Fixed, QSizePolicy::Expanding );
75  else
76  setSizePolicy( QSizePolicy::Expanding, QSizePolicy::Fixed );
77 
78  setAttribute( Qt::WA_WState_OwnSizePolicy, false );
79 
80  d_data = new QwtSlider::PrivateData;
81 
82  d_data->borderWidth = 2;
83  d_data->spacing = 4;
84  d_data->scalePos = scalePos;
85  d_data->bgStyle = bgStyle;
86 
87  const int handleThickness = 16;
88  d_data->handleSize.setWidth( 2 * handleThickness );
89  d_data->handleSize.setHeight( handleThickness );
90 
91  if ( !( bgStyle & QwtSlider::Trough ) )
92  d_data->handleSize.transpose();
93 
94  if ( orientation == Qt::Vertical )
95  d_data->handleSize.transpose();
96 
97  d_data->sliderRect.setRect( 0, 0, 8, 8 );
98 
99  QwtScaleDraw::Alignment align;
100  if ( orientation == Qt::Vertical )
101  {
102  // enforce a valid combination of scale position and orientation
103  if ( ( d_data->scalePos == QwtSlider::BottomScale )
104  || ( d_data->scalePos == QwtSlider::TopScale ) )
105  {
106  d_data->scalePos = NoScale;
107  }
108 
109  // adopt the policy of layoutSlider (NoScale lays out like Left)
110  if ( d_data->scalePos == QwtSlider::RightScale )
111  align = QwtScaleDraw::RightScale;
112  else
113  align = QwtScaleDraw::LeftScale;
114  }
115  else
116  {
117  // enforce a valid combination of scale position and orientation
118  if ( ( d_data->scalePos == QwtSlider::LeftScale )
119  || ( d_data->scalePos == QwtSlider::RightScale ) )
120  {
121  d_data->scalePos = QwtSlider::NoScale;
122  }
123 
124  // adopt the policy of layoutSlider (NoScale lays out like Bottom)
125  if ( d_data->scalePos == QwtSlider::TopScale )
126  align = QwtScaleDraw::TopScale;
127  else
128  align = QwtScaleDraw::BottomScale;
129  }
130 
131  scaleDraw()->setAlignment( align );
132  scaleDraw()->setLength( 100 );
133 
134  setRange( 0.0, 100.0, 1.0 );
135  setValue( 0.0 );
136 }

199 void QwtSlider::setScalePosition( ScalePos scalePos )
200 {

217  scaleDraw()->setAlignment( QwtScaleDraw::TopScale );

224  break;

228  QwtSlider::setOrientation( Qt::Vertical );
229  scaleDraw()->setAlignment( QwtScaleDraw::RightScale );
230  break;
231  }
232  default:

397 void QwtSlider::drawSlider(
398  QPainter *painter, const QRect &sliderRect ) const
399 {
400  QRect innerRect( sliderRect );
401 
402  if ( d_data->bgStyle & QwtSlider::Trough )
403  {
404  const int bw = d_data->borderWidth;
405 
406  qDrawShadePanel( painter, sliderRect, palette(), true, bw, NULL );
407 
408  innerRect = sliderRect.adjusted( bw, bw, -bw, -bw );
409  painter->fillRect( innerRect, palette().brush( QPalette::Mid ) );
410  }
411 
412  if ( d_data->bgStyle & QwtSlider::Groove )
413  {
414  int ws = 4;
415  int ds = d_data->handleSize.width() / 2 - 4;
416  if ( ds < 1 )
417  ds = 1;
418 
419  QRect rSlot;
420  if ( orientation() == Qt::Horizontal )
421  {
422  if ( innerRect.height() & 1 )
423  ws++;
424 
425  rSlot = QRect( innerRect.x() + ds,
426  innerRect.y() + ( innerRect.height() - ws ) / 2,
427  innerRect.width() - 2 * ds, ws );
428  }
429  else
430  {
431  if ( innerRect.width() & 1 )
432  ws++;
433 
434  rSlot = QRect( innerRect.x() + ( innerRect.width() - ws ) / 2,
435  innerRect.y() + ds,
436  ws, innerRect.height() - 2 * ds );
437  }
438 
439  QBrush brush = palette().brush( QPalette::Dark );
440  qDrawShadePanel( painter, rSlot, palette(), true, 1 , &brush );
441  }
442 
443  if ( isValid() )
444  drawHandle( painter, innerRect, transform( value() ) );
445 }

455  const QRect &sliderRect, int pos ) const
456 {
457  const int bw = d_data->borderWidth;
458 
459  pos++; // shade line points one pixel below

461  {

521 void QwtSlider::getScrollMode( const QPoint &p,
522  QwtAbstractSlider::ScrollMode &scrollMode, int &direction ) const
523 {
524  if ( !d_data->sliderRect.contains( p ) )
525  {
526  scrollMode = QwtAbstractSlider::ScrNone;
527  direction = 0;
528  return;
529  }
530 
531  const int pos = ( orientation() == Qt::Horizontal ) ? p.x() : p.y();
532  const int markerPos = transform( value() );
533 
534  if ( ( pos > markerPos - d_data->handleSize.width() / 2 )
535  && ( pos < markerPos + d_data->handleSize.width() / 2 ) )
536  {
537  scrollMode = QwtAbstractSlider::ScrMouse;
538  direction = 0;
539  return;
540  }
541 
542  scrollMode = QwtAbstractSlider::ScrPage;
543  direction = ( pos > markerPos ) ? 1 : -1;
544 
545  if ( scaleDraw()->scaleMap().p1() > scaleDraw()->scaleMap().p2() )
546  direction = -direction;
547 }

600 void QwtSlider::layoutSlider( bool update_geometry )
601 {
602  int handleThickness;
603  if ( orientation() == Qt::Horizontal )
604  handleThickness = d_data->handleSize.width();
605  else
606  handleThickness = d_data->handleSize.height();
607 
608  int sld1 = handleThickness / 2 - 1;
609  int sld2 = handleThickness / 2 + handleThickness % 2;
610 
611  if ( d_data->bgStyle & QwtSlider::Trough )
612  {
613  sld1 += d_data->borderWidth;
614  sld2 += d_data->borderWidth;
615  }
616 
617  int scd = 0;
618  if ( d_data->scalePos != QwtSlider::NoScale )
619  {
620  int d1, d2;
621  scaleDraw()->getBorderDistHint( font(), d1, d2 );
622  scd = qMax( d1, d2 );
623  }
624 
625  int slo = scd - sld1;
626  if ( slo < 0 )
627  slo = 0;
628 
629  int x, y, length;
630  QRect sliderRect;
631 
632  length = x = y = 0;
633 
634  const QRect cr = contentsRect();
635  if ( orientation() == Qt::Horizontal )
636  {
637  int sh = d_data->handleSize.height();
638  if ( d_data->bgStyle & QwtSlider::Trough )
639  sh += 2 * d_data->borderWidth;
640 
641  sliderRect.setLeft( cr.left() + slo );
642  sliderRect.setRight( cr.right() - slo );
643  sliderRect.setTop( cr.top() );
644  sliderRect.setBottom( cr.top() + sh - 1);
645 
646  if ( d_data->scalePos == QwtSlider::BottomScale )
647  {
648  y = sliderRect.bottom() + d_data->spacing;
649  }
650  else if ( d_data->scalePos == QwtSlider::TopScale )
651  {
652  sliderRect.setTop( cr.bottom() - sh + 1 );
653  sliderRect.setBottom( cr.bottom() );
654 
655  y = sliderRect.top() - d_data->spacing;
656  }
657 
658  x = sliderRect.left() + sld1;
659  length = sliderRect.width() - ( sld1 + sld2 );
660  }
661  else // Qt::Vertical
662  {
663  int sw = d_data->handleSize.width();
664  if ( d_data->bgStyle & QwtSlider::Trough )
665  sw += 2 * d_data->borderWidth;
666 
667  sliderRect.setLeft( cr.right() - sw + 1 );
668  sliderRect.setRight( cr.right() );
669  sliderRect.setTop( cr.top() + slo );
670  sliderRect.setBottom( cr.bottom() - slo );
671 
672  if ( d_data->scalePos == QwtSlider::LeftScale )
673  {
674  x = sliderRect.left() - d_data->spacing;
675  }
676  else if ( d_data->scalePos == QwtSlider::RightScale )
677  {
678  sliderRect.setLeft( cr.left() );
679  sliderRect.setRight( cr.left() + sw - 1);
680 
681  x = sliderRect.right() + d_data->spacing;
682  }
683 
684  y = sliderRect.top() + sld1;
685  length = sliderRect.height() - ( sld1 + sld2 );
686  }
687 
688  d_data->sliderRect = sliderRect;
689 
690  scaleDraw()->move( x, y );
691  scaleDraw()->setLength( length );
692 
693  d_data->map.setPaintInterval( scaleDraw()->scaleMap().p1(),
694  scaleDraw()->scaleMap().p2() );
695 
696  if ( update_geometry )
697  {
698  d_data->sizeHintCache = QSize(); // invalidate
699  updateGeometry();
700  update();
701  }
702 }

755 QSize QwtSlider::minimumSizeHint() const
756 {
757  if ( !d_data->sizeHintCache.isEmpty() )
758  return d_data->sizeHintCache;
759 
760  const int minLength = 84; // from QSlider
761 
762  int handleLength = d_data->handleSize.width();
763  int handleThickness = d_data->handleSize.height();
764 
765  if ( orientation() == Qt::Vertical )
766  qSwap( handleLength, handleThickness );
767 
768  int w = minLength;
769  int h = handleThickness;
770 
771  if ( d_data->scalePos != QwtSlider::NoScale )
772  {
773  int d1, d2;
774  scaleDraw()->getBorderDistHint( font(), d1, d2 );
775 
776  const int sdBorderDist = 2 * qMax( d1, d2 );
777  const int sdExtent = qCeil( scaleDraw()->extent( font() ) );
778  const int sdLength = scaleDraw()->minLength( font() );
779 
780  int l = sdLength;
781  if ( handleLength > sdBorderDist )
782  {
783  // We need additional space for the overlapping handle
784  l += handleLength - sdBorderDist;
785  }
786 
787  w = qMax( l, w );
788  h += sdExtent + d_data->spacing;
789  }
790 
791  if ( d_data->bgStyle & QwtSlider::Trough )
792  h += 2 * d_data->borderWidth;
793 
794  if ( orientation() == Qt::Vertical )
795  qSwap( w, h );
796 
797  int left, right, top, bottom;
798  getContentsMargins( &left, &top, &right, &bottom );
799 
800  w += left + right;
801  h += top + bottom;
802 
803  d_data->sizeHintCache = QSize( w, h );
804  return d_data->sizeHintCache;
805 }
