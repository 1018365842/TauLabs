782 quint32 OP_DFU::CRC32WideFast(quint32 Crc, quint32 Size, quint32 *Buffer)
783 {
784  //Size = Size >> 2; // /4 Size passed in as a byte count, assumed to be a multiple of 4
785 
786  while(Size--)
787  {
788  static const quint32 CrcTable[16] = { // Nibble lookup table for 0x04C11DB7 polynomial
789  0x00000000,0x04C11DB7,0x09823B6E,0x0D4326D9,0x130476DC,0x17C56B6B,0x1A864DB2,0x1E475005,
790  0x2608EDB8,0x22C9F00F,0x2F8AD6D6,0x2B4BCB61,0x350C9B64,0x31CD86D3,0x3C8EA00A,0x384FBDBD };
791 
792  Crc = Crc ^ *((quint32 *)Buffer); // Apply all 32-bits
793 
794  Buffer += 1;
795 
796  // Process 32-bits, 4 at a time, or 8 rounds
797 
798  Crc = (Crc << 4) ^ CrcTable[Crc >> 28]; // Assumes 32-bit reg, masking index to 4-bits
799  Crc = (Crc << 4) ^ CrcTable[Crc >> 28]; // 0x04C11DB7 Polynomial used in STM32
800  Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
801  Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
802  Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
803  Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
804  Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
805  Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
806  }
807 
808  return(Crc);
809 }
810 quint32 OP_DFU::CRCFromQBArray(QByteArray array, quint32 Size)
811 {
812  int pad=Size-array.length();
813  array.append(QByteArray(pad,255));
814  quint32 t[Size/4];
815  for(int x=0;x<array.length()/4;x++)
816  {
817  quint32 aux=0;
818  aux=(char)array[x*4+3]&0xFF;
819  aux=aux<<8;
820  aux+=(char)array[x*4+2]&0xFF;
821  aux=aux<<8;
822  aux+=(char)array[x*4+1]&0xFF;
823  aux=aux<<8;
824  aux+=(char)array[x*4+0]&0xFF;
825  t[x]=aux;
826  }
827  return CRC32WideFast(0xFFFFFFFF,Size/4,(quint32*)t);
