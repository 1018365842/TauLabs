120 // Normalize the plane
121 void GLC_Plane::normalize()
122 {
123  const double invMag= 1.0 / sqrt(m_Eq[0] * m_Eq[0] + m_Eq[1] * m_Eq[1] + m_Eq[2] * m_Eq[2]);
124 
125  m_Eq[0]= m_Eq[0] * invMag;
126  m_Eq[1]= m_Eq[1] * invMag;
127  m_Eq[2]= m_Eq[2] * invMag;
128  m_Eq[3]= m_Eq[3] * invMag;
129 }
