164 void ScopeGadgetWidget::wheelEvent(QWheelEvent *e)
165 {
166  //Change zoom on scroll wheel event
167  QwtInterval yInterval=axisInterval(QwtPlot::yLeft);
168  if (yInterval.minValue() != yInterval.maxValue()) //Make sure that the two values are never the same. Sometimes axisInterval returns (0,0)
169  {
170  //Determine what y value to zoom about. NOTE, this approach has a bug that the in that
171  //the value returned by Qt includes the legend, whereas the value transformed by Qwt
172  //does *not*. Thus, when zooming with a legend, there will always be a small bias error.
173  //In practice, this seems not to be a UI problem.
174  QPoint mouse_pos=e->pos(); //Get the mouse coordinate in the frame
175  double zoomLine=invTransform(QwtPlot::yLeft, mouse_pos.y()); //Transform the y mouse coordinate into a frame value.
176 
177  double zoomScale=1.1; //THIS IS AN ARBITRARY CONSTANT, AND PERHAPS SHOULD BE IN A DEFINE INSTEAD OF BURIED HERE
178 
179  mutex.lock(); //DOES THIS mutex.lock NEED TO BE HERE? I DON'T KNOW, I JUST COPIED IT FROM THE ABOVE CODE
180  // Set the scale
181  if (e->delta()<0){
182  setAxisScale(QwtPlot::yLeft,
183  (yInterval.minValue()-zoomLine)*zoomScale+zoomLine,
184  (yInterval.maxValue()-zoomLine)*zoomScale+zoomLine );
185  }
186  else{
187  setAxisScale(QwtPlot::yLeft,
188  (yInterval.minValue()-zoomLine)/zoomScale+zoomLine,
189  (yInterval.maxValue()-zoomLine)/zoomScale+zoomLine );
190  }
191  mutex.unlock();
192  }
193  QwtPlot::wheelEvent(e);
194 }

240 void ScopeGadgetWidget::addLegend()
241 {
242  if (legend())
243  return;
244 
245  // Show a legend at the top
246  m_legend = new QwtLegend;
247  m_legend->setItemMode(QwtLegend::CheckableItem);
248  m_legend->setFrameStyle(QFrame::Box | QFrame::Sunken);
249  m_legend->setToolTip(tr("Click legend to show/hide scope trace"));
250 
251  QPalette pal = m_legend->palette();
252  pal.setColor(m_legend->backgroundRole(), QColor(100, 100, 100)); // background colour

308 void ScopeGadgetWidget::replotNewData()
309 {
310  // If the plot is not visible or there is no scope, do not replot
311  if (!isVisible() || m_scope == NULL)
312  return;
313 
314  QMutexLocker locker(&mutex);
315 
316  // Update the data in the scopes

319  plotData->plotNewData(plotData, m_scope, this);
320  }
321 
322  // Repaint the scopes
323  replot();

353 QString ScopeGadgetWidget::getUavObjectFieldUnits(QString uavObjectName, QString uavObjectFieldName)
354 {
355  //Get the uav object
356  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
357  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
358  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(objManager->getObject(uavObjectName));
359  if(!obj) {
360  qDebug() << "In scope gadget, UAVObject " << uavObjectName << " is missing";
361  return "";
362  }
363  UAVObjectField* field = obj->getField(uavObjectFieldName);
364  if(!field) {
365  qDebug() << "In scope gadget, in fields loaded from GCS config file, field" << uavObjectFieldName << " of UAVObject " << uavObjectName << " is missing";
366  return "";
367  }
368 
369  //Get the units
370  QString units = field->getUnits();
371  if(units == 0)
372  units = QString();
373 
374  return units;
375 }
