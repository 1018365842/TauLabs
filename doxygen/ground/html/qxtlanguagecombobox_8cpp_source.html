35 static QStringList findQmFiles(const QString& pathToTranslations)
36 {
37  QDir dir(pathToTranslations);
38  QStringList fileNames = dir.entryList(QStringList("*.qm"), QDir::Files, QDir::Name);
39 
40  QMutableStringListIterator i(fileNames);
41  while (i.hasNext())
42  {
43  i.next();
44  int start = i.value().indexOf('_');
45  int end = i.value().lastIndexOf('.');
46  QString s = i.value().mid(start + 1, end - start - 1).toLower();
47 
48  i.setValue(s);
49  }
50 
51  return fileNames;
52 }

76  static const LanguageList& getAllLanguages()
77  {
78  if (_smAllLanguages.empty())
79  {
80 
81  for (int idx = 0; idx < QLocale::LastLanguage; ++idx)
82  {
83  QLocale::Language l = static_cast<QLocale::Language>(idx);
84  if (l == QLocale::LastLanguage)
85  continue;
86  // obsolete - NorwegianNynorsk is used instead
87  if (l == QLocale::Nynorsk)
88  continue;
89  if (l == QLocale::C)
90  continue;
91 
92  _smAllLanguages.push_back(Language(l));
93  }
94 
95  qSort(_smAllLanguages);
96  }
97  return _smAllLanguages;
98  };

100  static LanguageList getTrLanguages(const QString& translationPath)
101  {
102  LanguageList trLanguages;
103 
104  QStringList qms = findQmFiles(translationPath);
105  for (int i = 0; i < qms.size(); ++i)
106  {
107  QLocale locale(qms[i]);
108  if (locale.language() == QLocale::C)
109  continue;
110  trLanguages.push_back(Language(locale.language()));
111  }
112  qSort(trLanguages);
113  return trLanguages;
114  };

157  QVariant data(const QModelIndex& index, int role = Qt::DisplayRole) const
158  {
159  if (!index.isValid())
160  return QVariant();
161  if (_mLanguages.empty())
162  return QVariant();
163 
164  int idx = qMin(index.row(), _mLanguages.size());
165  const Language& l = _mLanguages[idx];
166  if (role == Qt::DecorationRole)
167  {
168  return QIcon(":/flags/" + l.countryName() + ".png");
169  }
170 
171  if (role == Qt::DisplayRole)
172  {
173  switch (index.column())
174  {
175  case 0:
176  return l.name();
177  case 1:
178  return l.language();
179  default:
180  return QVariant();
181  }
182  }
183  return QVariant();
184  }

217 void QxtLanguageComboBoxPrivate::reset()
218 {
219  if (_mModel != 0)
220  {
221  delete _mModel;
222  _mModel = 0;
223  }
224 
225  QLocale::Language currentLang = currentLanguage();
226  if (_mDisplayMode == QxtLanguageComboBox::AllLanguages)
227  _mModel = new LanguageModel(Language::getAllLanguages(), &qxt_p());
228  else
229  _mModel = new LanguageModel(Language::getTrLanguages(_mTranslationPath), &qxt_p());
230 
231  qxt_p().setModel(_mModel);
232  qxt_p().setModelColumn(0);
233 
234  setCurrentLanguage(currentLang);
235 }
