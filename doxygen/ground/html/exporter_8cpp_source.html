159  outSocket = new QUdpSocket();

193 void Exporter::receiveUpdate()
194 {
195  // Update connection timer and status
196  exporterTimer->setInterval(exporterTimeout);
197  exporterTimer->stop();
198  exporterTimer->start();
199  if ( !exporterConnectionStatus )
200  {
201  exporterConnectionStatus = true;
202  emit exporterConnected();
203  }
204 
205  // Process data
206  while(inSocket->hasPendingDatagrams()) {
207  // Receive datagram
208  QByteArray datagram;
209  datagram.resize(inSocket->pendingDatagramSize());
210  QHostAddress sender;
211  quint16 senderPort;
212  inSocket->readDatagram(datagram.data(), datagram.size(),
213  &sender, &senderPort);
214 
215  // Process incomming data
216  processUpdate(datagram);
217  }
218 }
