70 void PathListPlainTextEdit::insertFromMimeData(const QMimeData *source)
71 {
72  if (source->hasText()) {
73  // replace separator
74  QString text = source->text().trimmed();
75  text.replace(PathListEditor::separator(), QLatin1Char('\n'));
76  QSharedPointer<QMimeData> fixed(new QMimeData);
77  fixed->setText(text);

174 QStringList PathListEditor::pathList() const
175 {
176  const QString text = m_d->edit->toPlainText().trimmed();
177  if (text.isEmpty())
178  return QStringList();
179  // trim each line
180  QStringList rc = text.split(QLatin1Char('\n'), QString::SkipEmptyParts);
181  const QStringList::iterator end = rc.end();
182  for (QStringList::iterator it = rc.begin(); it != end; ++it)
183  *it = it->trimmed();
184  return rc;
185 }

268 void PathListEditor::insertPathAtCursor(const QString &path)
269 {
270  // If the cursor is at an empty line or at end(),
271  // just insert. Else insert line before
272  QTextCursor cursor = m_d->edit->textCursor();
273  QTextBlock block = cursor.block();
274  const bool needNewLine = !block.text().isEmpty();
275  if (needNewLine) {
276  cursor.movePosition(QTextCursor::StartOfLine, QTextCursor::MoveAnchor);
277  cursor.insertBlock();
278  cursor.movePosition(QTextCursor::PreviousBlock, QTextCursor::MoveAnchor);
279  }
280  cursor.insertText(path);
281  if (needNewLine) {
282  cursor.movePosition(QTextCursor::StartOfLine, QTextCursor::MoveAnchor);
283  m_d->edit->setTextCursor(cursor);
284  }
285 }

296 void PathListEditor::deletePathAtCursor()
297 {
298  // Delete current line
299  QTextCursor cursor = m_d->edit->textCursor();
300  if (cursor.block().isValid()) {
301  cursor.movePosition(QTextCursor::StartOfLine, QTextCursor::MoveAnchor);
302  // Select down or until end of [last] line
303  if (!cursor.movePosition(QTextCursor::Down, QTextCursor::KeepAnchor))
304  cursor.movePosition(QTextCursor::EndOfLine, QTextCursor::KeepAnchor);
305  cursor.removeSelectedText();
306  m_d->edit->setTextCursor(cursor);
307  }
