123  stdSum+= pow(yDataHistory->at(i)- boxcarAvg,2)/(meanSamples-1);

129  }
130  }

132  yData->append( currentValue );

137  } else //...otherwise, add a new y point at position xData

222

224  if (xData->size() == 0)
225  break;
226 
227  newestValue = xData->last();
228  oldestValue = xData->first();
229
