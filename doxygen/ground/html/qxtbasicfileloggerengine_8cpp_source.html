82 void QxtBasicFileLoggerEngine::writeToFile(const QString &level, const QVariantList &messages)
83 {
84  if (messages.isEmpty()) return;
85  QString header = '[' + QDateTime::currentDateTime().toString(qxt_d().dateFormat) + "] [" + level + "] ";
86  QString padding;
87  QIODevice* file = device();
88  Q_ASSERT(file);
89  file->write(header.toUtf8());
90  for (int i = 0; i < header.size(); i++) padding.append(" ");
91  int count = 0;
92  Q_FOREACH(const QVariant& out, messages)
93  {
94  if (!out.isNull())
95  {
96  if (count != 0) file->write(padding.toUtf8());
97  file->write(out.toString().toUtf8());
98  file->write("\n");
99  }
100  count++;
101  }
