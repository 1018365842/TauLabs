365 // Return true if the all instance's mesh are transparent
366 bool GLC_3DViewInstance::isTransparent() const
367 {
368  if (m_3DRep.isEmpty()) return false;
369  if (m_RenderProperties.renderingMode() == glc::OverwriteTransparency) return true;
370  if (m_RenderProperties.renderingMode() == glc::OverwriteMaterial)
371  {
372  return m_RenderProperties.overwriteMaterial()->isTransparent();
373  }
374  const int size= m_3DRep.numberOfBody();
375  bool result= true;
376  int i= 0;
377  while((i < size) && result)
378  {
379  result= result && m_3DRep.geomAt(i)->isTransparent();
380  ++i;
381  }
382  return result && m_RenderProperties.needToRenderWithTransparency();
383 }
384 
385 // Return true if the instance contains mesh which contains transparent material
386 bool GLC_3DViewInstance::hasTransparentMaterials() const
387 {
388  if (m_3DRep.isEmpty()) return false;
389  if (m_RenderProperties.needToRenderWithTransparency()) return true;
390  const int size= m_3DRep.numberOfBody();
391  bool result= false;
392  int i= 0;
393  while ((i < size) && !result)
394  {
395  result= result || m_3DRep.geomAt(i)->hasTransparentMaterials();
396  ++i;
397  }
398  return result;
399 }
401 bool GLC_3DViewInstance::setViewable(GLC_3DViewInstance::Viewable flag)
402 {
403  const int bodyCount= m_3DRep.numberOfBody();
404  if (bodyCount != m_ViewableGeomFlag.size())
405  {
406  m_ViewableGeomFlag.fill(true, bodyCount);
407  }
408  bool asChange= m_ViewableFlag != flag;
409  if (asChange)
410  {
411  m_ViewableFlag= flag;
412  if (flag != GLC_3DViewInstance::PartialViewable)
413  {
414  bool viewable= (flag == GLC_3DViewInstance::FullViewable);
415 
416  for (int i= 0; i < bodyCount; ++i)
417  {
418  m_ViewableGeomFlag[i]= viewable;
419  }
420  }
421  }
422  return asChange;
423 }
