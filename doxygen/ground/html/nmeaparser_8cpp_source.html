101 char NMEAParser::nmeaChecksum(char* gps_buffer)
102 {
103  char checksum=0;
104  char checksum_received=0;
105 
106  for(int x=0; x<NMEA_BUFFERSIZE; x++)
107  {
108  if(gps_buffer[x]=='*')
109  {
110  //Parsing received checksum...
111  checksum_received = strtol(&gps_buffer[x+1],NULL,16);
112 
113  break;
114  }
115  else
116  {
117  //XOR the received data...
118  checksum^=gps_buffer[x];
119  }
120  }
121  if(checksum == checksum_received)
122  {
123  ++numUpdates;
124  return 1;
125  }
126  else
127  {
128  ++numErrors;
129  return 0;
130  }
131 }

151 uint8_t NMEAParser::nmeaProcess(cBuffer* rxBuffer)
152 {
153  uint8_t foundpacket = NMEA_NODATA;
154  uint8_t startFlag = FALSE;
155  //u08 data;
156  uint16_t i,j;
157 
158  // process the receive buffer
159  // go through buffer looking for packets
160  while(rxBuffer->datalength)
161  {
162  // look for a start of NMEA packet
163  if(bufferGetAtIndex(rxBuffer,0) == '$')
164  {
165  // found start
166  startFlag = TRUE;
167  // when start is found, we leave it intact in the receive buffer
168  // in case the full NMEA string is not completely received. The
169  // start will be detected in the next nmeaProcess iteration.
170 
171  // done looking for start
172  break;
173  }
174  else
175  bufferGetFromFront(rxBuffer);
176  }
177 
178  // if we detected a start, look for end of packet
179  if(startFlag)
180  {
181  for(i=1; i<(rxBuffer->datalength)-1; i++)
182  {
183  // check for end of NMEA packet <CR><LF>
184  if((bufferGetAtIndex(rxBuffer,i) == '\r') && (bufferGetAtIndex(rxBuffer,i+1) == '\n'))
185  {
186  // have a packet end
187  // dump initial '$'
188  bufferGetFromFront(rxBuffer);
189  // copy packet to NmeaPacket
190  for(j=0; j<(i-1); j++)
191  {
192  // although NMEA strings should be 80 characters or less,
193  // receive buffer errors can generate erroneous packets.
194  // Protect against packet buffer overflow
195  if(j<(NMEA_BUFFERSIZE-1))
196  NmeaPacket[j] = bufferGetFromFront(rxBuffer);
197  else
198  bufferGetFromFront(rxBuffer);
199  }
200  // null terminate it
201  if (j<(NMEA_BUFFERSIZE-1)) {
202  NmeaPacket[j] = 0;
203  } else {
204  NmeaPacket[NMEA_BUFFERSIZE-1] = 0;
205  }
206  // dump <CR><LF> from rxBuffer
207  bufferGetFromFront(rxBuffer);
208  bufferGetFromFront(rxBuffer);
209  //DEBUG
210  #ifdef NMEA_DEBUG_PKT
211  qDebug() << NmeaPacket;
212  #endif
213  emit packet(QString(NmeaPacket));
214  // found a packet
215  // done with this processing session
216  foundpacket = NMEA_UNKNOWN;
217  break;
218  }
219  }
220  }
221 
222  if(foundpacket)
223  {
224  // check message type and process appropriately
225  if(!strncmp(NmeaPacket, "GPGGA", 5))
226  {
227  // process packet of this type
228  nmeaProcessGPGGA(NmeaPacket);
229  // report packet type
230  foundpacket = NMEA_GPGGA;
231  }
232  else if(!strncmp(NmeaPacket, "GPVTG", 5))
233  {
234  // process packet of this type
235  nmeaProcessGPVTG(NmeaPacket);
236  // report packet type
237  foundpacket = NMEA_GPVTG;
238  }
239  else if(!strncmp(NmeaPacket, "GPGSA", 5))
240  {
241  // process packet of this type
242  nmeaProcessGPGSA(NmeaPacket);
243  // report packet type
244  foundpacket = NMEA_GPGSA;
245  }
246  else if(!strncmp(NmeaPacket, "GPRMC", 5))
247  {
248  // process packet of this type
249  nmeaProcessGPRMC(NmeaPacket);
250  // report packet type
251  foundpacket = NMEA_GPRMC;
252  }
253  else if(!strncmp(NmeaPacket, "GPGSV", 5))
254  {
255  // Process packet of this type
256  nmeaProcessGPGSV(NmeaPacket);
257  // rerpot packet type
258  foundpacket = NMEA_GPGSV;
259  }
260  else if(!strncmp(NmeaPacket, "GPZDA", 5))
261  {
262  // Process packet of this type
263  nmeaProcessGPZDA(NmeaPacket);
264  // rerpot packet type
265  foundpacket = NMEA_GPZDA;
266  }
267  }
268  else if(rxBuffer->datalength >= rxBuffer->size)
269  {
270  // if we found no packet, and the buffer is full
271  // we're logjammed, flush entire buffer
272  bufferFlush(rxBuffer);
273  }
274  return foundpacket;
275 }

281 void NMEAParser::nmeaProcessGPGSV(char *packet)
282 {
283 
284  // start parsing just after "GPGSV,"
285  // attempt to reject empty packets right away
286  if(packet[6]==',' && packet[7]==',')
287  return;
288 
289  if(!nmeaChecksum(packet))
290  {
291  // checksum not valid
292  return;
293  }
294  nmeaTerminateAtChecksum(packet);
295 
296  QString nmeaString( packet );
297  QStringList tokenslist = nmeaString.split(",");
298 
299 
300  // Officially there should be a max of three sentences (12 sats), some gps receivers do more..
301 
302  const int sentence_total = tokenslist.at(1).toInt(); // Number of sentences for full data
303  const int sentence_index = tokenslist.at(2).toInt(); // sentence x of y
304 
305  int sats = (tokenslist.size() - 4) /4;
306  for(int sat = 0; sat < sats; sat++) {
307  int base = 4+sat*4;
308  const int id = tokenslist.at(base+0).toInt(); // Satellite PRN number
309  const int elv = tokenslist.at(base+1).toInt(); // Elevation, degrees
310  const int azimuth = tokenslist.at(base+2).toInt(); // Azimuth, degrees
311  const int sig = tokenslist.at(base+3).toInt(); // SNR - higher is better
312  const int index = (sentence_index-1) * 4 + sat;
313  emit satellite(index, id, elv, azimuth, sig);
314  }
315 
316  if(sentence_index == sentence_total) {
317  // Last sentence
318  int total_sats = (sentence_index-1) * 4 + sats;
319  for(int emptySatIndex = total_sats; emptySatIndex < 16; emptySatIndex++) {
320  // Wipe the rest.
321  emit satellite(emptySatIndex, 0, 0, 0, 0);
322  }
323  }
324 }

434 void NMEAParser::nmeaProcessGPGSA(char* packet)
435 {
436  // start parsing just after "GPGSA,"
437  // attempt to reject empty packets right away
438  if(packet[6]==',' && packet[7]==',')
439  return;
440 
441  if(!nmeaChecksum(packet)) {
442  // checksum not valid
443  return;
444  }
445  nmeaTerminateAtChecksum(packet);
446 
447  QString nmeaString( packet );
448  QStringList tokenslist = nmeaString.split(",");
449 
450  // M=Manual, forced to operate in 2D or 3D
451  // A=Automatic, 3D/2D
452  QString fixmodeValue = tokenslist.at(1);
453  if (fixmodeValue == "A") {
454  emit fixmode(QString("Auto"));
455  } else if (fixmodeValue == "B") {
456  emit fixmode(QString("Manual"));
457  }
458 
459  // Mode: 1=Fix not available, 2=2D, 3=3D
460  int fixtypeValue = tokenslist.at(2).toInt();
461  if (fixtypeValue == 1) {
462  emit fixtype(QString("NoFix"));
463  } else if (fixtypeValue == 2) {
464  emit fixtype(QString("Fix2D"));
465  } else if (fixtypeValue == 3) {
466  emit fixtype(QString("Fix3D"));
467  }
468 
469  // 3-14 = IDs of SVs used in position fix (null for unused fields)
470  QList<int> svList;
471  for(int pos = 0; pos < 12;pos ++) {
472  QString sv = tokenslist.at(3+pos);
473  if(!sv.isEmpty()) {
474  svList.append(sv.toInt());
475  }
476  }
477  emit fixSVs(svList);
478 
479  // 15 = PDOP
480  // 16 = HDOP
481  // 17 = VDOP
482  GpsData.PDOP = tokenslist.at(15).toDouble();
483  GpsData.HDOP = tokenslist.at(16).toDouble();
484  GpsData.VDOP = tokenslist.at(17).toDouble();
485  emit dop(GpsData.HDOP, GpsData.VDOP, GpsData.PDOP);
486 }
