112 template<typename LhsNested, typename RhsNested, int ProductMode>
113 struct ei_traits<Product<LhsNested, RhsNested, ProductMode> >
114 {
115  // clean the nested types:
116  typedef typename ei_cleantype<LhsNested>::type _LhsNested;
117  typedef typename ei_cleantype<RhsNested>::type _RhsNested;
118  typedef typename ei_scalar_product_traits<typename _LhsNested::Scalar, typename _RhsNested::Scalar>::ReturnType Scalar;
119 
120  enum {
121  LhsCoeffReadCost = _LhsNested::CoeffReadCost,
122  RhsCoeffReadCost = _RhsNested::CoeffReadCost,
123  LhsFlags = _LhsNested::Flags,
124  RhsFlags = _RhsNested::Flags,
125 
126  RowsAtCompileTime = _LhsNested::RowsAtCompileTime,
127  ColsAtCompileTime = _RhsNested::ColsAtCompileTime,
128  InnerSize = EIGEN_SIZE_MIN(_LhsNested::ColsAtCompileTime, _RhsNested::RowsAtCompileTime),
129 
130  MaxRowsAtCompileTime = _LhsNested::MaxRowsAtCompileTime,
131  MaxColsAtCompileTime = _RhsNested::MaxColsAtCompileTime,
132 
133  LhsRowMajor = LhsFlags & RowMajorBit,
134  RhsRowMajor = RhsFlags & RowMajorBit,
135 
136  CanVectorizeRhs = RhsRowMajor && (RhsFlags & PacketAccessBit)
137  && (ColsAtCompileTime % ei_packet_traits<Scalar>::size == 0),
138 
139  CanVectorizeLhs = (!LhsRowMajor) && (LhsFlags & PacketAccessBit)
140  && (RowsAtCompileTime % ei_packet_traits<Scalar>::size == 0),
141 
142  EvalToRowMajor = RhsRowMajor && (ProductMode==(int)CacheFriendlyProduct ? LhsRowMajor : (!CanVectorizeLhs)),
143 
144  RemovedBits = ~((EvalToRowMajor ? 0 : RowMajorBit)|DirectAccessBit),
145 
146  Flags = ((unsigned int)(LhsFlags | RhsFlags) & HereditaryBits & RemovedBits)
147  | EvalBeforeAssigningBit
148  | EvalBeforeNestingBit
149  | (CanVectorizeLhs || CanVectorizeRhs ? PacketAccessBit : 0)
150  | (LhsFlags & RhsFlags & AlignedBit),
151 
152  CoeffReadCost = InnerSize == Dynamic ? Dynamic
153  : InnerSize * (NumTraits<Scalar>::MulCost + LhsCoeffReadCost + RhsCoeffReadCost)
154  + (InnerSize - 1) * NumTraits<Scalar>::AddCost,
155 
156  /* CanVectorizeInner deserves special explanation. It does not affect the product flags. It is not used outside
157  * of Product. If the Product itself is not a packet-access expression, there is still a chance that the inner
158  * loop of the product might be vectorized. This is the meaning of CanVectorizeInner. Since it doesn't affect
159  * the Flags, it is safe to make this value depend on ActualPacketAccessBit, that doesn't affect the ABI.
160  */
161  CanVectorizeInner = LhsRowMajor && (!RhsRowMajor) && (LhsFlags & RhsFlags & ActualPacketAccessBit)
162  && (InnerSize % ei_packet_traits<Scalar>::size == 0)
163  };
164 };

267 template<typename Derived>
268 template<typename OtherDerived>
269 inline const typename ProductReturnType<Derived,OtherDerived>::Type
270 MatrixBase<Derived>::operator*(const MatrixBase<OtherDerived> &other) const
271 {
272  enum {
273  ProductIsValid = Derived::ColsAtCompileTime==Dynamic
274  || OtherDerived::RowsAtCompileTime==Dynamic
275  || int(Derived::ColsAtCompileTime)==int(OtherDerived::RowsAtCompileTime),
276  AreVectors = Derived::IsVectorAtCompileTime && OtherDerived::IsVectorAtCompileTime,
277  SameSizes = EIGEN_PREDICATE_SAME_MATRIX_SIZE(Derived,OtherDerived)
278  };
279  // note to the lost user:
280  // * for a dot product use: v1.dot(v2)
281  // * for a coeff-wise product use: v1.cwise()*v2
282  EIGEN_STATIC_ASSERT(ProductIsValid || !(AreVectors && SameSizes),
283  INVALID_VECTOR_VECTOR_PRODUCT__IF_YOU_WANTED_A_DOT_OR_COEFF_WISE_PRODUCT_YOU_MUST_USE_THE_EXPLICIT_FUNCTIONS)
284  EIGEN_STATIC_ASSERT(ProductIsValid || !(SameSizes && !AreVectors),
285  INVALID_MATRIX_PRODUCT__IF_YOU_WANTED_A_COEFF_WISE_PRODUCT_YOU_MUST_USE_THE_EXPLICIT_FUNCTION)
286  EIGEN_STATIC_ASSERT(ProductIsValid || SameSizes, INVALID_MATRIX_PRODUCT)
287  return typename ProductReturnType<Derived,OtherDerived>::Type(derived(), other.derived());
288 }
