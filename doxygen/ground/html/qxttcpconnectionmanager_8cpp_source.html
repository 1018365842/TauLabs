64 void QxtTcpConnectionManagerPrivate::incomingConnection(int socketDescriptor)
65 {
66  QIODevice* device = qxt_p().incomingConnection(socketDescriptor);
67  if (device)
68  {
69  qxt_p().addConnection(device, (quint64)static_cast<QObject*>(device));
70  mapper.setMapping(device, device);
71  QObject::connect(device, SIGNAL(destroyed()), &mapper, SLOT(map()));
72  QTcpSocket* sock = qobject_cast<QTcpSocket*>(device);
73  if (sock)
74  {
75  QObject::connect(sock, SIGNAL(error(QAbstractSocket::SocketError)), &mapper, SLOT(map()));
76  QObject::connect(sock, SIGNAL(disconnected()), &mapper, SLOT(map()));
77  }
78  }
