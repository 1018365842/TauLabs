108 void FancyTabBar::mouseMoveEvent(QMouseEvent *e)
109 {
110  if (!m_hoverRect.contains(e->pos())) {
111  int newHover = -1;
112  for (int i = 0; i < count(); ++i) {
113  QRect area = tabRect(i);
114  if (area.contains(e->pos())) {
115  newHover = i;
116  break;
117  }
118  }
119 
120  m_hoverControl.stop();
121  m_hoverIndex = newHover;
122  update(m_hoverRect);
123  m_hoverRect = QRect();
124 
125  if (m_hoverIndex >=0) {
126  QRect oldHoverRect = m_hoverRect;
127  m_hoverRect = tabRect(m_hoverIndex);
128  m_hoverControl.start();
129  }
130  }

194 {
195  QSize sh = tabSizeHint();
196

208  return QRect(index * sh.width(), 0, sh.width(), sh.height());

223 void FancyTabBar::paintTab(QPainter *painter, int tabIndex) const
224 {
225  painter->save();
226 
227  QRect rect = tabRect(tabIndex);
228 
229  bool selected = (tabIndex == m_currentIndex);
230  bool hover = (tabIndex == m_hoverIndex);
231 
232 #ifdef Q_WS_MAC
233  hover = false; // Dont hover on Mac
234 #endif
235 
236  QColor background = QColor(0, 0, 0, 10);
237  QColor hoverColor;
238 
239  if (hover) {
240  hoverColor = QColor(255, 255, 255, m_hoverControl.currentFrame());
241  }
242 
243  QColor light = QColor(255, 255, 255, 40);
244  QColor dark = QColor(0, 0, 0, 60);
245 
246  if (selected) {
247  QLinearGradient selectedGradient(rect.bottomRight(), QPoint(rect.center().x(), rect.top()));
248  selectedGradient.setColorAt(0, Qt::white);
249  selectedGradient.setColorAt(0.3, Qt::white);
250  selectedGradient.setColorAt(0.7, QColor(210, 210, 220)); //give a blue-ish color
251 
252  painter->fillRect(rect, selectedGradient);
253  painter->setPen(QColor(200, 200, 200));
254  painter->drawLine(rect.topLeft(), rect.bottomLeft());
255  painter->setPen(QColor(150, 160, 200));
256  painter->drawLine(rect.topRight(), rect.bottomRight());
257  } else {
258  painter->fillRect(rect, background);
259  if (hover)
260  painter->fillRect(rect, hoverColor);
261  painter->setPen(QPen(light, 0));
262  painter->drawLine(rect.topLeft(), rect.bottomLeft());
263  painter->setPen(QPen(dark, 0));
264  painter->drawLine(rect.topRight(), rect.bottomRight());
265  }
266 
267  QString tabText(this->tabText(tabIndex));
268  QRect tabTextRect(tabRect(tabIndex));
269  QRect tabIconRect(tabTextRect);
270  QFont boldFont(painter->font());
271  boldFont.setPointSizeF(Utils::StyleHelper::sidebarFontSize());
272  boldFont.setBold(true);
273  painter->setFont(boldFont);
274  painter->setPen(selected ? Utils::StyleHelper::panelTextColor() : QColor(30, 30, 30, 80));
275  int textFlags = Qt::AlignCenter | Qt::AlignBottom | Qt::ElideRight | Qt::TextWordWrap;
276  painter->drawText(tabTextRect, textFlags, tabText);
277  painter->setPen(selected ? QColor(60, 60, 60) : Utils::StyleHelper::panelTextColor());
278  int textHeight = painter->fontMetrics().boundingRect(QRect(0, 0, width(), height()), Qt::TextWordWrap, tabText).height();
279  tabIconRect.adjust(0, 4, 0, -textHeight);
280  int iconSize = qMin(tabIconRect.width(), tabIconRect.height());
281  if (iconSize > 4)
282  style()->drawItemPixmap(painter, tabIconRect, Qt::AlignCenter | Qt::AlignVCenter,
283  tabIcon(tabIndex).pixmap(tabIconRect.size()));
284  painter->translate(0, -1);
285  painter->drawText(tabTextRect, textFlags, tabText);
286  painter->restore();
287 }
