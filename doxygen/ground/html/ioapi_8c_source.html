69  voidpf opaque;

76  if ((mode & ZLIB_FILEFUNC_MODE_READWRITEFILTER)==ZLIB_FILEFUNC_MODE_READ)
77  mode_fopen = "rb";
78  else
79  if (mode & ZLIB_FILEFUNC_MODE_EXISTING)
