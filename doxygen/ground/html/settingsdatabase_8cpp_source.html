92 SettingsDatabase::SettingsDatabase(const QString &path,
93  const QString &application,
94  QObject *parent)
95  : QObject(parent)
96  , d(new SettingsDatabasePrivate)
97 {
98  const QLatin1Char slash('/');
99 
100  // TODO: Don't rely on a path, but determine automatically
101  QDir pathDir(path);
102  if (!pathDir.exists())
103  pathDir.mkpath(pathDir.absolutePath());
104 
105  QString fileName = path;
106  if (!fileName.endsWith(slash))
107  fileName += slash;
108  fileName += application;
109  fileName += QLatin1String(".db");
110 
111  d->m_db = QSqlDatabase::addDatabase("QSQLITE", QLatin1String("settings"));
112  d->m_db.setDatabaseName(fileName);
113  if (!d->m_db.open()) {
114  qWarning().nospace() << "Warning: Failed to open settings database at " << fileName << " ("
115  << d->m_db.lastError().driverText() << ")";
116  } else {
117  // Create the settings table if it doesn't exist yet
118  QSqlQuery query(d->m_db);
119  query.prepare(QLatin1String("CREATE TABLE IF NOT EXISTS settings ("
120  "key PRIMARY KEY ON CONFLICT REPLACE, "
121  "value)"));
122  if (!query.exec())
123  qWarning().nospace() << "Warning: Failed to prepare settings database! ("
124  << query.lastError().driverText() << ")";
125 
126  // Retrieve all available keys (values are retrieved lazily)
127  if (query.exec(QLatin1String("SELECT key FROM settings"))) {
128  while (query.next()) {
129  d->m_settings.insert(query.value(0).toString(), QVariant());
130  }
131  }
132  }

164 QVariant SettingsDatabase::value(const QString &key, const QVariant &defaultValue) const
165 {
166  const QString effectiveKey = d->effectiveKey(key);
167  QVariant value = defaultValue;
168 
169  SettingsMap::const_iterator i = d->m_settings.constFind(effectiveKey);
170  if (i != d->m_settings.constEnd() && i.value().isValid()) {
171  value = i.value();
172  } else if (d->m_db.isOpen()) {
173  // Try to read the value from the database
174  QSqlQuery query(d->m_db);
175  query.prepare(QLatin1String("SELECT value FROM settings WHERE key = ?"));
176  query.addBindValue(effectiveKey);
177  query.exec();
178  if (query.next()) {
179  value = query.value(0);
180 
181  if (debug_settings)
182  qDebug() << "Retrieved:" << effectiveKey << "=" << value;
183  }
184 
185  // Cache the result
186  d->m_settings.insert(effectiveKey, value);
187  }

197 void SettingsDatabase::remove(const QString &key)
198 {
199  const QString effectiveKey = d->effectiveKey(key);
200 
201  // Remove keys from the cache
202  foreach (const QString &k, d->m_settings.keys()) {
203  // Either it's an exact match, or it matches up to a /
204  if (k.startsWith(effectiveKey)
205  && (k.length() == effectiveKey.length()
206  || k.at(effectiveKey.length()) == QLatin1Char('/')))
207  {
208  d->m_settings.remove(k);
209  }
210  }
211 
212  if (!d->m_db.isOpen())
213  return;
214 
215  // Delete keys from the database
216  QSqlQuery query(d->m_db);
217  query.prepare(QLatin1String("DELETE FROM settings WHERE key = ? OR key LIKE ?"));
218  query.addBindValue(effectiveKey);
219  query.addBindValue(QString(effectiveKey + QLatin1String("/%")));
220  query.exec();
221 }
