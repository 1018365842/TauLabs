96 void WayPointCurve::refreshLocations()
97 {
98  double m_n, m_e, p_n, p_e, d, center_x, center_y;
99 
100  // Center between start and end
101  m_n = (m_start->pos().x() + m_dest->pos().x()) / 2;
102  m_e = (m_start->pos().y() + m_dest->pos().y()) / 2;
103 
104  // Normal vector the line between start and end.
105  if (!m_clockwise) {
106  p_n = -(m_dest->pos().y() - m_start->pos().y());
107  p_e = (m_dest->pos().x() - m_start->pos().x());
108  } else {
109  p_n = (m_dest->pos().y() - m_start->pos().y());
110  p_e = -(m_dest->pos().x() - m_start->pos().x());
111  }
112 
113  double radius_sign = (m_radius > 0) ? 1 : -1;
114  double pixels2meters = my_map->Projection()->GetGroundResolution(my_map->ZoomTotal(), m_start->Coord().Lat());
115  double radius = fabs(m_radius / pixels2meters);
116 
117  // Work out how far to go along the perpendicular bisector
118  d = sqrt(radius * radius / (p_n * p_n + p_e * p_e) - 0.25f);
119 
120  if (fabs(p_n) < 1e-3 && fabs(p_e) < 1e-3) {
121  center_x = m_n;
122  center_y = m_e;
123  } else {
124  center_x = m_n + p_n * d * radius_sign;
125  center_y = m_e + p_e * d * radius_sign;
126  }
127 
128  // Store the center
129  center.setX(center_x);
130  center.setY(center_y);
131 
132  // Compute the midpoint along the arc for the arrow
133  d = sqrt(radius * radius / (p_n * p_n + p_e * p_e));
134  midpoint.setX(center_x - p_n * d);
135  midpoint.setY(center_y - p_e * d);
136  midpoint_angle = -atan2(m_dest->pos().y() - m_start->pos().y(), m_dest->pos().x() - m_start->pos().x());
137 
138  double startAngle = atan2(-(m_start->pos().y() - center_y), m_start->pos().x() - center_x);
139  double endAngle = atan2(-(m_dest->pos().y() - center_y), m_dest->pos().x() - center_x);
140  double span = endAngle - startAngle;
141  if (!m_clockwise) {
142  if (span > 0)
143  span = span - 2 * M_PI;
144  } else {
145  if (span < 0)
146  span = span + 2 * M_PI;
147  }
148  setRect(center_x - radius, center_y - radius, 2 * radius, 2 * radius);
149  setStartAngle(startAngle * 180.0 / M_PI * 16.0);
150  setSpanAngle(span * 180.0 / M_PI * 16.0);
151  update();
152 }
