100 void UAVObjectTreeModel::addDataObject(UAVDataObject *obj, bool categorize)
101 {
102  //Determine if the root tree is the settings or dynamic data tree
103  TopTreeItem *root = obj->isSettings() ? m_settingsTree : m_nonSettingsTree;
104 
105  TreeItem* parent = root;
106 
107  if(categorize && obj->getCategory() != 0 && !obj->getCategory().isEmpty()) {
108  QStringList categoryPath = obj->getCategory().split('/');
109  parent = createCategoryItems(categoryPath, root);
110  }
111 
112  ObjectTreeItem* existing = root->findDataObjectTreeItemByObjectId(obj->getObjID());
113  if (existing) {
114  addInstance(obj, existing);
115  } else {
116  DataObjectTreeItem *dataTreeItem = new DataObjectTreeItem(obj->getName() + " (" + QString::number(obj->getNumBytes()) + " bytes)");
117  dataTreeItem->setHighlightManager(m_highlightManager);
118  connect(dataTreeItem, SIGNAL(updateHighlight(TreeItem*)), this, SLOT(updateHighlight(TreeItem*)));
119  parent->insertChild(dataTreeItem);
120  root->addObjectTreeItem(obj->getObjID(), dataTreeItem);
121  UAVMetaObject *meta = obj->getMetaObject();
122  MetaObjectTreeItem* metaTreeItem = addMetaObject(meta, dataTreeItem);
123  root->addMetaObjectTreeItem(meta->getObjID(), metaTreeItem);
124  addInstance(obj, dataTreeItem);
125  }
126 }

200 void UAVObjectTreeModel::addSingleField(int index, UAVObjectField *field, TreeItem *parent)
201 {
202  QList<QVariant> data;
203  if (field->getNumElements() == 1)
204  data.append(field->getName());
205  else
206  data.append( QString("[%1]").arg((field->getElementNames())[index]) );
207 
208  FieldTreeItem *item;
209  UAVObjectField::FieldType type = field->getType();
210  switch (type) {
211  case UAVObjectField::BITFIELD:
212  case UAVObjectField::ENUM: {
213  QStringList options = field->getOptions();
214  QVariant value = field->getValue();
215  data.append( options.indexOf(value.toString()) );
216  data.append(field->getUnits());
217  item = new EnumFieldTreeItem(field, index, data);
218  break;
219  }
220  case UAVObjectField::INT8:
221  case UAVObjectField::INT16:
222  case UAVObjectField::INT32:
223  case UAVObjectField::UINT8:
224  case UAVObjectField::UINT16:
225  case UAVObjectField::UINT32:
226  data.append(field->getValue(index));
227  data.append(field->getUnits());
228  item = new IntFieldTreeItem(field, index, data);
229  break;
230  case UAVObjectField::FLOAT32:
231  data.append(field->getValue(index));
232  data.append(field->getUnits());
233  item = new FloatFieldTreeItem(field, index, data, m_useScientificFloatNotation);
234  break;
235  default:
236  Q_ASSERT(false);
237  }
238  item->setHighlightManager(m_highlightManager);
239  connect(item, SIGNAL(updateHighlight(TreeItem*)), this, SLOT(updateHighlight(TreeItem*)));
240  parent->appendChild(item);
241 }

263 QModelIndex UAVObjectTreeModel::index(TreeItem *item)
264 {
265  if (item->parent() == 0)
266  return QModelIndex();
267 
268  QModelIndex root = index(item->parent());
269 
270  for (int i = 0; i < rowCount(root); ++i) {
271  QModelIndex childIndex = index(i, 0, root);
272  TreeItem *child = static_cast<TreeItem*>(childIndex.internalPointer());
273  if (child == item)
274  return childIndex;
275  }
276  Q_ASSERT(false);
277  return QModelIndex();

331 QVariant UAVObjectTreeModel::data(const QModelIndex &index, int role) const
332 {
333  if (!index.isValid())
334  return QVariant();
335 
336  if (index.column() == TreeItem::dataColumn && role == Qt::EditRole) {
337  TreeItem *item = static_cast<TreeItem*>(index.internalPointer());
338  return item->data(index.column());
339  }
340 
341  if (role == Qt::ToolTipRole) {
342  TreeItem *item = static_cast<TreeItem*>(index.internalPointer());
343  return item->description();
344  }
345 
346  TreeItem *item = static_cast<TreeItem*>(index.internalPointer());
347 
348  if (index.column() == 0 && role == Qt::BackgroundRole) {
349  if (!dynamic_cast<TopTreeItem*>(item) && item->highlighted())
350  return QVariant(m_recentlyUpdatedColor);
351  if (!dynamic_cast<TopTreeItem*>(item) && item->updatedOnly())
352  return QVariant(m_updatedOnlyColor);
353  }
354 
355  if (index.column() == TreeItem::dataColumn && role == Qt::BackgroundRole) {
356  FieldTreeItem *fieldItem = dynamic_cast<FieldTreeItem*>(item);
357  if (fieldItem && fieldItem->highlighted())
358  return QVariant(m_recentlyUpdatedColor);
359 
360  if (fieldItem && fieldItem->changed())
361  return QVariant(m_manuallyChangedColor);
362 
363  if (fieldItem && fieldItem->updatedOnly())
364  return QVariant(m_updatedOnlyColor);
365  }
366 
367  if (role != Qt::DisplayRole)
368  return QVariant();
369 
370  if (index.column() == TreeItem::dataColumn) {
371  EnumFieldTreeItem *fieldItem = dynamic_cast<EnumFieldTreeItem*>(item);
372  if (fieldItem) {
373  int enumIndex = fieldItem->data(index.column()).toInt();
374  return fieldItem->enumOptions(enumIndex);
375  }
376  }
377 
378  return item->data(index.column());
379 }
