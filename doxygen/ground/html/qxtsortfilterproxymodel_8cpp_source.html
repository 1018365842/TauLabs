52 bool QxtModelFilterPrivate::acceptsValue ( const QVariant & value )
53 {
54  uint matchType = m_flags & 0x0F;
55  //if we have no value we accept everything
56  if(!m_value.isValid() || !value.isValid())
57  return true;
58 
59  // QVariant based matching
60  if (matchType == Qt::MatchExactly ){
61  if(m_value == value)
62  return true;
63  }
64  // QString based matching
65  else {
66  Qt::CaseSensitivity cs = m_flags & Qt::MatchCaseSensitive ? Qt::CaseSensitive : Qt::CaseInsensitive;
67  QString filterText = m_value.toString();
68  QString modelText = value.toString();
69  switch (matchType){
70  case Qt::MatchRegExp:
71  if (QRegExp(filterText, cs).exactMatch(modelText))
72  return true;
73  break;
74  case Qt::MatchWildcard:
75  if (QRegExp(filterText, cs, QRegExp::Wildcard).exactMatch(modelText))
76  return true;
77  break;
78  case Qt::MatchStartsWith:
79  if (modelText.startsWith(filterText, cs))
80  return true;
81  break;
82  case Qt::MatchEndsWith:
83  if (modelText.endsWith(filterText, cs))
84  return true;
85  break;
86  case Qt::MatchFixedString:
87  if (modelText.compare(filterText, cs) == 0)
88  return true;
89  break;
90  default:
91  case Qt::MatchContains:
92  if (modelText.contains(filterText, cs))
93  return true;
94  }
95  }
96  return false;
97 }

160 bool QxtSortFilterProxyModel::filterAcceptsRow ( int source_row, const QModelIndex &source_parent ) const
161 {
162  QList<int> filterColumns = qxt_d().filters.keys();
163  foreach(int currCol,filterColumns){
164  QxtModelFilterPrivate filter = qxt_d().filters[currCol];
165  // if the column specified by the user is -1
166  // that means all columns need to pass the filter to get into the result
167  if (currCol == -1) {
168  int column_count = sourceModel()->columnCount(source_parent);
169  for (int column = 0; column < column_count; ++column) {
170  QModelIndex source_index = sourceModel()->index(source_row, column, source_parent);
171  QVariant key = sourceModel()->data(source_index, filter.m_role);
172  if (!filter.acceptsValue(key))
173  return false;
174  }
175  continue;
176  }
177  QModelIndex source_index = sourceModel()->index(source_row, currCol , source_parent);
178  if (!source_index.isValid()) // the column may not exist
179  continue;
180  QVariant key = sourceModel()->data(source_index, filter.m_role);
181  if(!filter.acceptsValue(key))
182  return false;
183  }
184  return true;
185 }
