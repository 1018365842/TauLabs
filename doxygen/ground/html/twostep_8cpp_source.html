79 void inv_fisher_information_matrix(Matrix3f& fisherInv,
80  Matrix3f& fisher,
81  const Vector3f samples[],
82  size_t n_samples,
83  const float noise)
84 {
85  MatrixXf column_samples(3, n_samples);
86  for (size_t i = 0; i < n_samples; ++i) {
87  column_samples.col(i) = samples[i];
88  }
89  fisher = 4 / noise *
90  column_samples * column_samples.transpose();
91  // Compute the inverse by taking advantage of the results symmetricness
92  fisher.ldlt().solve(Matrix3f::Identity(), &fisherInv);
93  return;
94 }
95 // Eq 14 of the original reference. Computes the gradiant of the cost function
96 // with respect to the bias offset.
97 // @param samples: The vector of measurement samples
98 // @param mags: The vector of sample delta magnitudes
99 // @param b: The estimate of the bias
100 // @param mu: The trace of the noise matrix (3*noise)
101 // @return the negative gradiant of the cost function with respect to the
102 // current value of the estimate, b
103 Vector3f
104 neg_dJdb(const Vector3f samples[],
105  const float mags[],
106  size_t n_samples,
107  const Vector3f& b,
108  float mu,
109  float noise)
110 {
111  Vector3f summation(Vector3f::Zero());
112  float b_norm2 = b.squaredNorm();
113 
114  for (size_t i = 0; i < n_samples; ++i) {
115  summation += (mags[i] - 2*samples[i].dot(b) + b_norm2 - mu)
116  *2* (samples[i] - b);
117  }
118 
119  return (1.0 / noise)*summation;
120 }

123 Vector3f twostep_bias_only(const Vector3f samples[],
124  size_t n_samples,
125  const Vector3f& referenceField,
126  const float noise)
127 {
128  // \tilde{H}
129  Vector3f centeredSamples[n_samples];
130  // z_k
131  float sampleDeltaMag[n_samples];
132  // eq 7 and 8 applied to samples
133  Vector3f avg = center(samples, n_samples);
134  float refSquaredNorm = referenceField.squaredNorm();
135  float sampleDeltaMagCenter = 0;
136  for (size_t i = 0; i < n_samples; ++i)
137  {
138  // eq 9 applied to samples
139  centeredSamples[i] = samples[i] - avg;
140  // eqn 2a
141  sampleDeltaMag[i] = samples[i].squaredNorm() - refSquaredNorm;
142  sampleDeltaMagCenter += sampleDeltaMag[i];
143  }
144  sampleDeltaMagCenter /= n_samples;
145 
146  Matrix3f P_bb;
147  Matrix3f P_bb_inv;
148  // Due to eq 12b
149  inv_fisher_information_matrix(P_bb, P_bb_inv, centeredSamples, n_samples, noise);
150  // Compute centered magnitudes
151  float sampleDeltaMagCentered[n_samples];
152  for (size_t i = 0; i < n_samples; ++i) {
153  sampleDeltaMagCentered[i] = sampleDeltaMag[i] - sampleDeltaMagCenter;
154  }
155 
156  // From eq 12a
157  Vector3f estimate( Vector3f::Zero());
158  for (size_t i = 0; i < n_samples; ++i) {
159  estimate += sampleDeltaMagCentered[i] * centeredSamples[i];
160  }
161  estimate = P_bb * ((2 / noise) * estimate);
162 
163  // Newton-Raphson gradient descent to the optimal solution
164  // eq 14a and 14b
165  float mu = -3*noise;
166  for (int i = 0; i < 6; ++i) {
167  Vector3f neg_gradiant = neg_dJdb(samples, sampleDeltaMag, n_samples,
168  estimate, mu, noise);
169  Matrix3f scale = P_bb_inv + 4/noise*(avg - estimate)*(avg - estimate).transpose();
170  Vector3f neg_increment;
171  scale.ldlt().solve(neg_gradiant, &neg_increment);
172  // Note that the negative has been done twice
173  estimate += neg_increment;
174  }
175  return estimate;
176 }

262 void twostep_bias_scale(Vector3f& bias,
263  Vector3f& scale,
264  const Vector3f samples[],
265  const size_t n_samples,
266  const Vector3f& referenceField,
267  const float noise)
268 {
269  // Initial estimate for gradiant descent starts at eq 37a of ref 2.
270 
271  // Define L_k by eq 30 and 28 for k = 1 .. n_samples
272  Matrix<double, Dynamic, 6> fullSamples(n_samples, 6);
273  // \hbar{L} by eq 33, simplified by obesrving that the
274  Matrix<double, 1, 6> centerSample = Matrix<double, 1, 6>::Zero();
275  // Define the sample differences z_k by eq 23 a)
276  double sampleDeltaMag[n_samples];
277  // The center value \hbar{z}
278  double sampleDeltaMagCenter = 0;
279  double refSquaredNorm = referenceField.squaredNorm();
280 
281  for (size_t i = 0; i < n_samples; ++i) {
282  fullSamples.row(i) << 2*samples[i].transpose().cast<double>(),
283  -(samples[i][0]*samples[i][0]),
284  -(samples[i][1]*samples[i][1]),
285  -(samples[i][2]*samples[i][2]);
286  centerSample += fullSamples.row(i);
287 
288  sampleDeltaMag[i] = samples[i].squaredNorm() - refSquaredNorm;
289  sampleDeltaMagCenter += sampleDeltaMag[i];
290  }
291  sampleDeltaMagCenter /= n_samples;
292  centerSample /= n_samples;
293 
294  // True for all k.
295  // double mu = -3*noise;
296  // The center value of mu, \bar{mu}
297  // double centerMu = mu;
298  // The center value of mu, \tilde{mu}
299  // double centeredMu = 0;
300 
301  // Define \tilde{L}_k for k = 0 .. n_samples
302  Matrix<double, Dynamic, 6> centeredSamples(n_samples, 6);
303  // Define \tilde{z}_k for k = 0 .. n_samples
304  double centeredMags[n_samples];
305  // Compute the term under the summation of eq 37a
306  Matrix<double, 6, 1> estimateSummation = Matrix<double, 6, 1>::Zero();
307  for (size_t i = 0; i < n_samples; ++i) {
308  centeredSamples.row(i) = fullSamples.row(i) - centerSample;
309  centeredMags[i] = sampleDeltaMag[i] - sampleDeltaMagCenter;
310  estimateSummation += centeredMags[i] * centeredSamples.row(i).transpose();
311  }
312  estimateSummation /= noise; // note: paper supplies 1/noise
313 
314  // By eq 37 b). Note, paper supplies 1/noise here
315  Matrix<double, 6, 6> P_theta_theta_inv = (1.0f/noise)*
316  centeredSamples.transpose()*centeredSamples;
317 #ifdef PRINTF_DEBUGGING
318  SelfAdjointEigenSolver<Matrix<double, 6, 6> > eig(P_theta_theta_inv);
319  std::cout << "P_theta_theta_inverse: \n" << P_theta_theta_inv << "\n\n";
320  std::cout << "P_\\tt^-1 eigenvalues: " << eig.eigenvalues().transpose()
321  << "\n";
322  std::cout << "P_\\tt^-1 eigenvectors:\n" << eig.eigenvectors() << "\n";
323 #endif
324 
325  // The Fisher information matrix has a small eigenvalue, with a
326  // corresponding eigenvector of about [1e-2 1e-2 1e-2 0.55, 0.55,
327  // 0.55]. This means that there is relatively little information
328  // about the common gain on the scale factor, which has a
329  // corresponding effect on the bias, too. The fixup is performed by
330  // the first iteration of the second stage of TWOSTEP, as documented in
331  // [1].
332  Matrix<double, 6, 1> estimate;
333  // By eq 37 a), \tilde{Fisher}^-1
334  P_theta_theta_inv.ldlt().solve(estimateSummation, &estimate);
335 
336 #ifdef PRINTF_DEBUGGING
337  Matrix<double, 6, 6> P_theta_theta;
338  P_theta_theta_inv.ldlt().solve(Matrix<double, 6, 6>::Identity(), &P_theta_theta);
339  SelfAdjointEigenSolver<Matrix<double, 6, 6> > eig2(P_theta_theta);
340  std::cout << "eigenvalues: " << eig2.eigenvalues().transpose() << "\n";
341  std::cout << "eigenvectors: \n" << eig2.eigenvectors() << "\n";
342  std::cout << "estimate summation: \n" << estimateSummation.normalized()
343  << "\n";
344 #endif
345 
346  // estimate i+1 = estimate_i - Fisher^{-1}(at estimate_i)*gradiant(theta)
347  // Fisher^{-1} = \tilde{Fisher}^-1 + \hbar{Fisher}^{-1}
348  size_t count = 3;
349  while (count --> 0) {
350  // eq 40
351  Matrix<double, 1, 6> db_dtheta = dnormb_dtheta(estimate);
352 
353  Matrix<double, 6, 1> dJ_dtheta = dJdb(centerSample,
354  sampleDeltaMagCenter,
355  estimate,
356  db_dtheta,
357  -3*noise,
358  noise/n_samples);
359 
360 
361  // Eq 39 (with double-negation on term inside parens)
362  db_dtheta = centerSample - db_dtheta;
363  Matrix<double, 6, 6> scale = P_theta_theta_inv +
364  (double(n_samples)/noise)*db_dtheta.transpose() * db_dtheta;
365 
366  // eq 14b, mutatis mutandis (grumble, grumble)
367  Matrix<double, 6, 1> increment;
368  scale.ldlt().solve(dJ_dtheta, &increment);
369  estimate -= increment;
370  }
371 
372  // Transform the estimated parameters from [c | e] back into [b | d].
373  for (size_t i = 0; i < 3; ++i) {
374  scale.coeffRef(i) = -1 + sqrt(1 + estimate.coeff(3+i));
375  bias.coeffRef(i) = estimate.coeff(i)/sqrt(1 + estimate.coeff(3+i));
376  }

422 // The gradient of the cost function with respect to theta, at a particular
423 // value of theta.
424 // @param centerL: The center of the samples theta'
425 // @param centerMag: The center of the magnitude data
426 // @param theta: The estimate of the bias and scale factor
427 // @return the gradient of the cost function with respect to the
428 // current value of the estimate, theta'
429 Matrix<double, 9, 1>
430 dJ_dtheta(const Matrix<double, 9, 1>& centerL,
431  double centerMag,
432  const Matrix<double, 9, 1>& theta,
433  const Matrix<double, 9, 1>& dbdtheta,
434  double mu,
435  double noise)
436 {
437  // \frac{\delta}{\delta \theta'} |b(\theta')|^2
438  Matrix<double, 9, 1> vectorPart = dbdtheta - centerL;
439 
440  // |b(\theta')|^2
441  Matrix3d E = E_theta(theta);
442  Vector3d rhs;
443  (Matrix3d::Identity() + E).ldlt().solve(theta.start<3>(), &rhs);
444  double normbthetaSquared = theta.start<3>().dot(rhs);
445 
446  double scalarPart = (centerMag - centerL.dot(theta) + normbthetaSquared
447  - mu)/noise;
448  return scalarPart * vectorPart;
449 }
