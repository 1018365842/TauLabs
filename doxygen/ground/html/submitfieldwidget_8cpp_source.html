140 int SubmitFieldWidgetPrivate::findSender(const QObject *o) const
141 {
142  const int count = fieldEntries.size();

144  const FieldEntry &fe = fieldEntries.at(i);

273 void SubmitFieldWidget::createField(const QString &f)
274 {
275  FieldEntry fe;
276  fe.createGui(m_d->removeFieldIcon);
277  fe.combo->addItems(m_d->fields);
278  if (!f.isEmpty()) {
279  const int index = fe.combo->findText(f);
280  if (index != -1) {
281  setComboBlocked(fe.combo, index);
282  fe.comboIndex = index;
283  }
284  }
285 
286  connect(fe.browseButton, SIGNAL(clicked()), this, SLOT(slotBrowseButtonClicked()));
287  if (!m_d->hasBrowseButton)
288  fe.browseButton->setVisible(false);
289 
290  if (m_d->completer)
291  fe.lineEdit->setCompleter(m_d->completer);
292 
293  connect(fe.combo, SIGNAL(currentIndexChanged(int)),
294  this, SLOT(slotComboIndexChanged(int)));
295  connect(fe.clearButton, SIGNAL(clicked()),
296  this, SLOT(slotRemove()));
297  m_d->layout->addLayout(fe.layout);
298  m_d->fieldEntries.push_back(fe);
299 }

301 void SubmitFieldWidget::slotRemove()
302 {
303  // Never remove first entry
304  const int index = m_d->findSender(sender());
305  switch (index) {
306  case -1:
307  break;

314  }

325 void SubmitFieldWidget::slotComboIndexChanged(int comboIndex)
326 {
327  const int pos = m_d->findSender(sender());
328  if (debug)
329  qDebug() << '>' << Q_FUNC_INFO << pos;
330  if (pos == -1)
331  return;
332  // Accept new index or reset combo to previous value?
333  int &previousIndex = m_d->fieldEntries[pos].comboIndex;
334  if (comboIndexChange(pos, comboIndex)) {
335  previousIndex = comboIndex;
336  } else {
337  setComboBlocked(m_d->fieldEntries.at(pos).combo, previousIndex);
338  }
339  if (debug)
340  qDebug() << '<' << Q_FUNC_INFO << pos;
341 }

343 // Handle change of a combo. Return "false" if the combo
344 // is to be reset (refuse new field).
345 bool SubmitFieldWidget::comboIndexChange(int pos, int index)
346 {
347  const QString newField = m_d->fieldEntries.at(pos).combo->itemText(index);
348  // If the field is visible elsewhere: focus the existing one and refuse
349  if (!m_d->allowDuplicateFields) {
350  const int existingFieldIndex = m_d->findField(newField, pos);
351  if (existingFieldIndex != -1) {
352  m_d->focusField(existingFieldIndex);
353  return false;
354  }
355  }
356  // Empty value: just change the field
357  if (m_d->fieldValue(pos).isEmpty())
358  return true;
359  // Non-empty: Create a new field and reset the triggering combo
360  createField(newField);
361  return false;
362 }
