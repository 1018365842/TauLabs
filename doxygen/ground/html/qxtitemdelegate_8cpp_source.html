45 void QxtItemDelegatePrivate::paintButton(QPainter* painter, const QStyleOptionViewItem& option, const QModelIndex& index, const QTreeView* view) const
46 {
47  // draw the button
48  QStyleOptionButton buttonOption;
49  buttonOption.state = option.state;
50 #ifdef Q_WS_MAC
51  buttonOption.state |= QStyle::State_Raised;
52 #endif
53  buttonOption.state &= ~QStyle::State_HasFocus;
54  if (view->isExpanded(index))
55  buttonOption.state |= QStyle::State_Sunken;
56  buttonOption.rect = option.rect;
57  buttonOption.palette = option.palette;
58  buttonOption.features = QStyleOptionButton::None;
59  view->style()->drawControl(QStyle::CE_PushButton, &buttonOption, painter, view);
60 
61  // draw the branch indicator
62  static const int i = 9;
63  const QRect& r = option.rect;
64  if (index.model()->hasChildren(index))
65  {
66  QStyleOption branchOption;
67  branchOption.initFrom(view);
68  if (branchOption.direction == Qt::LeftToRight)
69  branchOption.rect = QRect(r.left() + i / 2, r.top() + (r.height() - i) / 2, i, i);
70  else
71  branchOption.rect = QRect(r.right() - i / 2 - i, r.top() + (r.height() - i) / 2, i, i);
72  branchOption.palette = option.palette;
73  branchOption.state = QStyle::State_Children;
74  if (view->isExpanded(index))
75  branchOption.state |= QStyle::State_Open;
76  view->style()->drawPrimitive(QStyle::PE_IndicatorBranch, &branchOption, painter, view);
77  }
78 
79  // draw the text
80  QRect textrect = QRect(r.left() + i * 2, r.top(), r.width() - ((5 * i) / 2), r.height());
81  QString text = option.fontMetrics.elidedText(index.data().toString(), elide, textrect.width());
82  view->style()->drawItemText(painter, textrect, Qt::AlignCenter, option.palette, view->isEnabled(), text);
83 }

121 void QxtItemDelegatePrivate::paintProgress(QPainter* painter, const QStyleOptionViewItem& option, const QModelIndex& index) const
122 {
123  QVariant min = index.data(QxtItemDelegate::ProgressMinimumRole);
124  QVariant max = index.data(QxtItemDelegate::ProgressMaximumRole);
125 
126  QStyleOptionProgressBar opt;
127  opt.minimum = (min.isValid() && min.canConvert(QVariant::Int)) ? min.toInt() : 0;
128  opt.maximum = (max.isValid() && max.canConvert(QVariant::Int)) ? max.toInt() : 100;
129  opt.progress = index.data(QxtItemDelegate::ProgressValueRole).toInt();
130  opt.rect = option.rect;
131  opt.textVisible = textVisible;
132  opt.text = progressFormat.arg(opt.progress);
133  QApplication::style()->drawControl(QStyle::CE_ProgressBar, &opt, painter, 0);
134 
135  QWidget* viewport = dynamic_cast<QWidget*>(painter->device());
136  if (viewport)
137  {
138  if (opt.minimum == 0 && opt.maximum == 0)
139  {
140  if (!updatedItems.contains(viewport))
141  connect(viewport, SIGNAL(destroyed()), this, SLOT(viewDestroyed()));
142  updatedItems.replace(viewport, index);
143  }
144  else
145  {
146  updatedItems.remove(viewport, index);
147  if (!updatedItems.contains(viewport))
148  disconnect(viewport, SIGNAL(destroyed()), this, SLOT(viewDestroyed()));
149  }
150  }
151 
152  if (updatedItems.isEmpty())
153  {
154  if (updateTimer.isActive())
155  updateTimer.stop();
156  }
157  else
158  {
159  if (!updateTimer.isActive())
160  updateTimer.start(1000 / 25, const_cast<QxtItemDelegatePrivate*>(this));
161  }
162 }

170 void QxtItemDelegatePrivate::timerEvent(QTimerEvent* event)
171 {
172  if (event->timerId() == updateTimer.timerId())
173  {
174  QMutableHashIterator<QWidget*, QPersistentModelIndex> it(updatedItems);
175  while (it.hasNext())
176  {
177  it.next();
178  if (!it.key())
179  {
180  it.remove();
181  continue;
182  }
183 
184  // try to update the specific view item instead of the whole view if possible
185  if (QAbstractItemView* view = qobject_cast<QAbstractItemView*>(it.key()->parentWidget()))
186  view->update(it.value());
187  else
188  it.key()->update();
189  }
190  }

381 void QxtItemDelegate::paint(QPainter* painter, const QStyleOptionViewItem& option, const QModelIndex& index) const
382 {
383  const QAbstractItemModel* model = index.model();
384  const QTreeView* tree = qobject_cast<QTreeView*>(parent());
385  const bool topLevel = !index.parent().isValid();
386 
387  if (tree && model && topLevel && qxt_d().style != Qxt::NoDecoration)
388  {
389  QStyleOptionViewItem opt;
390  opt.QStyleOption::operator=(option);
391  opt.showDecorationSelected = false;
392 
393  QModelIndex valid = model->index(index.row(), 0);
394  QModelIndex sibling = valid;
395  while (sibling.isValid())
396  {
397  opt.rect |= tree->visualRect(sibling);
398  sibling = sibling.sibling(sibling.row(), sibling.column() + 1);
399  }
400 
401  switch (qxt_d().style)
402  {
403  case Qxt::Buttonlike:
404  qxt_d().paintButton(painter, opt, valid, tree);
405  break;
406  case Qxt::Menulike:
407  qxt_d().paintMenu(painter, opt, valid, tree);
408  break;
409  default:
410  qWarning("QxtItemDelegate::paint() unknown decoration style");
411  QItemDelegate::paint(painter, opt, valid);
412  break;
413  }
414  }
415  else
416  {
417  QItemDelegate::paint(painter, option, index);
418 
419  const QVariant data = index.data(ProgressValueRole);
420  if (data.isValid() && data.canConvert(QVariant::Int))
421  qxt_d().paintProgress(painter, option, index);
422  }
423 }

428 void QxtItemDelegate::drawDisplay(QPainter* painter, const QStyleOptionViewItem& option, const QRect& rect, const QString& text) const
429 {
430  if (!Qt::mightBeRichText(text))
431  {
432  QItemDelegate::drawDisplay(painter, option, rect, text);
433  return;
434  }
435 
436  QString key = QString(QLatin1String("QxtItemDelegate:%1")).arg(text);
437  QPixmap pixmap;
438  if (!QPixmapCache::find(key, pixmap))
439  {
440  if (!qxt_d().document)
441  qxt_d().document = new QTextDocument(const_cast<QxtItemDelegate*>(this));
442  qxt_d().document->setHtml(text);
443  qxt_d().document->adjustSize();
444 
445  pixmap = QPixmap(qxt_d().document->size().toSize());
446  pixmap.fill(Qt::transparent);
447  QPainter painter(&pixmap);
448  qxt_d().document->drawContents(&painter);
449  painter.end();
450  QPixmapCache::insert(key, pixmap);
451  }
452  painter->drawPixmap(option.rect.topLeft(), pixmap);
453 }

458 QSize QxtItemDelegate::sizeHint(const QStyleOptionViewItem& option, const QModelIndex& index) const
459 {
460  // something slightly bigger for top level indices if decorated
461  QSize size = QItemDelegate::sizeHint(option, index);
462  if (!index.parent().isValid() && qxt_d().style != Qxt::NoDecoration)
463  size += QSize(TOP_LEVEL_EXTENT, TOP_LEVEL_EXTENT);
464 
465  const QString text = index.data(Qt::DisplayRole).toString();
466  if (Qt::mightBeRichText(text))
467  {
468  if (!qxt_d().document)
469  qxt_d().document = new QTextDocument(const_cast<QxtItemDelegate*>(this));
470  qxt_d().document->setHtml(text);
471  qxt_d().document->adjustSize();
472  size = size.expandedTo(qxt_d().document->size().toSize()); // TODO: checkbox, icon, etc.
473  }
474  return size;
