133  template<typename OtherDerived>
134  inline SparseInnerVectorSet& operator=(const SparseMatrixBase<OtherDerived>& other)
135  {
136  if (IsRowMajor != ((OtherDerived::Flags&RowMajorBit)==RowMajorBit))
137  {
138  // need to transpose => perform a block evaluation followed by a big swap
139  DynamicSparseMatrix<Scalar,IsRowMajor?RowMajorBit:0> aux(other);
140  *this = aux.markAsRValue();
141  }
142  else
143  {
144  // evaluate/copy vector per vector
145  for (int j=0; j<m_outerSize.value(); ++j)
146  {
147  SparseVector<Scalar,IsRowMajor ? RowMajorBit : 0> aux(other.innerVector(j));
148  m_matrix.const_cast_derived()._data()[m_outerStart+j].swap(aux._data());
149  }
150  }
151  return *this;
152  }
153 
154  inline SparseInnerVectorSet& operator=(const SparseInnerVectorSet& other)
155  {
156  return operator=<SparseInnerVectorSet>(other);
157  }

372  inline Block(const MatrixType& matrix, int i)
373  : m_matrix(matrix),
374  // It is a row if and only if BlockRows==1 and BlockCols==MatrixType::ColsAtCompileTime,
375  // and it is a column if and only if BlockRows==MatrixType::RowsAtCompileTime and BlockCols==1,
376  // all other cases are invalid.
377  // The case a 1x1 matrix seems ambiguous, but the result is the same anyway.

379  m_startCol( (BlockRows==MatrixType::RowsAtCompileTime) && (BlockCols==1) ? i : 0),
380  m_blockRows(matrix.rows()), // if it is a row, then m_blockRows has a fixed-size of 1, so no pb to try to overwrite it
381  m_blockCols(matrix.cols()) // same for m_blockCols
382  {
383  ei_assert( (i>=0) && (
384  ((BlockRows==1) && (BlockCols==MatrixType::ColsAtCompileTime) && i<matrix.rows())
385  ||((BlockRows==MatrixType::RowsAtCompileTime) && (BlockCols==1) && i<matrix.cols())));
386  }
