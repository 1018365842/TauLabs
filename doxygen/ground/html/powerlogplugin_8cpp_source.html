205 void PowerlogThread::GetShowValue(QString label, DWORD Value, WORD Len, WORD Dot)
206 {
207  QString out;
208 
209  if (Value < 0) {
210  fileStream << "-";
211  Value = -Value;
212  }
213 
214  if(Dot==1)
215  fileStream << Value/10 << "." << Value%10; // printf("%ld.%01lu",Value/10,Value%10);
216  else if(Dot==2)
217  fileStream << Value/100 << "." << Value%100; // printf("%ld.%02lu",Value/100,Value%100);
218  else if(Dot==3)
219  fileStream << Value/1000 << "." << Value%1000; // printf("%ld.%03lu",Value/1000,Value%1000);
220  else if(Dot==4)
221  fileStream << Value/10000 << "." << Value%10000; // printf("%ld.%04lu",Value/10000,Value%10000);
222  else
223  fileStream << Value; // printf("%ld",Value);
224 
225  fileStream << out << ",";

288 void PowerlogPlugin::receiveLog()
289 {
290  if (logging) {
291  loggingThread->stopLogging();
292  logging = false;
293  cmd->action()->setText("Receive from PowerLog6S...");
294  } else {
295  QString fileName = QFileDialog::getSaveFileName(NULL, tr("Log filename"),
296  tr("PowerLog-%0.csv").arg(QDateTime::currentDateTime().toString("yyyy-MM-dd_hh-mm-ss")),
297  tr("Comma Separated Values (*.csv)"));
298  if (fileName.isEmpty())
299  return;
300 
301  loggingThread = new PowerlogThread();
302  if (loggingThread->openFile(fileName,this)) {
303  loggingThread->start();
304  cmd->action()->setText("Stop PowerLog6S reception");
305  logging = true;
306  }
307  }

331 void PowerlogPlugin::devRemoved(USBPortInfo port)
332 {
333  bool foundDevice;
334  QList<USBPortInfo> ports = USBMonitor::instance()->availableDevices();
335  foreach(USBPortInfo port, ports) {
336  if ((port.vendorID == 0x0483) && (port.productID==0x5750) &&
337  (devSerialNumber == port.serialNumber)) {
338  foundDevice = true;
339  break;
340  }
341  }
342  if (!foundDevice) {
343  devSerialNumber = QString("");
344  cmd->action()->setEnabled(false);
345  // Also stop logging in case we were logging:

348  }
