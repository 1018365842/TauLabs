330 template<typename MatrixType>
331 LU<MatrixType>::LU(const MatrixType& matrix)
332  : m_originalMatrix(matrix),
333  m_lu(matrix),
334  m_p(matrix.rows()),
335  m_q(matrix.cols())
336 {
337  const int size = matrix.diagonal().size();
338  const int rows = matrix.rows();
339  const int cols = matrix.cols();
340 
341  // this formula comes from experimenting (see "LU precision tuning" thread on the list)
342  // and turns out to be identical to Higham's formula used already in LDLt.
343  m_precision = machine_epsilon<Scalar>() * size;
344 
345  IntColVectorType rows_transpositions(matrix.rows());
346  IntRowVectorType cols_transpositions(matrix.cols());
347  int number_of_transpositions = 0;
348 
349  RealScalar biggest = RealScalar(0);
350  m_rank = size;
351  for(int k = 0; k < size; ++k)
352  {
353  int row_of_biggest_in_corner, col_of_biggest_in_corner;
354  RealScalar biggest_in_corner;
355 
356  biggest_in_corner = m_lu.corner(Eigen::BottomRight, rows-k, cols-k)
357  .cwise().abs()
358  .maxCoeff(&row_of_biggest_in_corner, &col_of_biggest_in_corner);
359  row_of_biggest_in_corner += k;
360  col_of_biggest_in_corner += k;
361  if(k==0) biggest = biggest_in_corner;
362 
363  // if the corner is negligible, then we have less than full rank, and we can finish early
364  if(ei_isMuchSmallerThan(biggest_in_corner, biggest, m_precision))
365  {
366  m_rank = k;
367  for(int i = k; i < size; i++)
368  {
369  rows_transpositions.coeffRef(i) = i;
370  cols_transpositions.coeffRef(i) = i;
371  }
372  break;
373  }
374 
375  rows_transpositions.coeffRef(k) = row_of_biggest_in_corner;
376  cols_transpositions.coeffRef(k) = col_of_biggest_in_corner;
377  if(k != row_of_biggest_in_corner) {
378  m_lu.row(k).swap(m_lu.row(row_of_biggest_in_corner));
379  ++number_of_transpositions;
380  }
381  if(k != col_of_biggest_in_corner) {
382  m_lu.col(k).swap(m_lu.col(col_of_biggest_in_corner));
383  ++number_of_transpositions;
384  }
385  if(k<rows-1)
386  m_lu.col(k).end(rows-k-1) /= m_lu.coeff(k,k);
387  if(k<size-1)
388  for(int col = k + 1; col < cols; ++col)
389  m_lu.col(col).end(rows-k-1) -= m_lu.col(k).end(rows-k-1) * m_lu.coeff(k,col);
390  }
391 
392  for(int k = 0; k < matrix.rows(); ++k) m_p.coeffRef(k) = k;
393  for(int k = size-1; k >= 0; --k)
394  std::swap(m_p.coeffRef(k), m_p.coeffRef(rows_transpositions.coeff(k)));
395 
396  for(int k = 0; k < matrix.cols(); ++k) m_q.coeffRef(k) = k;
397  for(int k = 0; k < size; ++k)
398  std::swap(m_q.coeffRef(k), m_q.coeffRef(cols_transpositions.coeff(k)));
399 
400  m_det_pq = (number_of_transpositions%2) ? -1 : 1;
401 }

409 template<typename MatrixType>
410 template<typename KernelMatrixType>
411 void LU<MatrixType>::computeKernel(KernelMatrixType *result) const
412 {
413  ei_assert(!isInvertible());
414  const int dimker = dimensionOfKernel(), cols = m_lu.cols();
415  result->resize(cols, dimker);
416 
417  /* Let us use the following lemma:
418  *
419  * Lemma: If the matrix A has the LU decomposition PAQ = LU,
420  * then Ker A = Q(Ker U).
421  *
422  * Proof: trivial: just keep in mind that P, Q, L are invertible.
423  */
424 
425  /* Thus, all we need to do is to compute Ker U, and then apply Q.
426  *
427  * U is upper triangular, with eigenvalues sorted so that any zeros appear at the end.
428  * Thus, the diagonal of U ends with exactly
429  * m_dimKer zero's. Let us use that to construct m_dimKer linearly
430  * independent vectors in Ker U.
431  */
432 
433  Matrix<Scalar, Dynamic, Dynamic, MatrixType::Options,
434  MatrixType::MaxColsAtCompileTime, MatrixType::MaxColsAtCompileTime>
435  y(-m_lu.corner(TopRight, m_rank, dimker));
436 
437  m_lu.corner(TopLeft, m_rank, m_rank)
438  .template marked<UpperTriangular>()
439  .solveTriangularInPlace(y);
440 
441  for(int i = 0; i < m_rank; ++i) result->row(m_q.coeff(i)) = y.row(i);
442  for(int i = m_rank; i < cols; ++i) result->row(m_q.coeff(i)).setZero();
443  for(int k = 0; k < dimker; ++k) result->coeffRef(m_q.coeff(m_rank+k), k) = Scalar(1);
444 }

474 template<typename MatrixType>
475 template<typename OtherDerived, typename ResultType>
476 bool LU<MatrixType>::solve(
477  const MatrixBase<OtherDerived>& b,
478  ResultType *result
479 ) const
480 {
481  /* The decomposition PAQ = LU can be rewritten as A = P^{-1} L U Q^{-1}.
482  * So we proceed as follows:
483  * Step 1: compute c = Pb.
484  * Step 2: replace c by the solution x to Lx = c. Exists because L is invertible.
485  * Step 3: replace c by the solution x to Ux = c. Check if a solution really exists.
486  * Step 4: result = Qc;
487  */
488 
489  const int rows = m_lu.rows(), cols = m_lu.cols();
490  ei_assert(b.rows() == rows);
491  const int smalldim = std::min(rows, cols);
492 
493  typename OtherDerived::PlainMatrixType c(b.rows(), b.cols());
494 
495  // Step 1
496  for(int i = 0; i < rows; ++i) c.row(m_p.coeff(i)) = b.row(i);
497 
498  // Step 2
499  m_lu.corner(Eigen::TopLeft,smalldim,smalldim).template marked<UnitLowerTriangular>()
500  .solveTriangularInPlace(
501  c.corner(Eigen::TopLeft, smalldim, c.cols()));
502  if(rows>cols)
503  {
504  c.corner(Eigen::BottomLeft, rows-cols, c.cols())
505  -= m_lu.corner(Eigen::BottomLeft, rows-cols, cols) * c.corner(Eigen::TopLeft, cols, c.cols());
506  }
507 
508  // Step 3
509  if(!isSurjective())
510  {
511  // is c is in the image of U ?
512  RealScalar biggest_in_c = m_rank>0 ? c.corner(TopLeft, m_rank, c.cols()).cwise().abs().maxCoeff() : RealScalar(0);
513  for(int col = 0; col < c.cols(); ++col)
514  for(int row = m_rank; row < c.rows(); ++row)
515  if(!ei_isMuchSmallerThan(c.coeff(row,col), biggest_in_c, m_precision))
516  return false;
517  }
518  if(m_rank>0)
519  m_lu.corner(TopLeft, m_rank, m_rank)
520  .template marked<UpperTriangular>()
521  .solveTriangularInPlace(c.corner(TopLeft, m_rank, c.cols()));
522 
523  // Step 4
524  result->resize(m_lu.cols(), b.cols());
525  for(int i = 0; i < m_rank; ++i) result->row(m_q.coeff(i)) = c.row(i);
526  for(int i = m_rank; i < m_lu.cols(); ++i) result->row(m_q.coeff(i)).setZero();
527  return true;
528 }
