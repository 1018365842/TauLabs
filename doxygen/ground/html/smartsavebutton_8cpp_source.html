64 void smartSaveButton::processOperation(QPushButton * button,bool save)
65 {
66  emit preProcessOperations();
67  if(button)
68  {
69  button->setEnabled(false);
70  button->setIcon(QIcon(":/uploader/images/system-run.svg"));
71  }
72  QTimer timer;
73  timer.setSingleShot(true);
74  bool error=false;
75  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
76  UAVObjectUtilManager* utilMngr = pm->getObject<UAVObjectUtilManager>();
77  foreach(UAVDataObject * obj,objects)
78  {
79  UAVObject::Metadata mdata= obj->getMetadata();
80  if(UAVObject::GetGcsAccess(mdata)==UAVObject::ACCESS_READONLY)
81  continue;
82  up_result=false;
83  current_object=obj;
84  for(int i=0;i<3;++i)
85  {
86  qDebug()<<"SMARTSAVEBUTTON"<<"Upload try number"<<i<<"Object"<<obj->getName();
87  connect(obj,SIGNAL(transactionCompleted(UAVObject*,bool)),this,SLOT(transaction_finished(UAVObject*, bool)));
88  connect(&timer,SIGNAL(timeout()),&loop,SLOT(quit()));
89  obj->updated();
90  timer.start(3000);
91  //qDebug()<<"begin loop";
92  loop.exec();
93  if(timer.isActive())
94  {
95  qDebug()<<"SMARTSAVEBUTTON"<<"Upload did not timeout"<<i<<"Object"<<obj->getName();
96  }
97  else
98  qDebug()<<"SMARTSAVEBUTTON"<<"Upload TIMEOUT"<<i<<"Object"<<obj->getName();
99  timer.stop();
100  disconnect(obj,SIGNAL(transactionCompleted(UAVObject*,bool)),this,SLOT(transaction_finished(UAVObject*, bool)));
101  disconnect(&timer,SIGNAL(timeout()),&loop,SLOT(quit()));
102  if(up_result)
103  break;
104  }
105  if(up_result==false)
106  {
107  qDebug()<<"SMARTSAVEBUTTON"<<"Object upload error:"<<obj->getName();
108  error=true;
109  continue;
110  }
111  sv_result=false;
112  current_objectID=obj->getObjID();
113  if(save && (obj->isSettings()))
114  {
115  for(int i=0;i<3;++i)
116  {
117  qDebug()<<"SMARTSAVEBUTTON"<<"Save try number"<<i<<"Object"<<obj->getName();
118  connect(utilMngr,SIGNAL(saveCompleted(int,bool)),this,SLOT(saving_finished(int,bool)));
119  connect(&timer,SIGNAL(timeout()),&loop,SLOT(quit()));
120  utilMngr->saveObjectToSD(obj);
121  timer.start(3000);
122  loop.exec();
123  if(timer.isActive())
124  {
125  qDebug()<<"SMARTSAVEBUTTON"<<"Saving did not timeout"<<i<<"Object"<<obj->getName();
126  }
127  else
128  qDebug()<<"SMARTSAVEBUTTON"<<"Saving TIMEOUT"<<i<<"Object"<<obj->getName();
129  timer.stop();
130  disconnect(utilMngr,SIGNAL(saveCompleted(int,bool)),this,SLOT(saving_finished(int,bool)));
131  disconnect(&timer,SIGNAL(timeout()),&loop,SLOT(quit()));
132  if(sv_result)
133  break;
134  }
135  if(sv_result==false)
136  {
137  qDebug()<<"SMARTSAVEBUTTON"<<"failed to save:"<<obj->getName();
138  error=true;
139  }
140  }
141  }
142  if(button)
143  button->setEnabled(true);
144  if(!error)
145  {
146  if(button)
147  button->setIcon(QIcon(":/uploader/images/dialog-apply.svg"));
148  emit saveSuccessfull();
149  }
150  else
151  {
152  if(button)
153  button->setIcon(QIcon(":/uploader/images/process-stop.svg"));
154  }
155  emit endOp();
156 }
