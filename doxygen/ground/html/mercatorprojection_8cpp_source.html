37 Point MercatorProjection::FromLatLngToPixel(double lat, double lng, const int &zoom)
38 {
39  Point ret;// = Point.Empty;
40 
41  lat = bound(lat, MinLatitude, MaxLatitude);
42  lng = bound(lng, MinLongitude, MaxLongitude);
43 
44  double x = (lng + 180) / 360;
45  double sinLatitude = sin(lat * M_PI / 180);
46  double y = 0.5 - log((1 + sinLatitude) / (1 - sinLatitude)) / (4 * M_PI);
47 
48  Size s = GetTileMatrixSizePixel(zoom);
49  qint64 mapSizeX = s.Width();
50  qint64 mapSizeY = s.Height();
51 
52  ret.SetX((qint64) round(bound(x * mapSizeX + 0.5, 0, mapSizeX - 1)));
53  ret.SetY((qint64) round(bound(y * mapSizeY + 0.5, 0, mapSizeY - 1)));
54 
55  return ret;
56 }

65 internals::PointLatLng MercatorProjection::FromPixelToLatLng(const qint64 &x,const qint64 &y,const int &zoom)
66 {
67  internals::PointLatLng ret;// = internals::PointLatLng.Empty;
68 
69  Size s = GetTileMatrixSizePixel(zoom);
70  double mapSizeX = s.Width();
71  double mapSizeY = s.Height();
72 
73  //Calculate the percentage distance between top and bottom, and left and right
74  double xx = (bound(x, 0, mapSizeX - 1) / mapSizeX) - 0.5;
75  double yy = 0.5 - (bound(y, 0, mapSizeY - 1) / mapSizeY);
76 
77  ret.SetLat(90 - 360 * atan(exp(-yy * 2 * M_PI)) / M_PI);
78  ret.SetLng(360 * xx);
79 
80  return ret;
81 }
