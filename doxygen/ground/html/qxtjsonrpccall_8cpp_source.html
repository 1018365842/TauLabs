127 void QxtJSONRpcCallPrivate::d_finished()
128 {
129  if (!reply->error())
130  {
131  QVariant m_=QxtJSON::parse(QString::fromUtf8(reply->readAll()));
132  if(m_.isNull()){
133  qWarning("QxtJSONRpcCall: invalid JSON received");
134  }
135 
136  QVariantMap m=m_.toMap();
137 
138  if(m["error"]!=QVariant()){
139  isFault=true;
140  result=m["error"];
141  }else{
142  result=m["result"];
143  }
144  }
145  emit pub->finished();
146 }
