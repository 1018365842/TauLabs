65 void QxtConfirmationMessagePrivate::init(const QString& message)
66 {
67  remember = false;
68  confirm = new QCheckBox(&qxt_p());
69  if (!message.isNull())
70  confirm->setText(message);
71  else
72  confirm->setText(QxtConfirmationMessage::tr("Do not show again."));
73 
74  QGridLayout* grid = qobject_cast<QGridLayout*>(qxt_p().layout());
75  QDialogButtonBox* buttons = qFindChild<QDialogButtonBox*>(&qxt_p());
76  if (grid && buttons)
77  {
78  const int idx = grid->indexOf(buttons);
79  int row, column, rowSpan, columnSpan = 0;
80  grid->getItemPosition(idx, &row, &column, &rowSpan, &columnSpan);
81  QLayoutItem* buttonsItem = grid->takeAt(idx);
82  grid->addWidget(confirm, row, column, rowSpan, columnSpan, Qt::AlignLeft | Qt::AlignTop);
83  grid->addItem(buttonsItem, ++row, column, rowSpan, columnSpan);
84  }
85 }

210 QMessageBox::StandardButton QxtConfirmationMessage::confirm(QWidget* parent,
211  const QString& title, const QString& text, const QString& confirmation,
212  QMessageBox::StandardButtons buttons, QMessageBox::StandardButton defaultButton)
213 {
214  QxtConfirmationMessage msgBox(QMessageBox::NoIcon, title, text, confirmation, QMessageBox::NoButton, parent);
215  QDialogButtonBox* buttonBox = qFindChild<QDialogButtonBox*>(&msgBox);
216  Q_ASSERT(buttonBox != 0);
217 
218  uint mask = QMessageBox::FirstButton;
219  while (mask <= QMessageBox::LastButton)
220  {
221  uint sb = buttons & mask;
222  mask <<= 1;
223  if (!sb)
224  continue;
225  QPushButton* button = msgBox.addButton((QMessageBox::StandardButton)sb);
226  // Choose the first accept role as the default
227  if (msgBox.defaultButton())
228  continue;
229  if ((defaultButton == QMessageBox::NoButton && buttonBox->buttonRole(button) == QDialogButtonBox::AcceptRole)
230  || (defaultButton != QMessageBox::NoButton && sb == uint(defaultButton)))
231  msgBox.setDefaultButton(button);
232  }
233  if (msgBox.exec() == -1)
234  return QMessageBox::Cancel;
235  return msgBox.standardButton(msgBox.clickedButton());
236 }

410 void QxtConfirmationMessage::done(int result)
411 {
412  QDialogButtonBox* buttons = qFindChild<QDialogButtonBox*>(this);
413  Q_ASSERT(buttons != 0);
414 
415  int role = buttons->buttonRole(clickedButton());
416  if (qxt_d().confirm->isChecked() &&
417  (qxt_d().remember || role != QDialogButtonBox::RejectRole))
418  {
419  qxt_d().doNotShowAgain(result);
420  }
421  QMessageBox::done(result);
