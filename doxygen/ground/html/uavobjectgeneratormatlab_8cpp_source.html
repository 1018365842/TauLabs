79 bool UAVObjectGeneratorMatlab::process_object(ObjectInfo* info, int numBytes)
80 {

82  return false;
83 
84  //Declare variables
85  QString objectName(info->name);
86  // QString objectTableName(objectName + "Objects");
87  QString objectTableName(objectName);
88  QString tableIdxName(objectName.toLower() + "Idx");
89  QString objectID(QString().setNum(info->id));
90  QString numBytesString=QString("%1").arg(numBytes);

92 
93  //=========================================================================//

97  QString instantiationFields;
98 
99  matlabInstantiationCode.append("\n\t" + tableIdxName + " = 0;\n");
100  matlabInstantiationCode.append("\t" + objectTableName + "=struct('timestamp', 0");
101  if (!info->isSingleInst) {
102  instantiationFields.append(",...\n\t\t 'instanceID', 0");
103  }
104  for (int n = 0; n < info->fields.length(); ++n) {
105  // Determine type
106  type = fieldTypeStrMatlab[info->fields[n]->type];
107  // Append field
108  if ( info->fields[n]->numElements > 1 )
109  instantiationFields.append(",...\n\t\t '" + info->fields[n]->name + "', zeros(" + QString::number(info->fields[n]->numElements, 10) + ",1)");
110  else
111  instantiationFields.append(",...\n\t\t '" + info->fields[n]->name + "', 0");
112  }
113  instantiationFields.append(");\n");
114 
115  matlabInstantiationCode.append(instantiationFields);
116  matlabInstantiationCode.append("\t" + objectTableName.toUpper() + "_OBJID=" + objectID + ";\n");
117  matlabInstantiationCode.append("\t" + objectTableName.toUpper() + "_NUMBYTES=" + numBytesString + ";\n");
118  matlabInstantiationCode.append("\t" + objectName + "FidIdx = [];\n");
119  matlabInstantiationCode.append("\n\tmultipleInstanceLookup(end+1,:) = [" + objectID + ", " + (info->isSingleInst ? "true" : "false") + "];\n");
120 
121  //==============================================================//
122  // Generate 'Switch:' code (will replace the $(SWITCHCODE) tag) //
123  //==============================================================//
124  matlabSwitchCode.append("\t\tcase " + objectTableName.toUpper() + "_OBJID\n");
125  matlabSwitchCode.append("\t\t\t" + tableIdxName + " = " + tableIdxName +" + 1;\n");

127  matlabSwitchCode.append("\t\t\t" + objectTableName + ".timestamp(" + tableIdxName + ") = timestamp; %#ok<*AGROW>\n");
128  matlabSwitchCode.append("\t\t\tbufferIdx=bufferIdx + " + objectTableName.toUpper() + "_NUMBYTES+1; %+1 is for CRC\n");
129  if(!info->isSingleInst){
130  matlabSwitchCode.append("\t\t\tbufferIdx = bufferIdx + 2; %An extra two bytes for the instance ID\n");
131  }
132  matlabSwitchCode.append("\t\t\tif " + tableIdxName + " >= length(" + objectTableName +"FidIdx) %Check to see if pre-allocated memory is exhausted\n");
133  matlabSwitchCode.append("\t\t\t\t" + objectTableName + "FidIdx(" + tableIdxName + "*2) = 0;\n");
134  matlabSwitchCode.append("\t\t\tend\n");
135 
136

169  ", " + objectName + "FidIdx + " + QString("%1").arg(currentIdx + size.toInt()*info->fields[n]->numElements - 1) + ")), '" + type + "')), "+ QString::number(info->fields[n]->numElements, 10) + ", [] );\n");
170  }
171  else{
172  allocationFields.append("\t" + objectName + "." + info->fields[n]->name + " = " +

174  ", " + objectName + "FidIdx + " + QString("%1").arg(currentIdx + size.toInt() - 1) + ")), '" + type + "'))';\n");
