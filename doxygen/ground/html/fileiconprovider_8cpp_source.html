64 QIcon FileIconProvider::icon(const QFileInfo &fileInfo)
65 {
66  const QString suffix = fileInfo.suffix();
67  QIcon icon = iconForSuffix(suffix);
68 
69  if (icon.isNull()) {
70  // Get icon from OS and store it in the cache
71 
72  // Disabled since for now we'll make sure that all icons fit with our
73  // own custom icons by returning an empty one if we don't know it.
74 #if defined(Q_WS_WIN) || defined(Q_WS_MAC)
75  // This is incorrect if the OS does not always return the same icon for the
76  // same suffix (Mac OS X), but should speed up the retrieval a lot ...
77  icon = m_systemIconProvider.icon(fileInfo);
78  if (!suffix.isEmpty())
79  registerIconOverlayForSuffix(icon, suffix);
80 #else
81  if (fileInfo.isDir()) {
82  icon = m_systemIconProvider.icon(fileInfo);
83  } else {
84  icon = m_unknownFileIcon;
85  }
86 #endif
87  }
88 
89  return icon;
90 }

107 void FileIconProvider::registerIconOverlayForSuffix(const QIcon &icon, const QString &suffix)
108 {
109  QPixmap fileIconPixmap = overlayIcon(QStyle::SP_FileIcon, icon, QSize(16, 16));
110  // delete old icon, if it exists
111  QList<QPair<QString,QIcon> >::iterator iter = m_cache.begin();
112  for (; iter != m_cache.end(); ++iter) {
113  if ((*iter).first == suffix) {
114  iter = m_cache.erase(iter);
115  break;
116  }
117  }
118 
119  QPair<QString,QIcon> newEntry(suffix, fileIconPixmap);
120  m_cache.append(newEntry);
121 }

135 QIcon FileIconProvider::iconForSuffix(const QString &suffix) const
136 {
137  QIcon icon;
138 #if defined(Q_WS_WIN) || defined(Q_WS_MAC) // On Windows and Mac we use the file system icons
139  Q_UNUSED(suffix)
140 #else
141  if (suffix.isEmpty())
142  return icon;
143 
144  QList<QPair<QString,QIcon> >::const_iterator iter = m_cache.constBegin();
145  for (; iter != m_cache.constEnd(); ++iter) {
146  if ((*iter).first == suffix) {
147  icon = (*iter).second;
148  break;
149  }
150  }
151 #endif
152  return icon;
153 }
