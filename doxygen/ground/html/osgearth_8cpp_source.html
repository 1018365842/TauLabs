118 void OsgEarthItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *style, QWidget *widget)
119 {
120  Q_UNUSED(painter);
121  Q_UNUSED(style);
122  QGLWidget *glWidget = qobject_cast<QGLWidget*>(widget);
123 
124  if (!m_renderer) {
125  m_renderer = new OsgEarthItemRenderer(this, glWidget);
126  connect(m_renderer, SIGNAL(frameReady()),
127  this, SLOT(updateView()), Qt::QueuedConnection);
128 
129  m_rendererThread = new QThread(this);
130  m_renderer->moveToThread(m_rendererThread);
131  m_rendererThread->start();
132 
133  QMetaObject::invokeMethod(m_renderer, "initScene", Qt::QueuedConnection);
134 
135  //osgEarth requires a number of frame updates to render the scene even from the cache.
136  //othervise it just looks gray if the scene is not updated.
137  QTimer *updateTimer = new QTimer(this);
138  connect(updateTimer, SIGNAL(timeout()), SLOT(updateFrame()));
139  updateTimer->start(500);
140 
141  return;
142  }
143 
144  QGLFramebufferObject *fbo = m_renderer->lastFrame();
145 
146  if (glWidget && fbo)
147  glWidget->drawTexture(boundingRect(), fbo->texture());
148 }

279 void OsgEarthItemRenderer::initScene()
280 {
281  if (m_viewer.get()) {
282  Q_ASSERT(0);
283  return; // No viewer pointer available, exit
284  }
285 
286  int w = m_currentSize.width();
287  int h = m_currentSize.height();
288 
289  QString sceneFile = m_item->resolvedSceneFile();
290  m_model = osgDB::readNodeFile(sceneFile.toStdString());
291 
292  //setup caching
293  osgEarth::MapNode *mapNode = osgEarth::MapNode::findMapNode(m_model.get());
294  if (!mapNode) {
295  qWarning() << Q_FUNC_INFO << sceneFile << " doesn't look like an osgEarth file";
296  }
297 
298  m_gw = new osgViewer::GraphicsWindowEmbedded(0,0,w,h);
299 
300  m_viewer = new osgViewer::Viewer();
301  m_viewer->setThreadingModel(osgViewer::Viewer::SingleThreaded);
302  m_viewer->setSceneData(m_model);
303  m_viewer->getDatabasePager()->setDoPreCompile(true);
304 
305  osg::Camera *camera = m_viewer->getCamera();
306  camera->setViewport(new osg::Viewport(0,0,w,h));
307  camera->setGraphicsContext(m_gw);
308  camera->setClearMask(GL_COLOR_BUFFER_BIT|GL_DEPTH_BUFFER_BIT);
309 
310  // configure the near/far so we don't clip things that are up close
311  camera->setNearFarRatio(0.00002);
312  camera->setProjectionMatrixAsPerspective(m_item->fieldOfView(), qreal(w)/h, 1.0f, 10000.0f);
313 
314  updateFrame();
315 }

317 void OsgEarthItemRenderer::updateFrame()
318 {
319  if (!m_cameraDirty || !m_viewer.get() || m_glWidget.isNull())
320  return;
321 
322  m_glWidget.data()->makeCurrent();
323 
324  //To find a camera view matrix, find placer matrixes for two points
325  //onr at requested coords and another latitude shifted by 0.01 deg
326  osgEarth::Util::ObjectPlacer placer(m_viewer->getSceneData());
327 
328  m_cameraDirty = false;
329 
330  osg::Matrixd positionMatrix;
331  placer.createPlacerMatrix(m_item->latitude(), m_item->longitude(), m_item->altitude(), positionMatrix);
332  osg::Matrixd positionMatrix2;
333  placer.createPlacerMatrix(m_item->latitude()+0.01, m_item->longitude(), m_item->altitude(), positionMatrix2);
334 
335  osg::Vec3d eye(0.0f, 0.0f, 0.0f);
336  osg::Vec3d viewVector(0.0f, 0.0f, 0.0f);
337  osg::Vec3d upVector(0.0f, 0.0f, 1.0f);
338 
339  eye = positionMatrix.preMult(eye);
340  upVector = positionMatrix.preMult(upVector);
341  upVector.normalize();
342  viewVector = positionMatrix2.preMult(viewVector) - eye;
343  viewVector.normalize();
344  viewVector *= 10.0;
345 
346  //TODO: clarify the correct rotation order,
347  //currently assuming yaw, pitch, roll
348  osg::Quat q;
349  q.makeRotate(-m_item->yaw()*M_PI/180.0, upVector);
350  upVector = q * upVector;
351  viewVector = q * viewVector;
352 
353  osg::Vec3d side = viewVector ^ upVector;
354  q.makeRotate(m_item->pitch()*M_PI/180.0, side);
355  upVector = q * upVector;
356  viewVector = q * viewVector;
357 
358  q.makeRotate(m_item->roll()*M_PI/180.0, viewVector);
359  upVector = q * upVector;
360  viewVector = q * viewVector;
361 
362  osg::Vec3d center = eye + viewVector;
363 
364 // qDebug() << "e " << eye.x() << eye.y() << eye.z();
365 // qDebug() << "c " << center.x() << center.y() << center.z();
366 // qDebug() << "up" << upVector.x() << upVector.y() << upVector.z();
367 
368  m_viewer->getCamera()->setViewMatrixAsLookAt(osg::Vec3d(eye.x(), eye.y(), eye.z()),
369  osg::Vec3d(center.x(), center.y(), center.z()),
370  osg::Vec3d(upVector.x(), upVector.y(), upVector.z()));
371 
372  {
373  QGLFramebufferObject *fbo = m_fbo[(m_lastFboNumber + 1) % FboCount];
374  QPainter fboPainter(fbo);
375  fboPainter.beginNativePainting();
376  m_viewer->frame();
377  fboPainter.endNativePainting();
378  }
379  m_glWidget.data()->doneCurrent();
380 
381  m_lastFboNumber = (m_lastFboNumber + 1) % FboCount;
382 
383  emit frameReady();
384 }
