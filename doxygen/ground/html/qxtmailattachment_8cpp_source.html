176 QByteArray QxtMailAttachment::mimeData()
177 {
178  QIODevice* c = content();
179  if (!c)
180  {
181  qWarning() << "QxtMailAttachment::mimeData(): Content not set or already output";
182  return QByteArray();
183  }
184  if (!c->isOpen() && !c->open(QIODevice::ReadOnly))
185  {
186  qWarning() << "QxtMailAttachment::mimeData(): Cannot open content for reading";
187  return QByteArray();
188  }
189 
190  QTextCodec* latin1 = QTextCodec::codecForName("latin1");
191  QByteArray rv = "Content-Type: " + qxt_d->contentType.toAscii() + "\r\nContent-Transfer-Encoding: base64\r\n";
192  foreach(const QString& r, qxt_d->extraHeaders.keys())
193  {
194  rv += qxt_fold_mime_header(r.toAscii(), extraHeader(r), latin1);
195  }
196  rv += "\r\n";
197 
198  while (!c->atEnd())
199  {
200  rv += c->read(57).toBase64() + "\r\n";
201  }
202  setContent((QIODevice*)0);
203  return rv;
204 }
