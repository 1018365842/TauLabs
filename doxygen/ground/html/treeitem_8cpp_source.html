70 /*
71  * Callback called periodically by the timer.
72  * This method checks for expired highlights and
73  * removes them if they are expired.
74  * Expired highlights are restored.
75  */
76 void HighLightManager::checkItemsExpired()
77 {
78  // Lock to ensure thread safety
79  QMutexLocker locker(&m_listMutex);
80 
81  // Get a mutable iterator for the list
82  QMutableLinkedListIterator<TreeItem*> iter(m_itemsList);
83 
84  // This is the timestamp to compare with
85  QTime now = QTime::currentTime();
86 
87  // Loop over all items, check if they expired.
88  while(iter.hasNext())
89  {
90  TreeItem* item = iter.next();
91  if(item->getHiglightExpires() < now)
92  {
93  // If expired, call removeHighlight
94  item->removeHighlight();
95 
96  // Remove from list since it is restored.
97  iter.remove();
98  }
99  }
100 }

185 /*
186  * Called after a value has changed to trigger highlightning of tree item.
187  */
188 void TreeItem::setHighlight(bool highlight) {
189  m_highlight = highlight;
190  m_changed = false;
191  if (highlight) {
192  // Update the expires timestamp
193  m_highlightExpires = QTime::currentTime().addMSecs(m_highlightTimeMs);
194 
195  // Add to highlightmanager
196  if(m_highlightManager->add(this))
197  {
198  // Only emit signal if it was added
199  emit updateHighlight(this);
200  }
201  }
202  else if(m_highlightManager->remove(this))
203  {
204  // Only emit signal if it was removed
205  emit updateHighlight(this);
206  }
207 
208  // If we have a parent, call recursively to update highlight status of parents.
209  // This will ensure that the root of a leaf that is changed also is highlighted.
210  // Only updates that really changes values will trigger highlight of parents.
211  if(m_parent)
212  {
213  m_parent->setHighlight(highlight);
214  }
215 }
